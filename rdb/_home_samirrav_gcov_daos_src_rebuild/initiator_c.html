401     struct rebuild_tgt_pool_tracker *rpt = arg;
402     struct rebuild_puller       *puller;
403     unsigned int            idx;
404
405 TF     while (daos_fail_check(DAOS_REBUILD_TGT_REBUILD_HANG))
406         ABT_thread_yield();
407
408     tls = rebuild_pool_tls_lookup(rpt->rt_pool_uuid,
409                       rpt->rt_rebuild_ver);
410     D_ASSERT(tls != NULL);
411     D_ASSERT(rpt->rt_pullers != NULL);
412     idx = dss_get_module_info()->dmi_tgt_id;
413     puller = &rpt->rt_pullers[idx];
414     puller->rp_ult_running = 1;
415     while (1) {
416         struct rebuild_one  *rdone;
417         struct rebuild_one  *tmp;
418         d_list_t        rebuild_list;
419         int         rc = 0;
420
421         D_INIT_LIST_HEAD(&rebuild_list);
422         ABT_mutex_lock(puller->rp_lock);
423         d_list_for_each_entry_safe(rdone, tmp, &puller->rp_one_list,
424                        ro_list) {
425             d_list_move_tail(&rdone->ro_list, &rebuild_list);
426             puller->rp_inflight++;
427         }
428         ABT_mutex_unlock(puller->rp_lock);
429
430         d_list_for_each_entry_safe(rdone, tmp, &rebuild_list, ro_list) {
431             d_list_del_init(&rdone->ro_list);
432 TF             if (!rpt->rt_abort) {
433                 rc = rebuild_dkey(rpt, rdone);
434                 D_DEBUG(DB_REBUILD, DF_UOID" rebuild dkey %d %s"
435                     " rc %d tag %d rpt %p\n",
436                     DP_UOID(rdone->ro_oid),
437                     (int)rdone->ro_dkey.iov_len,
438                     (char *)rdone->ro_dkey.iov_buf, rc,
439                     idx, rpt);
440             }
441
442             D_ASSERT(puller->rp_inflight > 0);
443             puller->rp_inflight--;
444
445 TF             if (rc == -DER_NOSPACE) {
446                 /* If there are no space on current VOS, let's
447                  * hang the rebuild ULT on the current xstream,
448                  * and waitting for the space is reclaimed or
449                  * the drive is replaced.
450                  *
451                  * If the space is reclaimed, then it will
452                  * resume the rebuild ULT.
453                  * If the drive is replaced, then it will
454                  * abort the current rebuild by other process.
455                  */
456                 rebuild_hang();
457                 ABT_thread_yield();
458                 D_DEBUG(DB_REBUILD, "%p rebuild got back.\n",
459                     rpt);
460                 rc = 0;
461                 /* Added it back to rdone */
462                 ABT_mutex_lock(puller->rp_lock);
463                 d_list_add_tail(&rdone->ro_list,
464                         &puller->rp_one_list);
465                 ABT_mutex_unlock(puller->rp_lock);
466                 continue;
467             }
468
469             /* Ignore nonexistent error because puller could race
470              * with user's container destroy:
471              * - puller got the container+oid from a remote scanner
472              * - user destroyed the container
473              * - puller try to open container or pulling data
474              *   (nonexistent)
475              * This is just a workaround...
476              */
477 TF             if (tls->rebuild_pool_status == 0 && rc != 0 &&
478                 rc != -DER_NONEXIST) {
479                 tls->rebuild_pool_status = rc;
480                 rpt->rt_abort = 1;
481             }
482             /* XXX If rebuild fails, Should we add this back to
483              * dkey list
484              */
485             rebuild_one_destroy(rdone);
486         }
487
488         /* check if it should exist */
489         ABT_mutex_lock(puller->rp_lock);
490 TF         if (d_list_empty(&puller->rp_one_list) && rpt->rt_finishing) {
491             ABT_mutex_unlock(puller->rp_lock);
492             break;
493         }
494         /* XXX exist if rebuild is aborted */
495         ABT_mutex_unlock(puller->rp_lock);
496         ABT_thread_yield();
497     }
498
499     ABT_mutex_lock(puller->rp_lock);
500     ABT_cond_signal(puller->rp_fini_cond);
501     puller->rp_ult_running = 0;
502     ABT_mutex_unlock(puller->rp_lock);
503     rpt_put(rpt);
504 }

585 static int
586 rebuild_one_queue(struct rebuild_iter_obj_arg *iter_arg, daos_unit_oid_t *oid,
587           daos_key_t *dkey, daos_epoch_t dkey_eph, daos_iod_t *iods,

589           daos_sg_list_t *sgls, uint32_t version)
590 {
591     struct rebuild_puller       *puller;
592     struct rebuild_tgt_pool_tracker *rpt = iter_arg->rpt;
593     struct rebuild_one      *rdone = NULL;
594     unsigned int            ephs_cnt = 0;
595     bool                inline_copy = true;
596     int             i;
597     int             rc;
598
599     D_DEBUG(DB_REBUILD, "rebuild dkey %d %s iod nr %d dkey_eph "DF_U64"\n",
600         (int)dkey->iov_buf_len, (char *)dkey->iov_buf, iod_eph_total,
601         dkey_eph);
602
603 TF     if (iod_eph_total == 0 || rpt->rt_rebuild_ver <= version) {
604         D_DEBUG(DB_REBUILD, "No need rebuild eph_total %d version %u"

606             rpt->rt_rebuild_ver);
607         return 0;
608     }
609
610     D_ALLOC_PTR(rdone);
611 TF     if (rdone == NULL)
612         return -DER_NOMEM;
613
614     D_ALLOC_ARRAY(rdone->ro_iods, iod_eph_total);
615 TF     if (rdone->ro_iods == NULL)
616         D_GOTO(free, rc = -DER_NOMEM);
617
618     D_ALLOC_ARRAY(rdone->ro_ephs, iod_eph_total);
619     D_ALLOC_ARRAY(rdone->ro_ephs_keys, iod_eph_total);

629 TF         if (sgls[i].sg_nr == 0 || sgls[i].sg_iovs == NULL) {
630             inline_copy = false;

632         }

635 TF             if (sgls[i].sg_iovs[j].iov_len == 0 ||
636                 sgls[i].sg_iovs[j].iov_buf == NULL) {
637                 inline_copy = false;
638                 break;
639             }
640         }
641
642 TF         if (!inline_copy)
643             break;
644     }
645
646 TF     for (i = 0; i < iod_eph_total; i++) {
647 TF         if (akey_ephs[i] != DAOS_EPOCH_MAX) {
648             /* Pack punched epoch here */
649             rdone->ro_ephs[ephs_cnt] = akey_ephs[i];
650             rc = daos_iov_copy(&rdone->ro_ephs_keys[ephs_cnt],
651                        &iods[i].iod_name);
652 TF             if (rc)
653                 D_GOTO(free, rc);
654
655             ephs_cnt++;
656             D_DEBUG(DB_REBUILD, "punched iod idx %d akey %d %s"
657                 " ephs "DF_U64" ephs_cnt %d\n", i,
658                 (int)iods[i].iod_name.iov_len,
659                 (char *)iods[i].iod_name.iov_buf,
660                 akey_ephs[i], ephs_cnt);
661         }
662
663 TF         if (iods[i].iod_nr == 0)
664             continue;
665
666 TF         if (iods[i].iod_size == 0)
667             rc = punch_iod_pack(rdone, &iods[i]);
668         else
669             rc = rw_iod_pack(rdone, &iods[i],
670 TF                      inline_copy ? &sgls[i] : NULL);
671     }
672
673     rdone->ro_ephs_num = ephs_cnt;
674     rdone->ro_max_eph = dkey_eph;
675     rdone->ro_version = version;
676     puller = &rpt->rt_pullers[iter_arg->tgt_idx];
677 TF     if (puller->rp_ult == NULL) {
678         /* Create puller ULT thread, and destroy ULT until
679          * rebuild finish in rebuild_fini().
680          */
681         D_ASSERT(puller->rp_ult_running == 0);
682         D_DEBUG(DB_REBUILD, "create rebuild dkey ult %d\n",
683             iter_arg->tgt_idx);
684         rpt_get(rpt);
685         rc = dss_ult_create(rebuild_one_ult, rpt, DSS_ULT_REBUILD,
686                     iter_arg->tgt_idx, PULLER_STACK_SIZE,
687                     &puller->rp_ult);
688 TF         if (rc) {
689             rpt_put(rpt);
690             D_GOTO(free, rc);
691         }
692     }
693
694     D_INIT_LIST_HEAD(&rdone->ro_list);
695     rc = daos_iov_copy(&rdone->ro_dkey, dkey);
696 TF     if (rc != 0)
697         D_GOTO(free, rc);
698
699     rdone->ro_oid = *oid;
700     uuid_copy(rdone->ro_cont_uuid, iter_arg->cont_uuid);
701
702     D_DEBUG(DB_REBUILD, DF_UOID" %p dkey %d %s rebuild on idx %d max eph"
703         " "DF_U64" iod_num %d\n", DP_UOID(rdone->ro_oid), rdone,
704         (int)dkey->iov_len, (char *)dkey->iov_buf, iter_arg->tgt_idx,
705         rdone->ro_max_eph, rdone->ro_iod_num);
706
707     ABT_mutex_lock(puller->rp_lock);
708     d_list_add_tail(&rdone->ro_list, &puller->rp_one_list);
709     ABT_mutex_unlock(puller->rp_lock);
710
711 free:
712 TF     if (rc != 0 && rdone != NULL)
713         rebuild_one_destroy(rdone);
714
715     return rc;
716 }
