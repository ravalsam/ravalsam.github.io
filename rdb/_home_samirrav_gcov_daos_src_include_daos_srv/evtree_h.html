474     EVT_ITER_COVERED    = (1 << 2),
475     /** Skip visible holes (Only valid with EVT_ITER_VISIBLE) */
476     EVT_ITER_SKIP_HOLES = (1 << 3),
477     /** Reverse iterator (ordered iterator only) */
478     EVT_ITER_REVERSE    = (1 << 4),
479     /* If either EVT_ITER_VISIBLE or EVT_ITER_COVERED are set,
480      * evt_iter_probe will calculate and cache visible extents and iterate
481      * through the cached extents.   Each rectangle will be marked as
482      * visible or covered.  The partial bit will be set if the rectangle
483      * returned differs from what is in the tree.  The state of this type
484      * of iterator is unaffected by tree insertion or deletion so reprobe
485      * isn't necessary.  One should probably not use the embedded iterator
486      * when holding such across yield boundaries.
487      * If neither flag is set, all rectangles in tree that intersect the
488      * search rectangle, including punched extents, are returned.
489      */
490
491     /** The iterator is for purge operation */
492     EVT_ITER_FOR_PURGE  = (1 << 5),
493     /** The iterator is for rebuild scan */
494     EVT_ITER_FOR_REBUILD    = (1 << 6),
495 };
