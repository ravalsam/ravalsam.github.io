794  *          identifier with the the low 96 bits untouched and the
795  *          DAOS private bits (the high 32 bits) encoded.
796  * \param[in]   ofeat   Feature bits specific to object
797  * \param[in]   cid Class Identifier
798  */
799 static inline void
800 daos_obj_generate_id(daos_obj_id_t *oid, daos_ofeat_t ofeats,
801              daos_oclass_id_t cid)
802 {
803     uint64_t hdr = cid;
804
805     oid->hi &= 0x00000000ffffffff;
806     /**
807      * | Upper bits contain
808      * | DAOS_OVERSION_BITS version        |
809      * | DAOS_OFEAT_BITS object features   |
810      * | DAOS_OCLASS_BITS object class     |
811      * | 96-bit for upper layer ...        |
812      */
813     hdr <<= DAOS_OCLASS_SHIFT;
814     hdr |= 0x1ULL << DAOS_OVERSION_SHIFT;
815     hdr |= ((uint64_t)ofeats << DAOS_OFEAT_SHIFT);
816     oid->hi |= hdr;
817 }
