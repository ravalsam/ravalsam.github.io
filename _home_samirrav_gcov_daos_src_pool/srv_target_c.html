514 static void
515 aggregate_pool_space(struct daos_pool_space *agg_ps,
516              struct daos_pool_space *ps)
517 {
518     int i;
519     bool    first;
520
521     D_ASSERT(agg_ps && ps);
522
523 TF     if (ps->ps_ntargets == 0) {
524         D_ERROR("Skip emtpy space info\n");
525         return;
526     }
527
528     first = (agg_ps->ps_ntargets == 0);
529     agg_ps->ps_ntargets += ps->ps_ntargets;
530
531 TF     for (i = DAOS_MEDIA_SCM; i < DAOS_MEDIA_MAX; i++) {
532         agg_ps->ps_space.s_total[i] += ps->ps_space.s_total[i];
533         agg_ps->ps_space.s_free[i] += ps->ps_space.s_free[i];
534
535 TF         if (agg_ps->ps_free_max[i] < ps->ps_free_max[i])
536             agg_ps->ps_free_max[i] = ps->ps_free_max[i];
537 TF         if (agg_ps->ps_free_min[i] > ps->ps_free_min[i] || first)
538             agg_ps->ps_free_min[i] = ps->ps_free_min[i];
539
540         agg_ps->ps_free_mean[i] = agg_ps->ps_space.s_free[i] /
541                       agg_ps->ps_ntargets;
542     }
543 }

583 {
584     struct dss_coll_stream_args *reduce = vin;
585     struct dss_stream_arg_type  *streams = reduce->csa_streams;
586     struct dss_module_info      *info = dss_get_module_info();
587     int              tid = info->dmi_tgt_id;

590     struct ds_pool_child        *pool_child;
591     struct daos_pool_space      *x_ps = &x_arg->qxa_space;

611
612 TF     for (i = DAOS_MEDIA_SCM; i < DAOS_MEDIA_MAX; i++) {
613         x_ps->ps_free_max[i] = x_ps->ps_space.s_free[i];
614         x_ps->ps_free_min[i] = x_ps->ps_space.s_free[i];
615     }
616 out:

658 {
659     struct pool_tgt_connect_in  *in = crt_req_get(rpc);
660     struct pool_tgt_connect_out *out = crt_reply_get(rpc);

666

668         DP_UUID(in->tci_uuid), rpc, DP_UUID(in->tci_hdl));
669
670     hdl = ds_pool_hdl_lookup(in->tci_hdl);
671 TF     if (hdl != NULL) {
672 TF         if (hdl->sph_capas == in->tci_capas) {
673             D_DEBUG(DF_DSMS, DF_UUID": found compatible pool "
674                 "handle: hdl="DF_UUID" capas="DF_U64"\n",
675                 DP_UUID(in->tci_uuid), DP_UUID(in->tci_hdl),
676                 hdl->sph_capas);

680                 DF_UUID" capas="DF_U64"\n",

682                 hdl->sph_capas);

686         D_GOTO(out, rc);
687     }
688
689     D_ALLOC_PTR(hdl);
690 TF     if (hdl == NULL)
691         D_GOTO(out, rc = -DER_NOMEM);
692
693     /* Init the pool map */
694 TF     if (rpc->cr_co_bulk_hdl != NULL) {
695         struct pool_buf     *buf;
696         daos_iov_t       iov = { 0 };
697         daos_sg_list_t       sgl;
698
699         sgl.sg_nr = 1;
700         sgl.sg_nr_out = 1;
701         sgl.sg_iovs = &iov;
702         rc = crt_bulk_access(rpc->cr_co_bulk_hdl, daos2crt_sg(&sgl));
703 TF         if (rc != 0) {
704             D_ERROR(DF_UUID": crt_bulk_access failed, rc %d.\n",
705                 DP_UUID(in->tci_uuid), rc);
706             D_GOTO(out, rc);
707         }
708
709         buf = iov.iov_buf;
710         D_ASSERT(buf != NULL);
711         rc = pool_map_create(buf, in->tci_map_version, &map);
712 TF         if (rc != 0) {
713             D_ERROR(DF_UUID" failed to create pool map: %d\n",
714                 DP_UUID(in->tci_uuid), rc);
715             D_GOTO(out, rc);
716         }
717     }
718
719     arg.pca_map = map;
720     arg.pca_map_version = in->tci_map_version;
721     arg.pca_need_group = 0;
722
723     rc = ds_pool_lookup_create(in->tci_uuid, &arg, &pool);
724 TF     if (rc != 0) {
725         D_FREE(hdl);
726         D_GOTO(out, rc);
727     }
728
729     uuid_copy(hdl->sph_uuid, in->tci_hdl);
730     hdl->sph_capas = in->tci_capas;
731     hdl->sph_pool = pool;
732
733     rc = pool_hdl_add(hdl);
734 TF     if (rc != 0) {
735         ds_pool_put(pool);
736         D_GOTO(out, rc);
737     }
738
739 TF     if (pool->sp_iv_ns == NULL) {
740         rc = ds_pool_iv_ns_update(pool, in->tci_master_rank,
741                       &in->tci_iv_ctxt,
742                       in->tci_iv_ns_id);
743 TF         if (rc) {
744             D_ERROR("attach iv ns failed rc %d\n", rc);
745             D_GOTO(out, rc);
746         }
747     }
748
749     rc = pool_tgt_query(pool, &out->tco_space);
750 out:
751 TF     if (rc != 0 && map != NULL)
752         pool_map_decref(map);
753
754 TF     out->tco_rc = (rc == 0 ? 0 : 1);
755     D_DEBUG(DF_DSMS, DF_UUID": replying rpc %p: %d (%d)\n",
756         DP_UUID(in->tci_uuid), rpc, out->tco_rc, rc);
757     crt_reply_send(rpc);
758 }

847 int
848 ds_pool_tgt_map_update(struct ds_pool *pool, struct pool_buf *buf,
849                unsigned int map_version)
850 {
851     struct pool_map *map = NULL;
852     int     rc = 0;
853
854 TF     if (buf != NULL) {
855         rc = pool_map_create(buf, map_version, &map);
856 TF         if (rc != 0) {
857             D_ERROR(DF_UUID" failed to create pool map: %d\n",
858                 DP_UUID(pool->sp_uuid), rc);
859             D_GOTO(out, rc);
860         }
861     }
862
863     ABT_rwlock_wrlock(pool->sp_lock);
864 TF     if (pool->sp_map_version < map_version ||
865         (pool->sp_map_version == map_version &&
866          (map != NULL && pool->sp_map == NULL))) {
867 TF         if (map != NULL) {
868             struct pool_map *tmp = pool->sp_map;
869
870             pool->sp_map = map;
871             map = tmp;
872         }
873
874         D_DEBUG(DF_DSMS, DF_UUID
875             ": changed cached map version: %u -> %u\n",
876             DP_UUID(pool->sp_uuid), pool->sp_map_version,
877             map_version);
878
879         pool->sp_map_version = map_version;
880         rc = dss_task_collective(update_child_map, pool, 0);
881         D_ASSERT(rc == 0);
882 TF     } else if (pool->sp_map != NULL &&
883            pool_map_get_version(pool->sp_map) < map_version &&
884            map != NULL) {
885         struct pool_map *tmp = pool->sp_map;
886
887         rc = pl_map_update(pool->sp_uuid, map,
888 TF                    pool->sp_map != NULL ? false : true);
889 TF         if (rc != 0) {
890             ABT_rwlock_unlock(pool->sp_lock);
891             D_ERROR(DF_UUID": failed to update pl_map: %d\n",
892                 DP_UUID(pool->sp_uuid), rc);
893             D_GOTO(out, rc);
894         }
895
896         /* drop the stale map */
897         pool->sp_map = map;
898         map = tmp;
899     } else {
900         D_WARN("Ignore old map version: cur=%u, input=%u pool %p\n",
