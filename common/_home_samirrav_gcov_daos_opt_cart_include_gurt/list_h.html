613 /**
614  * Circular queue definitions:
615  *
616  * A circular queue is headed by a structure defined by the CIRCLEQ_HEAD()
617  * macro. This structure contains a pair of pointers, one to the first element
618  * in the circular queue and the other to the last element in the circular
619  * queue. The elements are doubly linked so that an arbitrary element can be
620  * removed without traversing the queue. New elements can be added to the queue
621  * after an existing element, before an existing element, at the head of
622  * the queue, or at the end of the queue.
623  *
624  * A CIRCLEQ_HEAD structure is declared as follows:
625  */
626 #define D_CIRCLEQ_HEAD(name, type)                  \
627 struct name {                               \
628     struct type *cqh_first;     /* first element */     \
629     struct type *cqh_last;      /* last element */      \
630 }
631
632 #define D_CIRCLEQ_HEAD_INIT(head)               \
633     { (void *)&head, (void *)&head }
634
635 #define d_circleq_entry(type)                       \
636 struct {                                \
637     struct type *cqe_next;      /* next element */      \
638     struct type *cqe_prev;      /* previous element */      \
639 }

643  */
644 #define D_CIRCLEQ_INIT(head) do {                   \
645     (head)->cqh_first = (void *)(head);             \
646     (head)->cqh_last  = (void *)(head);             \
647 } while (0)

649 #define D_CIRCLEQ_INSERT_AFTER(head, listelm, elm, field) do {      \
650     (elm)->field.cqe_next = (listelm)->field.cqe_next;      \
651     (elm)->field.cqe_prev = (listelm);              \
