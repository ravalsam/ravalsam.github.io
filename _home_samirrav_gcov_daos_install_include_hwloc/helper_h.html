461 /** \brief Find the depth of cache objects matching cache depth and type.
462  *
463  * Return the depth of the topology level that contains cache objects
464  * whose attributes match \p cachedepth and \p cachetype. This function
465  * intends to disambiguate the case where hwloc_get_type_depth() returns
466  * ::HWLOC_TYPE_DEPTH_MULTIPLE.
467  *
468  * If no cache level matches, ::HWLOC_TYPE_DEPTH_UNKNOWN is returned.
469  *
470  * If \p cachetype is ::HWLOC_OBJ_CACHE_UNIFIED, the depth of the
471  * unique matching unified cache level is returned.
472  *
473  * If \p cachetype is ::HWLOC_OBJ_CACHE_DATA or ::HWLOC_OBJ_CACHE_INSTRUCTION,
474  * either a matching cache, or a unified cache is returned.
475  *
476  * If \p cachetype is \c -1, it is ignored and multiple levels may
477  * match. The function returns either the depth of a uniquely matching
478  * level or ::HWLOC_TYPE_DEPTH_MULTIPLE.
479  */
480 static __hwloc_inline int
481 hwloc_get_cache_type_depth (hwloc_topology_t topology,
482                 unsigned cachelevel, hwloc_obj_cache_type_t cachetype)
483 {
484   int depth;
485   int found = HWLOC_TYPE_DEPTH_UNKNOWN;
486   for (depth=0; ; depth++) {
487     hwloc_obj_t obj = hwloc_get_obj_by_depth(topology, depth, 0);
488 TF     if (!obj)
489       break;
490 TF     if (obj->type != HWLOC_OBJ_CACHE || obj->attr->cache.depth != cachelevel)
491       /* doesn't match, try next depth */
492       continue;
493 TF     if (cachetype == (hwloc_obj_cache_type_t) -1) {
494 TF       if (found != HWLOC_TYPE_DEPTH_UNKNOWN) {
495     /* second match, return MULTIPLE */
496         return HWLOC_TYPE_DEPTH_MULTIPLE;
497       }
498       /* first match, mark it as found */
499       found = depth;
500       continue;
501     }
502 TF     if (obj->attr->cache.type == cachetype || obj->attr->cache.type == HWLOC_OBJ_CACHE_UNIFIED)
503       /* exact match (either unified is alone, or we match instruction or data), return immediately */
504       return depth;
505   }
506   /* went to the bottom, return what we found */
507   return found;
508 }

699 /** \brief Distribute \p n items over the topology under \p roots
700  *
701  * Array \p set will be filled with \p n cpusets recursively distributed
702  * linearly over the topology under objects \p roots, down to depth \p until
703  * (which can be INT_MAX to distribute down to the finest level).
704  *
705  * \p n_roots is usually 1 and \p roots only contains the topology root object
706  * so as to distribute over the entire topology.
707  *
708  * This is typically useful when an application wants to distribute \p n
709  * threads over a machine, giving each of them as much private cache as
710  * possible and keeping them locally in number order.
711  *
712  * The caller may typically want to also call hwloc_bitmap_singlify()
713  * before binding a thread so that it does not move at all.
714  *
715  * \p flags should be 0 or a OR'ed set of ::hwloc_distrib_flags_e.
716  *
717  * \note This function requires the \p roots objects to have a CPU set.
718  *
719  * \note This function replaces the now deprecated hwloc_distribute()
720  * and hwloc_distributev() functions.
721  */
722 static __hwloc_inline int
723 hwloc_distrib(hwloc_topology_t topology,
724           hwloc_obj_t *roots, unsigned n_roots,
725           hwloc_cpuset_t *set,
726           unsigned n,
727           unsigned until, unsigned long flags)
728 {
729   unsigned i;
730   unsigned tot_weight;
731   unsigned given, givenweight;
732   hwloc_cpuset_t *cpusetp = set;
733
734 TF   if (flags & ~HWLOC_DISTRIB_FLAG_REVERSE) {
735     errno = EINVAL;
736     return -1;
737   }
738
739   tot_weight = 0;
740 TF   for (i = 0; i < n_roots; i++)
741 TF     if (roots[i]->cpuset)
742       tot_weight += hwloc_bitmap_weight(roots[i]->cpuset);
743
744 TF   for (i = 0, given = 0, givenweight = 0; i < n_roots; i++) {
745     unsigned chunk, weight;
746 TF     hwloc_obj_t root = roots[flags & HWLOC_DISTRIB_FLAG_REVERSE ? n_roots-1-i : i];
747     hwloc_cpuset_t cpuset = root->cpuset;
748 TF     if (!cpuset)
749       continue;
750     weight = hwloc_bitmap_weight(cpuset);
751 TF     if (!weight)
752       continue;
753     /* Give to root a chunk proportional to its weight.
754      * If previous chunks got rounded-up, we may get a bit less. */
755     chunk = (( (givenweight+weight) * n  + tot_weight-1) / tot_weight)
756           - ((  givenweight         * n  + tot_weight-1) / tot_weight);
757 TF     if (!root->arity || chunk <= 1 || root->depth >= until) {
758       /* We can't split any more, put everything there.  */
759 TF       if (chunk) {
760     /* Fill cpusets with ours */
761     unsigned j;
762 TF     for (j=0; j < chunk; j++)
763       cpusetp[j] = hwloc_bitmap_dup(cpuset);
764       } else {
765     /* We got no chunk, just merge our cpuset to a previous one
766      * (the first chunk cannot be empty)
767      * so that this root doesn't get ignored.
768      */
769     assert(given);
770     hwloc_bitmap_or(cpusetp[-1], cpusetp[-1], cpuset);
771       }
772     } else {
773       /* Still more to distribute, recurse into children */
774       hwloc_distrib(topology, root->children, root->arity, cpusetp, chunk, until, flags);
775     }
776     cpusetp += chunk;
777     given += chunk;
778     givenweight += weight;
779   }
780
781   return 0;
782 }

935  * indexes in the output node set, even if a full CPU set is given in input.

938  * as a single memory node, and the following behavior is used:
939  * If \p cpuset is empty, \p nodeset will be emptied as well.
940  * Otherwise \p nodeset will be entirely filled.
