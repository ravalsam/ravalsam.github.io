67  */

73
74 TF     if (term == RDB_NIL_TERM)

76     /*

274 static int
275 rdb_tx_append(struct rdb_tx *tx, struct rdb_tx_op *op)
276 {

482 static int
483 rdb_tx_apply_create(struct rdb *db, uint64_t index, rdb_oid_t parent,
484             daos_iov_t *key, enum rdb_kvs_class class)
485 {
486     daos_iov_t  value;
487     rdb_oid_t   oid_class;
488     rdb_oid_t   oid_number;
489     rdb_oid_t   oid;
490     int     rc;
491
492     /* Convert the KVS class into the object ID class. */
493     rc = rdb_oid_class(class, &oid_class);
494 TF     if (rc != 0) {
495         D_ERROR(DF_DB": unknown KVS class %x: %d\n", DP_DB(db), class,
496             rc);
497         return rc;
498     }
499
500     /* Does the KVS already exist? */
501     daos_iov_set(&value, NULL, sizeof(rdb_oid_t));
502     rc = rdb_lc_lookup(db->d_lc, index, parent, key, &value);
503 TF     if (rc == 0) {
504         return -DER_EXIST;
505 TF     } else if (rc != -DER_NONEXIST) {
506         D_ERROR(DF_DB": failed to check KVS existence: %d\n", DP_DB(db),
507             rc);
508         return rc;
509     }
510
511     /* Allocate an object for the new KVS. */
512     daos_iov_set(&value, &oid_number, sizeof(oid_number));
513     rc = rdb_lc_lookup(db->d_lc, index, RDB_LC_ATTRS, &rdb_lc_oid_next,
514                &value);
515 TF     if (rc == -DER_NONEXIST) {
516         oid_number = RDB_LC_OID_NEXT_INIT;
517         D_DEBUG(DB_MD, DF_DB": initialized rdb_lc_oid_next to "DF_U64
518             "\n", DP_DB(db), oid_number);
519 TF     } else if (rc != 0) {
520         D_ERROR(DF_DB": failed to look up next object number: %d\n",
521             DP_DB(db), rc);
522         return rc;
523     }
524 TF     if ((oid_number & RDB_OID_CLASS_MASK) != 0) {
525         D_ERROR(DF_DB": invalid next object number: "DF_X64"\n",
526             DP_DB(db), oid_number);
527         return -DER_IO;
528     }
529     oid = oid_class | oid_number;
530     oid_number += 1;
531
532     /* Update the next object number. */
533     rc = rdb_lc_update(db->d_lc, index, RDB_LC_ATTRS, 1 /* n */,
534                &rdb_lc_oid_next, &value);
535 TF     if (rc != 0) {
536         D_ERROR(DF_DB": failed to update next object number"DF_X64
537             ": %d\n", DP_DB(db), oid_number, rc);
538         return rc;
539     }
540
541     /* Update the key in the parent object. */
542     daos_iov_set(&value, &oid, sizeof(oid));
543     rc = rdb_lc_update(db->d_lc, index, parent, 1 /* n */, key, &value);
544 TF     if (rc != 0) {
545         D_ERROR(DF_DB": failed to update parent KVS: %d\n", DP_DB(db),
546             rc);
547         return rc;
548     }
549
550     return 0;
551 }

553 static int
554 rdb_tx_apply_destroy(struct rdb *db, uint64_t index, rdb_oid_t parent,
555              daos_iov_t *key)
556 {
557     daos_iov_t  value;
558     rdb_oid_t   oid;
559     int     rc;
560
561     /* Does the KVS exist? */
562     daos_iov_set(&value, &oid, sizeof(oid));
563     rc = rdb_lc_lookup(db->d_lc, index, parent, key, &value);
564 TF     if (rc != 0) {
565 TF         if (rc != -DER_NONEXIST)
566             D_ERROR(DF_DB": failed to check KVS existence: %d\n",

569     }

575             DP_DB(db), parent, rc);

577     }

581 TF     if (rc != 0) {
582         D_ERROR(DF_DB": failed to punch KVS "DF_X64": %d\n", DP_DB(db),
583             oid, rc);
584         return rc;
585     }
586
587     return 0;
588 }

616 static int
617 rdb_tx_apply_op(struct rdb *db, uint64_t index, struct rdb_tx_op *op)
618 {
619     struct rdb_kvs *kvs = NULL;
620     rdb_path_t  victim_path;
621     int     rc;
622
623     D_DEBUG(DB_TRACE, DF_DB": "DF_TX_OP"\n", DP_DB(db), DP_TX_OP(op));
624
625 TF     if (op->dto_opc != RDB_TX_CREATE_ROOT &&
626         op->dto_opc != RDB_TX_DESTROY_ROOT) {
627         /* Look up the KVS. */
628         rc = rdb_kvs_lookup(db, &op->dto_kvs, index, true /* alloc */,
629                     &kvs);
630 TF         if (rc != 0)
631             goto out;
632     }
633
634     /* If destroying a KVS, prepare a path to it. */
635 TF     if (op->dto_opc == RDB_TX_DESTROY_ROOT) {
636         rc = rdb_path_init(&victim_path);
637 TF         if (rc != 0)
638             goto out_kvs;
639         rc = rdb_path_push(&victim_path, &rdb_path_root_key);
640 TF         if (rc != 0)
641             goto out_victim_path;
642 TF     } else if (op->dto_opc == RDB_TX_DESTROY) {
643         rc = rdb_path_clone(&op->dto_kvs, &victim_path);
644 TF         if (rc != 0)
645             goto out_kvs;
646         rc = rdb_path_push(&victim_path, &op->dto_key);
647 TF         if (rc != 0)
648             goto out_victim_path;
649     }
650
651     switch (op->dto_opc) {
652 T     case RDB_TX_CREATE_ROOT:
653         rc = rdb_tx_apply_create(db, index, RDB_LC_ATTRS, &rdb_lc_root,
654                      op->dto_attr->dsa_class);
655         break;
656 T     case RDB_TX_CREATE:
657         rc = rdb_tx_apply_create(db, index, kvs->de_object,
658                      &op->dto_key, op->dto_attr->dsa_class);
659         break;
660 T     case RDB_TX_DESTROY_ROOT:
661         rc = rdb_tx_apply_destroy(db, index, RDB_LC_ATTRS,
662                       &rdb_lc_root);
663         break;
664 T     case RDB_TX_DESTROY:
665         rc = rdb_tx_apply_destroy(db, index, kvs->de_object,
666                       &op->dto_key);
667         break;
668 T     case RDB_TX_UPDATE:
669         rc = rdb_tx_apply_update(db, index, kvs->de_object,
670                      &op->dto_key, &op->dto_value);
671         break;
672 T     case RDB_TX_DELETE:
673         rc = rdb_tx_apply_delete(db, index, kvs->de_object,
674                      &op->dto_key);
675         break;
676 T     default:
677         D_ERROR(DF_DB": unknown update operation %u\n",
678             DP_DB(db), op->dto_opc);
679         rc = -DER_IO;
680     }
681 TF     if (rc != 0)
682         goto out_victim_path;
683
684 TF     if (op->dto_opc == RDB_TX_DESTROY_ROOT ||
685         op->dto_opc == RDB_TX_DESTROY) {
686         struct rdb_kvs *victim;
687         int     rc_tmp;
688
689         /*
690          * Evict the matching rdb_kvs object (if any). Since this TX
691          * destroys victim_path, no other TXs can query or update
692          * victim_path or its child KVSs (if any). Hence, until this TX
693          * releases the lock for victim_path after the rdb_tx_commit()
694          * call returns, no other TXs will look up victim_path in the
695          * rdb_kvs cache.
696          */
697         rc_tmp = rdb_kvs_lookup(db, &victim_path, index,
698                     false /* alloc */, &victim);
699 TF         if (rc_tmp == 0) {
700             D_DEBUG(DB_TRACE, DF_DB": evicting kvs %p\n",
701                 DP_DB(db), victim);
702             rdb_kvs_evict(db, victim);
703             rdb_kvs_put(db, victim);
704         }
705     }
706
707 out_victim_path:
708 TF     if (op->dto_opc == RDB_TX_DESTROY_ROOT ||
709         op->dto_opc == RDB_TX_DESTROY)
710         rdb_path_fini(&victim_path);
711 out_kvs:
712 TF     if (kvs != NULL)
713         rdb_kvs_put(db, kvs);
714 out:
715     return rc;
716 }

718 /* Is "error" deterministic? */
719 static inline bool
720 rdb_tx_deterministic_error(int error)
721 {
722     return error == -DER_NONEXIST || error == -DER_MISMATCH ||
723            error == -DER_INVAL || error == -DER_NO_PERM;
724 }
725
726 /*
727  * Apply an entry and return the error only if a nondeterministic error
728  * happens. This function tries to discard index if an error occurs.
729  */
730 int
731 rdb_tx_apply(struct rdb *db, uint64_t index, const void *buf, size_t len,
732          void *result)
733 {
734     const void     *p = buf;
735     int     rc = 0;
736
737     D_DEBUG(DB_TRACE, DF_DB": applying "DF_U64": buf=%p len="DF_U64"\n",
738         DP_DB(db), index, buf, len);
739
740 TF     while (p < buf + len) {
741         struct rdb_tx_op    op;
742         ssize_t         n;
743
744         n = rdb_tx_op_decode(p, buf + len - p, &op);
745 TF         if (n < 0) {
746             D_ERROR(DF_DB": invalid entry format: buf=%p len="DF_U64
747                 " p=%p\n", DP_DB(db), buf, len, p);
748             rc = n;
749             break;
750         }
751         rc = rdb_tx_apply_op(db, index, &op);
752 TF         if (rc != 0) {
753 TF             if (!rdb_tx_deterministic_error(rc))
754                 D_ERROR(DF_DB": failed to apply entry "DF_U64
755                     " op %u <%td, %zd>: %d\n", DP_DB(db),
756                     index, op.dto_opc, p - buf, n, rc);
757             break;
758         }
759         p += n;
760     }
761
762     /*
763      * If an error occurs, empty the rdb_kvs cache (to evict any rdb_kvs
764      * objects corresponding to KVSs created by this TX) and discard all
765      * updates in index. Don't bother with undoing the exact set of changes
766      * made by this TX, as nondeterministic errors must be rare and
767      * deterministic errors can be easily avoided by rdb callers.
768      */
769 TF     if (rc != 0) {
770         int rc_tmp;
771
772         rdb_kvs_cache_evict(db->d_kvss);
773         rc_tmp = rdb_lc_discard(db->d_lc, index, index);
774 TF         if (rc_tmp != 0) {
775             D_ERROR(DF_DB": failed to discard entry "DF_U64": %d\n",
776                 DP_DB(db), index, rc_tmp);
777 TF             if (rdb_tx_deterministic_error(rc))
778                 return rc_tmp;
779             else
780                 return rc;
781         }
782     }
783
784 TF     if (rc != 0 && !rdb_tx_deterministic_error(rc))
785         return rc;
786
787     /*
788      * Report the deterministic error to the result buffer, if there is
789      * one, and consider this entry applied.
790      */
791 TF     if (result != NULL)
792         *(int *)result = rc;
793
794     return 0;
795 }

818 /**
819  * Look up the value of \a key in \a kvs.
820  *
821  * If \a value->iov_len is nonzero and different from the actual value length,
822  * then -DER_MISMATCH is returned and \a value outputs the actual value.
823  *
824  * \param[in]       tx  transaction
825  * \param[in]       kvs path to KVS
826  * \param[in]       key key
827  * \param[in,out]   value   value
828  *
829  * \retval -DER_NOTLEADER   not current leader
830  * \retval -DER_MISMATCH    unexpected value length
831  */
832 int
833 rdb_tx_lookup(struct rdb_tx *tx, const rdb_path_t *kvs, const daos_iov_t *key,
834           daos_iov_t *value)
835 {
836     struct rdb     *db = tx->dt_db;
837     struct rdb_kvs *s;
838     int     rc;
839
840     rc = rdb_tx_query_pre(tx, kvs, &s);
841 TF     if (rc != 0)
842         return rc;
843     rc = rdb_lc_lookup(db->d_lc, db->d_applied, s->de_object,
844                (daos_iov_t *)key, value);
845     rdb_tx_query_post(tx, s);
846     return rc;
847 }
848
849 /**
850  * Perform a probe-and-fetch operation on \a kvs.
851  *
852  * If \a value->iov_len is nonzero and different from the actual value length,
853  * then -DER_MISMATCH is returned and \a value outputs the actual value.
854  *
855  * \param[in]       tx  transaction
856  * \param[in]       kvs path to KVS
857  * \param[in]       opc probe operation
858  * \param[in]       key_in  input key
859  * \param[out]      key_out output key
860  * \param[in,out]   value   value
861  *
862  * \retval -DER_NOTLEADER   not current leader
863  * \retval -DER_MISMATCH    unexpected value length
