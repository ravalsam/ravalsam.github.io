37 /*
38  * daos_iov_t encoding/decoding utilities
39  *
40  * These functions convert between a daos_iov_t object and a byte stream in a
41  * buffer. The format of such a byte stream is:
42  *
43  *   size_head (rdb_iov_size_t)
44  *   data
45  *   size_tail (rdb_iov_size_t)
46  *
47  * size_head and size_tail are identical, both indicates the size of data,
48  * which equals iov_len of the corresponding daos_iov_t object. The two sizes
49  * allows decoding from the tail as well as from the head.
50  */
51
52 typedef uint32_t rdb_iov_size_t;
53
54 /* Maximal buf_len and len of an iov */
55 const daos_size_t rdb_iov_max = (rdb_iov_size_t)-1LL;
56
57 /* If buf is NULL, then just calculate and return the length required. */
58 size_t
59 rdb_encode_iov(const daos_iov_t *iov, void *buf)
60 {
61     size_t len = sizeof(rdb_iov_size_t) * 2 + iov->iov_len;
62
63     D_ASSERTF(iov->iov_len <= rdb_iov_max, DF_U64"\n", iov->iov_len);
64     D_ASSERTF(iov->iov_buf_len <= rdb_iov_max, DF_U64"\n",
65           iov->iov_buf_len);
66 TF     if (buf != NULL) {
67         void *p = buf;
68
69         /* iov_len (head) */
70         *(rdb_iov_size_t *)p = iov->iov_len;
71         p += sizeof(rdb_iov_size_t);
72         /* iov_buf */
73         memcpy(p, iov->iov_buf, iov->iov_len);
74         p += iov->iov_len;
75         /* iov_len (tail) */
76         *(rdb_iov_size_t *)p = iov->iov_len;
77         p += sizeof(rdb_iov_size_t);
78         D_ASSERTF(p - buf == len, "%td == %zu\n", p - buf, len);
79     }
80     return len;
81 }

83 /* Returns the number of bytes processed or -DER_IO if the content is bad. */
84 ssize_t
85 rdb_decode_iov(const void *buf, size_t len, daos_iov_t *iov)

105 TF     if (v.iov_len != 0) {

107             D_ERROR("truncated iov_buf: %zu < %zu\n", buf + len - p,
108                 v.iov_len);
109             return -DER_IO;
110         }
111         v.iov_buf = (void *)p;
112         p += v.iov_len;
113     }
114     /* iov_len (tail) */
115 TF     if (p + sizeof(rdb_iov_size_t) > buf + len) {
116         D_ERROR("truncated iov_len (tail): %zu < %zu\n", buf + len - p,
117             sizeof(rdb_iov_size_t));

121         D_ERROR("inconsistent iov_lens: "DF_U64" != %u\n",

124     }
125     p += sizeof(rdb_iov_size_t);

274 static inline int
275 rdb_vos_fetch_check(daos_iov_t *value, daos_iov_t *value_orig)
276 {
277     /*
278      * An emtpy value represents nonexistence. Keep the caller value intact
279      * in this case.
280      */
281 TF     if (value->iov_len == 0) {

288      * necessary for compatibility reasons.)

290 TF     if (value_orig->iov_len > 0 && value->iov_len != value_orig->iov_len)
291         return -DER_MISMATCH;
