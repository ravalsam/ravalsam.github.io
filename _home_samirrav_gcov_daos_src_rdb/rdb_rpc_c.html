359 /* Daemon ULT for processing RPC replies */
360 void
361 rdb_recvd(void *arg)
362 {
363     struct rdb *db = arg;
364
365     D_DEBUG(DB_MD, DF_DB": recvd starting\n", DP_DB(db));
366     for (;;) {
367         struct rdb_raft_rpc    *rrpc = NULL;
368         bool            stop;
369
370         ABT_mutex_lock(db->d_mutex);
371         for (;;) {
372             stop = db->d_stop;
373 TF             if (!d_list_empty(&db->d_replies)) {
374                 rrpc = d_list_entry(db->d_replies.next,
375                             struct rdb_raft_rpc,
376                             drc_entry);
377                 d_list_del_init(&rrpc->drc_entry);
378                 break;
379             }
380 TF             if (stop)
381                 break;
382             ABT_cond_wait(db->d_replies_cv, db->d_mutex);
383         }
384         ABT_mutex_unlock(db->d_mutex);
385 TF         if (rrpc == NULL) {
386             D_ASSERT(stop);
387             /* The queue is empty and we are asked to stop. */
388             break;
389         }
390         /*
391          * The queue has pending replies. If we are asked to stop, skip
392          * the processing but still free the RPCs until the queue
393          * become empty.
394          */
395 TF         if (!stop)
396             rdb_raft_process_reply(db, rrpc->drc_rpc);
397         rdb_raft_free_request(db, rrpc->drc_rpc);
398         rdb_free_raft_rpc(rrpc);
399         ABT_thread_yield();
400     }
401     D_DEBUG(DB_MD, DF_DB": recvd stopping\n", DP_DB(db));
402 }
