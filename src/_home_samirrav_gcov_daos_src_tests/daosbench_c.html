488 {
489     struct a_ioreq      *ioreq;
490     int         i;
491     int         rc;
492     char            *valbuf = NULL;
493
494
495     rc = daos_eq_poll(eq, 0, DAOS_EQ_WAIT, test->t_naios,
496               events);
497     DBENCH_CHECK(rc, "Failed to poll event queue");
498     assert(rc <= test->t_naios - naios);
499
500 TF     if (fetch_flag && t_validate) {
501         D_ALLOC(valbuf, test->t_val_bufsize);
502 TF         if (valbuf == NULL)

507 TF     for (i = 0; i < rc; i++) {
508         ioreq = (struct a_ioreq *)
509               ((char *) events[i] -
510                (char *) (&((struct a_ioreq *) 0)->ev));
511
512         DBENCH_CHECK(ioreq->ev.ev_error,
513                  "Failed to transfer (%lu, %lu)",

515                  ioreq->iod.iod_recxs->rx_nr);

521
522 TF         if (fetch_flag && t_validate) {
523             kv_set_value(test, valbuf, comm_world_rank,
524                      ioreq->r_index, value);
525 TF             if (ioreq->val_iov.iov_len != test->t_val_bufsize ||
526                 memcmp(ioreq->val_iov.iov_buf, valbuf,
527                    test->t_val_bufsize)) {
528                 kv_print_value(test, ioreq->val_iov.iov_buf);
529                 kv_print_value(test, valbuf);

532                        ioreq->dkey_buf, ioreq->akey_buf,
533                        ioreq->r_index,

535                        test->t_val_bufsize);
536             }
537         }
538     }
539     D_FREE(valbuf);
540     DBENCH_INFO("Found %d completed AIOs (%d free %d busy)",
541             rc, naios, test->t_naios - naios);

1375 {
