457 TF         if (max != 0) {
458             memcpy(rgs, rsrvd_dma->brd_regions, max * size);
459             D_FREE(rsrvd_dma->brd_regions);
460         }
461
462         rsrvd_dma->brd_regions = rgs;
463         rsrvd_dma->brd_rg_max = new_cnt;
464     }
465
466     rsrvd_dma->brd_regions[cnt].brr_chk = chk;
467     rsrvd_dma->brd_regions[cnt].brr_pg_idx = chk_pg_idx;
468     rsrvd_dma->brd_regions[cnt].brr_off = off;
469     rsrvd_dma->brd_regions[cnt].brr_end = end;
470     rsrvd_dma->brd_rg_cnt++;
471     return 0;
472 }
473
474 /* Convert offset of @biov into memory pointer */
475 static int
476 dma_map_one(struct bio_desc *biod, struct bio_iov *biov,
477         struct bio_copy_args *arg)
478 {
479     struct bio_rsrvd_region *last_rg;
480     struct bio_dma_buffer *bdb;
481     struct bio_dma_chunk *chk = NULL;
482     uint64_t off, end;
483     unsigned int pg_cnt, pg_off, chk_pg_idx;
484     int rc;
485
486     D_ASSERT(arg == NULL);
487     D_ASSERT(biov && biov->bi_data_len != 0);
488
489 TF     if (bio_addr_is_hole(&biov->bi_addr)) {
490         biov->bi_buf = NULL;
491         return 0;
492     }
493
494 TF     if (biov->bi_addr.ba_type == DAOS_MEDIA_SCM) {
495         struct umem_instance *umem = biod->bd_ctxt->bic_umem;
496         umem_id_t ummid;
497
498         ummid.pool_uuid_lo = biod->bd_ctxt->bic_pmempool_uuid;
499         ummid.off = bio_iov2off(biov);
500
501         biov->bi_buf = umem_id2ptr(umem, ummid);
502         return 0;
503     }
504
505     D_ASSERT(biov->bi_addr.ba_type == DAOS_MEDIA_NVME);
506     bdb = iod_dma_buf(biod);
507
508     off = bio_iov2off(biov);
509     end = bio_iov2off(biov) + biov->bi_data_len;
510     pg_cnt = ((end + BIO_DMA_PAGE_SZ - 1) >> BIO_DMA_PAGE_SHIFT) -
511             (off >> BIO_DMA_PAGE_SHIFT);
512     pg_off = off & ((uint64_t)BIO_DMA_PAGE_SZ - 1);
513
514     /*
515      * For huge IOV, we'll bypass our per-xstream DMA buffer cache and
516      * allocate chunk from the SPDK reserved huge pages directly, this
517      * kind of huge chunk will be freed immediately on I/O completion.
518      *
519      * We assume the contiguous huge IOV is quite rare, so there won't
520      * be high contention over the SPDK huge page cache.
521      */
522 TF     if (pg_cnt > bio_chk_sz) {
523         chk = dma_alloc_chunk(pg_cnt);
524 TF         if (chk == NULL)
525             return -DER_NOMEM;
526
527         rc = iod_add_chunk(biod, chk);
528 TF         if (rc) {
529             dma_free_chunk(chk);
530             return rc;
531         }
532         biov->bi_buf = chk->bdc_ptr + pg_off;
533         chk_pg_idx = 0;
534
535         D_DEBUG(DB_IO, "Huge chunk:%p[%p], cnt:%u, off:%u\n",
536             chk, chk->bdc_ptr, pg_cnt, pg_off);
537
538         goto add_region;
539     }
540
541     last_rg = iod_last_region(biod);
542
543     /* First, try consecutive reserve from the last reserved region */
544 TF     if (last_rg) {
545         uint64_t cur_pg, prev_pg_start, prev_pg_end;
546
547         D_DEBUG(DB_IO, "Last region %p:%d ["DF_U64","DF_U64")\n",
548             last_rg->brr_chk, last_rg->brr_pg_idx,
549             last_rg->brr_off, last_rg->brr_end);
550
551         chk = last_rg->brr_chk;
552         chk_pg_idx = last_rg->brr_pg_idx;
553         D_ASSERT(chk_pg_idx < bio_chk_sz);
554
555         prev_pg_start = last_rg->brr_off >> BIO_DMA_PAGE_SHIFT;
556         prev_pg_end = last_rg->brr_end >> BIO_DMA_PAGE_SHIFT;
557         cur_pg = off >> BIO_DMA_PAGE_SHIFT;
558         D_ASSERT(prev_pg_start <= prev_pg_end);
559
560         /* Consecutive in page */
561 TF         if (cur_pg == prev_pg_end) {
562             chk_pg_idx += (prev_pg_end - prev_pg_start);
563             biov->bi_buf = chunk_reserve(chk, chk_pg_idx, pg_cnt,
564                              pg_off);
565 TF             if (biov->bi_buf != NULL) {
566                 D_DEBUG(DB_IO, "Consecutive reserve %p.\n",
567                     biov->bi_buf);
568                 last_rg->brr_end = end;
569                 return 0;
570             }
571         }
572     }
573
574     /* Try to reserve from the last DMA chunk in io descriptor */
575 TF     if (chk != NULL) {
576         chk_pg_idx = chk->bdc_pg_idx;
577         biov->bi_buf = chunk_reserve(chk, chk_pg_idx, pg_cnt, pg_off);
578 TF         if (biov->bi_buf != NULL) {
579             D_DEBUG(DB_IO, "Last chunk reserve %p.\n",
580                 biov->bi_buf);
581             goto add_region;
582         }
583     }
584
585     /*
586      * Try to reserve the DMA buffer from the 'current chunk' of the
587      * per-xstream DMA buffer. It could be different with the last chunk
588      * in io descripotr, because dma_map_one() may yield in the future.
589      */
590 TF     if (bdb->bdb_cur_chk != NULL && bdb->bdb_cur_chk != chk) {
591         chk = bdb->bdb_cur_chk;
592         chk_pg_idx = chk->bdc_pg_idx;
593         biov->bi_buf = chunk_reserve(chk, chk_pg_idx, pg_cnt, pg_off);
594 TF         if (biov->bi_buf != NULL) {
595             D_DEBUG(DB_IO, "Current chunk reserve %p.\n",
596                 biov->bi_buf);
597             goto add_chunk;
598         }
599     }
600
601     /*
602      * Switch to another idle chunk, if there isn't any idle chunk
603      * available, grow buffer.
604      */
605     chk = chunk_get_idle(bdb, biod);
606 TF     if (chk == NULL)
607         return -DER_OVERFLOW;
608
609     bdb->bdb_cur_chk = chk;
610     chk_pg_idx = chk->bdc_pg_idx;
611
612     D_ASSERT(chk_pg_idx == 0);
613     biov->bi_buf = chunk_reserve(chk, chk_pg_idx, pg_cnt, pg_off);
614 TF     if (biov->bi_buf != NULL) {
615         D_DEBUG(DB_IO, "New chunk reserve %p.\n", biov->bi_buf);
616         goto add_chunk;
617     }
618
619     return -DER_OVERFLOW;
620
621 add_chunk:
622     rc = iod_add_chunk(biod, chk);
623 TF     if (rc) {
624         /* Revert the reservation in chunk */
625         D_ASSERT(chk->bdc_pg_idx >= pg_cnt);
626         chk->bdc_pg_idx -= pg_cnt;
627         return rc;
628     }
629 add_region:
630     return iod_add_region(biod, chk, chk_pg_idx, off, end);
631 }

633 static void
634 rw_completion(void *cb_arg, int err)
635 {
636     struct bio_desc *biod = cb_arg;
637
638     ABT_mutex_lock(biod->bd_mutex);
639
640     D_ASSERT(biod->bd_inflights > 0);
641     biod->bd_inflights--;
642 TF     if (biod->bd_result == 0 && err != 0)
643         biod->bd_result = daos_errno2der(-err);
644
645 TF     if (biod->bd_inflights == 0 && biod->bd_dma_issued)
646         ABT_cond_broadcast(biod->bd_dma_done);
647
648     ABT_mutex_unlock(biod->bd_mutex);
649 }
650
651 static void
652 dma_rw(struct bio_desc *biod, bool prep)
653 {
654     struct spdk_io_channel  *channel;
655     struct spdk_blob    *blob;
656     struct bio_rsrvd_dma    *rsrvd_dma = &biod->bd_rsrvd;
657     struct bio_rsrvd_region *rg;
658     struct bio_xs_context   *xs_ctxt;
659     uint64_t         pg_idx, pg_cnt, pg_end;
660     void            *payload, *pg_rmw = NULL;
661     bool             rmw_read = (prep && biod->bd_update);
662     unsigned int         pg_off;
663     int          i;
664
665     D_ASSERT(biod->bd_ctxt->bic_xs_ctxt);
666     xs_ctxt = biod->bd_ctxt->bic_xs_ctxt;
667     blob = biod->bd_ctxt->bic_blob;
668     channel = xs_ctxt->bxc_io_channel;
669     D_ASSERT(blob != NULL && channel != NULL);
670
671     /* Bypass NVMe I/O, used by daos_perf for performance evaluation */
672 TF     if (daos_io_bypass & IOBP_NVME)
673         return;
674
675     D_DEBUG(DB_IO, "DMA start, blob:%p, update:%d, rmw:%d\n",
676         blob, biod->bd_update, rmw_read);
677
678     biod->bd_inflights = 0;
679     biod->bd_dma_issued = 0;
680     biod->bd_result = 0;
681
682 TF     for (i = 0; i < rsrvd_dma->brd_rg_cnt; i++) {
683         rg = &rsrvd_dma->brd_regions[i];
684
685         D_ASSERT(rg->brr_chk != NULL);
686         pg_idx = rg->brr_off >> BIO_DMA_PAGE_SHIFT;
687         payload = rg->brr_chk->bdc_ptr +
688             (rg->brr_pg_idx << BIO_DMA_PAGE_SHIFT);
689
690 TF         if (!rmw_read) {
691             pg_cnt = (rg->brr_end + BIO_DMA_PAGE_SZ - 1) >>
692                     BIO_DMA_PAGE_SHIFT;
693             D_ASSERT(pg_cnt > pg_idx);
694             pg_cnt -= pg_idx;
695
696             ABT_mutex_lock(biod->bd_mutex);
697             biod->bd_inflights++;
698             ABT_mutex_unlock(biod->bd_mutex);
699
700             D_DEBUG(DB_IO, "%s blob:%p payload:%p, "
701                 "pg_idx:"DF_U64", pg_cnt:"DF_U64"\n",
702                 biod->bd_update ? "Write" : "Read",
703                 blob, payload, pg_idx, pg_cnt);
704
705 TF             if (biod->bd_update)
706                 spdk_blob_io_write(blob, channel, payload,
707                            pg_idx, pg_cnt,
708                            rw_completion, biod);
709             else
710                 spdk_blob_io_read(blob, channel, payload,
711                           pg_idx, pg_cnt,
712                           rw_completion, biod);
713             continue;
714         }
715
716         /*
717          * Since DAOS doesn't support partial overwrite yet, we don't
718          * do RMW for partial update, only zeroing the page instead.
719          */
720         pg_off = rg->brr_off & ((uint64_t)BIO_DMA_PAGE_SZ - 1);
721
722 TF         if (pg_off != 0 && payload != pg_rmw) {
723             D_DEBUG(DB_IO, "Front partial blob:%p payload:%p, "
724                 "pg_idx:"DF_U64" pg_off:%d\n",
725                 blob, payload, pg_idx, pg_off);
726
727             memset(payload, 0, BIO_DMA_PAGE_SZ);
728             pg_rmw = payload;
729         }
730
731         pg_end = rg->brr_end >> BIO_DMA_PAGE_SHIFT;
732         D_ASSERT(pg_end >= pg_idx);
733         payload += (pg_end - pg_idx) << BIO_DMA_PAGE_SHIFT;
734         pg_off = rg->brr_end & ((uint64_t)BIO_DMA_PAGE_SZ - 1);
735
736 TF         if (pg_off != 0 && payload != pg_rmw) {
737             D_DEBUG(DB_IO, "Rear partial blob:%p payload:%p, "
738                 "pg_idx:"DF_U64" pg_off:%d\n",
739                 blob, payload, pg_idx, pg_off);
740
741             memset(payload, 0, BIO_DMA_PAGE_SZ);
742             pg_rmw = payload;
743         }
744     }
745
746 TF     if (xs_ctxt->bxc_xs_id == -1) {
747         D_DEBUG(DB_IO, "Self poll completion, blob:%p\n", blob);
748         xs_poll_completion(xs_ctxt, &biod->bd_inflights);
749     } else {
750         ABT_mutex_lock(biod->bd_mutex);
751         biod->bd_dma_issued = 1;
752 TF         if (biod->bd_inflights != 0)
753             ABT_cond_wait(biod->bd_dma_done, biod->bd_mutex);
754         ABT_mutex_unlock(biod->bd_mutex);
755     }
756
757     D_DEBUG(DB_IO, "DMA done, blob:%p, update:%d, rmw:%d\n",
758         blob, biod->bd_update, rmw_read);
759 }

778 static int
779 copy_one(struct bio_desc *biod, struct bio_iov *biov,
780      struct bio_copy_args *arg)
781 {
782     d_sg_list_t *sgl;
783     void        *addr = biov->bi_buf;
784     ssize_t      size = biov->bi_data_len;
785     uint16_t     media = biov->bi_addr.ba_type;
786
787     D_ASSERT(arg->ca_sgl_idx < arg->ca_sgl_cnt);
788     sgl = &arg->ca_sgls[arg->ca_sgl_idx];
789
790 TF     while (arg->ca_iov_idx < sgl->sg_nr) {
791         d_iov_t *iov;
792         ssize_t nob, buf_len;
793
794         iov = &sgl->sg_iovs[arg->ca_iov_idx];
795 TF         buf_len = biod->bd_update ? iov->iov_len : iov->iov_buf_len;
796
797 TF         if (buf_len <= arg->ca_iov_off) {
798             D_ERROR("Invalid iov[%d] "DF_U64"/"DF_U64" %d\n",
799                 arg->ca_iov_idx, arg->ca_iov_off,
800                 buf_len, biod->bd_update);
801             return -DER_INVAL;
802         }
803
804         nob = min(size, buf_len - arg->ca_iov_off);
805 TF         if (addr != NULL) {
806             D_DEBUG(DB_IO, "bio copy %p size %zd\n",
807                 addr, nob);
808             bio_memcpy(biod, media, addr, iov->iov_buf +
809                     arg->ca_iov_off, nob);
810             addr += nob;
811         } else {
812             /* fetch on hole */
813             D_ASSERT(!biod->bd_update);
814         }
815
816         arg->ca_iov_off += nob;
817 TF         if (!biod->bd_update) {
818             /* the first population for fetch */
819 TF             if (arg->ca_iov_off == nob)
820                 sgl->sg_nr_out++;
821
822             iov->iov_len = arg->ca_iov_off;
823             /* consumed an iov, move to the next */
824 TF             if (iov->iov_len == iov->iov_buf_len) {
825                 arg->ca_iov_off = 0;
826                 arg->ca_iov_idx++;
827             }
828         } else {
829             /* consumed an iov, move to the next */
830 TF             if (arg->ca_iov_off == iov->iov_len) {
831                 arg->ca_iov_off = 0;
832                 arg->ca_iov_idx++;
833             }
834         }
835
836         size -= nob;
837 TF         if (size == 0)
838             return 0;
839     }
840
841     D_DEBUG(DB_TRACE, "Consumed all iovs, "DF_U64" bytes left\n", size);
842     return -DER_OVERFLOW;
843 }

845 static void
846 dma_drop_iod(struct bio_dma_buffer *bdb)
847 {
848     D_ASSERT(bdb->bdb_active_iods > 0);
849     bdb->bdb_active_iods--;
850
851     ABT_mutex_lock(bdb->bdb_mutex);
852     ABT_cond_broadcast(bdb->bdb_wait_iods);
853     ABT_mutex_unlock(bdb->bdb_mutex);
854 }
855
856 int
857 bio_iod_prep(struct bio_desc *biod)
858 {
859     struct bio_dma_buffer *bdb;
860     int rc, retry_cnt = 0;
861
862 TF     if (biod->bd_buffer_prep)
863         return -DER_INVAL;
864
865 retry:
866     rc = iterate_biov(biod, dma_map_one, NULL);
867 TF     if (rc) {
868         /*
869          * To avoid deadlock, held buffers need be released
870          * before waiting for other active IODs.
871          */
872         iod_release_buffer(biod);
873
874 TF         if (!biod->bd_retry)
875             return rc;
876
877         biod->bd_retry = 0;
878         bdb = iod_dma_buf(biod);
879 TF         if (!bdb->bdb_active_iods) {
880             D_ERROR("Per-xstream DMA buffer isn't large enough "
881                 "to satisfy large IOD %p\n", biod);
882             return rc;
883         }
884
885         D_DEBUG(DB_IO, "IOD %p waits for active IODs. %d\n",
886             biod, retry_cnt++);
887
888         ABT_mutex_lock(bdb->bdb_mutex);
889         ABT_cond_wait(bdb->bdb_wait_iods, bdb->bdb_mutex);
890         ABT_mutex_unlock(bdb->bdb_mutex);
891
892         D_DEBUG(DB_IO, "IOD %p finished waiting. %d\n",
893             biod, retry_cnt);
894
895         goto retry;
896     }
897     biod->bd_buffer_prep = 1;
898
899     /* All SCM IOVs, no DMA transfer prepared */
900 TF     if (biod->bd_rsrvd.brd_rg_cnt == 0)
901         return 0;
902
903     bdb = iod_dma_buf(biod);
904     bdb->bdb_active_iods++;
905
906     rc = ABT_mutex_create(&biod->bd_mutex);
907 TF     if (rc != ABT_SUCCESS) {
908         rc = -DER_NOMEM;
909         goto failed;
910     }
911
912     rc = ABT_cond_create(&biod->bd_dma_done);
