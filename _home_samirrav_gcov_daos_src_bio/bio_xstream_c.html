442     bs_opts = nvme_glb.bd_bs_opts;
443     /*
444      * A little bit hacke here, we store a UUID in the 16 bytes 'bstype'
445      * and use it as the block device ID.
446      */
447     D_ASSERT(SPDK_BLOBSTORE_TYPE_LENGTH == 16);
448 TF     if (bs_uuid == NULL)
449         strncpy(bs_opts.bstype.bstype, "", SPDK_BLOBSTORE_TYPE_LENGTH);
450     else
451         memcpy(bs_opts.bstype.bstype, bs_uuid,
452                SPDK_BLOBSTORE_TYPE_LENGTH);
453
454     common_prep_arg(&cp_arg);
455 TF     if (create)
456         spdk_bs_init(bs_dev, &bs_opts, common_bs_cb, &cp_arg);
457     else
458         spdk_bs_load(bs_dev, &bs_opts, common_bs_cb, &cp_arg);
459     xs_poll_completion(ctxt, &cp_arg.cca_inflights);
460
461 TF     if (cp_arg.cca_rc != 0) {
462         D_CDEBUG(bs_uuid == NULL, DB_IO, DLOG_ERR,
463              "%s blobstore failed %d\n", create ? "init" : "load",
464              cp_arg.cca_rc);
465         return NULL;
466     }
467
468     D_ASSERT(cp_arg.cca_bs != NULL);
469     return cp_arg.cca_bs;
470 }

487 static int
488 create_bio_bdev(struct bio_xs_context *ctxt, struct spdk_bdev *bdev)
489 {
490     struct bio_bdev         *d_bdev;
491     struct spdk_blob_store      *bs = NULL;
492     struct spdk_bs_type      bstype;
493     struct smd_nvme_device_info  info;
494     uuid_t               bs_uuid;
495     int              rc;
496     bool                 new_bs = false;
497
498     D_ALLOC_PTR(d_bdev);
499 TF     if (d_bdev == NULL) {
500         D_ERROR("failed to allocate bio_bdev\n");
501         return -DER_NOMEM;
502     }
503     D_INIT_LIST_HEAD(&d_bdev->bb_link);
504
505     /* Try to load blobstore without specifying 'bstype' first */
506     bs = load_blobstore(ctxt, bdev, NULL, false);
507 TF     if (bs == NULL) {
508         /* Create blobstore if it wasn't created before */
509         uuid_generate(bs_uuid);
510         bs = load_blobstore(ctxt, bdev, &bs_uuid, true);
511 TF         if (bs == NULL) {
512             D_ERROR("Failed to create blobstore on dev: "
513                 ""DF_UUID"\n", DP_UUID(bs_uuid));
514             rc = -DER_INVAL;
515             goto error;
516         }
517         new_bs = true;
518     }
519
520     /* Get the 'bstype' (device ID) of blobstore */
521     bstype = spdk_bs_get_bstype(bs);
522     memcpy(bs_uuid, bstype.bstype, sizeof(bs_uuid));
523     D_DEBUG(DB_MGMT, "%s :"DF_UUID"\n",
524         new_bs ? "Created new blobstore" : "Loaded blobstore",
525         DP_UUID(bs_uuid));
526
527     rc = unload_blobstore(ctxt, bs);
528 TF     if (rc != 0) {
529         D_ERROR("Unable to unload blobstore\n");
530         goto error;
531     }
532
533     /* Find the initial xstream count per device */
534     rc = smd_nvme_get_device(bs_uuid, &info);
535 TF     if (rc == 0) {
536         d_bdev->bb_xs_cnt = info.ndi_xs_cnt;
537 TF     } else if (rc == -DER_NONEXIST) {
538         /* device not present in table, first xstream mapped to dev */
539         d_bdev->bb_xs_cnt = 0;
540     } else {
541         D_ERROR("Unable to get dev info for "DF_UUID"\n",
542             DP_UUID(bs_uuid));
543         goto error;
544     }
545     D_DEBUG(DB_MGMT, "Initial xstream count for "DF_UUID" set at %d\n",
546         DP_UUID(bs_uuid), d_bdev->bb_xs_cnt);
547
548     d_bdev->bb_bdev = bdev;
549     uuid_copy(d_bdev->bb_uuid, bs_uuid);
550     d_list_add(&d_bdev->bb_link, &nvme_glb.bd_bdevs);
551
552     return 0;
553
554 error:
555     D_FREE(d_bdev);
556     return rc;
557 }

577 static void
578 put_bio_blobstore(struct bio_blobstore *bb, struct bio_xs_context *ctxt)
579 {
580     struct spdk_blob_store *bs = NULL;
581     bool last = false;
582
583     /*
584      * Unload the blobstore within the same thread where is't loaded,
585      * all server xstreams which should have stopped using the blobstore.
586      */
587     ABT_mutex_lock(bb->bb_mutex);
588 TF     if (bb->bb_ctxt == ctxt && bb->bb_bs != NULL) {
589         bs = bb->bb_bs;
590         bb->bb_bs = NULL;
591     }
592
593     D_ASSERT(bb->bb_ref > 0);
594     bb->bb_ref--;
595 TF     if (bb->bb_ref == 0)
596         last = true;
597     ABT_mutex_unlock(bb->bb_mutex);
598
599 TF     if (bs != NULL)
600         unload_blobstore(ctxt, bs);
601
602 TF     if (last) {
603         ABT_mutex_free(&bb->bb_mutex);
604         D_FREE(bb);
605     }
606 }

655 /**
656  * Assign a device for xstream->device mapping. Device chosen will be the device
657  * with the least amount of mapped xstreams.
658  */
659 static int
660 assign_dev_to_xs(int xs_id)
661 {
662     struct bio_bdev         *d_bdev;
663     struct bio_bdev         *chosen_bdev;
664     struct smd_nvme_stream_bond  xs_bond;
665     int              lowest_xs_cnt;
666     int              rc;
667
668     D_ASSERT(!d_list_empty(&nvme_glb.bd_bdevs));
669     chosen_bdev = d_list_entry(nvme_glb.bd_bdevs.next, struct bio_bdev,
670                   bb_link);
671     lowest_xs_cnt = chosen_bdev->bb_xs_cnt;
672
673     /*
674      * Traverse the list and return the device with the least amount of
675      * mapped xstreams.
676      */
677     d_list_for_each_entry(d_bdev, &nvme_glb.bd_bdevs, bb_link) {
678 TF         if (d_bdev->bb_xs_cnt < lowest_xs_cnt) {
679             lowest_xs_cnt = d_bdev->bb_xs_cnt;
680             chosen_bdev = d_bdev;
681         }
682     }
683
684     /* Update mapping for this xstream in NVMe device table */
685     smd_nvme_set_stream_bond(xs_id, chosen_bdev->bb_uuid, &xs_bond);
686     rc = smd_nvme_add_stream_bond(&xs_bond);
687 TF     if (rc) {
688         D_ERROR("Failure adding entry to SMD stream table\n");
689         return rc;
690     }
691
692     chosen_bdev->bb_xs_cnt++;
693
694     D_DEBUG(DB_MGMT, "Successfully added entry to SMD stream table,"
695         " xs_id:%d, dev:"DF_UUID", xs_cnt:%d\n",
696         xs_id, DP_UUID(xs_bond.nsm_dev_id), chosen_bdev->bb_xs_cnt);
697
698     return 0;
699 }

701 static int
702 init_blobstore_ctxt(struct bio_xs_context *ctxt, int xs_id)
703 {
704     struct bio_bdev         *d_bdev;
705     struct spdk_blob_store      *bs;
706     struct smd_nvme_stream_bond  xs_bond;
707     int              rc;
708     bool                 found = false;
709
710     D_ASSERT(ctxt->bxc_desc == NULL);
711     D_ASSERT(ctxt->bxc_blobstore == NULL);
712     D_ASSERT(ctxt->bxc_io_channel == NULL);
713
714     /*
715      * Lookup @xs_id in the NVMe device table (per-server metadata),
716      * if found, create blobstore on the mapped device.
717      */
718 TF     if (d_list_empty(&nvme_glb.bd_bdevs)) {
719         D_ERROR("No available SPDK bdevs, please check whether "
720             "VOS_BDEV_CLASS is set properly.\n");
721         return -DER_UNINIT;
722     }
723
724     rc = smd_nvme_get_stream_bond(xs_id, &xs_bond);
725 TF     if (rc && rc != -DER_NONEXIST)
726         return rc;
727
728 TF     if (rc == -DER_NONEXIST) {
729         /*
730          * Assign a device to the xstream if there isn't existing
731          * mapping. Device chosen will be current device that is mapped
732          * to the least amount of xstreams.
733          */
734         rc = assign_dev_to_xs(xs_id);
735 TF         if (rc) {
736             D_ERROR("No device assigned to xs_id:%d\n", xs_id);
737             return rc;
738         }
739
740         rc = smd_nvme_get_stream_bond(xs_id, &xs_bond);
741 TF         if (rc)
742             return rc;
743
744     }
745
746     D_DEBUG(DB_MGMT, "SMD stream table entry found, xs_id:%d, "
747         "dev:"DF_UUID"\n", xs_id, DP_UUID(xs_bond.nsm_dev_id));
748
749     /* Iterate thru device list to find matching dev */
750     d_list_for_each_entry(d_bdev, &nvme_glb.bd_bdevs, bb_link) {
751 TF         if (uuid_compare(d_bdev->bb_uuid, xs_bond.nsm_dev_id) == 0) {
752             found = true;
753             break;
754         }
755     }
756 TF     if (!found) {
757         /* TODO
758          * Mapping between device table entry and device list
759          * is inconsistent, either device currently mapped to
760          * the xstream is not present in the device list or
761          * stream table entry is invalid. Call per-server
762          * metadata management tool to rectify.
763          */
764         D_ERROR("Failure finding dev: "DF_UUID"\n",
765             DP_UUID(xs_bond.nsm_dev_id));
766         return -DER_NONEXIST;
767     }
768
769     D_ASSERT(d_bdev->bb_bdev != NULL);
770     rc = spdk_bdev_open(d_bdev->bb_bdev, false, NULL, NULL,
771                 &ctxt->bxc_desc);
772 TF     if (rc != 0) {
773         D_ERROR("Failed to open bdev %s, %d\n",
774             spdk_bdev_get_name(d_bdev->bb_bdev), rc);
775         return daos_errno2der(-rc);
776     }
777
778 TF     if (d_bdev->bb_blobstore == NULL) {
779         d_bdev->bb_blobstore = alloc_bio_blobstore(ctxt);
780 TF         if (d_bdev->bb_blobstore == NULL)
781             return -DER_NOMEM;
782
783         /* Load blobstore with bstype specified for sanity check */
784         bs = load_blobstore(ctxt, d_bdev->bb_bdev, &d_bdev->bb_uuid,
785                     false);
786 TF         if (bs == NULL)
787             return -DER_INVAL;
788
789         d_bdev->bb_blobstore->bb_bs = bs;
790
791         D_DEBUG(DB_MGMT, "Loaded bs, xs_id:%d, xs:%p dev:%s\n",
792             xs_id, ctxt, spdk_bdev_get_name(d_bdev->bb_bdev));
793     }
794
795     ctxt->bxc_blobstore = get_bio_blobstore(d_bdev->bb_blobstore);
796     bs = ctxt->bxc_blobstore->bb_bs;
797     D_ASSERT(bs != NULL);
798     ctxt->bxc_io_channel = spdk_bs_alloc_io_channel(bs);
799 TF     if (ctxt->bxc_io_channel == NULL) {
800         D_ERROR("Failed to create io channel\n");
801         return -DER_NOMEM;
802     }
803
804     return 0;
805 }
806
807 /*
808  * Finalize per-xstream NVMe context and SPDK env.
809  *
810  * \param[IN] ctxt  Per-xstream NVMe context
811  *
812  * \returns     N/A
813  */
814 void
815 bio_xsctxt_free(struct bio_xs_context *ctxt)
816 {
817     /* NVMe context setup was skipped */
818 TF     if (ctxt == NULL)
819         return;
820
821 TF     if (ctxt->bxc_io_channel != NULL) {
822         spdk_bs_free_io_channel(ctxt->bxc_io_channel);
823         ctxt->bxc_io_channel = NULL;
824     }
825
826 TF     if (ctxt->bxc_blobstore != NULL) {
827         put_bio_blobstore(ctxt->bxc_blobstore, ctxt);
828         ctxt->bxc_blobstore = NULL;
829     }
830
831 TF     if (ctxt->bxc_desc != NULL) {
832         spdk_bdev_close(ctxt->bxc_desc);
833         ctxt->bxc_desc = NULL;
834     }
835
836     ABT_mutex_lock(nvme_glb.bd_mutex);
837     nvme_glb.bd_xstream_cnt--;
838
839 TF     if (nvme_glb.bd_init_thread != NULL) {
840 TF         if (nvme_glb.bd_init_thread == ctxt->bxc_thread) {
841             struct common_cp_arg    cp_arg;
842
843             /*
844              * The xstream initialized SPDK env will have to
845              * wait for all other xstreams finalized first.
846              */
847 TF             if (nvme_glb.bd_xstream_cnt != 0)
848                 ABT_cond_wait(nvme_glb.bd_barrier,
849                           nvme_glb.bd_mutex);
850
851             fini_bio_bdevs(ctxt);
852
853             common_prep_arg(&cp_arg);
854             spdk_copy_engine_finish(common_fini_cb, &cp_arg);
855             xs_poll_completion(ctxt, &cp_arg.cca_inflights);
856
857             common_prep_arg(&cp_arg);
858             spdk_bdev_finish(common_fini_cb, &cp_arg);
859             xs_poll_completion(ctxt, &cp_arg.cca_inflights);
860
861             nvme_glb.bd_init_thread = NULL;
862
863 TF         } else if (nvme_glb.bd_xstream_cnt == 0) {
864             ABT_cond_broadcast(nvme_glb.bd_barrier);
865         }
866     }
867
868     ABT_mutex_unlock(nvme_glb.bd_mutex);
869
870 TF     if (ctxt->bxc_thread != NULL) {
871         xs_poll_completion(ctxt, NULL);
872         spdk_free_thread();
873         ctxt->bxc_thread = NULL;
874     }
875
876 TF     if (ctxt->bxc_msg_ring != NULL) {
877         spdk_ring_free(ctxt->bxc_msg_ring);
878         ctxt->bxc_msg_ring = NULL;
879     }
880     D_ASSERT(d_list_empty(&ctxt->bxc_pollers));
881
882 TF     if (ctxt->bxc_dma_buf != NULL) {
883         dma_buffer_destroy(ctxt->bxc_dma_buf);
884         ctxt->bxc_dma_buf = NULL;
885     }
886
887     D_FREE(ctxt);
888 }

890 int
891 bio_xsctxt_alloc(struct bio_xs_context **pctxt, int xs_id)
892 {
893     struct spdk_conf *config = NULL;
894     struct bio_xs_context *ctxt;
895     char name[32];
896     int rc;
897
898     /* Skip NVMe context setup if the daos_nvme.conf isn't present */
899 TF     if (nvme_glb.bd_nvme_conf == NULL) {
900         *pctxt = NULL;
901         return 0;
902     }
903
904     D_ALLOC_PTR(ctxt);
905 TF     if (ctxt == NULL)
906         return -DER_NOMEM;
907
908     D_INIT_LIST_HEAD(&ctxt->bxc_pollers);
909     ctxt->bxc_xs_id = xs_id;
910
911     ABT_mutex_lock(nvme_glb.bd_mutex);
912
913     nvme_glb.bd_xstream_cnt++;
914
915     D_INFO("Initialize NVMe context, xs_id:%d, init_thread:%p\n",
916            xs_id, nvme_glb.bd_init_thread);
917
918     /* Initialize SPDK env in first started xstream */
919 TF     if (nvme_glb.bd_init_thread == NULL) {
920         struct spdk_env_opts opts;
921
922         D_ASSERTF(nvme_glb.bd_xstream_cnt == 1, "%d",
923               nvme_glb.bd_xstream_cnt);
924
925         config = spdk_conf_allocate();
926 TF         if (config == NULL) {
927             D_ERROR("failed to alloc SPDK config\n");
928             rc = -DER_NOMEM;
929             goto out;
930         }
931
932         rc = spdk_conf_read(config, nvme_glb.bd_nvme_conf);
933 TF         if (rc != 0) {
934             D_ERROR("failed to read %s, rc:%d\n",
