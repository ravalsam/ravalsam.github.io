479 static int
480 bio_rwv(struct bio_io_context *ioctxt, struct bio_sglist *bsgl_in,
481     daos_sg_list_t *sgl, bool update)
482 {
483     struct bio_sglist   *bsgl;
484     struct bio_desc     *biod;
485     int          i, rc;
486
487     /* allocate blob I/O descriptor */
488     biod = bio_iod_alloc(ioctxt, 1 /* single bsgl */, update);
489 TF     if (biod == NULL)
490         return -DER_NOMEM;
491
492     /*
493      * copy the passed in @bsgl_in to the bsgl attached on bio_desc,
494      * since we don't want following bio ops change caller's bsgl.
495      */

497
498     rc = bio_sgl_init(bsgl, bsgl_in->bs_nr);
499 TF     if (rc)

501
502 TF     for (i = 0; i < bsgl->bs_nr; i++) {
503         D_ASSERT(bsgl_in->bs_iovs[i].bi_buf == NULL);
504         D_ASSERT(bsgl_in->bs_iovs[i].bi_data_len != 0);
505         bsgl->bs_iovs[i] = bsgl_in->bs_iovs[i];
506     }
507     bsgl->bs_nr_out = bsgl->bs_nr;
508
509     /* map the biov to DMA safe buffer, fill DMA buffer if read operation */
510     rc = bio_iod_prep(biod);
511 TF     if (rc)
512         goto out;
513
514 TF     for (i = 0; i < bsgl->bs_nr; i++)
515         D_ASSERT(bsgl->bs_iovs[i].bi_buf != NULL);
516

518 TF     if (rc)

520
521     /* release DMA buffer, write data back to NVMe device for write */
522     rc = bio_iod_post(biod);
523
524 out:
525     bio_iod_free(biod); /* also calls bio_sgl_fini */
526
527     return rc;
