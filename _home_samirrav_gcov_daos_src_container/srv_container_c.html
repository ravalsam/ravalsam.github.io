393
394     /* Verify the pool handle capabilities. */
395 TF     if (!(pool_hdl->sph_capas & DAOS_PC_RW) &&
396         !(pool_hdl->sph_capas & DAOS_PC_EX))
397         D_GOTO(out, rc = -DER_NO_PERM);
398
399     /* Check if a container with this UUID already exists. */
400     daos_iov_set(&key, in->cci_op.ci_uuid, sizeof(uuid_t));
401     daos_iov_set(&value, NULL /* buf */, 0 /* size */);
402     rc = rdb_tx_lookup(tx, &svc->cs_conts, &key, &value);
403 TF     if (rc != -DER_NONEXIST) {
404 TF         if (rc == 0)
405             D_DEBUG(DF_DSMS, DF_CONT": container already exists\n",
406                 DP_CONT(pool_hdl->sph_pool->sp_uuid,
407                     in->cci_op.ci_uuid));
408         D_GOTO(out, rc);
409     }
410
411     /*
412      * Target-side creations (i.e., vos_cont_create() calls) are
413      * deferred to the time when the container is first successfully
414      * opened.
415      */
416
417     /* Create the container attribute KVS under the container KVS. */
418     daos_iov_set(&key, in->cci_op.ci_uuid, sizeof(uuid_t));
419     attr.dsa_class = RDB_KVS_GENERIC;
420     attr.dsa_order = 16;

424             "%d\n", rc);

426     }
427
428     /* Create a path to the container attribute KVS. */
429     rc = rdb_path_clone(&svc->cs_conts, &kvs);
430 TF     if (rc != 0)
431         D_GOTO(out, rc);
432     rc = rdb_path_push(&kvs, &key);
433 TF     if (rc != 0)
434         D_GOTO(out_kvs, rc);
435
436     /* Create the GHCE and GHPCE properties. */
437     daos_iov_set(&value, &ghce, sizeof(ghce));
438     rc = rdb_tx_update(tx, &kvs, &ds_cont_prop_ghce, &value);
439 TF     if (rc != 0)
440         D_GOTO(out_kvs, rc);
441     daos_iov_set(&value, &ghpce, sizeof(ghpce));
442     rc = rdb_tx_update(tx, &kvs, &ds_cont_prop_ghpce, &value);
443 TF     if (rc != 0)
444         D_GOTO(out_kvs, rc);
445     daos_iov_set(&value, &max_oid, sizeof(max_oid));
446     rc = rdb_tx_update(tx, &kvs, &ds_cont_prop_max_oid, &value);
447 TF     if (rc != 0)
448         D_GOTO(out_kvs, rc);

451      * parameter and then write to rdb.

454 TF     if (prop_dup == NULL) {

459 TF     if (rc != 0)

462 TF     if (rc != 0)

464

467     attr.dsa_order = 16;

472 TF     if (rc != 0)

476     attr.dsa_class = RDB_KVS_INTEGER;
477     attr.dsa_order = 16;
478     rc = rdb_tx_create_kvs(tx, &kvs, &ds_cont_prop_snapshots, &attr);
479
480     /* Create the user attribute KVS. */
481     attr.dsa_class = RDB_KVS_GENERIC;
482     attr.dsa_order = 16;
483     rc = rdb_tx_create_kvs(tx, &kvs, &ds_cont_attr_user, &attr);
484
485 out_kvs:
486     daos_prop_free(prop_dup);
487     rdb_path_fini(&kvs);
488 out:
489     return rc;
490 }

605 {

610
611     daos_iov_set(&key, (void *)uuid, sizeof(uuid_t));

679 err_attrs:

879 /* Close an array of handles, possibly belonging to different containers. */
880 static int
881 cont_close_hdls(struct cont_svc *svc, struct cont_tgt_close_rec *recs,
882         int nrecs, crt_context_t ctx)
883 {
884     int i;
885     int rc;
886
887     D_ASSERTF(nrecs > 0, "%d\n", nrecs);
888     D_DEBUG(DF_DSMS, DF_CONT": closing %d recs: recs[0].hdl="DF_UUID
889         " recs[0].hce="DF_U64"\n", DP_CONT(svc->cs_pool_uuid, NULL),
890         nrecs, DP_UUID(recs[0].tcr_hdl), recs[0].tcr_hce);
891
892     rc = cont_close_bcast(ctx, svc, recs, nrecs);
893 TF     if (rc != 0)
894         D_GOTO(out, rc);
895
896     /*
897      * Use one TX per handle to avoid calling ds_cont_epoch_fini_hdl() more
898      * than once in a TX, in which case we would be attempting to query
899      * uncommitted updates. This could be optimized by adding container
900      * UUIDs into recs[i] and sorting recs[] by container UUIDs. Then we
901      * could maintain a list of deleted LREs and a list of deleted LHEs for
902      * each container while looping, and use the lists to update the GHCE
903      * once for each container. This approach enables us to commit only
904      * once (or when a TX becomes too big).
905      */
906 TF     for (i = 0; i < nrecs; i++) {
907         struct rdb_tx tx;
908
909         rc = rdb_tx_begin(svc->cs_rsvc->s_db, svc->cs_rsvc->s_term,
910                   &tx);
911 TF         if (rc != 0)
912             break;
913         rc = cont_close_one_hdl(&tx, svc, ctx, recs[i].tcr_hdl);
914 TF         if (rc != 0) {
915             rdb_tx_end(&tx);
916             break;
917         }
918         rc = rdb_tx_commit(&tx);
919         rdb_tx_end(&tx);
920 TF         if (rc != 0)
921             break;
922     }
923
924 out:
925     D_DEBUG(DF_DSMS, DF_CONT": leaving: %d\n",
926         DP_CONT(svc->cs_pool_uuid, NULL), rc);
927     return rc;
928 }
