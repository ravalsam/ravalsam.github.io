317 /*
318  * Called via dss_collective() to destroy the ds_cont object as well as the vos
319  * container.
320  */
321 static int
322 cont_destroy_one(void *vin)
323 {
324     struct cont_tgt_destroy_in     *in = vin;
325     struct dsm_tls             *tls = dsm_tls_get();
326     struct ds_pool_child           *pool;
327     struct ds_cont             *cont;
328     int             rc;
329
330     pool = ds_pool_child_lookup(in->tdi_pool_uuid);
331 TF     if (pool == NULL)
332         D_GOTO(out, rc = -DER_NO_HDL);
333
334     rc = cont_lookup(tls->dt_cont_cache, in->tdi_uuid, NULL /* arg */,
335              &cont);
336 TF     if (rc == 0) {
337         /* Should evict if idle, but no such interface at the moment. */
338         cont_put(tls->dt_cont_cache, cont);
339         D_GOTO(out_pool, rc = -DER_BUSY);
340 TF     } else if (rc != -DER_NONEXIST) {
341         D_GOTO(out_pool, rc);
342     }
343
344     D_DEBUG(DF_DSMS, DF_CONT": destroying vos container\n",
345         DP_CONT(pool->spc_uuid, in->tdi_uuid));
346
347     rc = vos_cont_destroy(pool->spc_hdl, in->tdi_uuid);
348 TF     if (rc == -DER_NONEXIST)
349         /** VOS container creation is effectively delayed until
350          * container open time, so it might legitimately not exist if
351          * the container has never been opened */
352         rc = 0;
353
354 out_pool:
355     ds_pool_child_put(pool);
356 out:
357     return rc;
358 }

495 int
496 ds_cont_local_open(uuid_t pool_uuid, uuid_t cont_hdl_uuid, uuid_t cont_uuid,
497            uint64_t capas, struct ds_cont_hdl **cont_hdl)
498 {
499     struct dsm_tls      *tls = dsm_tls_get();
500     struct ds_cont_hdl  *hdl;
501     int         vos_co_created = 0;
502     int         rc = 0;
503
504     hdl = cont_hdl_lookup_internal(&tls->dt_cont_hdl_hash, cont_hdl_uuid);
505 TF     if (hdl != NULL) {
506 TF         if (capas != 0) {
507 TF             if (hdl->sch_capas != capas) {
508                 D_ERROR(DF_CONT": conflicting container : hdl="
509                     DF_UUID" capas="DF_U64"\n",
510                     DP_CONT(pool_uuid, cont_uuid),
511                     DP_UUID(cont_hdl_uuid), capas);
512                 rc = -DER_EXIST;
513             } else {
514                 D_DEBUG(DF_DSMS, DF_CONT": found compatible"
515                     " container handle: hdl="DF_UUID
516                     " capas="DF_U64"\n",
517                       DP_CONT(pool_uuid, cont_uuid),
518                       DP_UUID(cont_hdl_uuid), hdl->sch_capas);
519             }
520         }
521
522 TF         if (rc == 0)
523             hdl->sch_deleted = 0;
524
525 TF         if (cont_hdl != NULL && rc == 0)
526             *cont_hdl = hdl;
527         else
528             cont_hdl_put_internal(&tls->dt_cont_hdl_hash, hdl);
529         return rc;
530     }
531
532     D_ALLOC_PTR(hdl);
533 TF     if (hdl == NULL)
534         D_GOTO(err, rc = -DER_NOMEM);
535
536     hdl->sch_pool = ds_pool_child_lookup(pool_uuid);
537 TF     if (hdl->sch_pool == NULL)
538         D_GOTO(err_hdl, rc = -DER_NO_HDL);
539
540 TF     if (cont_uuid != NULL) {
541         rc = ds_cont_lookup_or_create(hdl, cont_uuid);
542 TF         if (rc == 1) {
543             vos_co_created = 1;
544             rc = 0;
545 TF         } else if (rc != 0) {
546             D_GOTO(err_pool, rc);
547         }
548     }
549     uuid_copy(hdl->sch_uuid, cont_hdl_uuid);
550     hdl->sch_capas = capas;
551
552     rc = cont_hdl_add(&tls->dt_cont_hdl_hash, hdl);
553 TF     if (rc != 0)
554         D_GOTO(err_cont, rc);
555
556 TF     if (cont_hdl != NULL) {
557         cont_hdl_get_internal(&tls->dt_cont_hdl_hash, hdl);
558         *cont_hdl = hdl;
559     }
560
561     /* It is possible to sync DTX status before destroy the CoS for close
562      * the container. But that may be not enough. Because the server may
563      * crashed before closing the container. Then the DTXs' status in the
564      * CoS cache will be lost. So we need to re-sync the DTXs status when
565      * open the container for the first time (not for cached open handle).
566      *
567      * On the other hand, even if we skip the DTX sync before destroy the
568      * CoS cache when close the container, resync DTX when open container
569      * is enough to guarantee related data records' visibility. That also
570      * simplify the DTX logic.
571      *
572      * XXX: The logic is related with DAOS server re-intergration, but we
573      *  do not support that currently. Then resync DTX when container
574      *  open will be used as temporary solution for DTX related logic.
575      *
576      * We do not trigger dtx_resync() when start the server. Because:
577      * 1. Currently, we do not support server re-integrate after restart.
578      * 2. A server may has multiple pools and each pool may has multiple
579      *    containers. These containers may not related with one another.
580      *    Make all the DTXs resync together during the server start will
581      *    cause the DTX resync time to be much longer than resync against
582      *    single container just when use (open) it. On the other hand, if
583      *    some servers are ready for dtx_resync, but others may not start
584      *    yet, then the ready ones may have to wait or failed dtx_resync.
585      *    Both cases are not expected.
586      */
587 TF     if (cont_uuid != NULL) {
588         struct ds_dtx_resync_args   *ddra = NULL;
589
590         rc = dtx_batched_commit_register(hdl);
591 TF         if (rc != 0) {
592             D_ERROR("Failed to register the container "DF_UUID
593                 " to the DTX batched commit list: rc = %d\n",
594                 DP_UUID(cont_uuid), rc);
595             D_GOTO(err_cont, rc);
596         }
597
598         D_ALLOC_PTR(ddra);
599 TF         if (ddra == NULL)
600             D_GOTO(err_cont, rc = -DER_NOMEM);
601
602         ddra->pool = ds_pool_child_get(hdl->sch_pool);
603         uuid_copy(ddra->co_uuid, cont_uuid);
604         rc = dss_ult_create(ds_dtx_resync, ddra, DSS_ULT_DTX_RESYNC,
605                     DSS_TGT_SELF, 0, NULL);
606 TF         if (rc != 0) {
607             ds_pool_child_put(hdl->sch_pool);
608             D_FREE(ddra);
609             D_GOTO(err_cont, rc);
610         }
611     }
612
613     return 0;
614
615 err_cont:
616     dtx_batched_commit_deregister(hdl);
617 TF     if (hdl->sch_cont)
618         cont_put(tls->dt_cont_cache, hdl->sch_cont);
619
620 TF     if (vos_co_created) {
621         D_DEBUG(DF_DSMS, DF_CONT": destroying new vos container\n",
622             DP_CONT(hdl->sch_pool->spc_uuid, cont_uuid));
623         vos_cont_destroy(hdl->sch_pool->spc_hdl, cont_uuid);
624     }
625 err_pool:
626     ds_pool_child_put(hdl->sch_pool);
627 err_hdl:
628     D_FREE(hdl);
629 err:
630     return rc;
631 }
