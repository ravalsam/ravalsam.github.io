403                     return -DER_NOMEM;
404                 i_args->sla_buf = ptr;
405             }
406         }
407
408 TF         if (i_args->sla_index < i_args->sla_count)
409             memcpy(&i_args->sla_buf[i_args->sla_index],
410                    key->iov_buf, sizeof(daos_epoch_t));
411     }
412     ++i_args->sla_index;
413     return 0;
414 }

416 static int
417 read_snap_list(struct rdb_tx *tx, struct cont *cont,
418            daos_epoch_t **buf, int *count)
419 {
420     struct snap_list_iter_args iter_args;
421     int rc;
422
423     iter_args.sla_index = 0;
424     iter_args.sla_max_count = *count;
425 TF     if (*count != 0) {
426         /* start with initial size then grow the buffer */
427 TF         iter_args.sla_count = *count > 0 && *count < 64 ? *count : 64;
428         D_ALLOC_ARRAY(iter_args.sla_buf, iter_args.sla_count);
429 TF         if (iter_args.sla_buf == NULL)
430             return -DER_NOMEM;
431     } else {
432         iter_args.sla_count = 0;
433         iter_args.sla_buf = NULL;
434     }
435     rc = rdb_tx_iterate(tx, &cont->c_snaps, false /* !backward */,
436                 snap_list_iter_cb, &iter_args);
437 TF     if (rc != 0) {
438         D_FREE(iter_args.sla_buf);
439         return rc;
440     }
441     *count = iter_args.sla_index;
442     *buf   = iter_args.sla_buf;
443     return 0;

446 static int
447 trigger_aggregation(struct rdb_tx *tx,
448             daos_epoch_t glre, daos_epoch_t glre_curr,
449             daos_epoch_t ghce, struct cont *cont, crt_context_t ctx)
450 {
451     daos_epoch_t        *snapshots;
452     int          snap_count;
453     daos_epoch_range_t   target_range;
454     daos_epoch_range_t  *ranges;
455     int          range_count;
456     int          start;
457     int          i;
458     int          rc = 0;
459
460     /* glre and glre_curr can take the value DAOS_EPOCH_MAX */
461 TF     target_range.epr_lo = glre < DAOS_EPOCH_MAX ? glre : 0;
462     target_range.epr_hi = MIN(ghce, glre_curr);
463 TF     if (target_range.epr_hi <= target_range.epr_lo)
464         goto out;
465     D_DEBUG(DF_DSMS, DF_CONT": Move GLRE { "DF_U64" => "DF_U64" }\n",
466              DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
467              target_range.epr_lo, target_range.epr_hi);
468
469     /* Aggregate between all snapshots less than GLRE */
470     snap_count = -1; /* No upper limit */
471     rc = read_snap_list(tx, cont, &snapshots, &snap_count);
472 TF     if (rc != 0)
473         goto out;
474     /* Start from highest snapshot less than old GLRE */
475 TF     for (i = 0; i < snap_count &&
476          snapshots[i] < target_range.epr_lo; ++i)
477         ;
478     start = i - 1;
479
480     /* range_count will be at least `1` and at most `snap_count + 1` */
481 TF     for (range_count = 1; i < snap_count &&
482          snapshots[i] < target_range.epr_hi; ++range_count, ++i)
483          ;
484
485     D_DEBUG(DF_DSMS, DF_CONT": snap_count=%d range_count=%d start=%d\n",
486              DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
487              snap_count, range_count, start);
488
489     D_ALLOC_ARRAY(ranges, range_count);
490 TF     if (ranges == NULL) {
491         rc = -DER_NOMEM;
492         goto out_snap;
493     }
494
495     /* If start == -1, old GLRE is less than lowest snapshot */
496 TF     ranges[0].epr_lo = start < 0 ? 0UL : snapshots[start];
497 TF     for (i = 1; i < range_count; ++i) {
498         ranges[i - 1].epr_hi = snapshots[start + i];
499         ranges[i].epr_lo = snapshots[start + i] + 1;
500     }
501     ranges[range_count - 1].epr_hi = target_range.epr_hi;
502     rc = epoch_aggregate_bcast(ctx, cont, ranges, range_count);
503     D_FREE(ranges);
504 out_snap:
505     D_FREE(snapshots);
506 out:
507     return rc;
508 }

510 /* Return 0 if the check passes, 1 otherwise. */
511 static inline int
512 check_global_epoch_invariant(struct cont *cont, struct epoch_prop *prop)
513 {
514 TF     if (!(prop->ep_ghce <= prop->ep_ghpce)) {
515         D_ERROR(DF_CONT": GHCE "DF_U64" > GHPCE "DF_U64"\n",
516             DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
517             prop->ep_ghce, prop->ep_ghpce);
518         return 1;
519     }
520
521 TF     if (!(prop->ep_glhe == DAOS_EPOCH_MAX ||
522           prop->ep_glhe > prop->ep_ghce)) {
523         D_ERROR(DF_CONT": GLHE "DF_U64" <= GHCE "DF_U64"\n",
524             DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
525             prop->ep_glhe, prop->ep_ghce);
526         return 1;
527     }
528
529     return 0;
530 }
531
532 /* Return 0 if the check passes, 1 otherwise. */
533 static inline int
534 check_epoch_invariant(struct cont *cont, struct epoch_prop *prop,
535               struct container_hdl *hdl)
536 {
537 TF     if (check_global_epoch_invariant(cont, prop) != 0)
538         return 1;
539
540 TF     if (!(hdl->ch_hce < hdl->ch_lhe)) {
541         D_ERROR(DF_CONT": HCE "DF_U64" >= LHE "DF_U64"\n",
542             DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
543             hdl->ch_hce, hdl->ch_lhe);
544         return 1;
545     }
546
547 TF     if (!(prop->ep_glre <= hdl->ch_lre)) {
548         D_ERROR(DF_CONT": GLRE "DF_U64" > LRE "DF_U64"\n",
549             DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
550             prop->ep_glre, hdl->ch_lre);
551         return 1;
552     }
553
554 TF     if (!(prop->ep_ghce < hdl->ch_lhe)) {
555         D_ERROR(DF_CONT": GHCE "DF_U64" >= LHE "DF_U64"\n",
556             DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
557             prop->ep_ghce, hdl->ch_lhe);
558         return 1;
559     }
560
561 TF     if (!(prop->ep_ghpce >= hdl->ch_hce)) {
562         D_ERROR(DF_CONT": GHPCE "DF_U64" < HCE "DF_U64"\n",
563             DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
564             prop->ep_ghpce, hdl->ch_hce);
565         return 1;
566     }
567
568     return 0;
569 }
570
571 /*
572  * Calculate GHCE afresh using prop->ep_ghpce and prop->ep_glhe, and if the
573  * result is higher than prop->ep_ghce, update prop->ep_ghce and CONT_GHCE.
574  *
575  * This function must be called by operations change GHPCE or GLHE. It may be
576  * called unnecessarily by other operations without compromising safety.
577  */
578 static int
579 update_ghce(struct rdb_tx *tx, struct cont *cont, struct epoch_prop *prop)
580 {
581     daos_epoch_t    ghce;
582     daos_iov_t  value;
583     int     rc;
584
585     /*
586      * GHCE shall be the highest epoch e satisfying both of these:
587      *   - e <= GHPCE (committed by some handle)
588      *   - e < GLHE (not held by any handle)
589      */
590     ghce = min(prop->ep_ghpce, prop->ep_glhe - 1);
591
592 TF     if (ghce < prop->ep_ghce) {
593         D_ERROR(DF_CONT": GHCE would decrease: GLRE="DF_U64" GHCE="
594             DF_U64" GHPCE="DF_U64 " GLHE="DF_U64"\n",
595             DP_CONT(cont->c_svc->cs_pool_uuid, cont->c_uuid),
596             prop->ep_glre, prop->ep_ghce, prop->ep_ghpce,
597             prop->ep_glhe);
598         return -DER_IO;
599 TF     } else if (ghce == prop->ep_ghce) {
600         return 0;
601     }
602
603     daos_iov_set(&value, &ghce, sizeof(ghce));
604     rc = rdb_tx_update(tx, &cont->c_prop, &ds_cont_prop_ghce, &value);
605 TF     if (rc != 0)
606         D_ERROR(DF_CONT": failed to update ghce: %d\n",
607             DP_CONT(cont->c_svc->cs_pool_uuid,
608                 cont->c_uuid), rc);
609
610     prop->ep_ghce = ghce;
611     return rc;
612 }

614 int
615 ds_cont_epoch_init_hdl(struct rdb_tx *tx, struct cont *cont, uuid_t c_hdl,
616                struct container_hdl *hdl)
617 {
618     struct epoch_prop   prop;
619     int         rc;
620
621     rc = read_epoch_prop(tx, cont, &prop);
622 TF     if (rc != 0)
623         return rc;
624
625 TF     if (check_global_epoch_invariant(cont, &prop) != 0)
626         return -DER_IO;
627
628     hdl->ch_hce = prop.ep_ghce;
629     hdl->ch_lre = prop.ep_ghce;
630     hdl->ch_lhe = DAOS_EPOCH_MAX;
631
632     /* Determine the new GLRE and update the LRE KVS. */
633     rc = ec_update_and_find_lowest(tx, cont, EC_LRE, NULL /* dec */,
634                        &hdl->ch_lre /* inc */,
635                        NULL /* emptyp */, &prop.ep_glre);
636 TF     if (rc != 0)
637         return rc;
638
639     /*
640      * TODO - For now, do an epoch hold on container open since
641      * daos_epoch_hold() is removed from API. Revisit when new epoch model
642      * implemented.
643      */
644 TF     if (hdl->ch_capas & DAOS_COO_RW) {
645         daos_iov_t  key;
646         daos_iov_t  value;
647
648         hdl->ch_lhe = prop.ep_ghpce + 1;
649
650         daos_iov_set(&key, c_hdl, sizeof(uuid_t));
651         daos_iov_set(&value, hdl, sizeof(*hdl));
652         rc = rdb_tx_update(tx, &cont->c_svc->cs_hdls, &key, &value);
653 TF         if (rc != 0)
654             return rc;
655     }
656
657     /* Determine the new GLHE and update the LHE KVS. */
658     rc = ec_update_and_find_lowest(tx, cont, EC_LHE, NULL /* dec */,
659                        &hdl->ch_lhe /* inc */,
660                        NULL /* emptyp */, &prop.ep_glhe);
661 TF     if (rc != 0)
662         return rc;
663
664 TF     if (check_epoch_invariant(cont, &prop, hdl) != 0)
665         return -DER_IO;
666
667     return 0;
668 }

670 int
671 ds_cont_epoch_fini_hdl(struct rdb_tx *tx, struct cont *cont,
672                crt_context_t ctx, struct container_hdl *hdl)
673 {
674     struct epoch_prop   prop;
675     daos_epoch_t        glre;
676     bool            empty;
677     bool            slip_flag;
678     int         rc;
679
680     rc = read_epoch_prop(tx, cont, &prop);
681 TF     if (rc != 0)
682         return rc;
683
684     slip_flag = auto_slip_enabled();
685     glre = prop.ep_glre;
686
687 TF     if (check_epoch_invariant(cont, &prop, hdl) != 0)
688         return -DER_IO;
689
690     /* Determine the new GLRE and update the LRE KVS. */
691     rc = ec_update_and_find_lowest(tx, cont, EC_LRE, &hdl->ch_lre /* dec */,
692                        NULL /* inc */, &empty, &prop.ep_glre);
693 TF     if (rc != 0)
694         return rc;
695 TF     if (empty)
696         prop.ep_glre = DAOS_EPOCH_MAX;
697
698     /* Determine the new GLHE and update the LHE KVS. */
699     rc = ec_update_and_find_lowest(tx, cont, EC_LHE, &hdl->ch_lhe /* dec */,
700                        NULL /* inc */, &empty, &prop.ep_glhe);
701 TF     if (rc != 0)
702         return rc;
703 TF     if (empty)
704         prop.ep_glhe = DAOS_EPOCH_MAX;
705
706     rc = update_ghce(tx, cont, &prop);
707 TF     if (rc != 0)
708         return rc;
709
710 TF     if (check_global_epoch_invariant(cont, &prop) != 0)
711         return -DER_IO;
712
713     /** once we have an aggregation daemon, mask this error message */
714 TF     if (slip_flag)
715         rc = trigger_aggregation(tx, glre, prop.ep_glre, prop.ep_ghce,
716                      cont, ctx);
717
718     return rc;
719 }

721 int
722 ds_cont_epoch_hold(struct rdb_tx *tx, struct ds_pool_hdl *pool_hdl,
723            struct cont *cont, struct container_hdl *hdl, crt_rpc_t *rpc)
724 {
725     struct cont_epoch_op_in        *in = crt_req_get(rpc);
726     struct epoch_prop       prop;
727     daos_epoch_t            lhe = hdl->ch_lhe;
728     daos_iov_t          key;
729     daos_iov_t          value;
730     int             rc;
731
732     D_DEBUG(DF_DSMS, DF_CONT": processing rpc %p: epoch="DF_U64"\n",
733         DP_CONT(pool_hdl->sph_pool->sp_uuid, in->cei_op.ci_uuid), rpc,
734         in->cei_epoch);
735
736     /* Verify the container handle capabilities. */
737 TF     if (!(hdl->ch_capas & DAOS_COO_RW))
738         D_GOTO(out, rc = -DER_NO_PERM);
739
740 TF     if (in->cei_epoch > DAOS_EPOCH_MAX)
741         D_GOTO(out, rc = -DER_OVERFLOW);
742
743     rc = read_epoch_prop(tx, cont, &prop);
744 TF     if (rc != 0)
745         D_GOTO(out, rc);
746
747 TF     if (check_epoch_invariant(cont, &prop, hdl) != 0)
748         D_GOTO(out, rc = -DER_IO);
749
750 TF     if (in->cei_epoch == 0)
751         /*
752          * No specific requirement, as defined by daos_epoch_hold().
753          * Give back an epoch that can be used to read and overwrite
754          * all (partially) committed epochs.
755          */
756         hdl->ch_lhe = prop.ep_ghpce + 1;
757 TF     else if (in->cei_epoch <= prop.ep_ghce)
758         /* Immutable epochs cannot be held. */
759         hdl->ch_lhe = prop.ep_ghce + 1;
760     else
761         hdl->ch_lhe = in->cei_epoch;
762
763 TF     if (hdl->ch_lhe == lhe)
764         D_GOTO(out_hdl, rc = 0);
765
766     D_DEBUG(DF_DSMS, "lhe="DF_U64" lhe'="DF_U64"\n", lhe, hdl->ch_lhe);
767
768     daos_iov_set(&key, in->cei_op.ci_hdl, sizeof(uuid_t));
769     daos_iov_set(&value, hdl, sizeof(*hdl));
770     rc = rdb_tx_update(tx, &cont->c_svc->cs_hdls, &key, &value);
771 TF     if (rc != 0)
772         D_GOTO(out_hdl, rc);
773
774     rc = ec_update_and_find_lowest(tx, cont, EC_LHE, &lhe /* dec */,
775                        &hdl->ch_lhe /* inc */,
776                        NULL /* emptyp */, &prop.ep_glhe);
777 TF     if (rc != 0)
778         D_GOTO(out_hdl, rc);
779
780     /* If we are releasing held epochs, then update GHCE. */
781 TF     if (hdl->ch_lhe > lhe) {
782         rc = update_ghce(tx, cont, &prop);
783 TF         if (rc != 0)
784             D_GOTO(out_hdl, rc);
785     }
786
787 TF     if (check_epoch_invariant(cont, &prop, hdl) != 0)
788         D_GOTO(out_hdl, rc = -DER_IO);
789
790 out_hdl:
791 TF     if (rc != 0)
792         hdl->ch_lhe = lhe;
793 out:
794     D_DEBUG(DF_DSMS, DF_CONT": replying rpc %p: %d\n",
795         DP_CONT(pool_hdl->sph_pool->sp_uuid, in->cei_op.ci_uuid), rpc,
796         rc);
797     return rc;
798 }

800 int
801 ds_cont_epoch_slip(struct rdb_tx *tx, struct ds_pool_hdl *pool_hdl,
802            struct cont *cont, struct container_hdl *hdl, crt_rpc_t *rpc)
803 {
804     struct cont_epoch_op_in        *in = crt_req_get(rpc);
805     struct epoch_prop       prop;
806     daos_epoch_t            lre = hdl->ch_lre;
807     daos_epoch_t            glre;
808     daos_iov_t          key;
809     daos_iov_t          value;
810     int             rc;
811
812     D_DEBUG(DF_DSMS, DF_CONT": processing rpc %p: epoch="DF_U64"\n",
813         DP_CONT(pool_hdl->sph_pool->sp_uuid, in->cei_op.ci_uuid), rpc,
814         in->cei_epoch);
815
816 TF     if (in->cei_epoch >= DAOS_EPOCH_MAX)
817         D_GOTO(out, rc = -DER_OVERFLOW);
818
819     rc = read_epoch_prop(tx, cont, &prop);
820 TF     if (rc != 0)
821         D_GOTO(out, rc);
822     glre = prop.ep_glre;
823
824 TF     if (check_epoch_invariant(cont, &prop, hdl) != 0)
825         D_GOTO(out, rc = -DER_IO);
826
827 TF     if (in->cei_epoch < hdl->ch_lre)
828         /*
829          * Since we don't allow LRE to decrease, let the new LRE be the
830          * old one.  (This is actually unnecessary; we only have to
831          * guarantee that the new LRE has not been aggregated away.)
832          */
833         ;
834     else
835         hdl->ch_lre = in->cei_epoch;
836
837 TF     if (hdl->ch_lre == lre)
838         D_GOTO(out_hdl, rc = 0);

1017         D_GOTO(out_hdl, rc);

1019     rc = ec_update_and_find_lowest(tx, cont, EC_LHE, &lhe /* dec */,
1020                        &hdl->ch_lhe /* inc */,

1023         D_GOTO(out_hdl, rc);

1026         rc = ec_update_and_find_lowest(tx, cont, EC_LRE, &lre /* dec */,
1027                            &hdl->ch_lre /* inc */,
1028                            NULL /* emptyp */,
1029                            &prop.ep_glre);
1030 TF         if (rc != 0)
1031             D_GOTO(out_hdl, rc);
1032     }
1033
1034 TF     if (hdl->ch_hce > prop.ep_ghpce) {
1035         prop.ep_ghpce = hdl->ch_hce;
1036         daos_iov_set(&value, &prop.ep_ghpce, sizeof(prop.ep_ghpce));

1044         }
1045     }

1048 TF     if (rc != 0)
1049         D_GOTO(out_hdl, rc);
1050
1051 TF     if (check_epoch_invariant(cont, &prop, hdl) != 0)
1052         D_GOTO(out_hdl, rc = -DER_IO);
1053
1054 TF     if (snapshot) {
1055         char        zero = 0;
1056
1057         daos_iov_set(&key, &in->cei_epoch, sizeof(in->cei_epoch));
1058         daos_iov_set(&value, &zero, sizeof(zero));
1059         rc = rdb_tx_update(tx, &cont->c_snaps, &key, &value);
1060 TF         if (rc != 0) {
1061             D_ERROR(DF_CONT": failed to create snapshot: %d\n",
1062                 DP_CONT(cont->c_svc->cs_pool_uuid,
1063                     cont->c_uuid), rc);
1064             goto out;
1065         }
1066     }

1068 TF     if (slip_flag) {

1070                      cont, rpc->cr_ctx);

1073                 rc);
1074             D_GOTO(out_hdl, rc);
1075         }
1076     }
1077
1078 out_hdl:
1079 TF     if (rc != 0) {
1080         hdl->ch_lre = lre;
1081         hdl->ch_lhe = lhe;

1086         DP_CONT(pool_hdl->sph_pool->sp_uuid, in->cei_op.ci_uuid), rpc,
1087         rc);
