350 TF     if (epr->epr_lo != epr->epr_hi) {

352             "because it is not clearly defined.\n");
353         return -DER_INVAL; /* XXX simplify it for now */
354     }
355
356     rc = key_tree_prepare(obj, ent->ie_epoch, obj->obj_toh, VOS_BTR_DKEY,
357                   &ent->ie_key, 0, vos_iter_intent(&oiter->it_iter),
358                   NULL, &toh);
359 TF     if (rc != 0) {
360         D_DEBUG(DB_IO, "can't load the akey tree: %d\n", rc);
361         return rc;
362     }
363
364     /* check if the akey exists */
365     tree_rec_bundle2iov(&rbund, &riov);
366     tree_key_bundle2iov(&kbund, &kiov);
367     kbund.kb_key = &oiter->it_akey;
368     kbund.kb_epoch = epr->epr_lo;
369
370     rc = dbtree_fetch(toh, BTR_PROBE_GT | BTR_PROBE_MATCHED,
371               vos_iter_intent(&oiter->it_iter), &kiov, NULL, &riov);
372     key_tree_release(toh, false);
373 TF     if (rc == 0)
374         return IT_OPC_NOOP; /* match the condition (akey), done */
375
376 TF     if (rc == -DER_NONEXIST)
377         return IT_OPC_NEXT; /* no matched akey */
378
379     return rc;
380 }

382 /**
383  * Check if the current item can match the provided condition (with the
384  * giving a-key). If the item can't match the condition, this function
385  * traverses the tree until a matched item is found.
386  */
387 static int
388 key_iter_match_probe(struct vos_obj_iter *oiter)
389 {
390     int rc;
391
392     while (1) {
393         vos_iter_entry_t    entry;
394         struct vos_key_bundle   kbund;
395         daos_iov_t      kiov;
396         int         opc = 0;
397
398         rc = key_iter_match(oiter, &entry, &opc);
399         switch (rc) {
400 T         default:
401             D_ASSERT(rc < 0);
402             D_ERROR("match failed, rc=%d\n", rc);
403             goto out;
404
405 T         case IT_OPC_NOOP:
406             /* already match the condition, no further operation */
407             rc = 0;
408             goto out;
409
410 T         case IT_OPC_PROBE:
411             /* probe the returned key and epoch range */
412             tree_key_bundle2iov(&kbund, &kiov);
413             kbund.kb_key    = &entry.ie_key;
414             kbund.kb_epoch  = entry.ie_epoch;
415             rc = dbtree_iter_probe(oiter->it_hdl, opc,
416                            vos_iter_intent(&oiter->it_iter),
417                            &kiov, NULL);
418 TF             if (rc)
419                 goto out;
420             break;
421
422 T         case IT_OPC_NEXT:
423             /* move to the next tree record */
424             rc = dbtree_iter_next_with_intent(oiter->it_hdl,
425                     vos_iter_intent(&oiter->it_iter));
426 TF             if (rc)
427                 goto out;
428             break;
429         }
430     }
431  out:
432     return rc;
433 }

584 /**
585  * Find the data that was written before/in the specified epoch of @oiter
586  * for the recx in @entry. If this recx has no data for this epoch, then
587  * this function will move on to the next recx and repeat this process.
588  */
589 static int
590 singv_iter_probe_epr(struct vos_obj_iter *oiter, vos_iter_entry_t *entry)
591 {
592     daos_epoch_range_t *epr = &oiter->it_epr;
593
594     while (1) {
595         int opc;
596         int rc;
597
598         switch (oiter->it_epc_expr) {
599 T         default:
600             return -DER_INVAL;
601
602 T         case VOS_IT_EPC_EQ:
603 TF             if (entry->ie_epoch > epr->epr_hi)
604                 return -DER_NONEXIST;
605
606 TF             if (entry->ie_epoch < epr->epr_lo) {
607                 entry->ie_epoch = epr->epr_lo;
608                 opc = BTR_PROBE_EQ;
609                 break;
610             }
611             return 0;
612
613 T         case VOS_IT_EPC_RE:
614 TF             if (entry->ie_epoch > epr->epr_hi)
615                 return -DER_NONEXIST;
616
617 TF             if (entry->ie_epoch < epr->epr_lo) {
618                 entry->ie_epoch = epr->epr_lo;
619                 opc = BTR_PROBE_GE;
620                 break;
621             }
622             return 0;
623
624 T         case VOS_IT_EPC_RR:
625 TF             if (entry->ie_epoch < epr->epr_lo)
626                 return -DER_NONEXIST; /* end of story */
627
628 TF             if (entry->ie_epoch > epr->epr_hi) {
629                 entry->ie_epoch = epr->epr_hi;
630                 opc = BTR_PROBE_LE;
631                 break;
632             }
633             return 0;
634
635 T         case VOS_IT_EPC_GE:
636 TF             if (entry->ie_epoch < epr->epr_lo) {
637                 entry->ie_epoch = epr->epr_lo;
638                 opc = BTR_PROBE_GE;
639                 break;
640             }
641             return 0;
642
643 T         case VOS_IT_EPC_LE:
644 TF             if (entry->ie_epoch > epr->epr_lo) {
645                 entry->ie_epoch = epr->epr_lo;
646                 opc = BTR_PROBE_LE;
647                 break;
648             }
649             return 0;
650         }
651         rc = singv_iter_probe_fetch(oiter, opc, entry);
652 TF         if (rc != 0)
653             return rc;
654     }
655 }

659 {
660     vos_iter_entry_t    entry;
661     daos_anchor_t       tmp;
662     int         opc;
663     int         rc;
664
665 TF     if (oiter->it_epc_expr == VOS_IT_EPC_RR)
666 TF         opc = anchor == NULL ? BTR_PROBE_LAST : BTR_PROBE_LE;

673         return rc;

677 TF     if (rc != 0)
678         return rc;

681 TF         if (memcmp(anchor, &tmp, sizeof(tmp)) == 0)

689          */
690     }

692     return rc;
693 }
