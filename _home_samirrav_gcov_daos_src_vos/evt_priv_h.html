269 /** Return true if a rectangle doesn't intersect the filter
270  *
271  * \param[IN]   filter  The optional input filter
272  * \param[IN]   rect    The rectangle to check
273  * \param[IN]   leaf    Indicates if the rectangle is a leaf entry
274  */
275 static inline bool
276 evt_filter_rect(const struct evt_filter *filter, const struct evt_rect *rect,
277         bool leaf)
278 {
279 TF     if (filter == NULL)
280         goto done;
281
282 TF     if (filter->fr_ex.ex_lo > rect->rc_ex.ex_hi ||
283         filter->fr_ex.ex_hi < rect->rc_ex.ex_lo ||
284         filter->fr_epr.epr_hi < rect->rc_epc)
285         return true; /* Rectangle is outside of filter */
286
287     /* In tree rectangle only includes lower bound.  For intermediate
288      * nodes, we can't filter based on lower bound.  For leaf nodes,
289      * we can because it represents a point in time.
290      */
291 TF     if (!leaf)

293
294 TF     if (filter->fr_epr.epr_lo > rect->rc_epc)
295         return true; /* Rectangle is outside of filter */
296 done:
297     return false;
298 }
