34  * EV tree sorted iterator returns logical entry in extent start order, and
35  * the information like: physical entry it belongs to, visibility, is it the
36  * last entry, etc. will be returned as well. One thing worth noting is that
37  * sorted iterator is working on an out-of-band auxiliary index, so it won't
38  * be affected by any in-tree modifications.
39  *
40  * EV tree aggregation is driven by sorted iterator: As the iterator moves on,
41  * the visible logical entries will be queued to form a merge window, and all
42  * the physical entries (no matter if it's fully covered or not) within the
43  * window will be queued too. When the window size reaches certain threshold,
44  * a procedure called merge window flush will be triggered to replace the old
45  * physical entries by the new coalesced physical entries.
46  *
47  * On merge window flush, the queued visible logical entries will be used to
48  * facilitate the data transfer from old physical entries to new coalesced
49  * physical entries. If any old physical entry (not fully covered in current
50  * window) straddles window end, it has to be head-truncated on window flush,
51  * and the remaining part will be processed in next merge window.
52  */

373     } else {
374         D_ERROR("Failed to prepare EV iterator: %d\n", rc);
375     }
376
377     return (sv_empty && ev_empty);
378 }

380 static int
381 vos_agg_akey(daos_handle_t ih, vos_iter_entry_t *entry,
382          struct vos_agg_param *agg_param, unsigned int *acts)
383 {
384     D_ASSERT(agg_param != NULL);
385 TF     if (vos_agg_key_compare(agg_param->ap_akey, entry->ie_key)) {
386         agg_param->ap_akey = entry->ie_key;
387 TF     } else if (!agg_param->ap_discard) {
388         D_DEBUG(DB_EPC, "Skip akey:[%s] aggregation on re-probe\n",
389             (char *)entry->ie_key.iov_buf);
390         *acts |= VOS_ITER_CB_SKIP;
391     }
392
393 TF     if (agg_param->ap_discard) {
394         /*
395          * SV or EV tree is emptied during discard, we need to
396          * verify again if the akey is empty, because new entries
397          * could be inserted while the dicard ULT yielding.
398          */
399 TF         if (agg_param->ap_sub_tree_empty)
400             agg_param->ap_sub_tree_empty = akey_empty(agg_param);
401         return agg_discard_parent(ih, entry, agg_param, acts);
402     }
403
404     /* Reset the max epoch for low-level SV tree iteration */
405     agg_param->ap_max_epoch = 0;
406     /* The merge window for EV tree aggregation should have been closed */
407 TF     if (merge_window_status(&agg_param->ap_window) != MW_CLOSED)
408         D_ASSERTF(false, "Merge window isn't closed.\n");
409
410     return 0;
411 }

413 static int
414 vos_agg_sv(daos_handle_t ih, vos_iter_entry_t *entry,
415        struct vos_agg_param *agg_param, unsigned int *acts)
416 {
417     int rc;
418
419     D_ASSERT(agg_param != NULL);
420     D_ASSERT(entry->ie_epoch != 0);
421
422     /* Discard */
423 TF     if (agg_param->ap_discard)
424         goto delete;
425
426     /*
427      * Aggregate: preserve the first recx which has highest epoch, because
428      * of re-probe, the highest epoch could be iterated multiple times.
429      */
430 TF     if (agg_param->ap_max_epoch == 0 ||
431         agg_param->ap_max_epoch == entry->ie_epoch) {
432         agg_param->ap_max_epoch = entry->ie_epoch;
433         return 0;
434     }
435
436     D_ASSERTF(entry->ie_epoch < agg_param->ap_max_epoch,
437           "max:"DF_U64", cur:"DF_U64"\n",
438           agg_param->ap_max_epoch, entry->ie_epoch);
439
440 delete:
441     rc = agg_del_entry(ih, agg_param->ap_umm, entry, acts);
442 TF     if (rc) {
443         D_ERROR("Failed to delete SV entry: %d\n", rc);
444 TF     } else if (vos_iter_empty(ih) == 1 && agg_param->ap_discard) {
445         agg_param->ap_sub_tree_empty = 1;
446         /* Trigger re-probe in akey iteration */
447         *acts |= VOS_ITER_CB_YIELD;
448     }
449
450     return rc;
451 }

453 static int
454 prepare_segments(struct agg_merge_window *mw)
455 {
456     struct agg_io_context   *io = &mw->mw_io_ctxt;
457     struct agg_phy_ent  *phy_ent;
458     struct agg_lgc_ent  *lgc_ent;
459     struct agg_lgc_seg  *lgc_seg;
460     struct evt_entry_in *ent_in;
461     struct evt_extent    ext;
462     unsigned int         i, seg_max;
463     bool             hole, coalesce;
464
465     /*
466      * Allocate large enough segments array to hold all the coalesced
467      * segments (at most mw_lgc_cnt) and truncated segments (at most
468      * mw_phy_cnt).
469      */
470     D_ASSERT(mw->mw_lgc_cnt > 0);
471     D_ASSERT(mw->mw_phy_cnt > 0);
472     seg_max = MAX((mw->mw_lgc_cnt + mw->mw_phy_cnt), 200);
473 TF     if (io->ic_seg_max < seg_max) {
474         D_REALLOC(lgc_seg, io->ic_segs, seg_max * sizeof(*lgc_seg));
475 TF         if (lgc_seg == NULL)
476             return -DER_NOMEM;
477
478         io->ic_segs = lgc_seg;
479         io->ic_seg_max = seg_max;
480     }
481     memset(io->ic_segs, 0, io->ic_seg_max * sizeof(*lgc_seg));
482     io->ic_seg_cnt = 0;
483
484     /* Generate coalesced segments according to visible logical entries */
485 TF     for (i = 0; i < mw->mw_lgc_cnt; i++) {
486
487         lgc_ent = &mw->mw_lgc_ents[i];
488         phy_ent = lgc_ent->le_phy_ent;
489
490         ext = lgc_ent->le_ext;
491         D_ASSERT(ext1_covers_ext2(&mw->mw_ext, &ext));
492
493 TF         if (i == 0) {
494             coalesce = false;
495 TF         } else if (hole != bio_addr_is_hole(&phy_ent->pe_addr)) {
496             coalesce = false;
497             io->ic_seg_cnt++;
498             D_ASSERT(io->ic_seg_cnt < io->ic_seg_max);
499         } else {
500             coalesce = true;
501         }
502
503         hole = bio_addr_is_hole(&phy_ent->pe_addr);
504         lgc_seg = &io->ic_segs[io->ic_seg_cnt];
505         ent_in = &lgc_seg->ls_ent_in;
506
507 TF         if (!coalesce) {
508             lgc_seg->ls_phy_ent = NULL;
509             lgc_seg->ls_idx_start = i;
510             ent_in->ei_inob = mw->mw_rsize;
511             ent_in->ei_rect.rc_ex.ex_lo = ext.ex_lo;
512             bio_addr_set_hole(&ent_in->ei_addr, hole);
513 TF             if (hole) {
514                 bio_addr_set(&ent_in->ei_addr, DAOS_MEDIA_SCM,
515                          0);
516                 ent_in->ei_inob = 0;
517             }
518         } else {
519             D_ASSERT(ext.ex_lo == ent_in->ei_rect.rc_ex.ex_hi + 1);
520         }
521
522         lgc_seg->ls_idx_end = i;
523         ent_in->ei_rect.rc_ex.ex_hi = ext.ex_hi;
524         /* Merge to highest epoch */
525 TF         if (ent_in->ei_rect.rc_epc < phy_ent->pe_rect.rc_epc)
526             ent_in->ei_rect.rc_epc = phy_ent->pe_rect.rc_epc;
527         /* Merge to highest pool map version */
528 TF         if (ent_in->ei_ver < phy_ent->pe_ver)
529             ent_in->ei_ver = phy_ent->pe_ver;
530     }
531
532     io->ic_seg_cnt++;
533     D_ASSERT(io->ic_seg_cnt < io->ic_seg_max);
534
535     /* Generate truncated segments according to physical entries */
536     d_list_for_each_entry(phy_ent, &mw->mw_phy_ents, pe_link) {
537
538         lgc_seg = &io->ic_segs[io->ic_seg_cnt];
539         ent_in = &lgc_seg->ls_ent_in;
540
541         ext = phy_ent->pe_rect.rc_ex;
542
543         /* The physical entry was truncated on prev window flush */
544 TF         if (phy_ent->pe_off != 0)
545             ext.ex_lo += phy_ent->pe_off;
546
547         D_ASSERT(ext.ex_lo <= ext.ex_hi);
548         D_ASSERT(ext.ex_lo <= mw->mw_ext.ex_hi);
549         D_ASSERT(ext.ex_hi >= mw->mw_ext.ex_lo);
550
551         /*
552          * Physical entry is in window, or it's fully covered (not
553          * visible) in current window.
554          */
555 TF         if (ext.ex_hi <= mw->mw_ext.ex_hi || phy_ent->pe_ref == 0)
556             continue;
557
558         lgc_seg->ls_phy_ent = phy_ent;
559         lgc_seg->ls_idx_start = 0;
560         lgc_seg->ls_idx_end = 0;
561
562         ent_in->ei_inob = mw->mw_rsize;
563         ent_in->ei_rect.rc_ex.ex_lo = mw->mw_ext.ex_hi + 1;
564         ent_in->ei_rect.rc_ex.ex_hi = ext.ex_hi;
565         ent_in->ei_rect.rc_epc = phy_ent->pe_rect.rc_epc;
566         ent_in->ei_ver = phy_ent->pe_ver;
567
568         hole = bio_addr_is_hole(&phy_ent->pe_addr);
569         bio_addr_set_hole(&ent_in->ei_addr, hole);
570 TF         if (hole) {
571             bio_addr_set(&ent_in->ei_addr, DAOS_MEDIA_SCM, 0);
572             ent_in->ei_inob = 0;
573         }
574
575         io->ic_seg_cnt++;
576         D_ASSERT(io->ic_seg_cnt <= io->ic_seg_max);
577     }
578
579     return 0;
580 }

582 static int
583 reserve_segment(struct vos_object *obj, struct agg_io_context *io,
584         daos_size_t size, bio_addr_t *addr)
585 {
586     struct vea_space_info   *vsi;
587     struct vea_hint_context *hint_ctxt;
588     struct vea_resrvd_ext   *nvme_ext;
589     uint32_t         blk_cnt;
590     uint16_t         media;
591     int          rc;
592
593     memset(addr, 0, sizeof(*addr));
594     media = vos_media_select(obj, DAOS_IOD_ARRAY, size);
595
596 TF     if (media == DAOS_MEDIA_SCM) {
597         struct pobj_action  *scm_ext;
598         umem_id_t        mmid;
599
600         D_ASSERT(io->ic_scm_max > io->ic_scm_cnt);
601         D_ASSERT(io->ic_scm_exts != NULL);
602         scm_ext = &io->ic_scm_exts[io->ic_scm_cnt];
603
604 TF         if (vos_obj2umm(obj)->umm_ops->mo_reserve != NULL)
605             mmid = umem_reserve(vos_obj2umm(obj), scm_ext, size);
606         else
607             mmid = umem_alloc(vos_obj2umm(obj), size);
608
609 TF         if (UMMID_IS_NULL(mmid)) {
610             D_ERROR("Reserve "DF_U64" bytes on SCM failed.\n",
611                 size);
612             return -DER_NOSPACE;
613         }
614
615         io->ic_scm_cnt++;
616         bio_addr_set(addr, media, mmid.off);
617         return 0;
618     }
619
620     D_ASSERT(media == DAOS_MEDIA_NVME);
621
622     vsi = obj->obj_cont->vc_pool->vp_vea_info;
623     D_ASSERT(vsi);
624     hint_ctxt = obj->obj_cont->vc_hint_ctxt;
625     D_ASSERT(hint_ctxt);
626     blk_cnt = vos_byte2blkcnt(size);
627
628     rc = vea_reserve(vsi, blk_cnt, hint_ctxt, &io->ic_nvme_exts);
629 TF     if (rc) {
630         D_ERROR("Reserve %u blocks on NVMe failed: %d\n", blk_cnt, rc);
631         return rc;
632     }
633
634     nvme_ext = d_list_entry(io->ic_nvme_exts.prev, struct vea_resrvd_ext,
635                 vre_link);
636     D_ASSERTF(nvme_ext->vre_blk_cnt == blk_cnt, "%u != %u\n",
637           nvme_ext->vre_blk_cnt, blk_cnt);
638     D_ASSERT(nvme_ext->vre_blk_off != 0);
639
640     bio_addr_set(addr, media, nvme_ext->vre_blk_off << VOS_BLK_SHIFT);
641     return 0;
642 }

652 static int
653 fill_one_segment(daos_handle_t ih, struct agg_merge_window *mw,
654          struct agg_lgc_seg *lgc_seg, unsigned int *acts)
655 {
656     struct vos_obj_iter *oiter = vos_hdl2oiter(ih);
657     struct vos_object   *obj = oiter->it_obj;
658     struct agg_io_context   *io = &mw->mw_io_ctxt;
659     struct evt_entry_in *ent_in = &lgc_seg->ls_ent_in;
660     struct agg_phy_ent  *phy_ent;
661     struct bio_io_context   *bio_ctxt;
662     struct bio_sglist    bsgl;
663     daos_sg_list_t       sgl;
664     daos_iov_t       iov;
665     bio_addr_t       addr_dst, addr_src;
666     daos_size_t      seg_size, copy_size, buf_max;
667     struct evt_extent    ext = { 0 };
668     daos_off_t       phy_lo;
669     unsigned int         i, biov_idx = 0;
670     int          rc;
671
672     D_ASSERT(obj != NULL);
673     D_ASSERT(mw->mw_rsize > 0);
674
675 TF     if (bio_addr_is_hole(&ent_in->ei_addr))
676         return 0;
677
678     seg_size = evt_rect_width(&ent_in->ei_rect) * mw->mw_rsize;
679     D_ASSERTF(seg_size > 0, "seg_size:"DF_U64"\n", seg_size);
680
681     buf_max = MAX(seg_size, merge_window_size(mw));
682     buf_max = MAX(buf_max, VOS_MW_FLUSH_THRESH);
683 TF     if (io->ic_buf_len < buf_max) {
684         void *buffer;
685
686         D_REALLOC(buffer, io->ic_buf, buf_max);
687 TF         if (buffer == NULL)
688             return -DER_NOMEM;
689
690         io->ic_buf = buffer;
691         io->ic_buf_len = buf_max;
692     }
693
694     rc = reserve_segment(obj, io, seg_size, &ent_in->ei_addr);
695 TF     if (rc) {
696         D_ERROR("Reserve "DF_U64" segment error: %d\n", seg_size, rc);
697         return rc;
698     }
699
700     /* Copy data from old logical entries into new segment */
701     D_ASSERT(lgc_seg->ls_idx_start <= lgc_seg->ls_idx_end);
702     D_ASSERT(lgc_seg->ls_idx_end < mw->mw_lgc_cnt);
703
704     bio_ctxt = obj->obj_cont->vc_pool->vp_io_ctxt;
705     D_ASSERT(bio_ctxt != NULL);
706
707     rc = bio_sgl_init(&bsgl, lgc_seg->ls_idx_end -
708               lgc_seg->ls_idx_start + 1);
709 TF     if (rc) {
710         D_ERROR("Init bsgl error: %d\n", rc);
711         return rc;
712     }
713
714     iov.iov_buf_len = io->ic_buf_len; /* for sanity check */
715
716     i = lgc_seg->ls_idx_start;
717 TF     while (i <= lgc_seg->ls_idx_end) {
718 TF         if (lgc_seg->ls_phy_ent != NULL) {
719             phy_ent = lgc_seg->ls_phy_ent;
720             ext = ent_in->ei_rect.rc_ex;
721         } else {
722             struct agg_lgc_ent *lgc_ent = &mw->mw_lgc_ents[i];
723
724             phy_ent = lgc_ent->le_phy_ent;
725             ext = lgc_ent->le_ext;
726         }
727         i++;
728
729         D_ASSERT(ext1_covers_ext2(&ent_in->ei_rect.rc_ex, &ext));
730         D_ASSERT(ext1_covers_ext2(&phy_ent->pe_rect.rc_ex, &ext));
731
732         phy_lo = phy_ent->pe_rect.rc_ex.ex_lo;
733 TF         if (phy_ent->pe_off != 0)
734             phy_lo += phy_ent->pe_off;
735
736         D_ASSERT(phy_lo <= phy_ent->pe_rect.rc_ex.ex_hi);
737         D_ASSERT(ext.ex_lo >= phy_lo);
738
739         copy_size = evt_extent_width(&ext) * ent_in->ei_inob;
740
741         addr_src = phy_ent->pe_addr;
742         addr_src.ba_off += (ext.ex_lo - phy_lo) * ent_in->ei_inob;
743
744         D_ASSERT(!bio_addr_is_hole(&addr_src));
745         D_ASSERT(iov.iov_buf_len >= copy_size);
746
747         mark_yield(&addr_src, acts);
748         D_ASSERT(biov_idx < bsgl.bs_nr);
749         bsgl.bs_iovs[biov_idx].bi_buf = NULL;
750         bsgl.bs_iovs[biov_idx].bi_addr = addr_src;
751         bsgl.bs_iovs[biov_idx].bi_data_len = copy_size;
752         biov_idx++;
753
754         D_ASSERT(iov.iov_buf_len >= copy_size);
755         iov.iov_buf_len -= copy_size;
756     }
757     D_ASSERT(seg_size == (io->ic_buf_len - iov.iov_buf_len));
758
759     iov.iov_buf = io->ic_buf;
760     iov.iov_len = 0;
761     iov.iov_buf_len = io->ic_buf_len;
762     sgl.sg_nr = 1;
763     sgl.sg_iovs = &iov;
764     rc = bio_readv(bio_ctxt, &bsgl, &sgl);
765 TF     if (rc) {
766         D_ERROR("Readv for "DF_RECT" error: %d\n",
767             DP_RECT(&ent_in->ei_rect), rc);
768         bio_sgl_fini(&bsgl);
769         return rc;
770     }
771     D_ASSERT(iov.iov_len == seg_size);
772
773     addr_dst = ent_in->ei_addr;
774     D_ASSERT(!bio_addr_is_hole(&addr_dst));
775     mark_yield(&addr_dst, acts);
776
777     iov.iov_buf = io->ic_buf;
778     iov.iov_buf_len = io->ic_buf_len;
779     iov.iov_len = seg_size;
780     rc = bio_write(bio_ctxt, addr_dst, &iov);
781 TF     if (rc)
782         D_ERROR("Write "DF_RECT" error: %d\n",
783             DP_RECT(&ent_in->ei_rect), rc);
784
785     bio_sgl_fini(&bsgl);
786     return rc;
787 }

795     unsigned int         i, scm_max;

815

818             DP_RECT(&lgc_seg->ls_ent_in.ei_rect));

820         rc = fill_one_segment(ih, mw, lgc_seg, acts);

827         }

833 static int
834 insert_segments(daos_handle_t ih, struct agg_merge_window *mw,
835         unsigned int *acts)
836 {
837     struct vos_obj_iter *oiter = vos_hdl2oiter(ih);
838     struct vos_object   *obj = oiter->it_obj;
839     struct agg_io_context   *io = &mw->mw_io_ctxt;
840     struct agg_phy_ent  *phy_ent, *tmp;
841     struct agg_lgc_ent  *lgc_ent;
842     struct agg_lgc_seg  *lgc_seg;
843     struct evt_entry_in *ent_in;
844     struct evt_rect      rect;
845     unsigned int         i, leftovers = 0;
846     int          rc;
847
848     D_ASSERT(obj != NULL);
849     rc = umem_tx_begin(vos_obj2umm(obj), NULL);
850 TF     if (rc)
851         return rc;
852
853     /* Publish SCM reservations */
854 TF     if (io->ic_scm_cnt) {
855         rc = umem_tx_publish(vos_obj2umm(obj), io->ic_scm_exts,
856                      io->ic_scm_cnt);
857         io->ic_scm_cnt = 0;
858 TF         if (rc) {
859             D_ERROR("Publish %u SCM extents error: %d\n",
860                 io->ic_scm_cnt, rc);
861             goto abort;
862         }
863     }
864
865     /* Adjust logical entry queue */
866 TF     for (i = 0; i < mw->mw_lgc_cnt; i++) {
867         lgc_ent = &mw->mw_lgc_ents[i];
868         phy_ent = lgc_ent->le_phy_ent;
869
870         D_ASSERT(ext1_covers_ext2(&mw->mw_ext, &lgc_ent->le_ext));
871         D_ASSERT(phy_ent->pe_ref > 0);
872         phy_ent->pe_ref--;
873         phy_ent->pe_trunc_head = true;
874     }
875     mw->mw_lgc_cnt = 0;
876
877     /* Adjust payload address of truncated physical entries */
878 TF     for (i = 0; i < io->ic_seg_cnt; i++) {
879         lgc_seg = &io->ic_segs[i];
880         ent_in = &io->ic_segs[i].ls_ent_in;
881         phy_ent = lgc_seg->ls_phy_ent;
882
883 TF         if (phy_ent != NULL && !bio_addr_is_hole(&ent_in->ei_addr))
884             phy_ent->pe_addr = ent_in->ei_addr;
885     }
886
887     /* Remove old physical entries from EV tree */
888     d_list_for_each_entry_safe(phy_ent, tmp, &mw->mw_phy_ents, pe_link) {
889         rect = phy_ent->pe_rect;
890
891         D_ASSERT(phy_ent->pe_ref == 0);
892         /* The physical entry was truncated on prev window flush */
893 TF         if (phy_ent->pe_off != 0)
894             rect.rc_ex.ex_lo += phy_ent->pe_off;
895
896         D_ASSERT(rect.rc_ex.ex_lo <= rect.rc_ex.ex_hi);
897         D_ASSERT(rect.rc_ex.ex_lo <= mw->mw_ext.ex_hi);
898         D_ASSERT(rect.rc_ex.ex_hi >= mw->mw_ext.ex_lo);
899
900         /*
901          * The physical entry spans window end, but is fully covered
902          * in current window, keep it intact.
903          */
904 TF         if (rect.rc_ex.ex_hi > mw->mw_ext.ex_hi &&
905             !phy_ent->pe_trunc_head) {
906             leftovers++;
907             continue;
908         }
909
910         mark_yield(&phy_ent->pe_addr, acts);
911
912         rc = evt_delete(oiter->it_hdl, &rect, NULL);
913 TF         if (rc) {
914             D_ERROR("Delete "DF_RECT" error: %d\n",
915                 DP_RECT(&rect), rc);
916             goto abort;
917         }
918
919         /* Physical entry is in window */
920 TF         if (rect.rc_ex.ex_hi <= mw->mw_ext.ex_hi) {
921             d_list_del(&phy_ent->pe_link);
922             D_FREE_PTR(phy_ent);
923             D_ASSERT(mw->mw_phy_cnt > 0);
924             mw->mw_phy_cnt--;
925             continue;
926         }
927
928         /* Update extent start of truncated physical entry */
929         rect.rc_ex.ex_lo = mw->mw_ext.ex_hi + 1;
930         phy_ent->pe_off = rect.rc_ex.ex_lo -
931                 phy_ent->pe_rect.rc_ex.ex_lo;
932         phy_ent->pe_trunc_head = false;
933         leftovers++;
934     }
935     D_ASSERT(leftovers == mw->mw_phy_cnt);
936
937     /* Clear window size */
938     mw->mw_ext.ex_lo = mw->mw_ext.ex_hi = 0;
939
940     /* Insert new segments into EV tree */
941 TF     for (i = 0; i < io->ic_seg_cnt; i++) {
942         ent_in = &io->ic_segs[i].ls_ent_in;
943
944         rc = evt_insert(oiter->it_hdl, ent_in);
945 TF         if (rc) {
946             D_ERROR("Insert segment "DF_RECT" error: %d\n",
947                 DP_RECT(&ent_in->ei_rect), rc);
948             goto abort;
949         }
950     }
951
952     /* Publish NVMe reservations */
953     rc = vos_publish_blocks(obj, &io->ic_nvme_exts, true);
954 TF     if (rc) {
955         D_ERROR("Publish NVMe extents error: %d\n", rc);
956         goto abort;
957     }
958 abort:
959 TF     if (rc)
960         rc = umem_tx_abort(vos_obj2umm(obj), rc);
961     else
962         rc = umem_tx_commit(vos_obj2umm(obj));
963
964     return rc;
965 }

1006 static bool
1007 need_flush(struct agg_merge_window *mw)
1008 {
1009     struct agg_phy_ent  *phy_ent;
1010     struct agg_lgc_ent  *lgc_ent;
1011     struct evt_extent    lgc_ext, phy_ext;
1012     int          i;
1013     bool             hole = false;
1014
1015 TF     for (i = 0; i < mw->mw_lgc_cnt; i++) {
1016         lgc_ent = &mw->mw_lgc_ents[i];
1017         phy_ent = lgc_ent->le_phy_ent;
1018         phy_ext = phy_ent->pe_rect.rc_ex;
1019         lgc_ext = lgc_ent->le_ext;
1020
1021         /*
1022          * Any physical entry is partially covered, or appeared
1023          * in other window
1024          */
1025 TF         if (lgc_ext.ex_lo != phy_ext.ex_lo ||
1026             lgc_ext.ex_hi != phy_ext.ex_hi)
1027             return true;
1028
1029         /* If any consecutive visible entries can be merged */
1030 TF         if (i != 0 && hole == bio_addr_is_hole(&phy_ent->pe_addr))
1031             return true;
1032
1033         hole = bio_addr_is_hole(&phy_ent->pe_addr);
1034     }
1035
1036     /* Any invisible physical entries ? */
1037 TF     if (mw->mw_lgc_cnt != mw->mw_phy_cnt)
1038         return true;
1039
1040     clear_merge_window(mw);
1041     D_DEBUG(DB_EPC, "Skip window flush "DF_EXT"\n", DP_EXT(&mw->mw_ext));
1042
1043     return false;
1044 }

1046 static int
1047 flush_merge_window(daos_handle_t ih, struct agg_merge_window *mw,
1048            unsigned int *acts)
1049 {
1050     int rc;
1051
1052     D_ASSERT(merge_window_status(mw) == MW_OPENED);
1053
1054     /*
1055      * If no new updates in an already aggregated window, window flush will
1056      * be skipped, otherwise, all the data within the window will be
1057      * migrated to a new location, such batch data migration is good for
1058      * anti-fragmentaion.
1059      */
1060 TF     if (!need_flush(mw))
1061         return 0;
1062
1063     /* Prepare the new segments to be inserted */
1064     rc = prepare_segments(mw);
1065 TF     if (rc) {
1066         D_ERROR("Prepare segments "DF_EXT" error: %d\n",
1067             DP_EXT(&mw->mw_ext), rc);
1068         goto out;
1069     }
1070
1071     /* Transfer data from old logical records to reserved new segments */
1072     rc = fill_segments(ih, mw, acts);
1073 TF     if (rc) {
1074         D_ERROR("Fill segments "DF_EXT" error: %d\n",
1075             DP_EXT(&mw->mw_ext), rc);
1076         goto out;
1077     }
1078
1079     /* Replace the old logical records with new segments in EV tree */
1080     rc = insert_segments(ih, mw, acts);
1081 TF     if (rc) {
1082         D_ERROR("Insert segments "DF_EXT" error: %d\n",
1083             DP_EXT(&mw->mw_ext), rc);
1084         goto out;
1085     }
1086 out:
1087     cleanup_segments(ih, mw, rc);
1088     return rc;
1089 }

1091 static bool
1092 trigger_flush(struct agg_merge_window *mw, struct evt_extent *lgc_ext)
1093 {
1094     struct evt_extent *w_ext = &mw->mw_ext;
1095
1096     D_ASSERT(w_ext->ex_lo <= lgc_ext->ex_lo);
1097     /* Empty or closed merge window */
1098 TF     if (merge_window_status(mw) == MW_CLOSED ||
1099         merge_window_status(mw) == MW_FLUSHED)
1100         return false;
1101
1102     /*
1103      * Window is formed by visible logical entries, must have no
1104      * overlapping.
1105      */
1106     D_ASSERTF(w_ext->ex_hi < lgc_ext->ex_lo,
1107           "win:"DF_EXT", lgc_ent:"DF_EXT"\n",
1108           DP_EXT(w_ext), DP_EXT(lgc_ext));
1109
1110     /* Window is large enough */
1111 TF     if (merge_window_size(mw) >= mw->mw_flush_thresh)
1112         return true;
1113
1114     /* Trigger flush when entry is disjoint with window */
1115     return !((w_ext->ex_hi + 1) == lgc_ext->ex_lo);
1116 }
