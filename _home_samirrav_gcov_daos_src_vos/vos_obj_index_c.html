314 /**
315  * Punch a durable object, it will generate a new incarnation with the same
316  * ID in OI table.
317  */
318 int
319 vos_oi_punch(struct vos_container *cont, daos_unit_oid_t oid,
320          daos_epoch_t epoch, uint32_t flags, struct vos_obj_df *obj)
321 {
322     struct oi_hkey  *hkey;
323     struct oi_key    key;
324     daos_iov_t   key_iov;
325     daos_iov_t   val_iov;
326     int      rc = 0;
327     bool         replay = (flags & VOS_OF_REPLAY_PC);
328
329     D_DEBUG(DB_TRACE, "Punch obj "DF_UOID", epoch="DF_U64".\n",
330         DP_UOID(oid), epoch);
331
332 TF     if (obj->vo_oi_attr & VOS_OI_PUNCHED &&
333         obj->vo_latest == epoch) {
334         D_DEBUG(DB_TRACE, "Punch the same epoch.\n");
335         goto out;
336     }
337
338 TF     if (obj->vo_latest >= epoch && !replay) {
339         D_ERROR("Underwrite is allowed only for replaying punch "
340             DF_U64" >= "DF_U64"\n", obj->vo_latest, epoch);
341         rc = -DER_NO_PERM;
342         goto out;
343     }
344
345     /* create a new incarnation for the punch */
346     hkey = &key.oi_hkey;
347     hkey->oi_oid    = oid;
348     key.oi_epc_lo = hkey->oi_epc = epoch;
349 TF     if (!replay) /* We steal the subtree from the max epoch */
350         key.oi_epc_lo = obj->vo_earliest;
351     daos_iov_set(&key_iov, &key, sizeof(key));
352
353     rc = dbtree_upsert(cont->vc_btr_hdl, BTR_PROBE_EQ, DAOS_INTENT_PUNCH,
354                &key_iov, &val_iov);
355 TF     if (rc != 0 || replay)
356         goto out;
357
358 TF     if (vos_dth_get() == NULL) {
359         struct vos_obj_df *tmp;
360
361         D_ASSERT((obj->vo_oi_attr & VOS_OI_PUNCHED) == 0);
362
363         tmp = (struct vos_obj_df *)val_iov.iov_buf;
364         D_ASSERT(tmp != obj);
365         /* the new incarnation should take over the subtree from
366          * the originally highest incarnation.
367          */
368         tmp->vo_tree = obj->vo_tree;
369         tmp->vo_incarnation = obj->vo_incarnation;
370         /* NB: this changed vos_object_df, which means cache might
371          * be stale, so other callers should invalidate cache.

373         umem_tx_add_ptr(&cont->vc_pool->vp_umm, obj, sizeof(*obj));
374         memset(&obj->vo_tree, 0, sizeof(obj->vo_tree));
375         obj->vo_latest = 0;
376         obj->vo_earliest = DAOS_EPOCH_MAX;
377         obj->vo_incarnation++; /* cache should be revalidated */
378     } else {
379         struct umem_instance    *umm = btr_hdl2umm(cont->vc_btr_hdl);
380
381         rc = vos_dtx_register_record(umm, umem_ptr2off(umm, obj),
382                          DTX_RT_OBJ, DTX_RF_EXCHANGE_SRC);
383     }
384
385 out:
386 TF     if (rc != 0)

499 /**
500  * This function checks if the current object can match the condition, it
501  * returns immediately on true, otherwise it will move the iterator cursor
502  * to the object matching the condition.
503  */
504 static int
505 oi_iter_match_probe(struct vos_iterator *iter)
506 {
507     struct vos_oi_iter  *oiter  = iter2oiter(iter);
508     daos_epoch_range_t  *epr    = &oiter->oit_epr;
509     char            *str    = NULL;
510     int          rc;
511
512 TF     if (oiter->oit_epr.epr_lo == 0 &&
513         oiter->oit_epr.epr_hi == DAOS_EPOCH_MAX)
514         goto out; /* no condition so it's a match */
515
516     while (1) {
517         struct vos_obj_df *obj;
518         struct oi_hkey     hkey;
519         int        probe;
520         daos_iov_t     iov;
521
522         rc = dbtree_iter_fetch(oiter->oit_hdl, NULL, &iov, NULL);
523 TF         if (rc != 0) {
524             str = "fetch";
525             goto failed;
526         }
527
528         D_ASSERT(iov.iov_len == sizeof(struct vos_obj_df));
529         obj = (struct vos_obj_df *)iov.iov_buf;
530
531         probe = 0;
532 TF         if (obj->vo_earliest > epr->epr_hi) {
533             /* NB: The object was created/updated only after the
534              * range in the condition so the object and all
535              * remaining incarnations of it can be skipped.
536              */
537             probe = BTR_PROBE_GT;
538             hkey.oi_epc = DAOS_EPOCH_MAX;
539 TF         } else if (obj->vo_oi_attr & VOS_OI_PUNCHED &&
540                obj->vo_latest <= epr->epr_lo) {
541             /* NB: Object was punched before our range.
542              * Probe again using our range condition to select
543              * either the next object or a subsequent incarnation
544              * of the current one.
545              */
546             probe = BTR_PROBE_GT;
547             hkey.oi_epc = epr->epr_lo;
548         } else {
549             /* Matches the condition. */
550             break;
551         }
552
553         hkey.oi_oid = obj->vo_id;
554         daos_iov_set(&iov, &hkey, sizeof(hkey));
555         rc = dbtree_iter_probe(oiter->oit_hdl, probe,
556                        vos_iter_intent(iter), &iov, NULL);
557 TF         if (rc != 0) {
558             str = "probe";
559             goto failed;
560         }
561     }
562  out:
563     return 0;
564  failed:
565 TF     if (rc == -DER_NONEXIST)
566         return 0; /* end of iteration, not a failure */
567
568     D_ERROR("iterator %s failed, rc=%d\n", str, rc);
569     return rc;
570 }
