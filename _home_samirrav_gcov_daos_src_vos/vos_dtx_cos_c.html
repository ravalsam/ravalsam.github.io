35 /* The record for the DTX CoS B+tree in DRAM. Each record contains current
36  * committable DTXs that modify (update or punch) something under the same
37  * object and the same dkey.
38  */
39 struct dtx_cos_rec {
40     daos_unit_oid_t      dcr_oid;
41     /* The list in DRAM for the DTXs that modify the same object/dkey.
42      * Per-dkey based.
43      */
44     d_list_t         dcr_update_list;
45     d_list_t         dcr_punch_list;
46     /* The number of the UPDATE DTXs in the dcr_update_list. */
47     int          dcr_update_count;
48     /* The number of the PUNCH DTXs in the dcr_punch_list. */
49     int          dcr_punch_count;

51     struct vos_container    *dcr_cont;

54 /* Above dtx_cos_rec is consisted of a series of dtx_cos_rec_child uints.
