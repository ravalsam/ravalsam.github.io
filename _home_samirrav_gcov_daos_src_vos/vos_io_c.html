383                    csum_nr * ent->en_csum.cs_len);
384             csum_copied += csum_nr * ent->en_csum.cs_len;
385
386             csum->cs_nr += csum_nr;
387
388             /** These should all be the same for each entry,
389              * so it's okay to copy over previously written
390              * value
391              */
392             csum->cs_len = ent->en_csum.cs_len;
393             csum->cs_type = ent->en_csum.cs_type;
394         }
395
396
397
398         biov.bi_data_len = nr * ent_array.ea_inob;
399         biov.bi_addr = ent->en_addr;
400         rc = iod_fetch(ioc, &biov);
401 TF         if (rc != 0)
402             goto failed;
403
404         index = lo + nr;
405     }
406
407     D_ASSERT(index <= end);
408 TF     if (index < end)
409         holes += end - index;
410
411 TF     if (holes != 0) { /* trailing holes */
412         biov_set_hole(&biov, holes * ent_array.ea_inob);
413         rc = iod_fetch(ioc, &biov);
414 TF         if (rc != 0)
415             goto failed;
416     }
417 TF     if (rsize_p)
418         *rsize_p = rsize;
419 failed:
420     evt_ent_array_fini(&ent_array);
421     return rc;
422 }

448 static int
449 akey_fetch(struct vos_io_context *ioc, daos_handle_t ak_toh)
450 {
451     daos_iod_t  *iod = &ioc->ic_iods[ioc->ic_sgl_at];
452     daos_epoch_t     epoch = ioc->ic_epoch;
453     struct vos_krec_df *krec = NULL;
454     daos_handle_t    toh = DAOS_HDL_INVAL;
455     int      i, rc;
456     int      flags = 0;
457
458     D_DEBUG(DB_IO, "akey %d %s fetch %s eph "DF_U64"\n",
459         (int)iod->iod_name.iov_len, (char *)iod->iod_name.iov_buf,
460         iod->iod_type == DAOS_IOD_ARRAY ? "array" : "single",
461         ioc->ic_epoch);
462
463 TF     if (iod->iod_type == DAOS_IOD_ARRAY)
464         flags |= SUBTR_EVT;
465
466 TF     if (iod->iod_type == DAOS_IOD_SINGLE) {
467 TF         if (iod->iod_eprs && iod->iod_eprs[0].epr_lo != 0)
468             epoch = iod->iod_eprs[0].epr_lo;
469
470         rc = key_tree_prepare(ioc->ic_obj, epoch, ak_toh, VOS_BTR_AKEY,
471                       &iod->iod_name, flags,
472                       DAOS_INTENT_DEFAULT, NULL, &toh);
473 TF         if (rc != 0) {
474 TF             if (rc == -DER_NONEXIST) {
475                 D_DEBUG(DB_IO, "Nonexistent akey %.*s\n",
476                     (int)iod->iod_name.iov_len,
477                     (char *)iod->iod_name.iov_buf);
478                 iod_empty_sgl(ioc, ioc->ic_sgl_at);
479                 rc = 0;
480             }
481             return rc;
482         }
483
484         rc = akey_fetch_single(toh, ioc->ic_epoch, &iod->iod_size, ioc);
485
486         key_tree_release(toh, false);
487
488         return rc;
489     }
490
491     iod->iod_size = 0;
492 TF     for (i = 0; i < iod->iod_nr; i++) {
493         daos_size_t rsize;
494 TF         if (iod->iod_eprs && iod->iod_eprs[i].epr_lo)
495             epoch = iod->iod_eprs[i].epr_lo;
496
497         /* If epoch on each iod_eprs are out of boundary, then it needs
498          * to re-prepare the key tree.
499          */
500 TF         if (daos_handle_is_inval(toh) || (epoch > krec->kr_latest ||
501                           epoch < krec->kr_earliest)) {
502 TF             if (!daos_handle_is_inval(toh)) {
503                 key_tree_release(toh, true);
504                 toh = DAOS_HDL_INVAL;
505             }
506
507             D_DEBUG(DB_IO, "repare the key tree for eph "DF_U64"\n",
508                 epoch);
509             rc = key_tree_prepare(ioc->ic_obj, epoch, ak_toh,
510                           VOS_BTR_AKEY, &iod->iod_name,
511                           flags, DAOS_INTENT_DEFAULT,
512                           &krec, &toh);
513 TF             if (rc != 0) {
514 TF                 if (rc == -DER_NONEXIST) {
515                     D_DEBUG(DB_IO, "Nonexist akey %.*s\n",
516                         (int)iod->iod_name.iov_len,
517                         (char *)iod->iod_name.iov_buf);
518                     rc = 0;
519                     continue;
520                 }
521                 return rc;
522             }
523         }
524
525         D_DEBUG(DB_IO, "fetch %d eph "DF_U64"\n", i, epoch);
526         rc = akey_fetch_recx(toh, epoch, &iod->iod_recxs[i],
527                      daos_iod_csum(iod, i), &rsize, ioc);
528 TF         if (rc != 0) {
529             D_DEBUG(DB_IO, "Failed to fetch index %d: %d\n", i, rc);
530             goto out;
531         }
532
533         /*
534          * Empty tree or all holes, DAOS array API relies on zero
535          * iod_size to see if an array cell is empty.
536          */
537 TF         if (rsize == 0)
538             continue;
539
540 TF         if (iod->iod_size == 0)
541             iod->iod_size = rsize;
542
543 TF         if (iod->iod_size != rsize) {
544             D_ERROR("Cannot support mixed record size "
545                 DF_U64"/"DF_U64"\n", iod->iod_size, rsize);
546             rc = -DER_INVAL;
547             goto out;
548         }
549     }
550
551     ioc_trim_tail_holes(ioc);
552 out:
553 TF     if (!daos_handle_is_inval(toh))
554         key_tree_release(toh, true);
555     return rc;
556 }

763 static int
764 akey_update(struct vos_io_context *ioc, uint32_t pm_ver, daos_handle_t ak_toh,
765         daos_epoch_range_t *dkey_epr)
766 {
767     struct vos_object  *obj = ioc->ic_obj;
768     struct vos_krec_df *krec = NULL;
769     daos_iod_t     *iod = &ioc->ic_iods[ioc->ic_sgl_at];
770     bool            is_array = (iod->iod_type == DAOS_IOD_ARRAY);
771     int         flags = SUBTR_CREATE;
772     daos_epoch_t        epoch = ioc->ic_epoch;
773     daos_epoch_range_t  akey_epr = {DAOS_EPOCH_MAX, 0};
774     daos_handle_t       toh = DAOS_HDL_INVAL;
775     int         i;
776     int         rc = 0;
777
778     D_DEBUG(DB_TRACE, "akey %d %s update %s value eph "DF_U64"\n",
779         (int)iod->iod_name.iov_len, (char *)iod->iod_name.iov_buf,
780         is_array ? "array" : "single", ioc->ic_epoch);
781
782 TF     if (is_array)
783         flags |= SUBTR_EVT;
784
785 TF     if (iod->iod_eprs == NULL || iod->iod_eprs[0].epr_lo == 0)
786         akey_epr.epr_hi = akey_epr.epr_lo = epoch;
787
788 TF     if (iod->iod_type == DAOS_IOD_SINGLE) {
789 TF         if (iod->iod_eprs && iod->iod_eprs[0].epr_lo != 0) {
790             epoch = iod->iod_eprs[0].epr_lo;
791             update_bounds(&akey_epr, &iod->iod_eprs[0]);
792         }
793         rc = key_tree_prepare(obj, epoch, ak_toh, VOS_BTR_AKEY,
794                       &iod->iod_name, flags, DAOS_INTENT_UPDATE,
795                       &krec, &toh);
796 TF         if (rc != 0)
797             return rc;
798
799         D_DEBUG(DB_IO, "Single update eph "DF_U64"\n", epoch);
800         rc = akey_update_single(toh, epoch, pm_ver, iod->iod_size, ioc);
801 TF         if (rc)
802             goto failed;
803         goto out;
804     } /* else: array */
805
806 TF     for (i = 0; i < iod->iod_nr; i++) {
807 TF         if (iod->iod_eprs && iod->iod_eprs[i].epr_lo != 0) {
808             update_bounds(&akey_epr, &iod->iod_eprs[i]);
809             epoch = iod->iod_eprs[i].epr_lo;
810         }
811
812 TF         if (daos_handle_is_inval(toh) ||
813             (epoch > krec->kr_latest || epoch < krec->kr_earliest)) {
814 TF             if (!daos_handle_is_inval(toh)) {
815                 key_tree_release(toh, is_array);
816                 toh = DAOS_HDL_INVAL;
817             }
818
819             /* re-prepare the tree if epoch is different */
820             rc = key_tree_prepare(obj, epoch, ak_toh, VOS_BTR_AKEY,
821                           &iod->iod_name, flags,
822                           DAOS_INTENT_UPDATE, &krec, &toh);
823 TF             if (rc != 0)
824                 return rc;
825         }
826
827         D_DEBUG(DB_IO, "Array update %d eph "DF_U64"\n", i, epoch);
828         daos_csum_buf_t *csum = daos_iod_csum(iod, i);
829         rc = akey_update_recx(toh, epoch, pm_ver, &iod->iod_recxs[i],
830                       csum, iod->iod_size, ioc);
831 TF         if (rc != 0)
832             goto failed;
833     }
834 out:
835     rc = vos_df_ts_update(ioc->ic_obj, &krec->kr_latest, &akey_epr);
836     update_bounds(dkey_epr, &akey_epr);
837 failed:
838 TF     if (!daos_handle_is_inval(toh))
839         key_tree_release(toh, is_array);
840
841     return rc;
842 }
