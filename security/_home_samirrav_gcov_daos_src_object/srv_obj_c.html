410 /**
411  * Pack nrs in sgls inside the reply, so the client can update
412  * sgls before it returns to application. Note: this is only
413  * needed for bulk transfer, for inline transfer, it will pack
414  * the complete sgls inside the req/reply, see obj_shard_rw().
415  */
416 static int
417 ds_obj_update_nrs_in_reply(crt_rpc_t *rpc, daos_handle_t ioh,
418                daos_sg_list_t *sgls)
419 {
420     struct obj_rw_in    *orw = crt_req_get(rpc);
421     struct obj_rw_out   *orwo = crt_reply_get(rpc);
422     uint32_t        *nrs;
423     uint32_t        nrs_count = orw->orw_nr;
424     int         i;
425
426 TF     if (nrs_count == 0)
427         return 0;
428
429     /* return num_out for sgl */
430     orwo->orw_nrs.ca_count = nrs_count;
431     D_ALLOC(orwo->orw_nrs.ca_arrays,
432         nrs_count * sizeof(uint32_t));
433
434 TF     if (orwo->orw_nrs.ca_arrays == NULL)
435         return -DER_NOMEM;
436
437     nrs = orwo->orw_nrs.ca_arrays;
438 TF     for (i = 0; i < nrs_count; i++) {
439         struct bio_sglist   *bsgl;
440         daos_sg_list_t      *sgl;
441
442 TF         if (sgls != NULL) {
443             sgl = &sgls[i];
444             D_ASSERT(sgl != NULL);
445             nrs[i] = sgl->sg_nr_out;
446         } else {
447             bsgl = vos_iod_sgl_at(ioh, i);
448             D_ASSERT(bsgl != NULL);
449             nrs[i] = bsgl->bs_nr_out;
450         }
451     }
452
453     return 0;
454 }

456 /**

458  * will never associate a particular container, then the contaier structure
459  * will be returned to \a contp.

461 static int
462 ds_check_container(uuid_t cont_hdl_uuid, uuid_t cont_uuid,
463            struct ds_cont_hdl **hdlp, struct ds_cont **contp)
464 {
465     struct ds_cont_hdl  *cont_hdl;

471              DP_UUID(cont_hdl_uuid));
472         D_GOTO(failed, rc = -DER_NO_HDL);
473     }
474
475 TF     if (cont_hdl->sch_cont != NULL) { /* a regular container */
476         *contp = cont_hdl->sch_cont;
477         D_GOTO(out, rc = 0);
478     }

480 TF     if (!is_rebuild_container(cont_hdl->sch_pool->spc_uuid,

492         D_GOTO(failed, rc = -DER_STALE);
493
494     D_DEBUG(DB_TRACE, DF_UUID"/%p is rebuild cont hdl\n",
495         DP_UUID(cont_hdl_uuid), cont_hdl);
496
497     /* load or create VOS container on demand */
498     rc = ds_cont_lookup(cont_hdl->sch_pool->spc_uuid, cont_uuid, contp);
499 TF     if (rc)
500         D_GOTO(failed, rc);
501 out:
502     *hdlp = cont_hdl;
503 failed:
504 TF     if (cont_hdl != NULL && rc != 0)
505         ds_cont_hdl_put(cont_hdl);
506
507     return rc;

510 void
511 ds_obj_rw_echo_handler(crt_rpc_t *rpc)
512 {
513     struct obj_rw_in    *orw = crt_req_get(rpc);
514     struct obj_rw_out   *orwo = crt_reply_get(rpc);
515     struct obj_tls      *tls;
516     daos_iod_t      *iod;
517     daos_sg_list_t      *p_sgl;
518     crt_bulk_op_t       bulk_op;
519     bool            bulk_bind;
520     int         i;
521     int         rc = 0;
522
523     D_DEBUG(DB_TRACE, "opc %d "DF_UOID" dkey %d %s tgt/xs %d/%d eph "
524         DF_U64".\n", opc_get(rpc->cr_opc), DP_UOID(orw->orw_oid),
525         (int)orw->orw_dkey.iov_len, (char *)orw->orw_dkey.iov_buf,
526         dss_get_module_info()->dmi_tgt_id,
527         dss_get_module_info()->dmi_xs_id,
528         orw->orw_epoch);
529
530 TF     if (opc_get(rpc->cr_opc) == DAOS_OBJ_RPC_FETCH) {
531         rc = ds_obj_update_sizes_in_reply(rpc);
532 TF         if (rc)
533             D_GOTO(out, rc);
534     }
535
536     /* Inline fetch/update */
537 TF     if (orw->orw_bulks.ca_arrays == NULL && orw->orw_bulks.ca_count == 0) {
538 TF         if (opc_get(rpc->cr_opc) == DAOS_OBJ_RPC_FETCH) {
539             orwo->orw_sgls.ca_count = orw->orw_sgls.ca_count;
540             orwo->orw_sgls.ca_arrays = orw->orw_sgls.ca_arrays;
541         }
542         D_GOTO(out, rc);
543     }
544
545     /* Only support 1 iod now */
546     D_ASSERT(orw->orw_iods.ca_count == 1);
547     iod = orw->orw_iods.ca_arrays;
548
549     tls = obj_tls_get();
550     p_sgl = &tls->ot_echo_sgl;
551
552     /* Let's check if tls already have enough buffer */
553 TF     if (p_sgl->sg_nr < iod->iod_nr) {
554         daos_sgl_fini(p_sgl, true);
555         rc = daos_sgl_init(p_sgl, iod->iod_nr);
556 TF         if (rc)
557             D_GOTO(out, rc);
558
559         p_sgl->sg_nr_out = p_sgl->sg_nr;
560     }
561
562 TF     for (i = 0; i < iod->iod_nr; i++) {
563         daos_size_t size = iod->iod_size;
564
565 TF         if (size == DAOS_REC_ANY)
566             size = sizeof(uint64_t);
567
568 TF         if (iod->iod_type == DAOS_IOD_ARRAY) {
569             D_ASSERT(iod->iod_recxs);
570             size *= iod->iod_recxs[i].rx_nr;
571         }
572
573         /* Check each vector */
574 TF         if (p_sgl->sg_iovs[i].iov_buf_len < size) {
575 TF             if (p_sgl->sg_iovs[i].iov_buf != NULL)
576                 D_FREE(p_sgl->sg_iovs[i].iov_buf);
577
578             D_ALLOC(p_sgl->sg_iovs[i].iov_buf, size);
579             /* obj_tls_fini() will free these buffer */
580 TF             if (p_sgl->sg_iovs[i].iov_buf == NULL)
581                 D_GOTO(out, rc = -DER_NOMEM);
582             p_sgl->sg_iovs[i].iov_buf_len = size;
583             p_sgl->sg_iovs[i].iov_len = size;
584         }
585     }
586
587     orwo->orw_sgls.ca_count = 0;
588     orwo->orw_sgls.ca_arrays = NULL;
589 TF     if (opc_get(rpc->cr_opc) == DAOS_OBJ_RPC_FETCH) {
590         rc = ds_obj_update_nrs_in_reply(rpc, DAOS_HDL_INVAL, p_sgl);
591 TF         if (rc != 0)
592             D_GOTO(out, rc);
593         bulk_op = CRT_BULK_PUT;
594     } else {
595         bulk_op = CRT_BULK_GET;
596     }
597
598     bulk_bind = orw->orw_flags & ORF_BULK_BIND;
599     rc = ds_bulk_transfer(rpc, bulk_op, bulk_bind, orw->orw_bulks.ca_arrays,
600                   DAOS_HDL_INVAL, &p_sgl, orw->orw_nr);
601
602 out:
603     orwo->orw_ret = rc;
604     orwo->orw_map_version = orw->orw_map_ver;
605 }

676 static int
677 ds_obj_rw_local_hdlr(crt_rpc_t *rpc, uint32_t tag, struct ds_cont_hdl *cont_hdl,
678              struct ds_cont *cont, daos_handle_t *ioh, bool update,
679              uint32_t map_version, struct dtx_handle *dth)
680 {
681     struct obj_rw_in    *orw = crt_req_get(rpc);
682     struct obj_rw_out   *orwo = crt_reply_get(rpc);
683     struct bio_desc     *biod;
684     crt_bulk_op_t        bulk_op;
685     bool             rma;
686     bool             bulk_bind;
687     int          rc, err;
688
689 TF     if (daos_oc_echo_type(daos_obj_id2class(orw->orw_oid.id_pub)) ||
690         (daos_io_bypass & IOBP_TARGET)) {
691         ds_obj_rw_echo_handler(rpc);
692         D_GOTO(out, rc = 0);
693     }
694
695     D_DEBUG(DB_TRACE, "opc %d "DF_UOID" dkey %d %s tag %d eph "DF_U64".\n",
696         opc_get(rpc->cr_opc), DP_UOID(orw->orw_oid),
697         (int)orw->orw_dkey.iov_len, (char *)orw->orw_dkey.iov_buf,
698         tag, orw->orw_epoch);
699     rma = (orw->orw_bulks.ca_arrays != NULL ||
700            orw->orw_bulks.ca_count != 0);
701
702     /* Prepare IO descriptor */
703 TF     if (update) {
704         bulk_op = CRT_BULK_GET;
705         rc = vos_update_begin(cont->sc_hdl, orw->orw_oid,
706                       orw->orw_epoch, &orw->orw_dkey,
707                       orw->orw_nr, orw->orw_iods.ca_arrays,
708                       ioh, dth);
709 TF         if (rc) {
710             D_ERROR(DF_UOID" Update begin failed: %d\n",
711                 DP_UOID(orw->orw_oid), rc);
712             goto out;
713         }
714     } else {
715         bool size_fetch = (!rma && orw->orw_sgls.ca_arrays == NULL);
716
717         bulk_op = CRT_BULK_PUT;
718         rc = vos_fetch_begin(cont->sc_hdl, orw->orw_oid, orw->orw_epoch,
719                      &orw->orw_dkey, orw->orw_nr,
720                      orw->orw_iods.ca_arrays, size_fetch, ioh);
721 TF         if (rc) {
722             D_ERROR(DF_UOID" Fetch begin failed: %d\n",
723                 DP_UOID(orw->orw_oid), rc);
724             goto out;
725         }
726
727         rc = ds_obj_update_sizes_in_reply(rpc);
728 TF         if (rc != 0)
729             goto out;
730
731 TF         if (rma) {
732             orwo->orw_sgls.ca_count = 0;
733             orwo->orw_sgls.ca_arrays = NULL;
734
735             rc = ds_obj_update_nrs_in_reply(rpc, *ioh, NULL);
736 TF             if (rc != 0)
737                 goto out;
738         } else {
739             orwo->orw_sgls.ca_count = orw->orw_sgls.ca_count;
740             orwo->orw_sgls.ca_arrays = orw->orw_sgls.ca_arrays;
741         }
742     }
743
744     biod = vos_ioh2desc(*ioh);
745     rc = bio_iod_prep(biod);
746 TF     if (rc) {
747         D_ERROR(DF_UOID" bio_iod_prep failed: %d.\n",
748             DP_UOID(orw->orw_oid), rc);
749         goto out;
750     }
751
752 TF     if (rma) {
753         bulk_bind = orw->orw_flags & ORF_BULK_BIND;
754         rc = ds_bulk_transfer(rpc, bulk_op, bulk_bind,
755             orw->orw_bulks.ca_arrays, *ioh, NULL, orw->orw_nr);
756 TF     } else if (orw->orw_sgls.ca_arrays != NULL) {
757         rc = bio_iod_copy(biod, orw->orw_sgls.ca_arrays, orw->orw_nr);
758     }
759
760 TF     if (rc == -DER_OVERFLOW) {
761         rc = -DER_REC2BIG;
762         D_ERROR(DF_UOID" ds_bulk_transfer/bio_iod_copy failed, rc %d",
763             DP_UOID(orw->orw_oid), rc);
764     }
765
766     err = bio_iod_post(biod);
767 TF     rc = rc ? : err;
768 out:
769     rc = ds_obj_rw_complete(rpc, cont_hdl, *ioh, rc, map_version, dth);
770     return rc;
771 }

821

824
825     rc = ds_check_container(orw->orw_co_hdl, orw->orw_co_uuid,
826                 &cont_hdl, &cont);
827 TF     if (rc)

830 TF     if (update && !(cont_hdl->sch_capas & DAOS_COO_RW)) {

839
840     D_DEBUG(DB_TRACE, "rpc %p opc %d "DF_UOID" dkey %d %s tag/xs %d/%d eph "

844         tag, dss_get_module_info()->dmi_xs_id, orw->orw_epoch,
845         orw->orw_map_ver, map_ver, DP_DTI(&orw->orw_dti));
846

848         D_DEBUG(DB_IO, "stale version req %d map_version %d\n",
849             orw->orw_map_ver, map_ver);
850 TF         if (update)
851             D_GOTO(out, rc = -DER_STALE);

854 TF     } else if (orw->orw_map_ver > map_ver) {
855         /* XXX: Client (or leader replica) has newer pool map than
856          *  current replica. Two possibile cases:

859          *     the old pool map version. According to current
860          *     leader election algorithm, it is still the new

862          *     leader switch, the unmatched pool version will

864          *
865          *  2. The current replica was NOT the old leader if
866          *     with the old pool map version. But it becomes
867          *     the new leader with the new pool map version.
868          *     In the subsequent modificaiton, it may hit
869          *     some 'prepared' DTX when make availability
870          *     check, it will return -DER_INPROGRESS that
871          *     will cause client to retry. It is possible
872          *     that the pool map version event arrives at
873          *     this server during the client retry. It is
874          *     inefficient, but harmless.
875          */
876     }
877
878 TF     if (!update) {
879         D_TIME_END(tls->ot_sp, OBJ_PF_UPDATE_PREP);
880         D_TIME_START(tls->ot_sp, OBJ_PF_UPDATE_LOCAL);
881
882         rc = ds_obj_rw_local_hdlr(rpc, tag, cont_hdl, cont,
883                       &ioh, false, map_ver, dth);
884 TF         if (rc != 0)
885             D_ERROR(DF_UOID": rw_local_hdlr (fetch) failed %d.\n",
886                 DP_UOID(orw->orw_oid), rc);
887
888         D_TIME_END(tls->ot_sp, OBJ_PF_UPDATE_LOCAL);
889
890         D_GOTO(out, rc);
891     }
892
893 TF     if (!daos_oc_echo_type(daos_obj_id2class(orw->orw_oid.id_pub))) {
894         /* Check DTX configuration. */
895         rc = ds_obj_check_dtx_config(orw->orw_flags, &orw->orw_dti);
896 TF         if (rc != 0)
897             D_GOTO(out, rc);
898
899         /* Handle resend. */
900 TF         if (srv_enable_dtx && orw->orw_flags & ORF_RESEND) {
901             rc = dtx_handle_resend(cont->sc_hdl, &orw->orw_oid,
902                            &orw->orw_dti,
903                            orw->orw_dkey_hash, false);
904 TF             if (rc == -DER_ALREADY)
905                 D_GOTO(out, rc = 0);
906 TF             if (rc == 0)
907                 resend = true;
908 TF             else if (rc == -DER_NONEXIST)
909                 rc = 0;
910             else
911                 D_GOTO(out, rc);
912         }
913     }
914
915 TF     if (srv_enable_dtx && (!resend || dispatch)) {
916         /* XXX: In fact, we do not need to start the DTX for
917          *  non-replicated object. But consider to handle
918          *  resend case, we need the DTX record. So start
919          *  the DTX unconditionally. It may be changed in
920          *  the future when we introduce more suitable
921          *  mechanism for handling resent modification.
922          *
923          *  For new leader, even though the local replica
924          *  has ever been modified before, but it doesn't
925          *  know whether other replicas have also done the
926          *  modification or not, so still need to dispatch
927          *  the RPC to other replicas.
928          */

930         rc = dtx_begin(&orw->orw_dti, &orw->orw_oid, cont->sc_hdl,
931                    orw->orw_epoch, orw->orw_dkey_hash,
932                    &conflict, orw->orw_dti_cos.ca_arrays,
933                    orw->orw_dti_cos.ca_count, orw->orw_map_ver,
934                    DAOS_INTENT_UPDATE,
935                    !(orw->orw_flags & ORF_FROM_LEADER), &dth);
936 TF         if (rc != 0) {
937             D_ERROR(DF_UOID": Failed to start DTX for update %d.\n",
938                 DP_UOID(orw->orw_oid), rc);
939             D_GOTO(out, rc);
940         }
941
942         dti_cos = dth->dth_dti_cos;
943         dti_cos_count = dth->dth_dti_cos_count;
944 TF         if (!(orw->orw_flags & ORF_FROM_LEADER) && !dispatch)

946     }

948     D_TIME_END(tls->ot_sp, OBJ_PF_UPDATE_PREP);

955                 orw->orw_shard_tgts.ca_arrays,
956                 orw->orw_shard_tgts.ca_count,

961                 DP_UOID(orw->orw_oid), rc);
962             D_GOTO(out, rc);
963         }
964         D_ASSERT(obj_arg != NULL);

968 TF         if (rc != 0) {

970                 DP_UOID(orw->orw_oid), rc);

972             D_GOTO(out, rc);
973         }
974
975         D_TIME_END(tls->ot_sp, OBJ_PF_UPDATE_DISPATCH);
976     }
977
978 TF     if (!resend) {
979         D_TIME_START(tls->ot_sp, OBJ_PF_UPDATE_LOCAL);
980
981         /* local RPC handler */
982         rc = ds_obj_rw_local_hdlr(rpc, tag, cont_hdl, cont,
983                       &ioh, true, map_ver, dth);
984 TF         if (rc != 0)
985             D_ERROR(DF_UOID": rw_local_hdlr (update) failed %d.\n",
986                 DP_UOID(orw->orw_oid), rc);
987
988         D_TIME_END(tls->ot_sp, OBJ_PF_UPDATE_LOCAL);
989     }
990
991     /* wait dispatched IO's completion when needed */
992 TF     if (dispatch) {
993         struct dtx_conflict_entry   *dces = NULL;
994
995         D_TIME_START(tls->ot_sp, OBJ_PF_UPDATE_WAIT);
996
997 TF         rc1 = ds_obj_req_disp_wait(obj_arg, rc >= 0 ? &dces : NULL);
998 TF         if (dces != NULL) {
999             /* XXX: The local modification has been done, but remote
1000              *  replica failed because of some uncommitted DTX,
1001              *  it may be caused by some garbage DTXs on remote
1002              *  replicas or leader has more information because
1003              *  of CoS cache. So handle (abort or commit) them
1004              *  firstly then retry.
1005              */
1006
1007             D_ASSERT(dth != NULL);
1008
1009             D_DEBUG(DB_TRACE, "Hit conflict DTX (%d)"DF_DTI" for "
1010                 DF_DTI", handle them and retry update.\n",
1011                 rc1, DP_DTI(&dces[0].dce_xid),
1012                 DP_DTI(&orw->orw_dti));
1013
1014             rc = dtx_conflict(cont->sc_hdl, dth,
1015                       cont_hdl->sch_pool->spc_uuid,
1016                       cont->sc_uuid, dces, rc1,
1017                       cont_hdl->sch_pool->spc_map_version);
1018             D_FREE(dces);
1019 TF             if (rc >= 0) {
1020                 resend = true;
1021                 flags |= ORF_RESEND;
1022                 dti_cos = dth->dth_dti_cos;
1023                 dti_cos_count = dth->dth_dti_cos_count;
1024
1025                 D_TIME_END(tls->ot_sp, OBJ_PF_UPDATE_WAIT);
1026
1027                 goto dispatch;
1028             }
1029 TF         } else if (rc >= 0) {
1030             rc = rc1;
1031         }
1032
1033         D_TIME_END(tls->ot_sp, OBJ_PF_UPDATE_WAIT);
1034     }
1035
1036 out:
1037 TF     if (dth != NULL)
1038         rc = dtx_end(dth, cont_hdl, cont, rc);
1039
1040     ds_obj_rw_reply(rpc, rc, map_ver, &conflict);
1041
1042     D_TIME_END(tls->ot_sp, OBJ_PF_UPDATE);
1043
1044 TF     if (cont_hdl) {
1045 TF         if (!cont_hdl->sch_cont)
1046             ds_cont_put(cont); /* -1 for rebuild container */
1047         ds_cont_hdl_put(cont_hdl);
1048     }
1049 }
