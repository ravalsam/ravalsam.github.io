429 static int
430 ut_setup(struct vea_ut_args *test_args)
431 {
432     daos_size_t pool_size = (50 << 20); /* 50MB */
433     struct umem_attr uma;
434     PMEMoid root;
435     void *root_addr;
436     int rc, i;
437
438     memset(test_args, 0, sizeof(struct vea_ut_args));
439     D_INIT_LIST_HEAD(&test_args->vua_alloc_list);
440
441     unlink(pool_file);
442
443     uma.uma_id = UMEM_CLASS_PMEM;
444     uma.uma_pool = pmemobj_create(pool_file, "vea_ut",
445                          pool_size, 0666);
446 TF     if (uma.uma_pool == NULL) {
447         fprintf(stderr, "create pmemobj pool error\n");
448         return -1;
449     }
450
451     root = pmemobj_root(uma.uma_pool,
452                 sizeof(struct vea_space_df) +
453                 sizeof(struct vea_hint_df) * IO_STREAM_CNT);
454 TF     if (OID_IS_NULL(root)) {
455         fprintf(stderr, "get root error\n");
456         rc = -1;
457         goto error;
458     }
459
460     rc = umem_class_init(&uma, &test_args->vua_umm);
461 TF     if (rc) {
462         fprintf(stderr, "initialize umm error %d\n", rc);
463         goto error;
464     }
465
466
467     root_addr = pmemobj_direct(root);
468     test_args->vua_md = root_addr;
469     root_addr += sizeof(struct vea_space_df);
470
471 TF     for (i = 0; i < IO_STREAM_CNT; i++) {
472         test_args->vua_hint[i] = root_addr;
473
474         test_args->vua_hint[i]->vhd_off = 0;
475         test_args->vua_hint[i]->vhd_seq = 0;
476         D_INIT_LIST_HEAD(&test_args->vua_resrvd_list[i]);
477
478         root_addr += sizeof(struct vea_hint_df);
479     }
480
481     umem_init_txd(&test_args->vua_txd);
482     return 0;
483 error:
484     pmemobj_close(uma.uma_pool);
485     test_args->vua_umm.umm_pool = NULL;
486
487     return rc;
488 }

548 static void
549 ut_reserve_special(void **state)
550 {
551     /* Use a temporary device instead of the main one the other tests use */
552     struct vea_ut_args args;
553     uint32_t blk_cnt = 0;
554     d_list_t *r_list;
555     uint32_t hdr_blks = 1;
556     uint64_t capacity = 2UL << 30; /* 2GB, 0.5M 4k blocks in total */
557     struct vea_unmap_context unmap_ctxt;
558     uint32_t blk_sz = 0; /* use the default size */
559     int rc;
560
561     ut_setup(&args);
562
563     rc = vea_format(&args.vua_umm, &args.vua_txd, args.vua_md, blk_sz,
564             hdr_blks, capacity, NULL, NULL, false);
565     assert_int_equal(rc, 0);
566
567     unmap_ctxt.vnc_unmap = NULL;
568     unmap_ctxt.vnc_data = NULL;
569     rc = vea_load(&args.vua_umm, &args.vua_txd, args.vua_md, &unmap_ctxt,
570               &args.vua_vsi);
571     assert_int_equal(rc, 0);
572
573     print_message("Try to reserve extent larger than available space\n");
574
575     r_list = &args.vua_resrvd_list[0];
576
577     /* reserve too big should fail */
578     blk_cnt = (1 << 20); /* 1M blocks */
579     rc = vea_reserve(args.vua_vsi, blk_cnt, NULL, r_list);
580     /* expect -DER_NOSPACE */
581     assert_int_equal(rc, -DER_NOSPACE);
582     print_message("correctly failed to reserve extent\n");
583
584     /* allocation should success */
585     blk_cnt = (500 * 1024); /* a bit less than 0.5M blocks */
586     rc = vea_reserve(args.vua_vsi, blk_cnt, NULL, r_list);
587     assert_int_equal(rc, 0);
588
589     rc = umem_tx_begin(&args.vua_umm, &args.vua_txd);
590     assert_int_equal(rc, 0);
591
592     rc = vea_tx_publish(args.vua_vsi, NULL, r_list);
593     assert_int_equal(rc, 0);
594
595     rc = umem_tx_commit(&args.vua_umm);
596     assert_int_equal(rc, 0);
597
598     /* free the allocated space */
599     rc = vea_free(args.vua_vsi, hdr_blks, blk_cnt);
600     assert_int_equal(rc, 0);
601
602     /*
603      * immediate reserve after free, the free extents should be made
604      * visible for allocation immediately, reserve should succeed.
605      */
606     rc = vea_reserve(args.vua_vsi, blk_cnt, NULL, r_list);
607     assert_int_equal(rc, 0);
608
609     vea_unload(args.vua_vsi);
610     ut_teardown(&args);
611 }

1148 static void
1149 ut_fragmentation(void **state)
1150 {
1151     struct vea_ut_args args;
1152     struct vea_unmap_context unmap_ctxt;
1153     struct vea_resrvd_ext *ext, *copy;
1154     struct vea_resrvd_ext *tmp_ext;
1155     d_list_t *r_list;
1156     d_list_t persist_list;
1157     uint64_t capacity = 32llu << 30; /* 32 GB */
1158     uint32_t block_size = 4096; /* use the default size */
1159     uint32_t header_blocks = 1;
1160     uint32_t block_count;
1161     uint32_t cur_stream;
1162     uint32_t max_blocks;
1163     int rc;
1164
1165     print_message("Test allocation on fragmented device\n");
1166     ut_setup(&args);
1167     rc = vea_format(&args.vua_umm, &args.vua_txd, args.vua_md, block_size,
1168             header_blocks, capacity, NULL, NULL, false);
1169     assert_int_equal(rc, 0);
1170
1171     unmap_ctxt.vnc_unmap = NULL;
1172     unmap_ctxt.vnc_data = NULL;
1173     rc = vea_load(&args.vua_umm, &args.vua_txd, args.vua_md, &unmap_ctxt,
1174               &args.vua_vsi);
1175     assert_int_equal(rc, 0);
1176
1177     /* Generate random fragments on the same I/O stream */
1178     srand(time(0));
1179     cur_stream = 0;
1180     r_list = &args.vua_resrvd_list[cur_stream];
1181     max_blocks = args.vua_vsi->vsi_class.vfc_large_thresh;
1182     /* Keep reserving until we run out of space */
1183 TF     while (rc == 0) {
1184         /* Get a random number greater than 2 */
1185         block_count = (rand() % max_blocks - 1) + 2;
1186         rc = vea_reserve(args.vua_vsi, block_count, NULL, r_list);
1187     }
1188
1189     /* Free some of the fragments. The cancelled ones remain in r_list */
1190     D_INIT_LIST_HEAD(&persist_list);
1191
1192     d_list_for_each_entry_safe(ext, tmp_ext, r_list, vre_link) {
1193         /* Copy the extents to keep to persist_list */
1194         /* Remove about every other fragment */
1195 TF         if (rand() % 2 == 0) {
1196             d_list_move_tail(&ext->vre_link, &persist_list);
1197
1198             D_ALLOC_PTR(copy);
1199             assert_ptr_not_equal(copy, NULL);
1200
1201             D_INIT_LIST_HEAD(&copy->vre_link);
1202             copy->vre_blk_off = ext->vre_blk_off;
1203             copy->vre_blk_cnt = ext->vre_blk_cnt;
1204             d_list_add(&copy->vre_link, &args.vua_alloc_list);
1205         }
1206     }
1207
1208     /* Publish the ones to persist */
1209     rc = umem_tx_begin(&args.vua_umm, &args.vua_txd);
1210     assert_int_equal(rc, 0);
1211     rc = vea_tx_publish(args.vua_vsi, NULL, &persist_list);
1212     assert_int_equal(rc, 0);
1213     rc = umem_tx_commit(&args.vua_umm);
1214     assert_int_equal(rc, 0);
1215
1216     /* Cancel the reservations */
1217     rc = vea_cancel(args.vua_vsi, NULL, r_list);
1218 TF     if (rc != 0) {
1219         print_message("vea_cancel() returned %d\n", rc);
1220         assert_int_equal(rc, 0);
1221     }
1222
1223     print_message("Fragments:\n");
1224     print_stats(&args, false);
1225
1226     /* Try to allocate on multiple I/O streams until no space available*/
1227 TF     while (rc == 0) {
1228 TF         for (cur_stream = 0; cur_stream < IO_STREAM_CNT; cur_stream++) {
1229             r_list = &args.vua_resrvd_list[cur_stream];
1230             /* Get a random number greater than 2 */
1231             block_count = (rand() % max_blocks - 1) + 2;
1232             rc = vea_reserve(args.vua_vsi, block_count, NULL,
1233                      r_list);
1234 TF             if (rc != 0) {
1235                 assert_true(rc == -DER_NOSPACE);
1236                 break;
1237             }
1238         }
1239     }
1240     print_message("Fragments after more reservations:\n");
1241     print_stats(&args, false);
1242
1243     /* Free allocated extents */
1244     r_list = &args.vua_alloc_list;
1245     d_list_for_each_entry(ext, r_list, vre_link) {
1246         uint64_t blk_off = ext->vre_blk_off;
1247         uint32_t blk_cnt = ext->vre_blk_cnt;
1248
1249         rc = vea_free(args.vua_vsi, blk_off, blk_cnt);
1250         assert_int_equal(rc, 0);
1251
1252         /* not immediately visual for allocation */
1253         rc = vea_verify_alloc(args.vua_vsi, true, blk_off, blk_cnt);
1254         assert_int_equal(rc, 0);
1255
1256         rc = vea_verify_alloc(args.vua_vsi, false, blk_off, blk_cnt);
1257         assert_int_equal(rc, 1);
1258     }
1259
1260     vea_unload(args.vua_vsi);
1261     ut_teardown(&args);
