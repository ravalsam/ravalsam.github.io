110 /* Helper function for migrating objects from pending list to free list.
111  *
112  * Migrates objects from the pending list to the free list.  Keeps going
113  * until either there are count objects on the free list or there are
114  * no more pending objects;
115  * This function should be called with the type lock held.
116  */
117 static int
118 restock(struct dfuse_da_type *type, int count)
119 {
120     d_list_t *entry, *enext;
121     int reset_calls = 0;
122
123 TF     if (type->free_count >= count)
124         return 0;
125
126 TF     if (type->reg.max_free_desc != 0 &&
127         type->free_count >= type->reg.max_free_desc) {
128         DFUSE_TRA_DEBUG(type, "free_count %d, max_free_desc %d, "
129                 "cannot append.",
130                 type->free_count, type->reg.max_free_desc);
131         return 0;
132     }
133
134     d_list_for_each_safe(entry, enext, &type->pending_list) {
135         void *ptr = (void *)entry - type->reg.offset;
136         bool rcb = true;
137
138         DFUSE_TRA_DEBUG(type, "Resetting %p", ptr);
139
140         d_list_del(entry);
141         type->pending_count--;
142
143 TF         if (type->reg.reset) {
144             type->reset_count++;
145             reset_calls++;
146             rcb = type->reg.reset(ptr);
147         }
148 TF         if (rcb) {
149             d_list_add(entry, &type->free_list);
150             type->free_count++;
151         } else {
152             DFUSE_TRA_INFO(ptr, "entry %p failed reset", ptr);
153             type->count--;
154             D_FREE(ptr);
155         }
156
157 TF         if (type->free_count == count)
158             return reset_calls;
159
160 TF         if (type->reg.max_free_desc != 0 &&
161             type->free_count >= type->reg.max_free_desc)
162             return reset_calls;
