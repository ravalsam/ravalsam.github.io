112 /**
113  * Get the target tag (context ID) for specific request type and target index.
114  *
115  * \param[in]   req_type    RPC request type (enum daos_rpc_type)
116  * \param[in]   tgt_idx     target index (VOS index, main xstream index)
117  *
118  * \return          target tag (context ID) to be used for the RPC
119  */
120 static inline int
121 daos_rpc_tag(int req_type, int tgt_idx)
122 {
123     switch (req_type) {
124     /* for normal IO request, send to the main service thread/context */
125 T     case DAOS_REQ_IO:
126 T     case DAOS_REQ_DTX:
127         return DAOS_IO_CTX_ID(tgt_idx);
128     /* target tag 0 is to handle below requests */
129 T     case DAOS_REQ_MGMT:
130 T     case DAOS_REQ_POOL:
131 T     case DAOS_REQ_RDB:
132 T     case DAOS_REQ_CONT:
133 T     case DAOS_REQ_REBUILD:
134 T     case DAOS_REQ_IV:
135 T     case DAOS_REQ_BCAST:
136 T     case DAOS_REQ_SWIM:
137         return 0;
138 T     default:
139         D_ASSERTF(0, "bad req_type %d.\n", req_type);
140         return -DER_INVAL;
141     };
142 }

160  * \param[in] cli_count count of RPCs to be registered in client.
161  * \param[in] handlers  RPC handlers to be registered, if
162  *                      it is NULL, then it is for registering

164  *                      server.
165  * \param[in] mod_id    module id of the module.

167  * \retval  0 if registration succeeds
168  * \retval  negative errno if registration fails.
169  */
170 static inline int
171 daos_rpc_register(struct crt_proto_format *proto_fmt, uint32_t cli_count,
172           struct daos_rpc_handler *handlers, int mod_id)
173 {
174     uint32_t i;
175
176 TF     if (proto_fmt == NULL)
177         return 0;
178
179 TF     if (handlers != NULL) {
180         /* walk through the RPC list and fill with handlers */
181 TF         for (i = 0; i < proto_fmt->cpf_count; i++) {
182             proto_fmt->cpf_prf[i].prf_hdlr =
183                         handlers[i].dr_hdlr;
184             proto_fmt->cpf_prf[i].prf_co_ops =
185                         handlers[i].dr_corpc_ops;
186         }
187     } else {
188         proto_fmt->cpf_count = cli_count;
189     }
190
191     return crt_proto_register(proto_fmt);
192 }
