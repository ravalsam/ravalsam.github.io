47 /** The offset of an object from the base address of the pool */
48 typedef uint64_t        umem_off_t;
49 /** Number of flag bits to reserve for encoding extra information in
50  *  a NULL umem_off_t entry.
51  */
52 #define UMOFF_NUM_BITS      (8)
53 /** The absolute value of a flag mask must be <= this value */
54 #define UMOFF_MAX_FLAG      (1ULL << UMOFF_NUM_BITS)
55 /** In theory and offset can be NULL but in practice, pmemobj_root
56  *  is not at offset 0 as pmdk reserves some space for its internal
57  *  use.   So, use 0 for NULL.   Invalid bits are also considered
58  *  NULL.
59  */
60 #define UMOFF_NULL      (0ULL)
61 /** Check for a NULL value including possible invalid flag bits */
62 #define UMOFF_IS_NULL(umoff)    (umoff < UMOFF_MAX_FLAG)
63
64 /** Up to UMOFF_NUM_BITS bits are available to the user to mark an
65  *  invalid offset with extra information.  If these fields are
66  *  set, the pointer resulting from such an offset will be
67  *  considered NULL.
68  *
69  *  \param  offset[IN,OUT]  The value is marked invalid with the flags
70  *  \param  flags[IN]   Auxilliarly information about the invalid entry
71  */
72 static inline void
73 umem_off_set_invalid(umem_off_t *offset, uint64_t flags)
74 {
75     D_ASSERTF(flags < UMOFF_MAX_FLAG,
76           "Attempt to set invalid flag bits on umem_off_t\n");
77     *offset = flags;
78 }

80 /** Retrieves any invalid flags that are set.  If the offset hasn't
