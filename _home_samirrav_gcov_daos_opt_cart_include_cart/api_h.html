492  * CRT_RPC_REGISTER(opcode, flags, my_rpc);

900 enum crt_tree_type {
901     CRT_TREE_INVALID    = 0,
902     CRT_TREE_MIN        = 1,
903     CRT_TREE_FLAT       = 1,
904     CRT_TREE_KARY       = 2,
905     CRT_TREE_KNOMIAL    = 3,
906     CRT_TREE_MAX        = 3,
907 };
908
909 #define CRT_TREE_TYPE_SHIFT (16U)
910 #define CRT_TREE_MAX_RATIO  (64)
911 #define CRT_TREE_MIN_RATIO  (2)
912
913 /*
914  * Calculate the tree topology. Can only be called on the server side.
915  *
916  * \param[in] tree_type        tree type
917  * \param[in] branch_ratio     branch ratio, be ignored for CRT_TREE_FLAT.
918  *                             for KNOMIAL tree or KARY tree, the valid value
919  *                             should within the range of
920  *                             [CRT_TREE_MIN_RATIO, CRT_TREE_MAX_RATIO], or
921  *                             will be treated as invalid parameter.
922  *
923  * \return                     tree topology value on success,
924  *                             negative value if error.
925  */
926 static inline int
927 crt_tree_topo(enum crt_tree_type tree_type, uint32_t branch_ratio)
928 {
929 TF     if (tree_type < CRT_TREE_MIN || tree_type > CRT_TREE_MAX)
930         return -DER_INVAL;
931
932     return (tree_type << CRT_TREE_TYPE_SHIFT) |
933            (branch_ratio & ((1U << CRT_TREE_TYPE_SHIFT) - 1));
934 }
