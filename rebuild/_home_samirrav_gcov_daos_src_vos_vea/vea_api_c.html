472 struct free_commit_cb_arg {
473     struct vea_space_info   *fca_vsi;
474     struct vea_free_extent   fca_vfe;
475 };
476
477 static void
478 free_commit_cb(void *data, bool noop)
479 {
480     struct free_commit_cb_arg *fca = data;
481     int rc;
482
483     /* Transaction aborted, only need to free callback arg */
484 TF     if (noop)
485         goto free;
486
487     /*
488      * Aggregated free will be executed on outermost transaction
489      * commit.
490      *
491      * If it fails, the freed space on persistent free tree won't
492      * be added in in-memory free tree, hence the space won't be
493      * visible for allocation until the tree sync up on next server
494      * restart. Such temporary space leak is tolerable, what we must
495      * avoid is the contrary case: in-memory tree update succeeds
496      * but persistent tree update fails, which risks data corruption.
497      */
498     rc = aggregated_free(fca->fca_vsi, &fca->fca_vfe);
499
500     D_CDEBUG(rc, DLOG_ERR, DB_IO, "Aggregated free on vsi:%p rc %d\n",
501          fca->fca_vsi, rc);
502 free:
503     D_FREE(fca);
504 }
505
506 /*
507  * Free allocated extent.
508  *
509  * The just recent freed extents won't be visible for allocation instantly,
510  * they will stay in vsi_agg_lru for a short period time, and being coalesced
511  * with each other there.
512  *
513  * Expired free extents in the vsi_agg_lru will be migrated to the allocation
514  * visible index (vsi_free_tree, vfc_heap or vfc_lrus) from time to time, this
515  * kind of migration will be triggered by vea_reserve() & vea_free() calls.
516  */
517 int
518 vea_free(struct vea_space_info *vsi, uint64_t blk_off, uint32_t blk_cnt)
519 {
520     D_ASSERT(vsi != NULL);
521     struct umem_instance *umem = vsi->vsi_umem;
522     struct free_commit_cb_arg *fca;
523     int rc;
524
525     D_ALLOC_PTR(fca);
526 TF     if (fca == NULL)
527         return -DER_NOMEM;
528
529     fca->fca_vsi = vsi;
530     fca->fca_vfe.vfe_blk_off = blk_off;
531     fca->fca_vfe.vfe_blk_cnt = blk_cnt;
532
533     rc = verify_free_entry(NULL, &fca->fca_vfe);
534 TF     if (rc)
535         goto error;
536
537     /*
538      * The transaction may have been started by caller already, here
539      * we start the nested transaction to ensure the stage callback
540      * and stage callback data being set to transaction properly.
541      */
542     rc = umem_tx_begin(umem, vsi->vsi_txd);
543 TF     if (rc != 0)
544         goto error;
545
546     /* Add the free extent in persistent free extent tree */
547     rc = persistent_free(vsi, &fca->fca_vfe);
548 TF     if (rc)
549         goto done;
550
551     rc = umem_tx_add_callback(umem, vsi->vsi_txd, TX_STAGE_ONCOMMIT,
552                   free_commit_cb, fca);
553 TF     if (rc == 0)
554         fca = NULL; /* Will be freed by commit callback */
555 done:
556     /* Commit/Abort transaction on success/error */
557 TF     rc = rc ? umem_tx_abort(umem, rc) : umem_tx_commit(umem);
558     /* Migrate the expired aggregated free extents to compound index */
559 TF     if (rc == 0)
560         migrate_free_exts(vsi);
561 error:
562 TF     if (fca != NULL)
563         D_FREE(fca);
564     return rc;
565 }
