560 /*
561  * Check rebuild status on the leader. Every other target sends
562  * its own rebuild status by IV.
563  */
564 static void
565 rebuild_leader_status_check(struct ds_pool *pool, uint32_t map_ver,
566                 struct rebuild_global_pool_tracker *rgt)
567 {
568     double      begin = ABT_get_wtime();
569     double      last_print = 0;
570     double      last_query = 0;
571     unsigned int    total;
572     int     rc;
573
574     rc = crt_group_size(pool->sp_group, &total);
575 TF     if (rc)
576         return;
577
578     while (1) {
579         struct daos_rebuild_status *rs = &rgt->rgt_status;
580         char    sbuf[RBLD_SBUF_LEN];
581         struct pool_target *targets;
582         unsigned int    failed_tgts_cnt;
583         double      now;
584         char        *str;
585
586         now = ABT_get_wtime();
587 TF         if (now - last_query < RBLD_BCAST_INTV) {
588             /* Yield to other ULTs */
589             ABT_thread_yield();

591         }
592
593         last_query = now;
594
595         rc = pool_map_find_failed_tgts(pool->sp_map, &targets,
596                            &failed_tgts_cnt);
597 TF         if (rc != 0) {
598             D_ERROR("failed to create failed tgt list rc %d\n",
599                 rc);
600             return;
601         }
602
603 TF         if (targets != NULL) {
604             struct pool_domain *dom;
605             int i;
606
607 TF             for (i = 0; i < failed_tgts_cnt; i++) {
608                 dom = pool_map_find_node_by_rank(pool->sp_map,
609                         targets[i].ta_comp.co_rank);
610
611                 D_ASSERT(dom != NULL);
612                 D_DEBUG(DB_REBUILD, "target %d failed\n",
613                     dom->do_comp.co_rank);
614                 setbit(rgt->rgt_scan_bits,
615                        dom->do_comp.co_rank);
616                 setbit(rgt->rgt_pull_bits,
617                        dom->do_comp.co_rank);
618             }
619
620             D_FREE(targets);
621         }
622
623 TF         if (!rgt->rgt_done && rgt->rgt_scan_done) {
624             struct rebuild_iv iv;
625
626             memset(&iv, 0, sizeof(iv));
627             uuid_copy(iv.riv_pool_uuid, rgt->rgt_pool_uuid);
628             iv.riv_master_rank = pool->sp_iv_ns->iv_master_rank;
629             iv.riv_global_scan_done = 1;
630             iv.riv_ver = rgt->rgt_rebuild_ver;
631             iv.riv_leader_term = rgt->rgt_leader_term;
632
633             /* Notify others the global scan is done, then
634              * each target can reliablly report its pull status
635              */
636             rc = rebuild_iv_update(pool->sp_iv_ns,
637                            &iv, CRT_IV_SHORTCUT_NONE,
638                            CRT_IV_SYNC_LAZY);
639 TF             if (rc)
640                 D_WARN("rebuild master iv update failed: %d\n",
641                        rc);
642         }
643
644         /* query the current rebuild status */
645 TF         if (rgt->rgt_done)
646             rs->rs_done = 1;
647
648 TF         if (rs->rs_done)
649 TF             str = rs->rs_errno ? "failed" : "completed";
650 TF         else if (rgt->rgt_abort || rebuild_gst.rg_abort)
651             str = "aborted";
652 TF         else if (rs->rs_obj_nr == 0 && rs->rs_rec_nr == 0)
653             str = "scanning";
654         else
655             str = "pulling";
656
657         snprintf(sbuf, RBLD_SBUF_LEN,
658             "Rebuild [%s] (pool "DF_UUID" ver=%u, toberb_obj="
659             DF_U64", rb_obj="DF_U64", rec= "DF_U64", done %d "
660             "status %d duration=%d secs)\n",
661             str, DP_UUID(pool->sp_uuid), map_ver,
662             rs->rs_toberb_obj_nr, rs->rs_obj_nr, rs->rs_rec_nr,
663             rs->rs_done, rs->rs_errno, (int)(now - begin));
664
665         D_DEBUG(DB_REBUILD, "%s", sbuf);
666 TF         if (rs->rs_done || rebuild_gst.rg_abort || rgt->rgt_abort) {
667             D_PRINT("%s", sbuf);
668             break;
669         }
670
671         /* print something at least for each 10 secons */
672 TF         if (now - last_print > 10) {
673             last_print = now;
674             D_PRINT("%s", sbuf);
675         }
676
677         ABT_thread_yield();
678     }
679 }
