65     int      rc;

67 TF     if (tsc->tsc_cred_inuse == 0)

69
70     while (1) {

72                   evs);
73 TF         if (rc < 0) {
74             fprintf(stderr, "failed to pool event: %d\n", rc);
75             return rc;
76         }
77
78 TF         for (i = 0; i < rc; i++) {
79             int err = evs[i]->ev_error;
80
81 TF             if (err != 0) {
82                 fprintf(stderr, "failed op: %d\n", err);
83                 return err;
84             }

86                container_of(evs[i], struct dts_io_credit, tc_ev);

96 TF         if (tsc->tsc_cred_inuse != 0 && drain)
97             continue;
