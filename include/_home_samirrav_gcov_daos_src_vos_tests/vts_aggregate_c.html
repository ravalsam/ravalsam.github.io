465 static void
466 aggregate_multi(struct io_test_args *arg, struct agg_tst_dataset *ds_sample)
467
468 {
469     daos_unit_oid_t      oid, oids[AT_OBJ_KEY_NR];
470     char             dkeys[AT_OBJ_KEY_NR][UPDATE_DKEY_SIZE];
471     char             akeys[AT_OBJ_KEY_NR][UPDATE_AKEY_SIZE];
472     char            *dkey, *akey, *buf_u;
473     struct agg_tst_dataset  *ds, *ds_arr;
474     daos_epoch_t         epoch;
475     daos_epoch_range_t  *epr_u, *epr_a;
476     daos_size_t      view_len;
477     daos_recx_t      recx, *recx_p;
478     int          oid_idx, dkey_idx, akey_idx;
479     int          i, key_nr, ds_nr, ds_idx, rc;
480     bool             random_type;
481
482     epr_u = &ds_sample->td_upd_epr;
483     epr_a = &ds_sample->td_agg_epr;
484
485 TF     for (i = 0; i < AT_OBJ_KEY_NR; i++) {
486         oids[i] = dts_unit_oid_gen(0, 0, 0);
487         dts_key_gen(dkeys[i], UPDATE_DKEY_SIZE, UPDATE_DKEY);
488         dts_key_gen(akeys[i], UPDATE_AKEY_SIZE, UPDATE_AKEY);
489     }
490
491     ds_nr = key_nr = AT_OBJ_KEY_NR * AT_OBJ_KEY_NR * AT_OBJ_KEY_NR;
492     random_type = (ds_sample->td_type == DAOS_IOD_NONE);
493 TF     if (random_type)
494         ds_nr *= 2;
495     D_ALLOC_ARRAY(ds_arr, ds_nr);
496     assert_non_null(ds_arr);
497
498 TF     for (i = 0; i < ds_nr; i++) {
499         ds = &ds_arr[i];
500         *ds = *ds_sample;
501         memset(&ds->td_upd_epr, 0, sizeof(*epr_u));
502     }
503
504     /* Set maximum value for random iod_size */
505 TF     if (ds_sample->td_iod_size == 0)
506         ds_sample->td_iod_size = AT_SV_IOD_SIZE_LARGE;
507
508     view_len = get_view_len(ds_sample, &recx);
509     D_ALLOC(buf_u, view_len);
510     assert_non_null(buf_u);
511
512     VERBOSE_MSG("Generate random updates over multiple objs/keys.\n");
513 TF     for (epoch = epr_u->epr_lo; epoch <= epr_u->epr_hi; epoch++) {
514         daos_iod_type_t iod_type;
515
516         oid_idx = rand() % AT_OBJ_KEY_NR;
517         dkey_idx = rand() % AT_OBJ_KEY_NR;
518         akey_idx = rand() % AT_OBJ_KEY_NR;
519
520         oid = oids[oid_idx];
521         dkey = dkeys[dkey_idx];
522         akey = akeys[akey_idx];
523
524         ds_idx = get_ds_index(oid_idx, dkey_idx, akey_idx,
525                       AT_OBJ_KEY_NR);
526 TF         if (random_type) {
527 TF             iod_type = (rand() % 2) ? DAOS_IOD_SINGLE :
528                           DAOS_IOD_ARRAY;
529 TF             if (iod_type == DAOS_IOD_ARRAY)
530                 ds_idx += key_nr;
531         } else {
532             iod_type = ds_sample->td_type;
533         }
534
535         ds = &ds_arr[ds_idx];
536         ds->td_type = iod_type;
537
538         /* First update can't be punched record */
539 TF         if ((rand() % 2) && (ds->td_iod_size != 0))
540             arg->ta_flags |= TF_PUNCH;
541
542 TF         if (ds->td_iod_size == 0)
543             ds->td_iod_size = (rand() % ds_sample->td_iod_size) + 1;
544
545 TF         if (ds->td_type == DAOS_IOD_SINGLE) {
546             recx_p = NULL;
547 TF             ds->td_expected_recs = ds->td_discard ? 0 : 1;
548         } else {
549             assert_true(ds->td_recx_nr == 1);
550             recx_p = &recx;
551             generate_recx(&ds->td_recx[0], recx_p);
552 TF             ds->td_expected_recs = ds->td_discard ? 0 : -1;
553         }
554         /*
555          * Amend update epr, it'll be used to setup correct expected
556          * physical record nr and expected view.
557          */
558 TF         if (ds->td_upd_epr.epr_lo == 0)
559             ds->td_upd_epr.epr_lo = epoch;
560         ds->td_upd_epr.epr_hi = epoch;
561
562         update_value(arg, oid, epoch, dkey, akey, ds->td_type,
563                  ds->td_iod_size, recx_p, buf_u);
564         arg->ta_flags &= ~TF_PUNCH;
565
566     }
567     D_FREE(buf_u);
568
569     /*
570      * Amend expected physical records nr when update epr has no
571      * itersection with aggregate/discard epr.
572      */
573 TF     for (i = 0; i < ds_nr; i++) {
574         ds = &ds_arr[i];
575
576 TF         if (ds->td_expected_recs <= 0)
577             continue;
578
579 TF         if (ds->td_upd_epr.epr_hi < epr_a->epr_lo ||
580             ds->td_upd_epr.epr_lo > epr_a->epr_hi)
581             ds->td_expected_recs = 0;
582     }
583
584     multi_view(arg, oids, dkeys, akeys, AT_OBJ_KEY_NR, ds_arr, random_type,
585            key_nr, false);
586
587     VERBOSE_MSG("%s multiple objs/keys\n", ds_sample->td_discard ?
588             "Discard" : "Aggregate");
589
590 TF     if (ds_sample->td_discard)
591         rc = vos_discard(arg->ctx.tc_co_hdl, epr_a);
592     else
593         rc = vos_aggregate(arg->ctx.tc_co_hdl, epr_a);
594     assert_int_equal(rc, 0);
595
596     multi_view(arg, oids, dkeys, akeys, AT_OBJ_KEY_NR, ds_arr, random_type,
597            key_nr, true);
598     D_FREE(ds_arr);
599 }
