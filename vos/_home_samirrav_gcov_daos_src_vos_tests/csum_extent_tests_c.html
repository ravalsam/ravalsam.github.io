710 /*
711  * This test verifies that csums aren't copied for holes
712  */
713 void
714 csum_test_holes(void **state)
715 {
716     const uint64_t       data_size = 1024 * 64; /** 64K */
717     const uint64_t       chunk_size = 1024 * 16; /** 16K */
718     struct extent_key    extent_key;
719     daos_recx_t      extent = {0, data_size};
720     daos_csum_buf_t      csum;
721     daos_csum_buf_t      read_csum;
722     uint8_t         *data_buf_1;
723     uint8_t         *data_buf_2;
724     uint8_t         *read_data_buf = calloc(data_size * 3, 1);
725     uint8_t         *csum_buf_1;
726     uint8_t         *csum_buf_2;
727     uint8_t         *csum_read_buf;
728
729     extent_key_from_test_args(&extent_key, (struct io_test_args *) *state);
730
731     csum.cs_len = 8; /** CRC64 maybe? */
732     csum.cs_type = 1;
733     csum.cs_chunksize = chunk_size;
734     csum.cs_nr = csum_needed_for_extent(chunk_size, extent, 1);
735     csum.cs_buf_len = csum.cs_len * csum.cs_nr;
736
737     csum_buf_1 = allocate_random(csum.cs_buf_len);
738     data_buf_1 = allocate_random(data_size);
739     sleep(1); /** Sleep so random seed is different */
740     data_buf_2 = allocate_random(data_size);
741     csum_buf_2 = allocate_random(csum.cs_buf_len);
742
743     assert_memory_not_equal(data_buf_1, data_buf_2, data_size);
744     assert_memory_not_equal(csum_buf_1, csum_buf_2, csum.cs_buf_len);
745     csum_read_buf = calloc(csum.cs_buf_len * 3, 1);
746
747     /** Write first 64K */
748     csum.cs_csum = csum_buf_1;
749     write_to_extent(&extent_key, &extent, data_buf_1, data_size, &csum);
750
751     /** Leave a 64K hole and write the following 64K */
752     extent.rx_idx = data_size * 2;
753     csum.cs_csum = csum_buf_2;
754     write_to_extent(&extent_key, &extent, data_buf_2, data_size, &csum);
755
756     /** Read from first to last written */
757     extent.rx_idx = 0;
758     extent.rx_nr *= 3;
759
760     memset(&read_csum, 0, sizeof(read_csum));
761     read_csum.cs_len = csum.cs_len;
762     read_csum.cs_chunksize = chunk_size;
763     read_csum.cs_nr = csum_needed_for_extent(chunk_size, extent, 1);
764     read_csum.cs_buf_len = read_csum.cs_len * read_csum.cs_nr;
765     read_csum.cs_csum = csum_read_buf;
766
767
768     read_from_extent(&extent_key, &extent,
769              read_data_buf, data_size * 3, &read_csum);
770
771     assert_memory_equal(data_buf_1, read_data_buf, data_size);
772     assert_memory_equal(data_buf_2, read_data_buf + data_size * 2,
773                 data_size);
774
775     assert_memory_equal(csum_buf_1, csum_read_buf, csum.cs_buf_len);
776     assert_memory_equal(csum_buf_2, csum_read_buf + csum.cs_buf_len * 2,
777                 csum.cs_buf_len);
778
779     free(data_buf_1);
780     free(data_buf_2);
781     free(read_data_buf);
782     free(csum_buf_1);
783     free(csum_buf_2);
784     free(csum_read_buf);
785 }
