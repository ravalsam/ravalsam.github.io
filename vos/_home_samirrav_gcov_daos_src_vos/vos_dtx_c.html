277 static void
278 dtx_obj_rec_exchange(struct umem_instance *umm, struct vos_obj_df *obj,
279              struct vos_dtx_entry_df *dtx,
280              struct vos_dtx_record_df *rec, bool abort)
281 {

283     struct vos_obj_df       *tgt_obj;
284
285 TF     if (rec->tr_flags == DTX_RF_EXCHANGE_TGT) {
286         /* For commit case, should already have been handled
287          * during handling the soruce.
288          *
289          * For abort case, set its vo_dtx as aborted. The tgt
290          * record will be removed via VOS aggregation or other
291          * tools some time later.
292          */
293 TF         if (abort)
294             dtx_set_aborted(&obj->vo_dtx);
295         return;
296     }
297
298 TF     if (rec->tr_flags != DTX_RF_EXCHANGE_SRC)
299         return;
300
301 TF     if (!(obj->vo_oi_attr & VOS_OI_REMOVED)) {
302         D_ERROR(DF_UOID" with OBJ DTX ("DF_DTI") missed REMOVED flag\n",
303             DP_UOID(dtx->te_oid), DP_DTI(&dtx->te_xid));
304         return;
305     }
306
307 TF     if (abort) { /* abort case */
308         /* Recover the availability of the exchange source. */
309         obj->vo_oi_attr &= ~VOS_OI_REMOVED;
310         obj->vo_dtx = UMOFF_NULL;
311         return;
312     }
313
314     /* XXX: If the exchange target still exist, it will be the next
315      *  record. If it does not exist, then either it is crashed
316      *  or it has already deregistered from the DTX records list.
317      *  We cannot commit the DTX under any the two cases. Fail
318      *  the DTX commit is meaningless, then some warnings.
319      */
320 TF     if (dtx_is_null(rec->tr_next)) {
321         D_ERROR(DF_UOID" miss OBJ DTX ("DF_DTI") exchange pairs (1)\n",
322             DP_UOID(dtx->te_oid), DP_DTI(&dtx->te_xid));
323         return;
324     }
325
326     tgt_rec = umem_off2ptr(umm, rec->tr_next);
327 TF     if (tgt_rec->tr_flags != DTX_RF_EXCHANGE_TGT) {
328         D_ERROR(DF_UOID" miss OBJ DTX ("DF_DTI") exchange pairs (2)\n",
329             DP_UOID(dtx->te_oid), DP_DTI(&dtx->te_xid));
330         return;
331     }
332
333     /* Exchange the sub-tree between max epoch record and the given
334      * epoch record. The record with max epoch will be removed when
335      * aggregation or some special cleanup.
336      */
337     tgt_obj = umem_off2ptr(umm, tgt_rec->tr_record);
338     umem_tx_add_ptr(umm, tgt_obj, sizeof(*tgt_obj));
339
340     /* The @tgt_obj which epoch is current DTX's epoch will be
341      * available to outside of VOS. Set its vo_earliest as @obj
342      * vo_earliest.
343      */
344     tgt_obj->vo_tree = obj->vo_tree;
345     tgt_obj->vo_earliest = obj->vo_earliest;
346     tgt_obj->vo_latest = dtx->te_epoch;
347     tgt_obj->vo_incarnation = obj->vo_incarnation;
348
349     /* The @obj which epoch is MAX will be removed later. */
350     memset(&obj->vo_tree, 0, sizeof(obj->vo_tree));
351     obj->vo_latest = 0;
352     obj->vo_earliest = DAOS_EPOCH_MAX;
353     obj->vo_incarnation++; /* cache should be revalidated */
354
355     D_DEBUG(DB_TRACE, "Exchanged OBJ DTX records for "DF_DTI"\n",
356         DP_DTI(&dtx->te_xid));
357 }

361             struct vos_dtx_record_df *rec, umem_off_t umoff, bool abort)
362 {
363     struct vos_dtx_entry_df     *dtx;
364
365     dtx = umem_off2ptr(umm, umoff);
366 TF     if (dtx->te_intent == DAOS_INTENT_PUNCH) {
367         /* Because PUNCH cannot share with others, the vo_dtx should
368          * reference current DTX.
369          */
370 TF         if (obj->vo_dtx != umoff)
371             D_ERROR("The OBJ "DF_UOID" should referece DTX "
372                 DF_DTI", but referenced %s\n",
373                 DP_UOID(dtx->te_oid), DP_DTI(&dtx->te_xid),
374                 dtx_is_null(obj->vo_dtx) ? "UNLL" : "other");
375         else
376             dtx_obj_rec_exchange(umm, obj, dtx, rec, abort);
377         return;
378     }
379
380     /* Because PUNCH and UPDATE cannot share, both current DTX
381      * and the obj former referenced DTX should be for UPDATE.
382      */
383
384     obj->vo_dtx_shares--;
385
386     /* If current DTX references the object with VOS_OI_REMOVED,
387      * then means that at least one of the former shared update
388      * DTXs has been committed before current update DTX commit
389      * or abort. Then in spite of whether the punch DTX will be
390      * or has been committed or not, current update DTX (in sipte
391      * of commit or abort) needs to do nothing.
392      *
393      * If the punch DTX is aborted, then the VOS_OI_REMOVED will
394      * be removed, and the obj::vo_dtx will be set as NULL.
395      */
396 TF     if (obj->vo_oi_attr & VOS_OI_REMOVED)
397         return;
398
399     /* Other DTX that shares the obj has been committed firstly.
400      * It must be for sharing of update. Subsequent modification
401      * may have seen the modification before current DTX commit
402      * or abort.
403      *
404      * Please NOTE that the obj::vo_latest and obj::vo_earliest
405      * have already been handled during vos_update_end().
406      */
407 TF     if (dtx_is_null(obj->vo_dtx))
408         return;
409
410 TF     if (abort) {
411 TF         if (obj->vo_dtx_shares == 0)
412             /* The last shared UPDATE DTX is aborted. */
413             dtx_set_aborted(&obj->vo_dtx);
414 TF         else if (obj->vo_dtx == umoff)
415             /* I am the original DTX that create the object that
416              * is still shared by others. Now, I will be aborted,
417              * set the reference as UNKNOWN for other left shares.
418              */
419             dtx_set_unknown(&obj->vo_dtx);
420     } else {
421         obj->vo_dtx = UMOFF_NULL;
422     }
423 }

425 static void
426 dtx_key_rec_exchange(struct umem_instance *umm, struct vos_krec_df *key,
427              struct vos_dtx_entry_df *dtx,
428              struct vos_dtx_record_df *rec, bool abort)
429 {
430     struct vos_dtx_record_df    *tgt_rec;
431     struct vos_krec_df      *tgt_key;
432
433 TF     if (rec->tr_flags == DTX_RF_EXCHANGE_TGT) {
434         /* For commit case, should already have been handled
435          * during handling the soruce.
436          *
437          * For abort case, set its kr_dtx as aborted. The tgt
438          * record will be removed via VOS aggregation or other
439          * tools some time later.
440          */
441 TF         if (abort)
442             dtx_set_aborted(&key->kr_dtx);
443         return;
444     }
445
446 TF     if (rec->tr_flags != DTX_RF_EXCHANGE_SRC)
447         return;
448
449 TF     if (!(key->kr_bmap & KREC_BF_REMOVED)) {
450         D_ERROR(DF_UOID" with KEY DTX ("DF_DTI") missed REMOVED flag\n",
451             DP_UOID(dtx->te_oid), DP_DTI(&dtx->te_xid));
452         return;
453     }
454
455 TF     if (abort) { /* abort case */
456         /* Recover the availability of the exchange source. */
457         key->kr_bmap &= ~KREC_BF_REMOVED;
458         key->kr_dtx = UMOFF_NULL;
459         return;
460     }
461
462     /* XXX: If the exchange target still exist, it will be the next
463      *  record. If it does not exist, then either it is crashed
464      *  or it has already deregistered from the DTX records list.
465      *  We cannot commit the DTX under any the two cases. Fail
466      *  the DTX commit is meaningless, then some warnings.
467      */
468 TF     if (dtx_is_null(rec->tr_next)) {
469         D_ERROR(DF_UOID" miss KEY DTX ("DF_DTI") exchange pairs (1)\n",
470             DP_UOID(dtx->te_oid), DP_DTI(&dtx->te_xid));
471         return;
472     }
473
474     tgt_rec = umem_off2ptr(umm, rec->tr_next);
475 TF     if (tgt_rec->tr_flags != DTX_RF_EXCHANGE_TGT) {
476         D_ERROR(DF_UOID" miss KEY DTX ("DF_DTI") exchange pairs (2)\n",
477             DP_UOID(dtx->te_oid), DP_DTI(&dtx->te_xid));
478         return;
479     }
480
481     /* Exchange the sub-tree between max epoch record and the given
482      * epoch record. The record with max epoch will be removed when
483      * aggregation or some special cleanup.
484      */
485     tgt_key = umem_off2ptr(umm, tgt_rec->tr_record);
486     umem_tx_add_ptr(umm, tgt_key, sizeof(*tgt_key));
487
488 TF     if (key->kr_bmap & KREC_BF_EVT) {
489         tgt_key->kr_evt = key->kr_evt;
490         /* The @key which epoch is MAX will be removed later. */
491         memset(&key->kr_evt, 0, sizeof(key->kr_evt));
492     } else {
493         D_ASSERT(key->kr_bmap & KREC_BF_BTR);
494
495         tgt_key->kr_btr = key->kr_btr;
496         /* The @key which epoch is MAX will be removed later. */
497         memset(&key->kr_btr, 0, sizeof(key->kr_btr));
498     }
499
500     /* The @tgt_key which epoch is current DTX's epoch will be
501      * available to outside of VOS. Set its kr_earliest as @key
502      * kr_earliest.
503      */
504     tgt_key->kr_earliest = key->kr_earliest;
505     tgt_key->kr_latest = dtx->te_epoch;
506     tgt_key->kr_bmap = (key->kr_bmap | KREC_BF_PUNCHED) & ~KREC_BF_REMOVED;
507
508     key->kr_latest = 0;
509     key->kr_earliest = DAOS_EPOCH_MAX;
510
511     D_DEBUG(DB_TRACE, "Exchanged KEY DTX records for "DF_DTI"\n",
512         DP_DTI(&dtx->te_xid));
513 }

581 static void
582 dtx_rec_release(struct umem_instance *umm, umem_off_t umoff,
583         bool abort, bool destroy)
584 {
585     struct vos_dtx_entry_df     *dtx;
586
587     dtx = umem_off2ptr(umm, umoff);
588 TF     if (!destroy &&
589         (!dtx_is_null(dtx->te_records) ||
590          dtx->te_flags & (DTX_EF_EXCHANGE_PENDING | DTX_EF_SHARES)))
591         umem_tx_add_ptr(umm, dtx, sizeof(*dtx));
592
593 TF     while (!dtx_is_null(dtx->te_records)) {
594         umem_off_t           rec_umoff = dtx->te_records;
595         struct vos_dtx_record_df    *rec;
596
597         rec = umem_off2ptr(umm, rec_umoff);
598         switch (rec->tr_type) {
599 T         case DTX_RT_OBJ: {
600             struct vos_obj_df   *obj;
601
602             obj = umem_off2ptr(umm, rec->tr_record);
603             umem_tx_add_ptr(umm, obj, sizeof(*obj));
604             dtx_obj_rec_release(umm, obj, rec, umoff, abort);
605             break;
606         }
607 T         case DTX_RT_KEY: {
608             struct vos_krec_df  *key;
609
610             key = umem_off2ptr(umm, rec->tr_record);
611             umem_tx_add_ptr(umm, key, sizeof(*key));
612             dtx_key_rec_release(umm, key, rec, umoff, abort);
613             break;
614         }
615 T         case DTX_RT_SVT: {
616             struct vos_irec_df  *svt;
617
618             svt = umem_off2ptr(umm, rec->tr_record);
619             umem_tx_add_ptr(umm, &svt->ir_dtx, sizeof(svt->ir_dtx));
620 TF             if (abort)
621                 dtx_set_aborted(&svt->ir_dtx);
622             else
623                 svt->ir_dtx = UMOFF_NULL;
624             break;
625         }
626 T         case DTX_RT_EVT: {
627             struct evt_desc     *evt;
628
629             evt = umem_off2ptr(umm, rec->tr_record);
630             umem_tx_add_ptr(umm, &evt->dc_dtx, sizeof(evt->dc_dtx));
631 TF             if (abort)
632                 dtx_set_aborted(&evt->dc_dtx);
633             else
634                 evt->dc_dtx = UMOFF_NULL;
635             break;
636         }
637 T         default:
638             D_ERROR(DF_UOID" unknown DTX "DF_DTI" type %u\n",
639                 DP_UOID(dtx->te_oid), DP_DTI(&dtx->te_xid),
640                 rec->tr_type);
641             break;
642         }
643
644         dtx->te_records = rec->tr_next;
645         umem_free_off(umm, rec_umoff);
646     }
647
648     D_DEBUG(DB_TRACE, "dtx_rec_release: %s/%s the DTX "DF_DTI"\n",
649         abort ? "abort" : "commit", destroy ? "destroy" : "keep",
650         DP_DTI(&dtx->te_xid));
651
652 TF     if (destroy)
653         umem_free_off(umm, umoff);
654     else
655         dtx->te_flags &= ~(DTX_EF_EXCHANGE_PENDING | DTX_EF_SHARES);
656 }

663
664 TF     if (dtx_is_null(dtx->te_next)) { /* The tail of the DTXs list. */

681         dtx->te_next = UMOFF_NULL;
682     } else {
683         ent = umem_off2ptr(umm, dtx->te_next);
684         umem_tx_add_ptr(umm, &ent->te_prev, sizeof(ent->te_prev));
685         ent->te_prev = dtx->te_prev;
686
687         ent = umem_off2ptr(umm, dtx->te_prev);
688         umem_tx_add_ptr(umm, &ent->te_next, sizeof(ent->te_next));
689         ent->te_next = dtx->te_next;
690
691         dtx->te_prev = UMOFF_NULL;
692         dtx->te_next = UMOFF_NULL;
693     }

696 static int
697 vos_dtx_commit_one(struct vos_container *cont, struct dtx_id *dti)
698 {
699     struct umem_instance        *umm = &cont->vc_pool->vp_umm;
700     struct vos_dtx_entry_df     *dtx;
701     struct vos_dtx_entry_df     *ent;
702     struct vos_dtx_table_df     *tab;
703     struct dtx_rec_bundle        rbund;
704     daos_iov_t           kiov;
705     daos_iov_t           riov;
706     umem_off_t           umoff;
707     int              rc = 0;
708
709     daos_iov_set(&kiov, dti, sizeof(*dti));
710     rc = dbtree_delete(cont->vc_dtx_active_hdl, &kiov, &umoff);
711 TF     if (rc == -DER_NONEXIST) {
712         daos_iov_set(&riov, NULL, 0);
713         rc = dbtree_lookup(cont->vc_dtx_committed_hdl, &kiov, &riov);
714         goto out;
715     }
716
717 TF     if (rc != 0)
718         goto out;
719
720     dtx = umem_off2ptr(umm, umoff);
721     umem_tx_add_ptr(umm, dtx, sizeof(*dtx));
722
723     dtx->te_state = DTX_ST_COMMITTED;
724     rbund.trb_umoff = umoff;
725     daos_iov_set(&riov, &rbund, sizeof(rbund));
726     rc = dbtree_upsert(cont->vc_dtx_committed_hdl, BTR_PROBE_EQ,
727                DAOS_INTENT_UPDATE, &kiov, &riov);
728 TF     if (rc != 0)
729         goto out;
730
731     tab = &cont->vc_cont_df->cd_dtx_table_df;
732     umem_tx_add_ptr(umm, tab, sizeof(*tab));
733
734     tab->tt_count++;
735 TF     if (dtx_is_null(tab->tt_entry_tail)) {
736         D_ASSERT(dtx_is_null(tab->tt_entry_head));
737
738         tab->tt_entry_head = umoff;
739         tab->tt_entry_tail = tab->tt_entry_head;
740     } else {
741         ent = umem_off2ptr(umm, tab->tt_entry_tail);
742         umem_tx_add_ptr(umm, &ent->te_next, sizeof(ent->te_next));
743
744         ent->te_next = umoff;
745         dtx->te_prev = tab->tt_entry_tail;
746         tab->tt_entry_tail = ent->te_next;
747     }
748
749     /* XXX: Only mark the DTX as DTX_ST_COMMITTED (when commit) is not
750      *  enough. Otherwise, some subsequent modification may change
751      *  related data record's DTX reference or remove related data
752      *  record as to the current DTX will have invalid reference(s)
753      *  via its DTX record(s).
754      */
755     dtx_rec_release(umm, umoff, false, false);
756     vos_dtx_del_cos(cont, &dtx->te_oid, dti, dtx->te_dkey_hash,
757 TF             dtx->te_intent == DAOS_INTENT_PUNCH ? true : false);
758
759 out:
760     D_DEBUG(DB_TRACE, "Commit the DTX "DF_DTI": rc = %d\n",
761         DP_DTI(dti), rc);
762
763     return rc;
764 }

766 static int
767 vos_dtx_abort_one(struct vos_container *cont, struct dtx_id *dti,
768           bool force)

855 static void
856 vos_dtx_append(struct umem_instance *umm, struct dtx_handle *dth,
857            umem_off_t rec_umoff, umem_off_t record, uint32_t type,
858            uint32_t flags, struct vos_dtx_entry_df **dtxp)
859 {
860     struct vos_dtx_entry_df     *dtx;
861     struct vos_dtx_record_df    *rec;

863
864     /* The @dtx must be new created via former
865      * vos_dtx_register_record(), no need umem_tx_add_ptr().
866      */

874 TF     if (flags == 0)

879      *  the target for punch {a,d}key. It is implemented as exchange

887      * vos_dtx_register_record(), no need umem_tx_add_ptr().

893     dtx->te_flags |= DTX_EF_EXCHANGE_PENDING;

895 TF     if (type == DTX_RT_OBJ) {
896         struct vos_obj_df   *obj;

907     D_DEBUG(DB_TRACE, "Register exchange source for %s DTX "DF_DTI"\n",
908         type == DTX_RT_OBJ ? "OBJ" : "KEY", DP_DTI(&dtx->te_xid));
909 }

1054 static int
1055 vos_dtx_check_shares(struct dtx_handle *dth, struct vos_dtx_entry_df *dtx,
1056              umem_off_t record, uint32_t intent, uint32_t type)
1057 {
1058     struct vos_dtx_share    *vds;
1059
1060 TF     if (dtx != NULL)
1061         D_ASSERT(dtx->te_intent == DAOS_INTENT_UPDATE);
1062
1063     /* PUNCH cannot share with others. */
1064 TF     if (intent == DAOS_INTENT_PUNCH) {
1065         /* XXX: One corner case: if some DTXs share the same
1066          *  object/key, and the original DTX that create
1067          *  the object/key is aborted, then when we come
1068          *  here, we do not know which DTX conflict with

1072         dtx_record_conflict(dth, dtx);

1078

1081         dtx_record_conflict(dth, dtx);
1082
1083         return dtx_inprogress(dtx, 5);
1084     }
1085
1086 TF     if (dth == NULL)
1087         return dtx_inprogress(dtx, 6);
1088

1090 TF     if (vds == NULL)
1091         return -DER_NOMEM;

1093     vds->vds_type = type;
1094     vds->vds_record = record;

1100 int
1101 vos_dtx_check_availability(struct umem_instance *umm, daos_handle_t coh,
1102                umem_off_t entry, umem_off_t record, uint32_t intent,
1103                uint32_t type)
1104 {
1105     struct dtx_handle       *dth = vos_dth_get();
1106     struct vos_dtx_entry_df     *dtx = NULL;
1107     bool                 hidden = false;
1108
1109     switch (type) {
1110 T     case DTX_RT_OBJ: {
1111         struct vos_obj_df   *obj;
1112
1113         obj = umem_off2ptr(umm, record);
1114
1115         /* I just created (or shared) the object,
1116          * so should be available unless aborted.
1117          */
1118 TF         if (dth != NULL && dth->dth_obj == record) {
1119 TF             if (!dtx_is_aborted(obj->vo_dtx))
1120                 return ALB_AVAILABLE_CLEAN;
1121
1122 TF             if (intent == DAOS_INTENT_PURGE)
1123                 return ALB_AVAILABLE_DIRTY;
1124
1125             return ALB_UNAVAILABLE;
1126         }
1127
1128 TF         if (obj->vo_oi_attr & VOS_OI_REMOVED)
1129             hidden = true;
1130         break;
1131     }
1132 T     case DTX_RT_KEY: {
1133         struct vos_krec_df  *key;
1134
1135         key = umem_off2ptr(umm, record);
1136 TF         if (key->kr_bmap & KREC_BF_REMOVED)
1137             hidden = true;
1138         break;
1139     }
1140 T     case DTX_RT_SVT:
1141 T     case DTX_RT_EVT:
1142         break;
1143 T     default:
1144         D_ERROR("Unexpected DTX type %u\n", type);
1145         /* Everything is available to PURGE, even if it belongs to some
1146          * uncommitted DTX that may be garbage because of corruption.
1147          */
1148 TF         if (intent == DAOS_INTENT_PURGE)
1149             return ALB_AVAILABLE_DIRTY;
1150
1151         return -DER_INVAL;
1152     }
1153
1154 TF     if (intent == DAOS_INTENT_CHECK) {
1155         /* Check whether the DTX is aborted or not. */
1156 TF         if (dtx_is_aborted(entry))
1157             return ALB_UNAVAILABLE;
1158         else
1159             return ALB_AVAILABLE_CLEAN;
1160     }
1161
1162     /* Committed */
1163 TF     if (dtx_is_null(entry)) {
1164 TF         if (!hidden)
1165             return ALB_AVAILABLE_CLEAN;
1166
1167 TF         if (intent == DAOS_INTENT_PURGE)
1168             return ALB_AVAILABLE_DIRTY;
1169
1170         return ALB_UNAVAILABLE;
1171     }
1172
1173 TF     if (intent == DAOS_INTENT_PURGE)
1174 TF         return hidden ? ALB_AVAILABLE_CLEAN : ALB_AVAILABLE_DIRTY;
1175
1176     /* Aborted */
1177 TF     if (dtx_is_aborted(entry))
1178 TF         return hidden ? ALB_AVAILABLE_CLEAN : ALB_UNAVAILABLE;
1179
1180 TF     if (dtx_is_unknown(entry)) {
1181         /* The original DTX must be with DAOS_INTENT_UPDATE, then
1182          * it has been shared by other UPDATE DTXs. And then the
1183          * original DTX was aborted, but other DTXs still are not
1184          * 'committable' yet.
1185          */
1186
1187 TF         if (intent == DAOS_INTENT_DEFAULT ||
1188             intent == DAOS_INTENT_REBUILD)
1189 TF             return hidden ? ALB_AVAILABLE_CLEAN : ALB_UNAVAILABLE;
1190
1191         return vos_dtx_check_shares(dth, NULL, record, intent, type);
1192     }
1193
1194     /* The DTX owner can always see the DTX. */
1195 TF     if (dth != NULL && entry == dth->dth_ent)
1196         return ALB_AVAILABLE_CLEAN;
1197
1198     dtx = umem_off2ptr(umm, entry);
1199     switch (dtx->te_state) {
1200 T     case DTX_ST_COMMITTED:
1201 TF         return hidden ? ALB_UNAVAILABLE : ALB_AVAILABLE_CLEAN;
1202 T     case DTX_ST_PREPARED: {
1203         struct vos_container    *cont = vos_hdl2cont(coh);
1204         int          rc;
1205
1206         rc = vos_dtx_lookup_cos(coh, &dtx->te_oid, &dtx->te_xid,
1207             dtx->te_dkey_hash,
1208 TF             dtx->te_intent == DAOS_INTENT_PUNCH ? true : false);
1209 TF         if (rc == 0) {
1210             /* XXX: For the committable punch DTX, if there is
1211              *  pending exchange (of sub-trees) operation,
