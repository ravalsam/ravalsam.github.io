410     D_ASSERT(ns != NULL);
411
412     iv_key_unpack(&key, iv_key);
413 TF     if (priv_entry == NULL || priv_entry->entry == NULL) {
414         /* find and prepare entry */
415         rc = iv_entry_lookup_or_create(ns, &key, &entry);
416 TF         if (rc < 0)
417             return rc;
418     } else {
419         entry = priv_entry->entry;
420     }
421
422 TF     if (iv_value && iv_value->sg_iovs != NULL) {
423 TF         if (refresh)
424             rc = refresh_iv_value(entry, &entry->iv_value, iv_value,
425                       ref_rc,
426 TF                       priv_entry ? priv_entry->priv : NULL);
427         else
428             rc = update_iv_value(entry, &entry->iv_value, iv_value,
429 TF                       priv_entry ? priv_entry->priv : NULL);
430 TF         if (rc != -DER_IVCB_FORWARD && rc != 0) {
431             D_ERROR("key id %d update failed: rc = %d\n",
432                 key.class_id, rc);
433             return rc;
434         }
435     }
436
437 TF     if (invalidate)
438         entry->iv_valid = false;
439     else
440         entry->iv_valid = true;
441
442     D_DEBUG(DB_TRACE, "key id %d rank %d myrank %d valid %s\n",
443         key.class_id, key.rank, dss_self_rank(),
444         invalidate ? "no" : "yes");
445
446     return rc;
447 }

827 static int
828 iv_internal(struct ds_iv_ns *ns, struct ds_iv_key *key_iv, d_sg_list_t *value,
829         crt_iv_sync_t *sync, unsigned int shortcut, int opc)
830 {
831     struct iv_cb_info   cb_info;
832     ABT_future      future;
833     crt_iv_key_t        key_iov;
834     struct ds_iv_class  *class;
835     int         rc;
836
837     rc = ABT_future_create(1, NULL, &future);
838 TF     if (rc)
839         return rc;
840
841     key_iv->rank = ns->iv_master_rank;
842     class = iv_class_lookup(key_iv->class_id);
843     D_ASSERT(class != NULL);
844     D_DEBUG(DB_TRACE, "class_id %d crt class id %d opc %d\n",
845         key_iv->class_id, class->iv_cart_class_id, opc);
846
847     iv_key_pack(&key_iov, key_iv);
848     memset(&cb_info, 0, sizeof(cb_info));
849     cb_info.future = future;
850     cb_info.key = key_iv;
851     cb_info.value = value;
852     cb_info.opc = opc;
853     cb_info.ns = ns;
854     switch (opc) {
855 T     case IV_FETCH:
856         rc = crt_iv_fetch(ns->iv_ns, class->iv_cart_class_id,
857                   (crt_iv_key_t *)&key_iov, 0,
858                   0, ds_iv_done, &cb_info);
859         break;
860 T     case IV_UPDATE:
861         rc = crt_iv_update(ns->iv_ns, class->iv_cart_class_id,
862                    (crt_iv_key_t *)&key_iov, 0,
863                    (d_sg_list_t *)value, shortcut,
864                    *sync, ds_iv_done, &cb_info);
865         break;
866 T     case IV_INVALIDATE:
867         rc = crt_iv_invalidate(ns->iv_ns, class->iv_cart_class_id,
868                        (crt_iv_key_t *)&key_iov, 0, 0, *sync,
869                        ds_iv_done, &cb_info);
870         break;
871 T     default:
872         D_ASSERT(0);
873     }
874
875 TF     if (rc)
876         D_GOTO(out, rc);
877
878     ABT_future_wait(future);
879     rc = cb_info.result;
880     D_DEBUG(DB_TRACE, "class_id %d opc %d rc %d\n", key_iv->class_id, opc,
881         rc);
882 out:
883     ABT_future_free(&future);
884     return rc;
885 }
