206 /**
207  * Get the appropriate number of main XS based on the number of cores and
208  * passed in preferred number of threads.

210 static int
211 dss_tgt_nr_get(int ncores, int nr)
212 {
213     int nr_default;
214
215     D_ASSERT(ncores >= 1);
216     /* Each system XS uses one core, and each main XS with
217      * dss_tgt_offload_xs_nr offload XS. Calculate the nr_default
218      * as the number of main XS based on number of cores.
219      */
220     nr_default = (ncores - dss_sys_xs_nr) / DSS_XS_NR_PER_TGT;
221 TF     if (nr_default == 0)
222         nr_default = 1;
223
224     /* If user requires less target threads then set it as dss_tgt_nr,
225      * if user requires more then uses the number calculated above
226      * as creating more threads than #cores may hurt performance.
227      */
228 TF     if (nr >= 1 && nr < nr_default)
229         nr_default = nr;
230
231 TF     if (nr_default != nr)
232         D_PRINT("%d target XS(xstream) requested (#cores %d); "
233             "use (%d) target XS\n", nr, ncores, nr_default);
234
235     return nr_default;
236 }

760     /** block all possible signals but faults */
