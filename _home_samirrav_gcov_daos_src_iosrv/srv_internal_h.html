68 /** To schedule ULT on caller's self XS */

90 /**
91  * Get the service xstream xs_id to schedule the ULT for specific ULT type and
92  * target index.
93  *
94  * \param[in]   ult_type    ULT type (enum daos_ult_type)
95  * \param[in]   tgt_idx     target VOS index (main xstream index)
96  *
97  * \return          XS (xstream) xs_id to be used for the ULT
98  */
99 static inline int
100 dss_ult_xs(int ult_type, int tgt_id)
101 {
102 TF     if (tgt_id == DSS_TGT_SELF || ult_type == DSS_ULT_DTX_RESYNC)
103         return DSS_XS_SELF;
104
105     D_ASSERT(tgt_id >= 0 && tgt_id < dss_tgt_nr);
106     switch (ult_type) {
107 T     case DSS_ULT_IOFW:
108 T     case DSS_ULT_MISC:
109         return (DSS_MAIN_XS_ID(tgt_id) + 1) % DSS_XS_NR_TOTAL;
110 T     case DSS_ULT_EC:
111 T     case DSS_ULT_CHECKSUM:
112 T     case DSS_ULT_COMPRESS:
113         return DSS_MAIN_XS_ID(tgt_id) + dss_tgt_offload_xs_nr;
114 T     case DSS_ULT_POOL_SRV:
115 T     case DSS_ULT_RDB:
116 T     case DSS_ULT_DRPC:
117         return 0;
118 T     case DSS_ULT_REBUILD:
119 T     case DSS_ULT_AGGREGATE:
120         return DSS_MAIN_XS_ID(tgt_id);
121 T     default:
122         D_ASSERTF(0, "bad ult_type %d.\n", ult_type);
123         return -DER_INVAL;
124     }
125 }
