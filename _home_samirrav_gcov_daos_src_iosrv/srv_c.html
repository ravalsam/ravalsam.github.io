384                 D_ASSERTF(dx->dx_ctx_id ==

386                       "incorrect ctx_id %d for xs_id %d\n",
387                       dx->dx_ctx_id, dx->dx_xs_id);
388             else
389                 D_ASSERTF(dx->dx_ctx_id ==
390                       (dss_sys_xs_nr + dss_tgt_nr +
391                        dx->dx_tgt_id),
392                       "incorrect ctx_id %d for xs_id %d\n",
393                       dx->dx_ctx_id, dx->dx_xs_id);
394         }
395     }
396
397     /* Prepare the scheduler */
398     rc = tse_sched_init(&dmi->dmi_sched, NULL, dmi->dmi_ctx);
399 TF     if (rc != 0) {
400         D_ERROR("failed to init the scheduler\n");
401         goto crt_destroy;
402     }
403
404     /* Initialize NVMe context for main XS which accesses NVME */
405 TF     if (dx->dx_main_xs) {
406         rc = bio_xsctxt_alloc(&dmi->dmi_nvme_ctxt, dmi->dmi_xs_id);
407 TF         if (rc != 0) {
408             D_ERROR("failed to init spdk context for xstream(%d) "
409                 "rc:%d\n", dmi->dmi_xs_id, rc);
410             D_GOTO(tse_fini, rc);
411         }
412     }
413
414     dmi->dmi_xstream = dx;
415     ABT_mutex_lock(xstream_data.xd_mutex);
416     /* initialized everything for the ULT, notify the creater */
417     D_ASSERT(!xstream_data.xd_ult_signal);
418     xstream_data.xd_ult_signal = true;
419     xstream_data.xd_ult_init_rc = 0;
420     ABT_cond_signal(xstream_data.xd_ult_init);
421
422     /* wait until all xstreams are ready, otherwise it is not safe
423      * to run lock-free dss_collective, althought this race is not
424      * realistically possible in the DAOS stack.
425      */
426     ABT_cond_wait(xstream_data.xd_ult_barrier, xstream_data.xd_mutex);
427     ABT_mutex_unlock(xstream_data.xd_mutex);
428
429     signal_caller = false;
430     /* main service progress loop */
431     for (;;) {
432         ABT_bool state;
433
434 TF         if (dx->dx_comm) {
435             rc = crt_progress(dmi->dmi_ctx, 0 /* no wait */, NULL,
436                       NULL);
437 TF             if (rc != 0 && rc != -DER_TIMEDOUT) {
438                 D_ERROR("failed to progress CART context: %d\n",
439                     rc);
440                 /* XXX Sometimes the failure might be just
441                  * temporary, Let's keep progressing for now.
442                  */
443             }
444         }
445
446 TF         if (dx->dx_main_xs)
447             bio_nvme_poll(dmi->dmi_nvme_ctxt);
448
449         rc = ABT_future_test(dx->dx_shutdown, &state);
450         D_ASSERTF(rc == ABT_SUCCESS, "%d\n", rc);
451 TF         if (state == ABT_TRUE)
452             break;
453
454         ABT_thread_yield();
455     }
456
457     /* Let's wait until all of queue ULTs has been executed, in case dmi_ctx
458      * might be used by some other ULTs.
459      */
460     while (1) {
461         size_t total_size = 0;
462         int i;
463
464 TF         for (i = 0; i < DSS_POOL_CNT; i++) {
465             size_t pool_size;
466
467             rc = ABT_pool_get_total_size(dx->dx_pools[i],
468                              &pool_size);
469             D_ASSERTF(rc == ABT_SUCCESS, "%d\n", rc);
470             total_size += pool_size;
471         }
472 TF         if (total_size == 0)
473             break;
474
475         ABT_thread_yield();
476     }
477
478 TF     if (dx->dx_main_xs)
479         bio_xsctxt_free(dmi->dmi_nvme_ctxt);
480 tse_fini:
481     tse_sched_fini(&dmi->dmi_sched);
482 crt_destroy:
483 TF     if (dx->dx_comm)
484         crt_context_destroy(dmi->dmi_ctx, true);
485 tls_fini:
486     dss_tls_fini(dtc);
487 signal:
488 TF     if (signal_caller) {
489         ABT_mutex_lock(xstream_data.xd_mutex);
490         /* initialized everything for the ULT, notify the creater */
491         D_ASSERT(!xstream_data.xd_ult_signal);
492         xstream_data.xd_ult_signal = true;
493         xstream_data.xd_ult_init_rc = rc;
494         ABT_cond_signal(xstream_data.xd_ult_init);
495         ABT_mutex_unlock(xstream_data.xd_mutex);
496     }
497 }

506     D_ALLOC_PTR(dx);

547 /**
548  * Start one xstream.
549  *
550  * \param[in] cpus  the cpuset to bind the xstream
551  * \param[in] xs_id the xs_id of xstream (start from 0)
552  *
553  * \retval  = 0 if starting succeeds.
554  * \retval  negative errno if starting fails.
555  */
556 static int
557 dss_start_one_xstream(hwloc_cpuset_t cpus, int xs_id)
558 {
559     struct dss_xstream  *dx;
560     ABT_thread_attr     attr = ABT_THREAD_ATTR_NULL;
561     int         rc = 0;
562     bool            comm; /* true to create cart ctx for RPC */
563     int         xs_offset;
564     int         i;
565
566     /** allocate & init xstream configuration data */
567     dx = dss_xstream_alloc(cpus);
568 TF     if (dx == NULL)
569         return -DER_NOMEM;
570
571     /** create pools */
572 TF     for (i = 0; i < DSS_POOL_CNT; i++) {
573         ABT_pool_access access;
574
575         /* for DSS_POOL_URGENT, now the only usage is for dtx_resync,
576          * that creates ULT in DSS_XS_SELF. So ABT_POOL_ACCESS_PRIV
577          * is fine.
578          */
579 TF         access = (i == DSS_POOL_SHARE || i == DSS_POOL_REBUILD) ?
580              ABT_POOL_ACCESS_MPSC : ABT_POOL_ACCESS_PRIV;
581
582         rc = ABT_pool_create_basic(ABT_POOL_FIFO, access, ABT_TRUE,
583                        &dx->dx_pools[i]);
584 TF         if (rc != ABT_SUCCESS)
585             D_GOTO(out_pool, rc = dss_abterr2der(rc));
586     }
587
588     /* Partial XS need the RPC communication ability - system XS, each
589      * main XS and its first offload XS (for IO dispatch).
590      * The 2nd offload XS(if exists) does not need RPC communication
591      * as it is only for EC/checksum/compress offloading.
592      */
593 TF     xs_offset = xs_id < dss_sys_xs_nr ? -1 : DSS_XS_OFFSET_IN_TGT(xs_id);
594     comm = (xs_id < dss_sys_xs_nr) || xs_offset == 0 || xs_offset == 1;
595     dx->dx_tgt_id   = dss_xs2tgt(xs_id);
596 TF     if (xs_id < dss_sys_xs_nr) {
597         snprintf(dx->dx_name, DSS_XS_NAME_LEN, DSS_SYS_XS_NAME_FMT,
598              xs_id);
599     } else {
600         snprintf(dx->dx_name, DSS_XS_NAME_LEN, DSS_TGT_XS_NAME_FMT,
601              dx->dx_tgt_id, xs_offset + 1);
602     }
603     dx->dx_xs_id    = xs_id;
604     dx->dx_ctx_id   = -1;
605     dx->dx_comm = comm;
606     dx->dx_main_xs  = xs_id >= dss_sys_xs_nr && xs_offset == 0;
607
608     rc = dss_sched_create(dx->dx_pools, DSS_POOL_CNT, &dx->dx_sched);
609 TF     if (rc != 0) {
610         D_ERROR("create scheduler fails: %d\n", rc);
611         D_GOTO(out_pool, rc);
612     }
613
614     /** start XS, ABT rank 0 is reserved for the primary xstream */
615     rc = ABT_xstream_create_with_rank(dx->dx_sched, xs_id + 1,
616                       &dx->dx_xstream);
617 TF     if (rc != ABT_SUCCESS) {
618         D_ERROR("create xstream fails %d\n", rc);
619         D_GOTO(out_sched, rc = dss_abterr2der(rc));
620     }
621
622     rc = ABT_thread_attr_create(&attr);
623 TF     if (rc != ABT_SUCCESS) {
624         D_ERROR("ABT_thread_attr_create fails %d\n", rc);
625         D_GOTO(out_xstream, rc = dss_abterr2der(rc));
626     }
627
628     rc = ABT_thread_attr_set_stacksize(attr, 65536);
629 TF     if (rc != ABT_SUCCESS) {
630         D_ERROR("ABT_thread_attr_set_stacksize fails %d\n", rc);
631         D_GOTO(out_xstream, rc = dss_abterr2der(rc));
632     }
633
634     /** start progress ULT */
635     rc = ABT_thread_create(dx->dx_pools[DSS_POOL_SHARE],
636                    dss_srv_handler, dx, attr,
637                    &dx->dx_progress);
638 TF     if (rc != ABT_SUCCESS) {
639         D_ERROR("create xstream failed: %d\n", rc);
640         D_GOTO(out_xstream, rc = dss_abterr2der(rc));
641     }
642
643     ABT_mutex_lock(xstream_data.xd_mutex);
644
645 TF     if (!xstream_data.xd_ult_signal)
646         ABT_cond_wait(xstream_data.xd_ult_init, xstream_data.xd_mutex);
647     xstream_data.xd_ult_signal = false;
648     rc = xstream_data.xd_ult_init_rc;
649 TF     if (rc != 0) {
650         ABT_mutex_unlock(xstream_data.xd_mutex);
651         goto out_xstream;
652     }
653     xstream_data.xd_xs_ptrs[xs_id] = dx;
654     ABT_mutex_unlock(xstream_data.xd_mutex);
655     ABT_thread_attr_free(&attr);
656
657     D_DEBUG(DB_TRACE, "created xstream name(%s)xs_id(%d)/tgt_id(%d)/"
658         "ctx_id(%d)/comm(%d)/is_main_xs(%d).\n",
659         dx->dx_name, dx->dx_xs_id, dx->dx_tgt_id, dx->dx_ctx_id,
660         dx->dx_comm, dx->dx_main_xs);
661
662     return 0;
663 out_xstream:
664 TF     if (attr != ABT_THREAD_ATTR_NULL)
665         ABT_thread_attr_free(&attr);
666     ABT_xstream_join(dx->dx_xstream);
667     ABT_xstream_free(&dx->dx_xstream);
668     dss_xstream_free(dx);
669     return rc;
670 out_sched:
671     ABT_sched_free(&dx->dx_sched);
672 out_pool:
673 TF     for (i = 0; i < DSS_POOL_CNT; i++) {
674 TF         if (dx->dx_pools[i] != ABT_POOL_NULL)
675             ABT_pool_free(&dx->dx_pools[i]);
676     }
677     dss_xstream_free(dx);
678     return rc;
679 }

681 static void
682 dss_xstreams_fini(bool force)
683 {
684     struct dss_xstream  *dx;
685     int          i;
686     int          rc;
687
688     D_DEBUG(DB_TRACE, "Stopping execution streams\n");
689
690     /** Stop & free progress ULTs */
691 TF     for (i = 0; i < xstream_data.xd_xs_nr; i++) {
692         dx = xstream_data.xd_xs_ptrs[i];
693 TF         if (dx == NULL)
694             continue;
695         ABT_future_set(dx->dx_shutdown, dx);
696     }
697 TF     for (i = 0; i < xstream_data.xd_xs_nr; i++) {
698         dx = xstream_data.xd_xs_ptrs[i];
699 TF         if (dx == NULL)
700             continue;
701         ABT_thread_join(dx->dx_progress);
702         ABT_thread_free(&dx->dx_progress);
703         ABT_future_free(&dx->dx_shutdown);
704     }
705
706     /** Wait for each execution stream to complete */
707 TF     for (i = 0; i < xstream_data.xd_xs_nr; i++) {
708         dx = xstream_data.xd_xs_ptrs[i];
709 TF         if (dx == NULL)
710             continue;
711         ABT_xstream_join(dx->dx_xstream);
712         ABT_xstream_free(&dx->dx_xstream);
713     }
714
715     /** housekeeping ... */
716 TF     for (i = 0; i < xstream_data.xd_xs_nr; i++) {
717         dx = xstream_data.xd_xs_ptrs[i];
718 TF         if (dx == NULL)
719             continue;
720         ABT_sched_free(&dx->dx_sched);
721         dss_xstream_free(dx);
722         xstream_data.xd_xs_ptrs[i] = NULL;
723     }
724
725     /* All other xstreams have terminated. */
726     xstream_data.xd_xs_nr = 0;
727     dss_tgt_nr = 0;
728
729     /* release local storage */
730     rc = pthread_key_delete(dss_tls_key);
731 TF     if (rc)
732         D_ERROR("failed to delete dtc: %d\n", rc);
733
734     D_DEBUG(DB_TRACE, "Execution streams stopped\n");
735 }

771 static int
772 dss_xstreams_init()
773 {
774     int rc;
775     int i, xs_id;
776
777     D_ASSERT(dss_tgt_nr >= 1);
778     D_ASSERT(dss_tgt_offload_xs_nr == 0 || dss_tgt_offload_xs_nr == 1 ||
779          dss_tgt_offload_xs_nr == 2);
780
781     /* initialize xstream-local storage */
782     rc = pthread_key_create(&dss_tls_key, NULL);
783 TF     if (rc) {
784         D_ERROR("failed to create dtc: %d\n", rc);
785         return -DER_NOMEM;
786     }
787
788     /* start the execution streams */
789     D_DEBUG(DB_TRACE, "%d cores detected, starting %d main xstreams\n",
790         dss_core_nr, dss_tgt_nr);
791
792     xstream_data.xd_xs_nr = DSS_XS_NR_TOTAL;
793     /* start system service XS */
794 TF     for (i = 0; i < dss_sys_xs_nr; i++) {
795         xs_id = i;
796         rc = dss_start_xs_id(xs_id);
797 TF         if (rc)
798             D_GOTO(out, rc);
799     }
800
801     /* start main IO service XS */
802 TF     for (i = 0; i < dss_tgt_nr; i++) {
803         xs_id = DSS_MAIN_XS_ID(i);
804         rc = dss_start_xs_id(xs_id);
805 TF         if (rc)
806             D_GOTO(out, rc);
807     }
808
809     /* start offload XS if any */
810 TF     if (dss_tgt_offload_xs_nr == 0)
811         D_GOTO(out, rc);
812 TF     for (i = 0; i < dss_tgt_nr; i++) {
813         int j;
814
815 TF         for (j = 0; j < dss_tgt_offload_xs_nr; j++) {
816             xs_id = DSS_MAIN_XS_ID(i) + j + 1;
817             rc = dss_start_xs_id(xs_id);
818 TF             if (rc)
819                 D_GOTO(out, rc);
820         }
821     }
822
823     D_DEBUG(DB_TRACE, "%d execution streams successfully started "
824         "(first core %d)\n", dss_tgt_nr, dss_core_offset);
825 out:
826     dss_xstreams_open_barrier();
827 TF     if (dss_xstreams_empty()) /* started nothing */
828         pthread_key_delete(dss_tls_key);
829
830     return rc;
831 }
