398  * fail out (if leader also failed because of confilict) or abort them

477             }

501                sizeof(struct dtx_id) * commit_cnt);
502         dth->dth_dti_cos_count = dti_cos_count;
503         dth->dth_dti_cos = dti_cos;
504     }

506 TF     if (abort_cnt > 0) {

560 {

567 TF     if (cont == NULL)

629 int
630 dtx_handle_resend(daos_handle_t coh, daos_unit_oid_t *oid,
631           struct dtx_id *dti, uint64_t dkey_hash, bool punch)
632 {

634

636     switch (rc) {
637 T     case DTX_ST_PREPARED:
638         return 0;
639 T     case DTX_ST_INIT:
640         /* XXX: The INIT DTX in SCM must be for an in-updating
641          *  object/key that was waiting for the bulk transfer.
642          *
643          *  Currently, we do NOT support server re-integration,
644          *  then we will ignore the case of client resending
645          *  RPC to the restarted server. So we can handle the
646          *  DTX_ST_INIT case the same as DTX_ST_PREPARED.
647          *

649          *  older than the server re-integration time or not
650          *  in the furture.

653 T     case DTX_ST_COMMITTED:
654         return -DER_ALREADY;
655 T     case -DER_NONEXIST:
656 TF         if (time(NULL) - dti->dti_sec > DTX_AGG_THRESHOLD_AGE_LOWER) {
657             D_DEBUG(DB_IO, "Not sure about whether the old RPC "
658                 DF_DTI" is resent or not.\n", DP_DTI(dti));
659             return -DER_TIMEDOUT;
