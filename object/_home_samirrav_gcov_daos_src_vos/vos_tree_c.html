374         krec->kr_bmap |= KREC_BF_PUNCHED;
375     }
376     rbund->rb_krec = krec;
377
378     /** Subtree will be created later */
379
380     rc = vos_dtx_register_record(&tins->ti_umm, rec->rec_off,
381                      DTX_RT_KEY, 0);
382 TF     if (rc == 0)
383         ktr_rec_store(tins, rec, kbund, rbund);
384
385     return rc;
386 }

388 static int
389 ktr_rec_free(struct btr_instance *tins, struct btr_record *rec, void *args)
390 {
391     struct vos_krec_df *krec;
392     struct umem_attr    uma;
393     daos_handle_t       toh;
394     int         rc = 0;
395
396 TF     if (UMOFF_IS_NULL(rec->rec_off))
397         return 0;
398
399     krec = vos_rec2krec(tins, rec);
400     umem_attr_get(&tins->ti_umm, &uma);
401
402     vos_dtx_deregister_record(&tins->ti_umm, krec->kr_dtx, rec->rec_off,
403                   DTX_RT_KEY);
404 TF     if (krec->kr_dtx_shares > 0) {
405         D_ERROR("There are some unknown DTXs (%d) share the key rec\n",
406             krec->kr_dtx_shares);
407         return -DER_BUSY;
408     }
409
410     /* has subtree? */
411 TF     if (krec->kr_bmap & KREC_BF_EVT) {
412 TF         if (krec->kr_evt.tr_order == 0)
413             goto exit; /* No subtree */
414
415         rc = evt_open(&krec->kr_evt, &uma, tins->ti_coh,
416                   tins->ti_blks_info, &toh);
417 TF         if (rc != 0)
418             D_ERROR("Failed to open evtree: %d\n", rc);
419         else
420             evt_destroy(toh);
421 TF     } else if (krec->kr_bmap & KREC_BF_BTR) {
422         D_ASSERT(krec->kr_bmap & KREC_BF_BTR);
423 TF         if (krec->kr_btr.tr_order == 0)
424             goto exit; /* No subtree */
425         rc = dbtree_open_inplace_ex(&krec->kr_btr, &uma, tins->ti_coh,
426                         tins->ti_blks_info, &toh);
427 TF         if (rc != 0)
428             D_ERROR("Failed to open btree: %d\n", rc);
429         else
430             dbtree_destroy(toh);
431     } /* It's possible that neither tree is created in case of punch only */
432 exit:
433     umem_free_off(&tins->ti_umm, rec->rec_off);
434     return rc;
435 }

517 /**
518  * Set size for the record and returns write buffer address of the record,
519  * so caller can copy/rdma data into it.
520  */
521 static int
522 svt_rec_store(struct btr_instance *tins, struct btr_record *rec,
523           struct vos_key_bundle *kbund, struct vos_rec_bundle *rbund)
524 {
525     struct vos_irec_df  *irec   = vos_rec2irec(tins, rec);
526     daos_csum_buf_t     *csum   = rbund->rb_csum;
527     struct bio_iov      *biov   = rbund->rb_biov;
528
529 TF     if (biov->bi_data_len != rbund->rb_rsize)
530         return -DER_IO_INVAL;
531
532     irec->ir_cs_size = csum->cs_len;
533     irec->ir_cs_type = csum->cs_type;
534     irec->ir_size    = biov->bi_data_len;
535     irec->ir_ex_addr = biov->bi_addr;
536     irec->ir_ver     = rbund->rb_ver;
537
538 TF     if (irec->ir_size == 0) { /* it is a punch */
539         csum->cs_csum = NULL;
540         return 0;
541     }
542     /** at this point, it's assumed that enough was allocated for the irec
543      *  to hold a checksum of length csum->cs_len
544      */
545     memcpy(vos_irec2csum(irec), csum->cs_csum, csum->cs_len);
546
547     return 0;
548 }
549
550 /**
551  * Return memory address of data and checksum of this record.
552  */
553 static int
554 svt_rec_load(struct btr_instance *tins, struct btr_record *rec,
555          struct vos_key_bundle *kbund, struct vos_rec_bundle *rbund)
556 {
557     struct svt_hkey    *skey = (struct svt_hkey *)&rec->rec_hkey[0];
558     struct vos_irec_df *irec = vos_rec2irec(tins, rec);
559     daos_csum_buf_t    *csum = rbund->rb_csum;
560     struct bio_iov     *biov = rbund->rb_biov;
561
562 TF     if (kbund != NULL) /* called from iterator */
563         kbund->kb_epoch = skey->sv_epoch;
564
565     /* NB: return record address, caller should copy/rma data for it */
566     biov->bi_data_len = irec->ir_size;
567     biov->bi_addr = irec->ir_ex_addr;
568     biov->bi_buf = NULL;
569
570 TF     if (irec->ir_size != 0 && csum) {
571         csum->cs_len        = irec->ir_cs_size;
572         csum->cs_buf_len    = irec->ir_cs_size;
573         csum->cs_type       = irec->ir_cs_type;
574 TF         if (csum->cs_csum)
575             memcpy(csum->cs_csum,
576                    vos_irec2csum(irec), csum->cs_len);
577         else
578             csum->cs_csum = (uint8_t *) vos_irec2csum(irec);
579     }
580
581     rbund->rb_rsize = irec->ir_size;
582     rbund->rb_ver   = irec->ir_ver;
583     return 0;
584 }

633 /** allocate a new record and fetch data */
634 static int
635 svt_rec_alloc(struct btr_instance *tins, daos_iov_t *key_iov,
636            daos_iov_t *val_iov, struct btr_record *rec)
637 {
638     struct vos_rec_bundle   *rbund;
639     struct vos_key_bundle   *kbund;
640     int          rc = 0;
641
642     kbund = iov2key_bundle(key_iov);
643     rbund = iov2rec_bundle(val_iov);
644
645 TF     if (UMMID_IS_NULL(rbund->rb_mmid)) {
646 TF         rec->rec_off = umem_alloc_off(&tins->ti_umm,
646 TF         rec->rec_off = umem_alloc_off(&tins->ti_umm,
646 TF         rec->rec_off = umem_alloc_off(&tins->ti_umm,
646 TF         rec->rec_off = umem_alloc_off(&tins->ti_umm,
647                        vos_irec_size(rbund));
648 TF         if (UMOFF_IS_NULL(rec->rec_off))
649             return -DER_NOMEM;
650     } else {
651         umem_tx_add(&tins->ti_umm, rbund->rb_mmid,
652                 vos_irec_msize(rbund));
653         rec->rec_off = umem_id2off(&tins->ti_umm, rbund->rb_mmid);
654         rbund->rb_mmid = UMMID_NULL; /* taken over by btree */
655     }
656
657     rc = vos_dtx_register_record(&tins->ti_umm, rec->rec_off,
658                      DTX_RT_SVT, 0);
659 TF     if (rc != 0)
660         /* It is unnecessary to free the PMEM that will be dropped
661          * automatically when the PMDK transaction is aborted.
662          */
663         return rc;
664
665     rc = svt_rec_store(tins, rec, kbund, rbund);
666     return rc;
667 }

669 static int
670 svt_rec_free(struct btr_instance *tins, struct btr_record *rec,
671           void *args)
672 {
673     struct vos_irec_df *irec = vos_rec2irec(tins, rec);
674     bio_addr_t *addr = &irec->ir_ex_addr;
675
676 TF     if (UMOFF_IS_NULL(rec->rec_off))
677         return 0;
678
679     vos_dtx_deregister_record(&tins->ti_umm, irec->ir_dtx, rec->rec_off,
680                   DTX_RT_SVT);
681 TF     if (args != NULL) {
682         *(umem_id_t *)args = umem_off2id(&tins->ti_umm, rec->rec_off);
683         rec->rec_off = UMOFF_NULL; /** taken over by user */
684         return 0;
685     }
686
687 TF     if (addr->ba_type == DAOS_MEDIA_NVME && !bio_addr_is_hole(addr)) {
688         struct vea_space_info *vsi = tins->ti_blks_info;
689         uint64_t blk_off;
690         uint32_t blk_cnt;
691         int rc;
692
693         D_ASSERT(vsi != NULL);
694
695         blk_off = vos_byte2blkoff(addr->ba_off);
696         blk_cnt = vos_byte2blkcnt(irec->ir_size);
697
698         rc = vea_free(vsi, blk_off, blk_cnt);
699 TF         if (rc)
700             D_ERROR("Error on block free. %d\n", rc);
701     }
702
703     umem_free_off(&tins->ti_umm, rec->rec_off);
704     return 0;
705 }

722 static int
723 svt_rec_update(struct btr_instance *tins, struct btr_record *rec,
724         daos_iov_t *key_iov, daos_iov_t *val_iov)
725 {
726     struct svt_hkey     *skey;
727     struct vos_key_bundle   *kbund;
728     struct vos_rec_bundle   *rbund;
729
730     kbund = iov2key_bundle(key_iov);
731     rbund = iov2rec_bundle(val_iov);

734         !vos_irec_size_equal(vos_rec2irec(tins, rec), rbund)) {
735         /* This function should return -DER_NO_PERM to dbtree if:
736          * - it is a rdma, the original record should be replaced.

739          *

741          * rdma-ed record, or just allocate a new one.
742          */
743         return -DER_NO_PERM;
744     }
745
746     skey = (struct svt_hkey *)&rec->rec_hkey[0];
747     D_DEBUG(DB_IO, "Overwrite epoch "DF_U64"\n", skey->sv_epoch);
748
749     umem_tx_add_off(&tins->ti_umm, rec->rec_off, vos_irec_size(rbund));
750     return svt_rec_store(tins, rec, kbund, rbund);
751 }

808 static int
809 tree_open_create(struct vos_object *obj, enum vos_tree_class tclass, int flags,
810          struct vos_krec_df *krec, daos_handle_t *sub_toh)
811 {
812     struct umem_attr    *uma = vos_obj2uma(obj);
813     struct vea_space_info   *info = obj->obj_cont->vc_pool->vp_vea_info;
814     daos_handle_t        coh = vos_cont2hdl(obj->obj_cont);
815     int          expected_flag;
816     int          unexpected_flag;
817     int          rc = 0;
818
819 TF     if (flags & SUBTR_EVT) {
820         expected_flag = KREC_BF_EVT;
821         unexpected_flag = KREC_BF_BTR;
822     } else {
823         expected_flag = KREC_BF_BTR;
824         unexpected_flag = KREC_BF_EVT;
825     }
826
827 TF     if (krec->kr_bmap & unexpected_flag) {
