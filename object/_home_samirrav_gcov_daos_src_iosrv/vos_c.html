354 /**

368  * \retval      1   buffer(s) full
369  * \retval      -DER_*  error
370  */
371 int
372 dss_enum_pack(vos_iter_param_t *param, vos_iter_type_t type, bool recursive,
373           struct vos_iter_anchors *anchors, struct dss_enum_arg *arg)
374 {
375     int rc;
376
377     D_ASSERT(!arg->fill_recxs ||
378          type == VOS_ITER_SINGLE || type == VOS_ITER_RECX);
379
380     rc = vos_iterate(param, type, recursive, anchors, enum_pack_cb, arg);
381
382     D_DEBUG(DB_IO, "enum type %d tag %d rc %d\n", type,
383         dss_get_module_info()->dmi_tgt_id, rc);
384     return rc;
385 }

387 static int
388 grow_array(void **arrayp, size_t elem_size, int old_len, int new_len)
389 {
390     void *p;
391
392     D_ASSERTF(old_len < new_len, "%d < %d\n", old_len, new_len);
393     D_REALLOC(p, *arrayp, elem_size * new_len);
394 TF     if (p == NULL)
395         return -DER_NOMEM;
396     /* Until D_REALLOC does this, zero the new segment. */
397     memset(p + elem_size * old_len, 0, elem_size * (new_len - old_len));
398     *arrayp = p;
399     return 0;
400 }

402 enum {
403     UNPACK_COMPLETE_IO = 1, /* Only finish current I/O */
404     UNPACK_COMPLETE_IOD = 2,    /* Only finish current IOD */
405 };

407 /* Parse recxs in <*data, len> and append them to iod and sgl. */
408 static int
409 unpack_recxs(daos_iod_t *iod, int *recxs_cap, daos_sg_list_t *sgl,
410          daos_key_t *akey, daos_key_desc_t *kds, void **data,
411          daos_size_t len, uint32_t *version)
412 {
413     int rc = 0;
414     int type;
415
416 TF     if (iod->iod_name.iov_len == 0)
417         daos_iov_copy(&iod->iod_name, akey);
418     else
419         D_ASSERT(daos_key_match(&iod->iod_name, akey));
420
421 TF     if (kds == NULL)
422         return 0;
423
424 TF     if (kds->kd_val_types == VOS_ITER_SINGLE)
425         type = DAOS_IOD_SINGLE;
426     else
427         type = DAOS_IOD_ARRAY;
428
429 TF     while (len > 0) {
430         struct obj_enum_rec *rec = *data;
431
432         D_DEBUG(DB_REBUILD, "data %p len "DF_U64"\n", *data, len);
433
434         /* Every recx begins with an obj_enum_rec. */
435 TF         if (len < sizeof(*rec)) {
436             D_ERROR("invalid recxs: <%p, %zu>\n", *data, len);
437             rc = -DER_INVAL;
438             break;
439         }
440
441         /* Check if the version is changing. */
442 TF         if (*version == 0) {
443             *version = rec->rec_version;
444 TF         } else if (*version != rec->rec_version) {
445             D_DEBUG(DB_REBUILD, "different version %u != %u\n",
446                 *version, rec->rec_version);
447             rc = UNPACK_COMPLETE_IO;
448             break;
449         }
450
451 TF         if (iod->iod_nr > 0 &&
452             (iod->iod_type == DAOS_IOD_SINGLE ||
453              iod->iod_type != type || rec->rec_size == 0 ||
454              iod->iod_size == 0)) {
455             rc = UNPACK_COMPLETE_IOD;
456             break;
457         }
458
459 TF         if (iod->iod_nr == 0)
460             iod->iod_type = type;
461
462         /* If the arrays are full, grow them as if all the remaining
463          * recxs have no inline data.
464          */
465 TF         if (iod->iod_nr + 1 > *recxs_cap) {
466             int cap = *recxs_cap + len / sizeof(*rec);
467
468             rc = grow_array((void **)&iod->iod_recxs,
469                     sizeof(*iod->iod_recxs), *recxs_cap,
470                     cap);
471 TF             if (rc != 0)
472                 break;
473             rc = grow_array((void **)&iod->iod_eprs,
474                     sizeof(*iod->iod_eprs), *recxs_cap,
475                     cap);
476 TF             if (rc != 0)
477                 break;
478 TF             if (sgl != NULL) {
479                 rc = grow_array((void **)&sgl->sg_iovs,
480                         sizeof(*sgl->sg_iovs),
481                         *recxs_cap, cap);
482 TF                 if (rc != 0)
483                     break;
484             }
485             /* If we execute any of the three breaks above,
486              * *recxs_cap will be < the real capacities of some of
487              * the arrays. This is harmless, as it only causes the
488              * diff segment to be copied and zeroed unnecessarily
489              * next time we grow them.
490              */
491             *recxs_cap = cap;
492         }
493
494         /* Append one more recx. */
495         iod->iod_eprs[iod->iod_nr] = rec->rec_epr;
496         /* Iteration does not fill the high epoch, so let's reset
497          * the high epoch with EPOCH_MAX to make vos fetch/update happy.
498          */
499         iod->iod_eprs[iod->iod_nr].epr_hi = DAOS_EPOCH_MAX;
500         iod->iod_recxs[iod->iod_nr] = rec->rec_recx;
501         iod->iod_nr++;
502         iod->iod_size = rec->rec_size;
503         *data += sizeof(*rec);
504         len -= sizeof(*rec);
505
506         /* Append the data, if inline. */
507 TF         if (sgl != NULL && rec->rec_size > 0) {
508             daos_iov_t *iov = &sgl->sg_iovs[sgl->sg_nr];
509
510 TF             if (rec->rec_flags & RECX_INLINE) {
511                 daos_iov_set(iov, *data, rec->rec_size *
512                              rec->rec_recx.rx_nr);
513             } else {
514                 daos_iov_set(iov, NULL, 0);
515             }
516
517             sgl->sg_nr++;
518             D_ASSERTF(sgl->sg_nr <= iod->iod_nr, "%u == %u\n",
519                   sgl->sg_nr, iod->iod_nr);
520             *data += iov->iov_len;
521             len -= iov->iov_len;
522         }
523
524         D_DEBUG(DB_REBUILD, "unpack %p idx/nr "DF_U64"/"DF_U64 "ver %u"
525             " epr lo/hi "DF_U64"/"DF_U64" size %zd\n",
526             *data, iod->iod_recxs[iod->iod_nr - 1].rx_idx,
527             iod->iod_recxs[iod->iod_nr - 1].rx_nr, rec->rec_version,
528             iod->iod_eprs[iod->iod_nr - 1].epr_lo,
529             iod->iod_eprs[iod->iod_nr - 1].epr_hi, iod->iod_size);
530     }
531
532     D_DEBUG(DB_REBUILD, "pack nr %d version/type /%u/%d rc %d\n",
533         iod->iod_nr, *version, iod->iod_type, rc);
534     return rc;
535 }

537 /**
538  * Initialize \a io with \a iods[\a iods_cap], \a recxs_caps[\a iods_cap], and
539  * \a sgls[\a iods_cap].
540  *
541  * \param[in,out]   io      I/O descriptor
542  * \param[in]       iods        daos_iod_t array
543  * \param[in]       recxs_caps  recxs capacity array
544  * \param[in]       sgls        optional sgl array for inline recxs
545  * \param[in]       ephs        epoch array
546  * \param[in]       iods_cap    maximal number of elements in \a iods,
547  *                  \a recxs_caps, \a sgls, and \a ephs
548  */
549 static void
550 dss_enum_unpack_io_init(struct dss_enum_unpack_io *io, daos_iod_t *iods,
551             int *recxs_caps, daos_sg_list_t *sgls,
552             daos_epoch_t *ephs, int iods_cap)
553 {
554     int i;
555
556     memset(io, 0, sizeof(*io));
557
558     io->ui_dkey_eph = DAOS_EPOCH_MAX;
559
560     D_ASSERTF(iods_cap > 0, "%d\n", iods_cap);
561     io->ui_iods_cap = iods_cap;
562
563     D_ASSERT(iods != NULL);
564     memset(iods, 0, sizeof(*iods) * iods_cap);
565     io->ui_iods = iods;
566
567     D_ASSERT(recxs_caps != NULL);
568     memset(recxs_caps, 0, sizeof(*recxs_caps) * iods_cap);
569     io->ui_recxs_caps = recxs_caps;
570
571 TF     if (sgls != NULL) {
572         memset(sgls, 0, sizeof(*sgls) * iods_cap);
573         io->ui_sgls = sgls;
574     }
575
576 TF     for (i = 0; i < iods_cap; i++)
577         ephs[i] = DAOS_EPOCH_MAX;
578
579     io->ui_akey_ephs = ephs;
580 }

608 {
609     int i;
610
611 TF     for (i = 0; i < io->ui_iods_len; i++) {

630  */

632 dss_enum_unpack_io_fini(struct dss_enum_unpack_io *io)
633 {
634     D_ASSERTF(io->ui_iods_len == 0, "%d\n", io->ui_iods_len);
635     daos_iov_free(&io->ui_dkey);
636 }

638 /*
639  * Close the current iod (i.e., io->ui_iods[io->ui_iods_len]). If it contains
640  * recxs, append it to io by incrementing io->ui_iods_len. If it doesn't
641  * contain any recx, clear it.
642  */
643 static void
644 close_iod(struct dss_enum_unpack_io *io)
645 {
646     D_ASSERTF(io->ui_iods_cap > 0, "%d > 0\n", io->ui_iods_cap);
647     D_ASSERTF(io->ui_iods_len < io->ui_iods_cap, "%d < %d\n",
648           io->ui_iods_len, io->ui_iods_cap);
649 TF     if (io->ui_iods[io->ui_iods_len].iod_nr > 0) {
650         io->ui_iods_len++;
651     } else {
652         daos_sg_list_t *sgl = NULL;
653
654         D_DEBUG(DB_TRACE, "iod without recxs: %d\n", io->ui_iods_len);
655 TF         if (io->ui_sgls != NULL)
656             sgl = &io->ui_sgls[io->ui_iods_len];
657         clear_iod(&io->ui_iods[io->ui_iods_len], sgl,
658               &io->ui_recxs_caps[io->ui_iods_len]);
659     }
660 }

662 /* Close io, pass it to cb, and clear it. */
663 static int
664 complete_io(struct dss_enum_unpack_io *io, dss_enum_unpack_cb_t cb, void *arg)
665 {
666     int rc = 0;
667
668 TF     if (io->ui_iods_len == 0) {
669         D_DEBUG(DB_TRACE, "io empty\n");
670         goto out;
671     }
672     rc = cb(io, arg);
673 out:
674     dss_enum_unpack_io_clear(io);
675     return rc;
676 }

678 /**
679  * Unpack the result of a dss_enum_pack enumeration into \a io, which can then
680  * be used to issue a VOS update. \a arg->*_anchor are ignored currently. \a cb
681  * will be called, for the caller to consume the recxs accumulated in \a io.
682  *
683  * \param[in]       type    enumeration type
684  * \param[in]       arg enumeration argument
685  * \param[in]       cb  callback
686  * \param[in]       cb_arg  callback argument
687  */
688 int
689 dss_enum_unpack(vos_iter_type_t type, struct dss_enum_arg *arg,
690         dss_enum_unpack_cb_t cb, void *cb_arg)
691 {
692     struct dss_enum_unpack_io   io;
693     daos_iod_t          iods[DSS_ENUM_UNPACK_MAX_IODS];
694     int             recxs_caps[DSS_ENUM_UNPACK_MAX_IODS];
695     daos_epoch_t            ephs[DSS_ENUM_UNPACK_MAX_IODS];
696     daos_sg_list_t          sgls[DSS_ENUM_UNPACK_MAX_IODS];
697     daos_key_t          akey = {0};
698     daos_epoch_range_t      *eprs = arg->eprs;
699     void                *ptr;
700     unsigned int            i;
701     int             rc = 0;
702
703     /* Currently, this function is only for unpacking recursive
704      * enumerations from arg->kds and arg->sgl.
705      */
706     D_ASSERT(arg->chk_key2big && !arg->fill_recxs);
707
708     D_ASSERT(arg->kds_len > 0);
709     D_ASSERT(arg->kds != NULL);
710 TF     if (arg->kds[0].kd_val_types != type) {
711         D_ERROR("the first kds type %d != %d\n",
712             arg->kds[0].kd_val_types, type);
713         return -DER_INVAL;
714     }
715
716     dss_enum_unpack_io_init(&io, iods, recxs_caps, sgls, ephs,
717                 DSS_ENUM_UNPACK_MAX_IODS);
718 TF     if (type > VOS_ITER_OBJ)
719         io.ui_oid = arg->oid;
720
721     D_ASSERTF(arg->sgl->sg_nr > 0, "%u\n", arg->sgl->sg_nr);
722     D_ASSERT(arg->sgl->sg_iovs != NULL);
723     ptr = arg->sgl->sg_iovs[0].iov_buf;
724
725 TF     for (i = 0; i < arg->kds_len; i++) {
726         D_DEBUG(DB_REBUILD, "process %d type %d ptr %p len "DF_U64
727             " total %zd\n", i, arg->kds[i].kd_val_types, ptr,
728             arg->kds[i].kd_key_len, arg->sgl->sg_iovs[0].iov_len);
729
730         D_ASSERT(arg->kds[i].kd_key_len > 0);
731 TF         if (arg->kds[i].kd_val_types == VOS_ITER_OBJ) {
732             daos_unit_oid_t *oid = ptr;
733
734 TF             if (arg->kds[i].kd_key_len != sizeof(*oid)) {
735                 D_ERROR("Invalid object ID size: "DF_U64
736                     " != %zu\n", arg->kds[i].kd_key_len,
737                     sizeof(*oid));
738                 rc = -DER_INVAL;
739                 break;
740             }
741 TF             if (daos_unit_oid_is_null(io.ui_oid)) {
742                 io.ui_oid = *oid;
743 TF             } else if (daos_unit_oid_compare(io.ui_oid, *oid) !=
744                    0) {
745                 close_iod(&io);
746                 rc = complete_io(&io, cb, cb_arg);
747 TF                 if (rc != 0)
748                     break;
749                 daos_iov_free(&io.ui_dkey);
750                 io.ui_oid = *oid;
751             }
752             D_DEBUG(DB_REBUILD, "process obj "DF_UOID"\n",
753                 DP_UOID(io.ui_oid));
754 TF         } else if (arg->kds[i].kd_val_types == VOS_ITER_DKEY) {
755             daos_key_t tmp_key;
756
757             tmp_key.iov_buf = ptr;
758             tmp_key.iov_buf_len = arg->kds[i].kd_key_len;
759             tmp_key.iov_len = arg->kds[i].kd_key_len;
760
761 TF             if (io.ui_dkey.iov_len == 0) {
762                 daos_iov_copy(&io.ui_dkey, &tmp_key);
763 TF             } else if (!daos_key_match(&io.ui_dkey, &tmp_key) ||
764                    (eprs != NULL &&
765                     io.ui_dkey_eph != eprs[i].epr_lo)) {
766                 close_iod(&io);
767                 rc = complete_io(&io, cb, cb_arg);
768 TF                 if (rc != 0)
769                     break;
770
771 TF                 if (!daos_key_match(&io.ui_dkey, &tmp_key)) {
772                     daos_iov_free(&io.ui_dkey);
773                     daos_iov_copy(&io.ui_dkey, &tmp_key);
774                 }
775             }
776
777 TF             if (eprs != NULL)
778                 io.ui_dkey_eph = eprs[i].epr_lo;
779
780             D_DEBUG(DB_REBUILD, "process dkey %d %s eph "DF_U64"\n",
781                 (int)io.ui_dkey.iov_len,
782                 (char *)io.ui_dkey.iov_buf,
783                 eprs ? io.ui_dkey_eph : 0);
784 TF         } else if (arg->kds[i].kd_val_types == VOS_ITER_AKEY) {
785             daos_key_t *iod_akey;
786
787             akey.iov_buf = ptr;
788             akey.iov_buf_len = arg->kds[i].kd_key_len;
789             akey.iov_len = arg->kds[i].kd_key_len;
790 TF             if (io.ui_dkey.iov_buf == NULL) {
791                 D_ERROR("No dkey for akey %*.s invalid buf.\n",
792                       (int)akey.iov_len, (char *)akey.iov_buf);
793                 rc = -DER_INVAL;
794                 break;
795             }
796
797             D_DEBUG(DB_REBUILD, "process akey %d %s eph "DF_U64"\n",
