423 static int
424 bootstrap_self(struct ds_rsvc *svc, void *arg)
425 {
426     int rc;
427
428     D_DEBUG(DB_MD, "%s: bootstrapping\n", svc->s_name);
429     ABT_mutex_lock(svc->s_mutex);
430
431     /*
432      * This single-replica DB shall change from DS_RSVC_DOWN to
433      * DS_RSVC_UP_EMPTY state promptly.
434      */
435 TF     while (svc->s_state == DS_RSVC_DOWN)
436         ABT_cond_wait(svc->s_state_cv, svc->s_mutex);
437     D_ASSERTF(svc->s_state == DS_RSVC_UP_EMPTY, "%d\n", svc->s_state);
438
439     rc = rsvc_class(svc->s_class)->sc_bootstrap(svc, arg);
440 TF     if (rc != 0)
441         goto out_mutex;
442
443     /* Try stepping up again. */
444     rc = rsvc_class(svc->s_class)->sc_step_up(svc);
445 TF     if (rc != 0) {
446         D_ASSERT(rc != DER_UNINIT);
447         goto out_mutex;
448     }
449
450     change_state(svc, DS_RSVC_UP);
451 out_mutex:
452     ABT_mutex_unlock(svc->s_mutex);
453     D_DEBUG(DB_MD, "%s: bootstrapped: %d\n", svc->s_name, rc);
454     return rc;
455 }

457 static void
458 rsvc_step_down_cb(struct rdb *db, uint64_t term, void *arg)
459 {
460     struct ds_rsvc *svc = arg;
461
462     D_DEBUG(DB_MD, "%s: stepping down from "DF_U64"\n", svc->s_name, term);
463     ABT_mutex_lock(svc->s_mutex);
464     D_ASSERTF(svc->s_term == term, DF_U64" == "DF_U64"\n", svc->s_term,
465           term);
466     D_ASSERT(svc->s_state == DS_RSVC_UP_EMPTY ||
467          svc->s_state == DS_RSVC_UP);
468
469 TF     if (svc->s_state == DS_RSVC_UP) {
470         /* Stop accepting new leader references. */
471         change_state(svc, DS_RSVC_DRAINING);
472
473         rsvc_class(svc->s_class)->sc_drain(svc);
474
475         /* TODO: Abort all in-flight RPCs we sent. */
476
477         /* Wait for all leader references to be released. */
478         for (;;) {
479 TF             if (svc->s_leader_ref == 0)
480                 break;
481             D_DEBUG(DB_MD, "%s: waiting for %d leader refs\n",
482                 svc->s_name, svc->s_leader_ref);
483             ABT_cond_wait(svc->s_leader_ref_cv, svc->s_mutex);
484         }
485
486         rsvc_class(svc->s_class)->sc_step_down(svc);
487     }
488
489     change_state(svc, DS_RSVC_DOWN);
490     ABT_mutex_unlock(svc->s_mutex);
491     D_DEBUG(DB_MD, "%s: stepped down from "DF_U64"\n", svc->s_name, term);
492 }

539 static int
540 start(enum ds_rsvc_class_id class, daos_iov_t *id, uuid_t db_uuid, bool create,
541       size_t size, d_rank_list_t *replicas, void *arg, struct ds_rsvc **svcp)
542 {
543     struct ds_rsvc *svc;
544     int     rc;
545
546     rc = alloc_init(class, id, db_uuid, &svc);
547 TF     if (rc != 0)
548         goto err;
549
550 TF     if (create) {
551         rc = rdb_create(svc->s_db_path, svc->s_db_uuid, size, replicas);
552 TF         if (rc != 0)
553             goto err_svc;
554     }
555
556     rc = rdb_start(svc->s_db_path, svc->s_db_uuid, &rsvc_rdb_cbs, svc,
557                &svc->s_db);
558 TF     if (rc != 0)
559         goto err_creation;
560
561 TF     if (create && self_only(replicas) &&
562         rsvc_class(class)->sc_bootstrap != NULL) {
563         rc = bootstrap_self(svc, arg);
564 TF         if (rc != 0)
565             goto err_db;
566     }
567
568     svc->s_ref = 1;
569     *svcp = svc;
570     return 0;
571
572 err_db:
573     rdb_stop(svc->s_db);
574 err_creation:
575 TF     if (create)
576         rdb_destroy(svc->s_db_path, svc->s_db_uuid);
577 err_svc:
578     fini_free(svc);
579 err:
580     return rc;
581 }

583 /**
584  * Start a replicated service. If \a create is false, all remaining input
585  * parameters are ignored; otherwise, create the replica first. If \a replicas
586  * is NULL, all remaining input parameters are ignored; otherwise, bootstrap
587  * the replicated service.
588  *
589  * \param[in]   class       replicated service class
590  * \param[in]   id      replicated service ID
591  * \param[in]   db_uuid     DB UUID
592  * \param[in]   create      whether to create the replica before starting
593  * \param[in]   size        replica size in bytes
594  * \param[in]   replicas    optional initial membership
595  * \param[in]   arg     argument for cbs.sc_bootstrap
596  *
597  * \retval -DER_ALREADY     replicated service already started
598  * \retval -DER_CANCELED    replicated service stopping
599  */
600 int
601 ds_rsvc_start(enum ds_rsvc_class_id class, daos_iov_t *id, uuid_t db_uuid,
602           bool create, size_t size, d_rank_list_t *replicas, void *arg)
603 {
604     uuid_t           db_uuid_buf;
605     struct ds_rsvc      *svc;
606     struct ds_rsvc_class    *impl;
607     d_list_t        *entry;
608     int          rc;
609
610     impl = rsvc_class(class);
611 TF     if (!create) {
612         rc = impl->sc_load_uuid(id, db_uuid_buf);
613 TF         if (rc != 0)
614             goto out;
615         db_uuid = db_uuid_buf;
616     }
617
618     entry = d_hash_rec_find(&rsvc_hash, id->iov_buf, id->iov_len);
619 TF     if (entry != NULL) {
620         svc = rsvc_obj(entry);
621         D_DEBUG(DB_MD, "%s: found: stop=%d\n", svc->s_name,
622             svc->s_stop);
623 TF         if (svc->s_stop)
624             rc = -DER_CANCELED;
625         else
626             rc = -DER_ALREADY;
627         ds_rsvc_put(svc);
628         goto out;
629     }
630
631     rc = start(class, id, db_uuid, create, size, replicas, arg, &svc);
632 TF     if (rc != 0)
633         goto out;
634
635     rc = d_hash_rec_insert(&rsvc_hash, svc->s_id.iov_buf, svc->s_id.iov_len,
636                    &svc->s_entry, true /* exclusive */);
637 TF     if (rc != 0) {
638         D_DEBUG(DB_MD, "%s: insert: %d\n", svc->s_name, rc);
639         stop(svc, create /* destroy */);
640         goto out;
641     }
642
643 TF     if (create) {
644         rc = impl->sc_store_uuid(id, db_uuid);
645 TF         if (rc != 0) {
646             stop(svc, create /* destroy */);
647             goto out;
648         }
649     }
650     D_DEBUG(DB_MD, "%s: started replicated service\n", svc->s_name);
651     ds_rsvc_put(svc);
652 out:
653 TF     if (rc != 0 && rc != -DER_ALREADY && !(create && rc == -DER_EXIST))
654         D_ERROR("Failed to start replicated service: %d\n", rc);
655     return rc;
656 }

658 static int
659 stop(struct ds_rsvc *svc, bool destroy)
660 {
661     int rc = 0;
662
663     ABT_mutex_lock(svc->s_mutex);
664
665 TF     if (svc->s_stop) {
666         ABT_mutex_unlock(svc->s_mutex);
667         D_DEBUG(DB_MD, "%s: stopping already\n", svc->s_name);
668         return -DER_CANCELED;
669     }
670     svc->s_stop = true;
671     D_DEBUG(DB_MD, "%s: stopping\n", svc->s_name);
672
673 TF     if (svc->s_state == DS_RSVC_UP || svc->s_state == DS_RSVC_UP_EMPTY)
674         /*
675          * The service has stepped up. If it is still the leader of
676          * svc->s_term, the following rdb_resign() call will trigger
677          * the matching rsvc_step_down_cb() callback in svc->s_term;
678          * otherwise, the callback must already be pending. Either way,
679          * the service shall eventually enter the DS_RSVC_DOWN state.
680          */
681         rdb_resign(svc->s_db, svc->s_term);
682 TF     while (svc->s_state != DS_RSVC_DOWN)
683         ABT_cond_wait(svc->s_state_cv, svc->s_mutex);
684
685 TF     if (destroy)
686         rc = remove(svc->s_db_path);
687
688     ABT_mutex_unlock(svc->s_mutex);
689     ds_rsvc_put(svc);
690     return rc;
691 }

694  * Stop a replicated service. If destroy is false, all remaining parameters are
695  * ignored; otherwise, destroy the service afterward.
696  *
697  * \param[in]   class       replicated service class
698  * \param[in]   id      replicated service ID
699  * \param[in]   destroy     whether to destroy the replica after stopping
700  *
701  * \retval -DER_ALREADY     replicated service already stopped
702  * \retval -DER_CANCELED    replicated service stopping
703  */
704 int
705 ds_rsvc_stop(enum ds_rsvc_class_id class, daos_iov_t *id, bool destroy)
706 {
707     struct ds_rsvc      *svc;
708     struct ds_rsvc_class    *impl;
709     d_list_t        *entry;
710     int          rc;
711
712     impl = rsvc_class(class);
713     entry = d_hash_rec_find(&rsvc_hash, id->iov_buf, id->iov_len);
714 TF     if (entry == NULL)
715         return -DER_ALREADY;
716     svc = rsvc_obj(entry);

723     return rc;
724 }

847  * all replicas of a database with \a dbid spanning \a ranks. This method can
848  * be called on any rank. If \a create is false, \a ranks may be NULL.
849  *
850  * \param[in]   class       replicated service class
851  * \param[in]   id      replicated service ID
852  * \param[in]   dbid        database UUID
853  * \param[in]   ranks       list of replica ranks
854  * \param[in]   create      create replicas first
855  * \param[in]   bootstrap   start with an initial list of replicas
856  * \param[in]   size        size of each replica in bytes if \a create
857  */
858 int
859 ds_rsvc_dist_start(enum ds_rsvc_class_id class, daos_iov_t *id,
860            const uuid_t dbid, const d_rank_list_t *ranks, bool create,
861            bool bootstrap, size_t size)
862 {
863     crt_rpc_t       *rpc;
864     struct rsvc_start_in    *in;
865     struct rsvc_start_out   *out;
866     int          rc;
867
868     D_ASSERT(!create || ranks != NULL);
869     D_DEBUG(DB_MD, DF_UUID": %s DB\n",
870         DP_UUID(dbid), create ? "creating" : "starting");
871
872     /*
873      * If ranks doesn't include myself, creating a group with ranks will
874      * fail; bcast to the primary group instead.
875      */
876     rc = bcast_create(RSVC_START, NULL /* group */, &rpc);
877 TF     if (rc != 0)
878         goto out;
879     in = crt_req_get(rpc);
880     in->sai_class = class;
881     rc = daos_iov_copy(&in->sai_svc_id, id);
882 TF     if (rc != 0)
883         goto out_rpc;
884     uuid_copy(in->sai_db_uuid, dbid);
885 TF     if (create)
886         in->sai_flags |= RDB_AF_CREATE;
887 TF     if (bootstrap)
888         in->sai_flags |= RDB_AF_BOOTSTRAP;
889     in->sai_size = size;
890     in->sai_ranks = (d_rank_list_t *)ranks;
891
892     rc = dss_rpc_send(rpc);
893 TF     if (rc != 0)
894         goto out_mem;
895
896     out = crt_reply_get(rpc);
897     rc = out->sao_rc;
898 TF     if (rc != 0) {
