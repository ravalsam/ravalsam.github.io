348 {
349     struct attr_list_iter_args *i_args = arg;
350
351     i_args->length += key->iov_len;
352
353 TF     if (i_args->available > key->iov_len && key->iov_len > 0) {
354         /*
355          * Exponentially grow the array of IOVs if insufficient.
356          * Considering the pathological case where each key is just
357          * a single character, with one additional trailing '\0',
358          * if the client buffer is 'N' bytes, it can hold at the most
359          * N/2 keys, which requires that many IOVs to be allocated.
360          * Thus, the upper limit on the space required for IOVs is:
361          * sizeof(daos_iov_t) * N/2 = 24 * N/2 = 12*N bytes.
362          */
363 TF         if (i_args->iov_index == i_args->iov_count) {
364             void *ptr;
365
366             D_REALLOC(ptr, i_args->iovs,
367                   i_args->iov_count * 2 * sizeof(daos_iov_t));
368             /*
369              * TODO: Fail or continue transferring
370              *   iteratively using available memory?
371              */
372 TF             if (ptr == NULL)
373                 return -DER_NOMEM;
374             i_args->iovs = ptr;
375             i_args->iov_count *= 2;
376         }
377
378         memcpy(&i_args->iovs[i_args->iov_index],
379                key, sizeof(daos_iov_t));
380         i_args->iovs[i_args->iov_index]

382         i_args->available -= key->iov_len;

384     }
