393
394     orw = crt_req_get(req);
395     D_ASSERT(orw != NULL);
396
397 TF     if (fw_shard_tgts != NULL) {
398         D_ASSERT(fw_cnt >= 1);
399         orw->orw_shard_tgts.ca_count = fw_cnt;
400         orw->orw_shard_tgts.ca_arrays = fw_shard_tgts;
401     } else {
402         orw->orw_shard_tgts.ca_count = 0;
403         orw->orw_shard_tgts.ca_arrays = NULL;
404     }
405     orw->orw_map_ver = *map_ver;
406     orw->orw_oid = shard->do_id;
407     uuid_copy(orw->orw_co_hdl, cont_hdl_uuid);
408     uuid_copy(orw->orw_co_uuid, cont_uuid);
409     daos_dti_copy(&orw->orw_dti, dti);
410     orw->orw_flags = flags;
411     orw->orw_dti_cos.ca_count = 0;
412     orw->orw_dti_cos.ca_arrays = NULL;
413
414     orw->orw_epoch = epoch;
415     orw->orw_dkey_hash = dkey_hash;
416     orw->orw_nr = nr;
417     orw->orw_dkey = *dkey;
418     orw->orw_iods.ca_count = nr;
419     orw->orw_iods.ca_arrays = iods;
420
421     data_size = daos_iods_len(iods, nr);
422     sgls_size = daos_sgls_packed_size(sgls, nr, &buf_size);
423     /* If the sgl buffer is not big enough, let's return -REC2BIG
424      * then user can provide appropriate buffer and redo it.
425      */
426 TF     if (data_size != -1 && data_size > buf_size) {
427         rc = -DER_REC2BIG;
428         D_ERROR("Object "DF_UOID", iod_size "DF_U64", sg_buf"
429             " "DF_U64", failed %d.\n",
430             DP_UOID(shard->do_id), data_size, buf_size,
431             rc);
432         D_GOTO(out_req, rc);
433     }
434     /* inline fetch needs to pack sgls buffer into RPC so uses it to check
435      * if need bulk transferring.
436      */
437     data_size = sgls_size;
438
439     D_DEBUG(DB_TRACE, "opc %d "DF_UOID" %d %s rank %d tag %d eph "
440         DF_U64" data_size "DF_U64", DTI = "DF_DTI"\n",
441         opc, DP_UOID(shard->do_id), (int)dkey->iov_len,
442         (char *)dkey->iov_buf, tgt_ep.ep_rank,
443         tgt_ep.ep_tag, epoch, data_size, DP_DTI(&orw->orw_dti));
444
445     do_bulk = data_size >= OBJ_BULK_LIMIT;
446 TF     if (do_bulk) {
447         bool forward = fw_shard_tgts != NULL;
448
449         /* Transfer data by bulk */
450         rc = obj_shard_rw_bulk_prep(req, nr, sgls, forward, task);
451 TF         if (rc != 0)
452             D_GOTO(out_req, rc);
453         orw->orw_sgls.ca_count = 0;
454         orw->orw_sgls.ca_arrays = NULL;
455     } else {
456         /* Transfer data inline */
457 TF         if (sgls != NULL)
458             orw->orw_sgls.ca_count = nr;
459         else
460             orw->orw_sgls.ca_count = 0;
461         orw->orw_sgls.ca_arrays = sgls;
462         orw->orw_bulks.ca_count = 0;
463         orw->orw_bulks.ca_arrays = NULL;
464     }
465
466     crt_req_addref(req);
467     rw_args.rpc = req;
468     rw_args.hdlp = (daos_handle_t *)pool;
469     rw_args.map_ver = map_ver;
470     rw_args.dobj = shard;
471     /* remember the sgl to copyout the data inline for fetch */
472 TF     rw_args.rwaa_sgls = (opc == DAOS_OBJ_RPC_FETCH) ? sgls : NULL;
473
474 TF     if (DAOS_FAIL_CHECK(DAOS_SHARD_OBJ_RW_CRT_ERROR))
475         D_GOTO(out_args, rc = -DER_HG);
476
477     rc = tse_task_register_comp_cb(task, dc_rw_cb, &rw_args,
478                        sizeof(rw_args));
479 TF     if (rc != 0)
480         D_GOTO(out_args, rc);
481     cb_registered = true;
482
483 TF     if (daos_io_bypass & IOBP_CLI_RPC) {
484         rc = daos_rpc_complete(req, task);
485     } else {
486         rc = daos_rpc_send(req, task);
487 TF         if (rc != 0) {
488             D_ERROR("update/fetch rpc failed rc %d\n", rc);
489             D_GOTO(out_args, rc);
490         }
491     }
492     return rc;
493
494 out_args:
495     crt_req_decref(req);
496 TF     if (do_bulk)
497         obj_shard_rw_bulk_fini(req);
498 out_req:
499     crt_req_decref(req);
500 out_pool:
501     dc_pool_put(pool);
502 out_obj:
503 TF     if (!cb_registered)
504         obj_shard_decref(shard);
505     tse_task_complete(task, rc);
506     return rc;
507 }

656 static int
657 dc_enumerate_cb(tse_task_t *task, void *arg)
658 {
659     struct obj_enum_args    *enum_args = (struct obj_enum_args *)arg;
660     struct obj_key_enum_in  *oei;
661     struct obj_key_enum_out *oeo;
662     int          ret = task->dt_result;
663     int          rc = 0;
664
665     oei = crt_req_get(enum_args->rpc);
666     D_ASSERT(oei != NULL);
667
668 TF     if (ret != 0) {
669         /* If any failure happens inside Cart, let's reset
670          * failure to TIMEDOUT, so the upper layer can retry
671          **/
672         D_ERROR("RPC %d failed: %d\n",
673             opc_get(enum_args->rpc->cr_opc), ret);
674         D_GOTO(out, ret);
675     }
676
677     oeo = crt_reply_get(enum_args->rpc);
678     rc = obj_reply_get_status(enum_args->rpc);
679 TF     if (rc != 0) {
680 TF         if (rc == -DER_KEY2BIG) {
681             D_DEBUG(DB_IO, "key size "DF_U64" too big.\n",
682                 oeo->oeo_size);
683             enum_args->eaa_kds[0].kd_key_len = oeo->oeo_size;
684         } else {
685             D_ERROR("rpc %p RPC %d failed: %d\n", enum_args->rpc,
686                  opc_get(enum_args->rpc->cr_opc), rc);
687         }
688         D_GOTO(out, rc);
689     }
690     *enum_args->eaa_map_ver = obj_reply_map_version_get(enum_args->rpc);
691
692 TF     if (enum_args->eaa_size)
693         *enum_args->eaa_size = oeo->oeo_size;
694
695 TF     if (*enum_args->eaa_nr < oeo->oeo_num) {
696         D_ERROR("key enumerate get %d > %d more kds, %d\n",
697             oeo->oeo_num, *enum_args->eaa_nr, -DER_PROTO);
698         D_GOTO(out, rc = -DER_PROTO);
699     }
700
701     *enum_args->eaa_nr = oeo->oeo_num;
702
703 TF     if (enum_args->eaa_kds && oeo->oeo_kds.ca_count > 0)
704         memcpy(enum_args->eaa_kds, oeo->oeo_kds.ca_arrays,
705                sizeof(*enum_args->eaa_kds) *
706                oeo->oeo_kds.ca_count);
707
708 TF     if (enum_args->eaa_eprs && oeo->oeo_eprs.ca_count > 0) {
709         D_ASSERT(*enum_args->eaa_nr >= oeo->oeo_eprs.ca_count);
710         memcpy(enum_args->eaa_eprs, oeo->oeo_eprs.ca_arrays,
711                sizeof(*enum_args->eaa_eprs) *
712                oeo->oeo_eprs.ca_count);
713     }
714
715 TF     if (enum_args->eaa_recxs && oeo->oeo_recxs.ca_count > 0) {
716         D_ASSERT(*enum_args->eaa_nr >= oeo->oeo_recxs.ca_count);
717         memcpy(enum_args->eaa_recxs, oeo->oeo_recxs.ca_arrays,
718                sizeof(*enum_args->eaa_recxs) *
719                oeo->oeo_recxs.ca_count);
720     }
721
722 TF     if (enum_args->eaa_sgl && oeo->oeo_sgl.sg_nr > 0) {
723         rc = daos_sgl_copy_data_out(enum_args->eaa_sgl, &oeo->oeo_sgl);
724 TF         if (rc)
725             D_GOTO(out, rc);
726     }
727
728     /* Update dkey hash and tag */
729 TF     if (enum_args->eaa_dkey_anchor)
730         enum_anchor_copy(enum_args->eaa_dkey_anchor,
731                  &oeo->oeo_dkey_anchor);
732
733 TF     if (enum_args->eaa_akey_anchor)
734         enum_anchor_copy(enum_args->eaa_akey_anchor,
735                  &oeo->oeo_akey_anchor);
736
737 TF     if (enum_args->eaa_anchor)
738         enum_anchor_copy(enum_args->eaa_anchor,
739                  &oeo->oeo_anchor);
740 out:
741 TF     if (enum_args->eaa_obj != NULL)
742         obj_shard_decref(enum_args->eaa_obj);
743
744 TF     if (oei->oei_bulk != NULL)
745         crt_bulk_free(oei->oei_bulk);
746 TF     if (oei->oei_kds_bulk != NULL)
747         crt_bulk_free(oei->oei_kds_bulk);
748     crt_req_decref(enum_args->rpc);
749     dc_pool_put((struct dc_pool *)enum_args->hdlp);
750
751 TF     if (ret == 0 || obj_retry_error(rc))
752         ret = rc;
753     return ret;
754 }
