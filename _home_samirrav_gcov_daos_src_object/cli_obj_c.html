378             D_RWLOCK_UNLOCK(&obj->cob_lock);
379             return -DER_INVAL;
380         }

386         }
387     } else {

389     }
390
391 TF     for (i = 0; i < grp_size;
392          i++, idx = idx_first + (idx + 1 - idx_first) % grp_size) {

394 TF         if (op != DAOS_OBJ_RPC_UPDATE &&

396             continue;
397
398 TF         if (obj->cob_shards->do_shards[idx].do_target_id != -1)
399             break;
400     }
401
402     D_RWLOCK_UNLOCK(&obj->cob_lock);
403
404 TF     if (i == grp_size)
405         return -DER_NONEXIST;

408 }

505 static int

507             uint32_t *start_shard, uint32_t *shard_cnt,
508             struct daos_obj_shard_tgt **fw_shard_tgts, uint32_t *fw_cnt)

511     uint32_t             lead_shard;
512     uint32_t             i, j;

516 TF     if (!srv_io_dispatch || *shard_cnt == 1)

523             DP_OID(obj->cob_md.omd_id), rc);

529         /* fw_cnt possibly changed per progressive layout */

541     }

555                 rc);
556             return rc;
557         }
558     }
559

561     *shard_cnt = 1;
562
563     return 0;
564 }

927          struct obj_auxi_args *obj_auxi)
928 {
929     tse_sched_t  *sched = tse_task2sched(task);
930     tse_task_t   *pool_task = NULL;
931     int       result = task->dt_result;
932     int       rc;
933
934 TF     if (result == -DER_INPROGRESS)
935         obj_auxi->retry_with_leader = 1;
936
937     /* For the case of retry with leader, if it is for modification,
938      * since we always send modification RPC to the leader, then no
939      * need to refresh the pool map. Because if the client used old
940      * pool map and sent the modification RPC to non-leader replica,
941      * then the replied errno will be -DER_STALE (assume that there
942      * will be at least one replica will have the latest pool map).
943      *
944      * For read-only RPC (fetch/list/query), retry with leader case
945      * only can happen when the server to which we just sent the RPC
946      * is not the leader. To guarantee the next retry can find the
947      * right leader, we need to refresh the pool map before retry.
948      */
949 TF     if (!obj_auxi->retry_with_leader ||
950         !obj_is_modification_opc(obj_auxi->opc)) {
951         rc = obj_pool_query_task(sched, obj, &pool_task);
952 TF         if (rc != 0)
953             D_GOTO(err, rc);
954     }
955
956 TF     if (obj_auxi->io_retry) {
957         /* Let's reset task result before retry */
958         rc = dc_task_resched(task);
959 TF         if (rc != 0) {
960             D_ERROR("Failed to re-init task (%p)\n", task);
961             D_GOTO(err, rc);
962         }
963
964 TF         if (pool_task != NULL) {
965             rc = dc_task_depend(task, 1, &pool_task);
966 TF             if (rc != 0) {
967                 D_ERROR("Failed to add dependency on pool "
968                      "query task (%p)\n", pool_task);
969                 D_GOTO(err, rc);
970             }
971         }
972     }
973
974     D_DEBUG(DB_IO, "Retrying task=%p for err=%d, io_retry=%d\n",
975          task, result, obj_auxi->io_retry);
976
977 TF     if (pool_task != NULL)
978         /* ignore returned value, error is reported by comp_cb */
979         dc_task_schedule(pool_task, obj_auxi->io_retry);
980
981     return 0;
982 err:
983 TF     if (pool_task)
984         dc_task_decref(pool_task);
985
986     task->dt_result = result; /* restore the orignal error */
987     D_ERROR("Failed to retry task=%p(err=%d), io_retry=%d, rc %d.\n",
988         task, result, obj_auxi->io_retry, rc);
989     return rc;
990 }
