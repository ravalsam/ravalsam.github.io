403 int
404 obj_ec_codec_init()
405 {
406     struct obj_ec_codec *ec_codec;
407     struct daos_obj_class   *oc;
408     unsigned char       *encode_matrix = NULL;
409     int          ocnr;
410     int          i;
411     int          k, p, m;
412     int          rc;
413
414 TF     if (oc_ec_codecs != NULL)
415         return 0;
416
417     ocnr = 0;
418 TF     for (oc = &daos_obj_classes[0]; oc->oc_id != DAOS_OC_UNKNOWN; oc++) {
419 TF         if (oc->oc_attr.ca_resil == DAOS_RES_EC)
420             ocnr++;
421     }
422 TF     if (ocnr == 0)
423         return 0;
424
425     D_ALLOC_ARRAY(oc_ec_codecs, ocnr);
426 TF     if (oc_ec_codecs == NULL)
427         D_GOTO(failed, rc = -DER_NOMEM);
428     oc_ec_codec_nr = ocnr;
429
430     i = 0;
431 TF     for (oc = &daos_obj_classes[0]; oc->oc_id != DAOS_OC_UNKNOWN; oc++) {
432 TF         if (oc->oc_attr.ca_resil != DAOS_RES_EC)
433             continue;
434
435         oc_ec_codecs[i].ec_oc_id = oc->oc_id;
436         ec_codec = &oc_ec_codecs[i++].ec_codec;
437         k = oc->oc_attr.u.ec.e_k;
438         p = oc->oc_attr.u.ec.e_p;
439         m = k + p;
440         /* 32B needed for data generated for each input coefficient */
441         D_ALLOC(ec_codec->ec_gftbls, k * p * 32);
442 TF         if (ec_codec->ec_gftbls == NULL)
443             D_GOTO(failed, rc = -DER_NOMEM);
444         D_ALLOC(encode_matrix, m * k);
445 TF         if (encode_matrix == NULL)
446             D_GOTO(failed, rc = -DER_NOMEM);
447         ec_codec->ec_en_matrix = encode_matrix;
448         /* A Cauchy matrix is always invertible, the recovery rule is
449          * simpler than gf_gen_rs_matrix (vandermonde matrix).
450          */
451         gf_gen_cauchy1_matrix(encode_matrix, m, k);
452         /* Initialize gf tables from encode matrix */
453         ec_init_tables(k, p, &encode_matrix[k * k],
454                    ec_codec->ec_gftbls);
455     }
456
457     D_ASSERT(i == ocnr);
458     return 0;
459
460 failed:
461     obj_ec_codec_fini();
462     return rc;
463 }
