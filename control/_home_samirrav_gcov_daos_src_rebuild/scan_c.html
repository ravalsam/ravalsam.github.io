436     rc = dbtree_lookup(cont_root->root_hdl, &key_iov, &val_iov);
437     D_DEBUG(DB_REBUILD, "lookup "DF_UOID" in cont "DF_UUID" eph "
438         DF_U64" rc %d\n", DP_UOID(oid), DP_UUID(co_uuid), eph, rc);
439 TF     if (rc == -DER_NONEXIST) {
440         rc = dbtree_update(cont_root->root_hdl, &key_iov, &val_iov);
441 TF         if (rc < 0) {
442             D_ERROR("failed to insert "DF_UOID": rc %d\n",
443                 DP_UOID(oid), rc);
444             D_GOTO(out, rc);
445         }
446         cont_root->count++;
447         D_DEBUG(DB_REBUILD, "update "DF_UOID"/"DF_UUID" eph "DF_U64" in"
448             " cont_root %p count %d\n", DP_UOID(oid),
449             DP_UUID(co_uuid), eph, cont_root, cont_root->count);
450         return 1;
451     }
452
453 out:
454     return rc;
455 }

493 /**

496  **/
497 static int
498 rebuild_object_insert(struct rebuild_scan_arg *arg, unsigned int tgt_id,
499               unsigned int shard, uuid_t pool_uuid, uuid_t co_uuid,
500               daos_unit_oid_t oid, daos_epoch_t epoch)
501 {

503     daos_iov_t      val_iov;

505     daos_handle_t       toh = arg->rebuild_tree_hdl;
506     int         rc;
507
508     /* look up the target tree */
509     daos_iov_set(&key_iov, &tgt_id, sizeof(tgt_id));
510     daos_iov_set(&val_iov, NULL, 0);
511     ABT_mutex_lock(arg->scan_lock);

513 TF     if (rc < 0) {

517             ABT_mutex_unlock(arg->scan_lock);
518             D_GOTO(out, rc);
519         }
520     } else {
521         tgt_root = val_iov.iov_buf;
522     }

525                      shard, tgt_id, NULL, 0,
526                      rebuild_obj_insert_cb);
527 TF     if (rc <= 0) {
528         ABT_mutex_unlock(arg->scan_lock);
529         D_GOTO(out, rc);
530     }
531
532 TF     if (rc == 1) {

534 TF         if (++tgt_root->count >= REBUILD_SEND_LIMIT) {

536 TF             if (rc < 0)
537                 D_GOTO(out, rc);
538         }
539         rc = 0;
540     }
541     ABT_mutex_unlock(arg->scan_lock);
542
543     D_DEBUG(DB_REBUILD, "insert "DF_UOID"/"DF_UUID" tgt %u cnt %d rc %d\n",
544         DP_UOID(oid), DP_UUID(co_uuid), tgt_id, tgt_root->count, rc);
545 out:
546     return rc;
547 }

549 #define LOCAL_ARRAY_SIZE    128
550 static int
551 placement_check(uuid_t co_uuid, vos_iter_entry_t *ent, void *data)
552 {
553     struct rebuild_scan_arg *arg = data;
554     struct rebuild_tgt_pool_tracker *rpt = arg->rpt;
555     struct pl_map       *map = NULL;
556     struct daos_obj_md  md;
557     daos_unit_oid_t     oid = ent->ie_oid;
558     unsigned int        tgt_array[LOCAL_ARRAY_SIZE];
559     unsigned int        shard_array[LOCAL_ARRAY_SIZE];
560     unsigned int        *tgts = NULL;
561     unsigned int        *shards = NULL;
562     int         rebuild_nr;
563     d_rank_t        myrank;
564     int         i;
565     int         rc;
566
567 TF     if (rpt->rt_abort)
568         return 1;
569
570     map = pl_map_find(rpt->rt_pool_uuid, oid.id_pub);
571 TF     if (map == NULL) {
572         D_ERROR(DF_UOID"Cannot find valid placement map"
573             DF_UUID"\n", DP_UOID(oid),
574             DP_UUID(rpt->rt_pool_uuid));
575         D_GOTO(out, rc = -DER_INVAL);
576     }
577
578     dc_obj_fetch_md(oid.id_pub, &md);
579     crt_group_rank(rpt->rt_pool->sp_group, &myrank);
580     md.omd_ver = rpt->rt_rebuild_ver;
581 TF     if (arg->rebuild_tgt_nr > LOCAL_ARRAY_SIZE) {
582         D_ALLOC_ARRAY(tgts, arg->rebuild_tgt_nr);
583         D_ALLOC_ARRAY(shards, arg->rebuild_tgt_nr);
584 TF         if (tgts == NULL || shards == NULL)
585             D_GOTO(out, rc = -DER_NOMEM);
586     } else {
587         tgts = tgt_array;
588         shards = shard_array;
589     }
590
591     rebuild_nr = pl_obj_find_rebuild(map, &md, NULL, rpt->rt_rebuild_ver,
592                      tgts, shards, arg->rebuild_tgt_nr,
593                      myrank);
594 TF     if (rebuild_nr <= 0) /* No need rebuild */
595         D_GOTO(out, rc = rebuild_nr);
596
597     D_ASSERT(rebuild_nr <= arg->rebuild_tgt_nr);
598 TF     for (i = 0; i < rebuild_nr; i++) {
599         D_DEBUG(DB_REBUILD, "rebuild obj "DF_UOID"/"DF_UUID"/"DF_UUID
600             " on %d for shard %d\n", DP_UOID(oid), DP_UUID(co_uuid),
601             DP_UUID(rpt->rt_pool_uuid), tgts[i], shards[i]);
602
603         /* During rebuild test, it will manually exclude some target to
604          * trigger the rebuild, then later add it back, so some objects
605          * might exist on some illegal target, so they might use its
606          * "own" target as the spare target, let's skip these object
607          * now. When we have better support from CART exclude/addback,
608          * myrank should always not equal to tgt_rebuild. XXX
609          */
610 TF         if (myrank != tgts[i]) {
611             rc = rebuild_object_insert(arg, tgts[i], shards[i],
612                            rpt->rt_pool_uuid, co_uuid,
613                            oid, ent->ie_epoch);
614 TF             if (rc)
615                 D_GOTO(out, rc);
616         } else {
617             D_DEBUG(DB_REBUILD, "skip "DF_UOID".\n", DP_UOID(oid));
618             rc = 0;
619         }
620     }
621 out:
622 TF     if (tgts != tgt_array && tgts != NULL)
623         D_FREE(tgts);
624
625 TF     if (shards != shard_array && shards != NULL)
626         D_FREE(shards);
627
628 TF     if (map != NULL)
629         pl_map_decref(map);
630
631     return rc;
632 }

670 /**
671  * Wait for pool map and setup global status, then spawn scanners for all
672  * service xsteams
673  */
674 static void
675 rebuild_scan_leader(void *data)
676 {
677     struct rebuild_scan_arg   *arg = data;
678     struct pool_map       *map;
679     struct rebuild_tgt_pool_tracker *rpt;
680     struct rebuild_pool_tls   *tls;
681     struct rebuild_iter_arg    iter_arg;
682     int            rc;
683
684     D_ASSERT(arg != NULL);
685     D_ASSERT(!daos_handle_is_inval(arg->rebuild_tree_hdl));
686
687     rpt = arg->rpt;
688     /* refresh placement for the server stack */
689     ABT_mutex_lock(rpt->rt_lock);
690     map = rebuild_pool_map_get(rpt->rt_pool);
691     D_ASSERT(map != NULL);
692     rc = pl_map_update(rpt->rt_pool_uuid, map, true);
693 TF     if (rc != 0) {
694         ABT_mutex_unlock(rpt->rt_lock);
695         D_GOTO(out_map, rc = -DER_NOMEM);
696     }
697     ABT_mutex_unlock(rpt->rt_lock);
698
699     iter_arg.arg = arg;
700     iter_arg.callback = placement_check;
701
702     rc = dss_thread_collective(rebuild_scanner, &iter_arg, 0);
703 TF     if (rc)
704         D_GOTO(put_plmap, rc);
705
706     D_DEBUG(DB_REBUILD, "rebuild scan collective "DF_UUID" done.\n",
707         DP_UUID(rpt->rt_pool_uuid));
708
709     /* NB: only leading xstream will operate the scan tree since then,
710      * so no need lock rebuild tgt tree.
711      */
712 TF     while (!dbtree_is_empty(arg->rebuild_tree_hdl)) {
713         /* walk through the rebuild tree and send the rebuild objects */
714         rc = dbtree_iterate(arg->rebuild_tree_hdl, DAOS_INTENT_REBUILD,
715                     false, rebuild_tgt_fini_obj_send_cb, arg);
716 TF         if (rc)
717             D_GOTO(put_plmap, rc);
718     }
719
720     ABT_mutex_lock(rpt->rt_lock);
721     rc = dss_task_collective(rebuild_scan_done, rpt, 0);
722     ABT_mutex_unlock(rpt->rt_lock);
723 TF     if (rc) {
724         D_ERROR(DF_UUID" send rebuild object list failed:%d\n",
725             DP_UUID(rpt->rt_pool_uuid), rc);
726         D_GOTO(put_plmap, rc);
727     }
728
729     D_DEBUG(DB_REBUILD, DF_UUID" sent objects to initiator %d\n",
730         DP_UUID(rpt->rt_pool_uuid), rc);
731
732 put_plmap:
733     pl_map_disconnect(rpt->rt_pool_uuid);
734 out_map:
735     rebuild_pool_map_put(map);
736     dbtree_destroy(arg->rebuild_tree_hdl);
737     tls = rebuild_pool_tls_lookup(rpt->rt_pool_uuid, rpt->rt_rebuild_ver);
738     D_ASSERT(tls != NULL);
739 TF     if (tls->rebuild_pool_status == 0 && rc != 0)
740         tls->rebuild_pool_status = rc;
741     D_DEBUG(DB_REBUILD, DF_UUID"scan leader done %d\n",
742         DP_UUID(rpt->rt_pool_uuid), rc);
743     ABT_mutex_free(&arg->scan_lock);
744     D_FREE(arg);
745     rpt_put(rpt);
746 }

748 /* Scan the local target and generate rebuild object list */
749 void
750 rebuild_tgt_scan_handler(crt_rpc_t *rpc)
751 {
752     struct rebuild_scan_in      *rsi;
753     struct rebuild_scan_out     *ro;
754     struct rebuild_scan_arg     *scan_arg;
755     struct umem_attr         uma;
756     struct rebuild_tgt_pool_tracker *rpt = NULL;
757     int              rc;
758
759     rsi = crt_req_get(rpc);
760     D_ASSERT(rsi != NULL);
761
762     D_DEBUG(DB_REBUILD, "%d scan rebuild for "DF_UUID" ver %d/%d\n",
763         dss_get_module_info()->dmi_tgt_id, DP_UUID(rsi->rsi_pool_uuid),
764         rsi->rsi_pool_map_ver, rsi->rsi_rebuild_ver);
765
766     /* check if the rebuild is already started */
767     rpt = rpt_lookup(rsi->rsi_pool_uuid, rsi->rsi_rebuild_ver);
768 TF     if (rpt != NULL) {
769         /* Rebuild should never skip the version */
770         D_ASSERTF(rsi->rsi_rebuild_ver == rpt->rt_rebuild_ver,
771               "rsi_rebuild_ver %d != rt_rebuild_ver %d\n",
772               rsi->rsi_rebuild_ver, rpt->rt_rebuild_ver);
773
774         D_DEBUG(DB_REBUILD, DF_UUID" already started.\n",
775             DP_UUID(rsi->rsi_pool_uuid));
776
777         /* Ignore the rebuild trigger request if it comes from
778          * an old or same leader.
779          */
780 TF         if (rsi->rsi_leader_term <= rpt->rt_leader_term)
781             D_GOTO(out, rc = 0);
782
783 TF         if (rpt->rt_pool->sp_iv_ns != NULL &&
784             rpt->rt_pool->sp_iv_ns->iv_master_rank !=
785                     rsi->rsi_master_rank) {
786             D_DEBUG(DB_REBUILD, DF_UUID" master rank"
787                 " %d -> %d term "DF_U64" -> "DF_U64"\n",
788                 DP_UUID(rpt->rt_pool_uuid),
789                 rpt->rt_pool->sp_iv_ns->iv_master_rank,
790                 rsi->rsi_master_rank,
791                 rpt->rt_leader_term,
792                 rsi->rsi_leader_term);
793             /* re-report the #rebuilt cnt next time */
794             rpt->rt_re_report = 1;
795             /* Update master rank */
796             rc = ds_pool_iv_ns_update(rpt->rt_pool,
797                           rsi->rsi_master_rank,
798                           &rsi->rsi_ns_iov,
799                           rsi->rsi_ns_id);
800 TF             if (rc)
801                 D_GOTO(out, rc);
802
803             /* If this is the old leader, then also stop the rebuild
804              * tracking ULT.
805              */
806             ds_rebuild_leader_stop(rsi->rsi_pool_uuid,
807                            rsi->rsi_rebuild_ver);
808         }
809
810         rpt->rt_leader_term = rsi->rsi_leader_term;
811
812         D_GOTO(out, rc = 0);
813     }
814
815 TF     if (daos_fail_check(DAOS_REBUILD_TGT_START_FAIL))
816         D_GOTO(out, rc = -DER_INVAL);
817
818     rc = rebuild_tgt_prepare(rpc, &rpt);
819 TF     if (rc)
820         D_GOTO(out, rc);
821
822     rpt_get(rpt);
823     rc = dss_ult_create(rebuild_tgt_status_check, rpt, DSS_ULT_REBUILD,
824                 DSS_TGT_SELF, 0, NULL);
825 TF     if (rc) {
826         rpt_put(rpt);
827         D_GOTO(out, rc);
828     }
829
830     /* step-1: parameters for scanner */
831     D_ALLOC_PTR(scan_arg);
832 TF     if (scan_arg == NULL)
833         D_GOTO(out, rc = -DER_NOMEM);
834
835     rc = ABT_mutex_create(&scan_arg->scan_lock);
836 TF     if (rc != ABT_SUCCESS) {
837         rc = dss_abterr2der(rc);
838         D_GOTO(out_arg, rc);
839     }
840
841     /* step-2: Create the btree root for global object scan list */
842     memset(&uma, 0, sizeof(uma));
843     uma.uma_id = UMEM_CLASS_VMEM;
844     rc = dbtree_create(DBTREE_CLASS_NV, 0, 4, &uma, NULL,
845                &scan_arg->rebuild_tree_hdl);
846 TF     if (rc != 0) {
847         D_ERROR("failed to create rebuild tree: %d\n", rc);
848         D_GOTO(out_lock, rc);
849     }
850
851     scan_arg->rebuild_tgt_nr = rsi->rsi_tgts_num;
852     rpt_get(rpt);
853     scan_arg->rpt = rpt;
854     /* step-3: start scann leader */
855     rc = dss_ult_create(rebuild_scan_leader, scan_arg, DSS_ULT_REBUILD,
856                 DSS_TGT_SELF, 0, NULL);
857 TF     if (rc != 0) {
858         rpt_put(rpt);
859         D_GOTO(out_tree, rc);
860     }
861
862     D_GOTO(out, rc);
863 out_tree:
864     dbtree_destroy(scan_arg->rebuild_tree_hdl);
865 out_lock:
866     ABT_mutex_free(&scan_arg->scan_lock);
867 out_arg:
868     D_FREE(scan_arg);
869 out:
870 TF     if (rpt)
871         rpt_put(rpt);
872     ro = crt_reply_get(rpc);
873     ro->rso_status = rc;
874 TF     if (rc) {
875         d_rank_list_t *fail_list;
876
877         /* If it failed, tell the master the target can not
878          * start the rebuild, so master will put the target
