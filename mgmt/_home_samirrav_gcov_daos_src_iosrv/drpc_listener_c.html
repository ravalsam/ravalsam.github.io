77 static void
78 drpc_listener_run(void *arg)
79 {
80     struct drpc_progress_context *ctx;
81
82     D_ASSERT(arg != NULL);
83     ctx = (struct drpc_progress_context *)arg;
84
85     D_INFO("Starting dRPC listener\n");
86     set_listener_running(true);

91         rc = drpc_progress(ctx, 0);
92 TF         if (rc != DER_SUCCESS && rc != -DER_TIMEDOUT) {
93             D_ERROR("dRPC listener progress error: %d\n", rc);

97     }

99     D_INFO("Closing down dRPC listener\n");
100     drpc_progress_context_close(ctx);
