440                 vos_iter_param_t *param);

442 static int
443 ts_iterate_internal(uint32_t type, vos_iter_param_t *param,
444             iterate_cb_t iter_cb)
445 {
446     daos_anchor_t       *probe_hash = NULL;
447     vos_iter_entry_t    key_ent;
448     daos_handle_t       ih;
449     int         rc;
450
451     rc = vos_iter_prepare(type, param, &ih);
452 TF     if (rc != 0) {
453 TF         if (rc == -DER_NONEXIST)
454             rc = 0;
455         else
456             D_ERROR("Failed to prepare d-key iterator: %d\n", rc);
457         D_GOTO(out, rc);
458     }
459
460     rc = vos_iter_probe(ih, probe_hash);
461 TF     if (rc != 0) {
462 TF         if (rc == -DER_NONEXIST || rc == -DER_AGAIN)
463             rc = 0;
464         D_GOTO(out_iter_fini, rc);
465     }
466
467     while (1) {
468         rc = vos_iter_fetch(ih, &key_ent, NULL);
469 TF         if (rc != 0)
470             break;
471
472         /* fill the key to iov if there are enough space */
473 TF         if (iter_cb) {
474             rc = iter_cb(ih, &key_ent, param);
475 TF             if (rc != 0)
476                 break;
477         }
478
479         rc = vos_iter_next(ih);
480 TF         if (rc)
481             break;
482     }
483
484 TF     if (rc == -DER_NONEXIST)
485         rc = 0;
486
487 out_iter_fini:
488     vos_iter_finish(ih);
489 out:
490     return rc;
491 }

876 void show_result(double now, double then, int vsize, char *test_name)
877 {
878     double      duration, agg_duration;
879     double      first_start;
880     double      last_end;
881     double      duration_max;
882     double      duration_min;
883     double      duration_sum;
884
885     duration = now - then;
886
887 TF     if (ts_ctx.tsc_mpi_size > 1) {
888         MPI_Reduce(&then, &first_start, 1, MPI_DOUBLE,
889                MPI_MIN, 0, MPI_COMM_WORLD);
890         MPI_Reduce(&now, &last_end, 1, MPI_DOUBLE,
891                MPI_MAX, 0, MPI_COMM_WORLD);
892     } else {
893         first_start = then;
894         last_end = now;
895     }
896
897     agg_duration = last_end - first_start;
898
899 TF     if (ts_ctx.tsc_mpi_size > 1) {
900         MPI_Reduce(&duration, &duration_max, 1, MPI_DOUBLE,
901                MPI_MAX, 0, MPI_COMM_WORLD);
902         MPI_Reduce(&duration, &duration_min, 1, MPI_DOUBLE,
903                MPI_MIN, 0, MPI_COMM_WORLD);
904         MPI_Reduce(&duration, &duration_sum, 1, MPI_DOUBLE,
905                MPI_SUM, 0, MPI_COMM_WORLD);
906     } else {
907         duration_max = duration_min = duration_sum = duration;
908     }
909
910 TF     if (ts_ctx.tsc_mpi_rank == 0) {
911         unsigned long   total;
912         double      bandwidth;
913         double      latency;
914         double      rate;
915
916         total = ts_ctx.tsc_mpi_size *
917             ts_obj_p_cont * ts_dkey_p_obj *
918             ts_akey_p_dkey * ts_recx_p_akey;
919
920         rate = total / agg_duration;
921         latency = (agg_duration * 1000 * 1000) / total;
922         bandwidth = (rate * vsize) / (1024 * 1024);
923
924         fprintf(stdout, "%s successfully completed:\n"
925             "\tduration : %-10.6f sec\n"
926             "\tbandwith : %-10.3f MB/sec\n"
927             "\trate     : %-10.2f IO/sec\n"
928             "\tlatency  : %-10.3f us "
929             "(nonsense if credits > 1)\n",
930             test_name, agg_duration, bandwidth, rate, latency);
931
932         fprintf(stdout, "Duration across processes:\n");
933         fprintf(stdout, "\tMAX duration : %-10.6f sec\n",
934             duration_max);
935         fprintf(stdout, "\tMIN duration : %-10.6f sec\n",
936             duration_min);
937         fprintf(stdout, "\tAverage duration : %-10.6f sec\n",
938             duration_sum / ts_ctx.tsc_mpi_size);
939     }
