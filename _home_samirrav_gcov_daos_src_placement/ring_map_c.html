437 TF                 if (&scratch[buf->rb_domain_nr] - merged > 0) {
438                     *dst = *merged;
439                     dst++;
440                     merged++;
441                 }
442             } else {
443 TF                 if (nr > 0) {
444                     *dst = buf->rb_domains[start + j];
445                     dst++;
446                     nr--;
447                     j++;
448                 }
449             }
450         }
451         merged = dst2;
452
453 TF         if (ver != comp->co_ver) {
454             ver = comp->co_ver;
455             start = i;
456         }
457     }
458
459     D_DEBUG(DB_PL, "Copy scratch buffer\n");
460     memcpy(buf->rb_domains, scratch, buf->rb_domain_nr * sizeof(*scratch));
461     D_FREE(scratch);
462     return 0;

466  * build a ring with pseudo-randomly ordered domains and targets
467  */

470         struct ring_buf *buf)

475     int         i;

580 /* each target has at least 10 bits for the key range of consistent hash */
581 #define TARGET_BITS     10
582 /* one million for domains */
583 #define DOMAIN_BITS     20
584 /* maximum bits for a ring */
585 #define TARGET_HASH_BITS    45
586 /* max to 8 million rings */
587 #define RING_HASH_BITS      23

590 #define RING_PRECISION      0.00001

592 /**
593  * create consistent hashes for rimap
594  */
595 static int
596 ring_map_hash_build(struct pl_ring_map *rimap)
597 {
598     uint64_t    range;
599     double      stride;
600     double      hash;
601     int     i;
602     unsigned    tg_per_dom;
603
604     D_DEBUG(DB_PL, "Build consistent hash for ring map\n");
605     D_ALLOC_ARRAY(rimap->rmp_target_hashes, rimap->rmp_target_nr);
606 TF     if (rimap->rmp_target_hashes == NULL)
607         return -DER_NOMEM;
608
609     D_ALLOC_ARRAY(rimap->rmp_ring_hashes, rimap->rmp_ring_nr);
610 TF     if (rimap->rmp_ring_hashes == NULL)
611         return -DER_NOMEM;
612
613     tg_per_dom = rimap->rmp_target_nr / rimap->rmp_domain_nr;
614     rimap->rmp_target_hbits = DOMAIN_BITS + TARGET_BITS +
615                   daos_power2_nbits(tg_per_dom);
616 TF     if (rimap->rmp_target_hbits > TARGET_HASH_BITS)
617         rimap->rmp_target_hbits = TARGET_HASH_BITS;
618
619     range = 1ULL << rimap->rmp_target_hbits;
620
621     D_DEBUG(DB_PL, "domanis %d, targets %d, hash range is 0-0x"DF_X64"\n",
622         rimap->rmp_domain_nr, rimap->rmp_target_nr, range);
623
624     /* create consistent hash for targets */
625     stride = (double)range / rimap->rmp_target_nr;
626     rimap->rmp_stride = stride;
627
628     hash = 0;
629 TF     for (i = 0; i < rimap->rmp_target_nr; i++) {
630         rimap->rmp_target_hashes[i] = hash;
631         hash += stride;
632     }
633
634     /* create consistent hash for rings */
635     range = 1ULL << RING_HASH_BITS;
636     stride = (double)range / rimap->rmp_ring_nr;
637     hash = 0;
638
639 TF     for (i = 0; i < rimap->rmp_ring_nr; i++) {
640         rimap->rmp_ring_hashes[i] = hash;
641         hash += stride;
642     }
643     return 0;
644 }

742 static unsigned int

744 {

747     /* mix bits */

755
756     return d_hash_srch_u64(rimap->rmp_target_hashes,
757                 rimap->rmp_target_nr, hash);

972 /**
973  * Given object placement @rop, calculate the next spare target start
974  * from @spare_idx. Return false if no available spare, otherwise, return
975  * true and next spare index in @spare_idx.
976  */
977 static bool
978 ring_remap_next_spare(struct pl_ring_map *rimap,
979               struct ring_obj_placement *rop, unsigned int *spare_idx)
980 {
981     unsigned int dist, max_dist, total_dist;
982
983     D_ASSERTF(rop->rop_grp_size <= rimap->rmp_domain_nr,
984           "grp_size: %u > domain_nr: %u\n",
985           rop->rop_grp_size, rimap->rmp_domain_nr);
986
987     /*
988      * Please be aware that we want to relocate the shard of
989      * non-replicated object (grp_size == 1) as well.
990      */
991 TF     if (rop->rop_grp_size == rimap->rmp_domain_nr &&
992         rop->rop_grp_size > 1)
993         return false;
994
995     /* Assume the ring consists of all pool targets. */
996     total_dist = rimap->rmp_target_nr;
997
998     /* The max distance from spare index to rop->rop_begin */
999     max_dist = total_dist - rop->rop_grp_size * rop->rop_grp_nr;
1000     /* Current distance from spare index to rop->rop_begin */
1001 TF     dist = (*spare_idx <= rop->rop_begin) ?
1002             rop->rop_begin - *spare_idx :
1003             rop->rop_begin + total_dist - *spare_idx;
1004     /*
1005      * Move the spare index forward, skip the domains where
1006      * the original shards located on. Revise this when the
1007      * rop_dist can be other values rather than 1.
1008      */
1009 TF     if (!((dist + rop->rop_grp_size) % rimap->rmp_domain_nr))
1010         dist += rop->rop_grp_size;
1011
1012     dist++;
1013 TF     if (dist > max_dist)
1014         return false;
1015
1016     /* Convert distance to spare index */
1017 TF     *spare_idx = (rop->rop_begin >= dist) ?
1018             rop->rop_begin - dist :
1019             total_dist - (dist - rop->rop_begin);
1020     return true;
1021 }

1154              * and try next spare on the ring.
1155              */

1157                 continue; /* try next spare */
1158
1159             /*

1161              * add the spare target to the fail list for remap, and

1168                       f_shard->rfs_fseq,
1169                       spare_tgt->ta_comp.co_fseq);

1172             f_shard->rfs_status = spare_tgt->ta_comp.co_status;
1173
1174             current = current->next;
1175             d_list_del_init(&f_shard->rfs_list);
1176             ring_remap_add_one(remap_list, f_shard);
1177
1178             /* Continue with the failed shard has minimal fseq */
1179 TF             if (current == remap_list) {
1180                 current = &f_shard->rfs_list;
1181             } else {
1182                 f_tmp = d_list_entry(current,
1183                              struct ring_failed_shard,
1184                              rfs_list);
1185 TF                 if (f_shard->rfs_fseq < f_tmp->rfs_fseq)
1186                     current = &f_shard->rfs_list;
1187             }
1188             continue; /* try next spare */
1189         }
1190 next_fail:
1191 TF         if (spare_avail) {
1192             /* The selected spare target is up and ready */
1193             l_shard->po_target = spare_tgt->ta_comp.co_id;
1194             l_shard->po_fseq = spare_tgt->ta_comp.co_fseq;
1195
1196             /*
1197              * Mark the shard as 'rebuilding' so that read will
1198              * skip this shard.
1199              */
1200 TF             if (f_shard->rfs_status == PO_COMP_ST_DOWN) {
1201                 l_shard->po_rebuilding = 1;
1202                 f_shard->rfs_tgt_id = spare_tgt->ta_comp.co_id;
1203             }
1204         } else {
1205             l_shard->po_shard = -1;
1206             l_shard->po_target = -1;
1207         }
1208         current = current->next;
1209     }
1210
1211     ring_remap_dump(remap_list, md, "after remap:");
1212 }
