535 /**
536  * Select leader replica for the given object's shard.
537  *
538  * \param [IN]  oid     The object identifier.
539  * \param [IN]  shard_idx   The shard index.
540  * \param [IN]  shards_ns   Total count of the object's shards.
541  * \param [IN]  for_tgt_id  Require leader target id or leader shard index.
542  * \param [IN]  pl_get_shard    The callback function to parse out pl_obj_shard
543  *              from the given @data.
544  * \param [IN]  data        The parameter used by the @pl_get_shard.
545  *
546  * \return          The selected leader on success: its tgt_id or
547  *              shard index. Negative value if error.
548  */
549 int
550 pl_select_leader(daos_obj_id_t oid, uint32_t shard_idx, int shards_nr,
551          bool for_tgt_id, pl_get_shard_t pl_get_shard, void *data)
552 {
553     struct pl_obj_shard     *shard;
554     struct daos_oclass_attr     *oc_attr;
555     uint32_t             replicas;
556     int              preferred;
557     int              rdg_idx;
558     int              start;
559     int              pos;
560     int              off;
561     int              i;
562
563 TF     if (shards_nr <= shard_idx)
564         return -DER_INVAL;
565
566     oc_attr = daos_oclass_attr_find(oid);
567 TF     if (oc_attr->ca_resil != DAOS_RES_REPL) {
568         /* For non-replicated object, elect current shard as leader. */
569         shard = pl_get_shard(data, shard_idx);
570 TF         if (for_tgt_id)
571             return shard->po_target;
572
573         return shard->po_shard;
574     }
575
576     replicas = oc_attr->u.repl.r_num;
577 TF     if (replicas == DAOS_OBJ_REPL_MAX)
578         replicas = shards_nr;
579
580 TF     if (replicas < 1)
581         return -DER_INVAL;
582
583 TF     if (replicas == 1) {
584         shard = pl_get_shard(data, shard_idx);
585 TF         if (shard->po_target == -1)
586             return -DER_NONEXIST;
587
588         /* Single replicated object will not rebuild. */
589         D_ASSERT(!shard->po_rebuilding);
590         D_ASSERT(shard->po_shard == shard_idx);
591
592 TF         if (for_tgt_id)
593             return shard->po_target;
594
595         return shard->po_shard;
596     }
597
598     /* XXX: The shards within [start, start + replicas) will search from
599      *  the same @preferred position, then they will have the same
600      *  leader. The shards (belonging to the same object) in
601      *  other redundancy group may get different leader node.
602      *
603      *  The one with the lowest f_seq will be elected as the leader
604      *  to avoid leader switch.
605      */
606     rdg_idx = shard_idx / replicas;
607     start = rdg_idx * replicas;
608     preferred = start + (oid.lo + rdg_idx) % replicas;
609 TF     for (i = 0, off = preferred, pos = -1; i < replicas;
610          i++, off = (off + 1) % replicas + start) {
611         shard = pl_get_shard(data, off);
612 TF         if (shard->po_target == -1 || shard->po_rebuilding)
613             continue;
614
615 TF         if (pos == -1 ||
616             pl_get_shard(data, pos)->po_fseq > shard->po_fseq)
617             pos = off;
618     }
619
620 TF     if (pos != -1) {
621         D_ASSERT(pl_get_shard(data, pos)->po_shard == pos);
622
623 TF         if (for_tgt_id)
624             return pl_get_shard(data, pos)->po_target;
625
626         return pl_get_shard(data, pos)->po_shard;
627     }
628
629     /* If all the replicas are failed or in-rebuilding, then NONEXIST. */
630     return -DER_NONEXIST;
631 }
