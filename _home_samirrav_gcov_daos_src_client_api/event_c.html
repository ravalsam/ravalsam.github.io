410         D_ERROR("Event status should be INIT: %d\n", evx->evx_status);
411         return -DER_NO_PERM;

415         D_ERROR("Launch all children before launch the parent.\n");

447  out:

450
451 TF     if (eqx != NULL)
452         daos_eq_putref(eqx);
453
454     return rc;

501 struct ev_progress_arg {
502     struct daos_eq_private      *eqx;
503     struct daos_event_private   *evx;
504 };
505
506 static int
507 ev_progress_cb(void *arg)
508 {
509     struct ev_progress_arg      *epa = (struct ev_progress_arg  *)arg;
510     struct daos_event_private       *evx = epa->evx;
511     struct daos_eq_private      *eqx = epa->eqx;
512
513     tse_sched_progress(evx->evx_sched);
514
515     /** If another thread progressed this, get out now. */
516 TF     if (evx->evx_status == DAOS_EVS_READY)
517         return 1;
518
519     /** Event is still in-flight */
520 TF     if (evx->evx_status != DAOS_EVS_COMPLETED &&
521         evx->evx_status != DAOS_EVS_ABORTED)
522         return 0;
523
524     /** If there are children in flight, then return in-flight */
525 TF     if (evx->evx_nchild_running > 0)
526         return 0;
527
528     /*
529      * Change status of event to INIT only if event is not in EQ and get
530      * out.
531      */
532 TF     if (daos_handle_is_inval(evx->evx_eqh)) {
533         evx->evx_status = DAOS_EVS_READY;
534         return 1;
535     }
536
537     /** Grab the lock so we don't race with eq_progress_cb. */
538     D_MUTEX_LOCK(&eqx->eqx_lock);
539
540     /*
541      * if the EQ was finalized from under us, just update the event status
542      * and return.
543      */
544 TF     if (eqx->eqx_finalizing) {
545         evx->evx_status = DAOS_EVS_READY;
546         D_ASSERT(d_list_empty(&evx->evx_link));
547         D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
548         return 1;
549     }
550
551     /*
552      * Check again if the event is still in completed/aborted state, then
553      * remove it from the event queue.
554      */
555 TF     if (evx->evx_status == DAOS_EVS_COMPLETED ||
556         evx->evx_status == DAOS_EVS_ABORTED) {
557         struct daos_eq *eq = daos_eqx2eq(eqx);
558
559         evx->evx_status = DAOS_EVS_READY;
560         D_ASSERT(eq->eq_n_comp > 0);
561         eq->eq_n_comp--;
562         d_list_del_init(&evx->evx_link);
563     }
564
565     D_ASSERT(evx->evx_status == DAOS_EVS_READY);
566     D_MUTEX_UNLOCK(&eqx->eqx_lock);
567
568     return 1;
569 }

571 int
572 daos_event_test(struct daos_event *ev, int64_t timeout, bool *flag)
573 {
574     struct ev_progress_arg      epa;
575     struct daos_event_private   *evx = daos_ev2evx(ev);
576     int             rc;
577
578     /** Can't call test on a Child event */
579 TF     if (evx->evx_parent != NULL)
580         return -DER_NO_PERM;
581
582     epa.evx = evx;
583     epa.eqx = NULL;
584
585 TF     if (!daos_handle_is_inval(evx->evx_eqh)) {
586         epa.eqx = daos_eq_lookup(evx->evx_eqh);
587 TF         if (epa.eqx == NULL) {
588             D_ERROR("Can't find eq from handle %"PRIu64"\n",
589                 evx->evx_eqh.cookie);
590             return -DER_NONEXIST;
591         }
592     }
593
594     /* pass the timeout to crt_progress() with a conditional callback */
595     rc = crt_progress(evx->evx_ctx, timeout, ev_progress_cb, &epa);
596
597     /** drop ref grabbed in daos_eq_lookup() */
598 TF     if (epa.eqx)
599         daos_eq_putref(epa.eqx);
600
601 TF     if (rc != 0 && rc != -DER_TIMEDOUT) {
602         D_ERROR("crt progress failed with %d\n", rc);
603         return rc;
604     }
605
606 TF     if (evx->evx_status == DAOS_EVS_READY)
607         *flag = true;
608     else
609         *flag = false;
610
611     return 0;
612 }

640 struct eq_progress_arg {
641     struct daos_eq_private   *eqx;
642     unsigned int          n_events;
643     struct daos_event   **events;
644     int           wait_running;
645     int           count;
646 };
647
648 static int
649 eq_progress_cb(void *arg)
650 {
651     struct eq_progress_arg      *epa = (struct eq_progress_arg  *)arg;
652     struct daos_event       *ev;
653     struct daos_eq          *eq;
654     struct daos_event_private   *evx;
655     struct daos_event_private   *tmp;
656
657     eq = daos_eqx2eq(epa->eqx);
658
659     tse_sched_progress(&epa->eqx->eqx_sched);
660
661     D_MUTEX_LOCK(&epa->eqx->eqx_lock);
662     d_list_for_each_entry_safe(evx, tmp, &eq->eq_comp, evx_link) {
663         D_ASSERT(eq->eq_n_comp > 0);
664
665         /** don't poll out a parent if it has inflight events */
666 TF         if (evx->evx_nchild_running > 0)
667             continue;
668
669         eq->eq_n_comp--;
670
671         d_list_del_init(&evx->evx_link);
672         D_ASSERT(evx->evx_status == DAOS_EVS_COMPLETED ||
673              evx->evx_status == DAOS_EVS_ABORTED);
674         evx->evx_status = DAOS_EVS_READY;
675
676 TF         if (epa->events != NULL) {
677             ev = daos_evx2ev(evx);
678             epa->events[epa->count++] = ev;
679         }
680
681         D_ASSERT(epa->count <= epa->n_events);
682 TF         if (epa->count == epa->n_events)
683             break;
684     }
685
686     /* exit once there are completion events */
687 TF     if (epa->count > 0) {
688         D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
689         return 1;
690     }
691
692     /* no completion event, eq::eq_comp is empty */
693 TF     if (epa->eqx->eqx_finalizing) { /* no new event is coming */
694         D_ASSERT(d_list_empty(&eq->eq_running));
695         D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
696         return -DER_NONEXIST;
697     }
698
699     /* wait only if there are running events? */
700 TF     if (epa->wait_running && d_list_empty(&eq->eq_running)) {
701         D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
702         return 1;
703     }
704
705     D_MUTEX_UNLOCK(&epa->eqx->eqx_lock);
706
707     /** continue waiting */
708     return 0;
709 }

711 int
712 daos_eq_poll(daos_handle_t eqh, int wait_running, int64_t timeout,
713          unsigned int n_events, struct daos_event **events)
714 {
715     struct eq_progress_arg  epa;
716     int         rc;
717
718 TF     if (n_events == 0)
719         return -DER_INVAL;
720
721     /** look up private eq */
722     epa.eqx = daos_eq_lookup(eqh);
723 TF     if (epa.eqx == NULL)
724         return -DER_NONEXIST;
725
726     epa.n_events    = n_events;
727     epa.events  = events;
728     epa.wait_running = wait_running;
729     epa.count   = 0;
730
731     /* pass the timeout to crt_progress() with a conditional callback */
732     rc = crt_progress(epa.eqx->eqx_ctx, timeout, eq_progress_cb, &epa);
733
734     /* drop ref grabbed in daos_eq_lookup() */
735     daos_eq_putref(epa.eqx);
736
737 TF     if (rc != 0 && rc != -DER_TIMEDOUT) {
738         D_ERROR("crt progress failed with %d\n", rc);
739         return rc;
740     }
741
742     return epa.count;
743 }

745 int
746 daos_eq_query(daos_handle_t eqh, daos_eq_query_t query,
747           unsigned int n_events, struct daos_event **events)
748 {
749     struct daos_eq_private      *eqx;
750     struct daos_eq          *eq;
751     struct daos_event_private   *evx;
752     struct daos_event       *ev;
753     int              count;
754
755     eqx = daos_eq_lookup(eqh);
756 TF     if (eqx == NULL)
757         return -DER_NONEXIST;
758
759     eq = daos_eqx2eq(eqx);
760
761     count = 0;
762     D_MUTEX_LOCK(&eqx->eqx_lock);
763
764 TF     if (n_events == 0 || events == NULL) {
765 TF         if ((query & DAOS_EQR_COMPLETED) != 0)
766             count += eq->eq_n_comp;
767
768 TF         if ((query & DAOS_EQR_WAITING) != 0)
769             count += eq->eq_n_running;
770         goto out;
771     }
772
773 TF     if ((query & DAOS_EQR_COMPLETED) != 0) {
774         d_list_for_each_entry(evx, &eq->eq_comp, evx_link) {
775             ev = daos_evx2ev(evx);
776             events[count++] = ev;
777 TF             if (count == n_events)
778                 goto out;
779         }
780     }
781
782 TF     if ((query & DAOS_EQR_WAITING) != 0) {
783         d_list_for_each_entry(evx, &eq->eq_running, evx_link) {
784             ev = daos_evx2ev(evx);
785             events[count++] = ev;
786 TF             if (count == n_events)
787                 goto out;
788         }
789     }
790 out:
791     D_MUTEX_UNLOCK(&eqx->eqx_lock);
792     daos_eq_putref(eqx);
793     return count;
794 }

796 static void
797 daos_event_abort_one(struct daos_event_private *evx)
798 {
799 TF     if (evx->evx_status != DAOS_EVS_RUNNING)
800         return;
801
802     /* NB: ev::ev_error will be set by daos_event_complete(),
803      * so user can decide to not set error if operation has already
804      * finished while trying to abort */
805     /* NB: always set ev_status to DAOS_EVS_ABORTED even w/o callback,
806      * so aborted parent event can be marked as COMPLETE right after
807      * completion all launched events other than completion of all
808      * children. See daos_parent_event_can_complete for details. */
809     evx->evx_status = DAOS_EVS_ABORTED;
810     daos_event_complete_cb(evx, -DER_CANCELED);
811 }
812
813 static void
814 daos_event_abort_locked(struct daos_eq_private *eqx,
815             struct daos_event_private *evx)
816 {
817     struct daos_event_private *child;
818
819     D_ASSERT(evx->evx_status == DAOS_EVS_RUNNING);
820
821     daos_event_abort_one(evx);
822     /* abort all children if he has */
823     d_list_for_each_entry(child, &evx->evx_child, evx_link)
824         daos_event_abort_one(child);
825
826     /* if aborted event is not a child event, move it to the
827      * head of launched list */
828 TF     if (evx->evx_parent == NULL && eqx != NULL) {
829         struct daos_eq *eq = daos_eqx2eq(eqx);
830
831         d_list_del(&evx->evx_link);
832         d_list_add(&evx->evx_link, &eq->eq_comp);
833         eq->eq_n_running--;
834         eq->eq_n_comp++;
835     }
836 }

838 int
839 daos_eq_destroy(daos_handle_t eqh, int flags)
840 {
841     struct daos_eq_private      *eqx;
842     struct daos_eq          *eq;
843     struct daos_event_private   *evx;
844     struct daos_event_private   *tmp;
845     int              rc = 0;
846
847     eqx = daos_eq_lookup(eqh);
848 TF     if (eqx == NULL) {
849         D_ERROR("eqh nonexist.\n");
850         return -DER_NONEXIST;
851     }
852
853     D_MUTEX_LOCK(&eqx->eqx_lock);
854 TF     if (eqx->eqx_finalizing) {
855         D_ERROR("eqx_finalizing.\n");
856         rc = -DER_NONEXIST;
857         goto out;
858     }
859
860     eq = daos_eqx2eq(eqx);
861
862     /* If it is not force destroyed, then we need check if
863      * there are still events linked here */
864 TF     if (((flags & DAOS_EQ_DESTROY_FORCE) == 0) &&
865         (!d_list_empty(&eq->eq_running) ||
866          !d_list_empty(&eq->eq_comp))) {
867         rc = -DER_BUSY;
868         goto out;
869     }
870
871     /* prevent other threads to launch new event */
872     eqx->eqx_finalizing = 1;
873
874     /* abort all launched events */
875     d_list_for_each_entry_safe(evx, tmp, &eq->eq_running, evx_link) {
876         D_ASSERT(evx->evx_parent == NULL);
877         daos_event_abort_locked(eqx, evx);
878     }
879
880     D_ASSERT(d_list_empty(&eq->eq_running));
881
882     d_list_for_each_entry_safe(evx, tmp, &eq->eq_comp, evx_link) {
883         d_list_del(&evx->evx_link);
884         D_ASSERT(eq->eq_n_comp > 0);
885         eq->eq_n_comp--;
886     }
887     eqx->eqx_ctx = NULL;
888
889     tse_sched_complete(&eqx->eqx_sched, rc, true);
890
891 out:
892     D_MUTEX_UNLOCK(&eqx->eqx_lock);
893 TF     if (rc == 0)
894         daos_eq_delete(eqx);
895     daos_eq_putref(eqx);
896     return rc;
897 }

902 /**

928 daos_event_destroy_children(struct daos_event *ev, bool force)
929 {
930     struct daos_event_private   *evp = daos_ev2evx(ev);
931     struct daos_event_private   *sub_evx;
932     struct daos_event_private   *tmp;
933     int              rc = 0;
934
935     /* Destroy all of sub events */
936     d_list_for_each_entry_safe(sub_evx, tmp, &evp->evx_child,
937                    evx_link) {
938         struct daos_event *sub_ev = daos_evx2ev(sub_evx);
939         daos_ev_status_t ev_status = sub_evx->evx_status;
940

942         rc = daos_event_destroy(sub_ev, force);
943 TF         if (rc != 0) {
944             d_list_add(&sub_evx->evx_link,
945                    &evp->evx_child);
946             break;
947         }
948 TF         if (ev_status == DAOS_EVS_COMPLETED)
949             evp->evx_nchild_comp--;
950 TF         else if (ev_status == DAOS_EVS_RUNNING)
951             evp->evx_nchild_running--;
952         evp->evx_nchild--;
953     }
954
955     return rc;

958 /**
959  * Add the event to the event queue, and if there is parent, add
960  * it to its child list as well.
961  */
