255 /**
256  * log a message using stdarg list without checking filtering
257  *
258  * \param[in] flags     facility+level+misc flags
259  *
260  * \return          flags to pass to d_vlog, 0 indicates no log
261  */
262 static inline int d_log_check(int flags)
263 {
264     int fac = flags & DLOG_FACMASK;
265     int lvl = flags & DLOG_PRIMASK;
266     int msk;
267
268 TF     if (!d_log_xst.tag) /* Log isn't open */
269         return 0;
270
271     /*
272      * Immediately skip if facility is disabled and the log level is less
273      * severe than DLOG_ERR. Otherwise, the mask will be checked again
274      * below. Essentially all error messages should be logged from all
275      * facilities, except if the user specifies a mask > DLOG_ERR.
276      */
277 TF     if (!d_log_xst.dlog_facs[fac].is_enabled && lvl < DLOG_ERR)
278         return 0;
279
280     /* Use default facility if it is malformed */
281 TF     if (fac >= d_log_xst.fac_cnt)
282         fac = 0;
283
284     /*
285      * first, see if we can ignore the log messages because it is
286      * masked out.  if debug messages are masked out, then we just
287      * directly compare levels.  if debug messages are not masked,
288      * then we allow all non-debug messages and for debug messages we
289      * check to make sure the proper bit is on.  [apps that don't use
290      * the debug bits just log with DLOG_DBG which has them all set]
291      */
292     msk = d_log_xst.dlog_facs[fac].fac_mask;
293 TF     if (lvl >= DLOG_INFO) {
294 TF         if (lvl < msk)
295             return 0; /* Skip it */
296     } else { /* debug clog message */
297         /*
298          * note: if (msk >= DLOG_INFO), then all the mask's debug bits
299          * are zero (meaning debugging messages are masked out).  thus,
300          * for messages with the debug level we only have to do a bit
301          * test.
302          */
303 TF         if ((lvl & msk) == 0)   /* do we want this type of debug msg? */
304             return 0; /* Skip it */
305     }
306
307     return lvl | fac;
308 }
