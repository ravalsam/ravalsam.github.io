644 /**
645  * Record
646  *
647  * A record is an atomic blob of arbitrary length which is always
648  * fetched/updated as a whole. The size of a record can change over time.
649  * A record is uniquely identified by the following composite key:
650  * - the distribution key (aka dkey) denotes a set of arrays co-located on the
651  *   same storage targets. The dkey has an arbitrary size.
652  * - the attribute key (aka akey) distinguishes individual arrays. Likewise, the
653  *   akey has a arbitrary size.
654  * - the index within an array discriminates individual records. The index
655  *   is an integer that ranges from zero to infinity. A range of indices
656  *   identifies a contiguous set of records called extent. All records inside an
657  *   extent must have the same size.
658  */
659
660 /**
661  * A record extent is a range of contiguous records of the same size inside an
662  * array. \a rx_idx is the first array index of the extent and \a rx_nr is the
663  * number of records covered by the extent.
664  */
665 typedef struct {
666     /** Indice of the first record in the extent */
667     uint64_t    rx_idx;
668     /**
669      * Number of contiguous records in the extent
670      * If \a rx_nr is equal to 1, the extent is composed of a single record
671      * at indice \a rx_idx
672      */
673     uint64_t    rx_nr;
674 } daos_recx_t;

686 /**

697      * Type of the value in an iod can be either a single type that is
698      * always overwritten when updated, or it can be an array of EQUAL sized
699      * records where the record is updated atomically. Note that an akey can
700      * only support one type of value which is set on the first update. If
701      * user attempts to mix types in the same akey, the behavior is
702      * undefined, even after the object, key, or value is punched. If
703      * \a iod_type == DAOS_IOD_SINGLE, then iod_nr has to be 1, and

714     unsigned int        iod_nr;
715     /*
