441             principal_name, &ace_to_remove);
442 TF     if (rc != 0) {
443         /* requested principal not in the list */
444         return rc;
445     }
446
447     new_len = (*acl)->dal_len - daos_ace_get_size(ace_to_remove);
448
449     D_ALLOC(new_acl, get_total_acl_size(new_len));
450 TF     if (new_acl == NULL) {
451         return -DER_NOMEM;
452     }
453
454     new_acl->dal_len = new_len;
455     new_acl->dal_ver = (*acl)->dal_ver;
456
457     pen = new_acl->dal_ace;
458     current = daos_acl_get_next_ace(*acl, NULL);
459 TF     while (current != NULL) {
460 TF         if (!ace_matches_principal(current, type, principal_name)) {
461             pen = write_ace(current, pen);
462         }
463
464         current = daos_acl_get_next_ace(*acl, current);
465     }
466
467     daos_acl_free(*acl);
468     *acl = new_acl;
469
470     return 0;
471 }

482 struct daos_ace *
483 daos_acl_get_next_ace(struct daos_acl *acl, struct daos_ace *current_ace)
484 {
485     size_t offset;
486
487 TF     if (acl == NULL) {
488         return NULL;
489     }
490
491     /* requested the first ACE */
492 TF     if (current_ace == NULL && acl->dal_len > 0) {
493         return (struct daos_ace *)acl->dal_ace;
494     }
495
496     /* already at/beyond the end */
497 TF     if (!is_in_ace_list((uint8_t *)current_ace, acl)) {
498         return NULL;
499     }
500
501     /* there is no next item */
502     offset = sizeof(struct daos_ace) + current_ace->dae_principal_len;
503 TF     if (!is_in_ace_list((uint8_t *)current_ace + offset, acl)) {
504         return NULL;
505     }
506
507     return (struct daos_ace *)((uint8_t *)current_ace + offset);
508 }

626 /*
627  * Checks if the given daos_ace is a duplicate of one already we've already
628  * seen.
629  */
630 static int
631 check_ace_is_duplicate(struct daos_ace *ace, struct d_hash_table *found_aces)
632 {
633     struct ace_hash_entry   *entry;
634     d_list_t        *link;
635     int         rc;
636
637     D_ALLOC_PTR(entry);
638 TF     if (entry == NULL) {
639         D_ERROR("Failed to allocate hash table entry\n");
640         return -DER_NOMEM;
641     }
642
643     D_INIT_LIST_HEAD(&entry->entry);
644     entry->ace = ace;
645
646     link = d_hash_rec_find(found_aces, ace, daos_ace_get_size(ace));
647 TF     if (link != NULL) { /* Duplicate */
648         hash_ace_dec_ref(found_aces, link);
649         D_FREE(entry);
650         return -DER_INVAL;
651     }
652
653     rc = d_hash_rec_insert(found_aces, ace,
654             daos_ace_get_size(ace),
655             &entry->entry, true);
656 TF     if (rc != 0) {
657         D_ERROR("Failed to insert new hash entry, rc=%d\n", rc);
658         D_FREE(entry);
659     }
660
661     return rc;
662 }

664 /*
665  * Walks the list of ACEs and checks them for validity.
666  */
667 static int
668 validate_aces(struct daos_acl *acl)
669 {
670     struct daos_ace     *current;
671     int         last_type;
672     int         rc;
673     struct d_hash_table found;
674     d_hash_table_ops_t  ops = {
675             .hop_key_cmp = hash_ace_key_cmp,
676             .hop_rec_addref = hash_ace_add_ref,
677             .hop_rec_decref = hash_ace_dec_ref,
678             .hop_rec_free = hash_ace_free
679     };
680
681     last_type = -1;
682     rc = d_hash_table_create_inplace(D_HASH_FT_NOLOCK,
683             8, NULL, &ops, &found);
684 TF     if (rc != 0) {
685         D_ERROR("Failed to create hash table, rc=%d\n", rc);
686         return rc;
687     }
688
689     current = daos_acl_get_next_ace(acl, NULL);
690 TF     while (current != NULL) {
691 TF         if (!daos_ace_is_valid(current)) {
692             rc = -DER_INVAL;
693             goto out;
694         }
695
696         /* Type order is in order of the enum */
697 TF         if (current->dae_principal_type < last_type) {
698             rc = -DER_INVAL;
699             goto out;
700         }
701
702         rc = check_ace_is_duplicate(current, &found);
703 TF         if (rc != 0) {
704             goto out;
705         }
706
707         last_type = current->dae_principal_type;
708         current = daos_acl_get_next_ace(acl, current);
709     }
710
711 out:
712     d_hash_table_destroy_inplace(&found, true);
713     return rc;
714 }
