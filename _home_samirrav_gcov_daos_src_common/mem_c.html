109 static void
110 pmem_process_cb_vec(struct umem_tx_stage_item *vec, unsigned int *cnt,
111             bool noop)
112 {
113     struct umem_tx_stage_item   *txi;
114     int              i;
115
116 TF     for (i = 0; i < *cnt; i++) {
117         txi = &vec[i];
118
119         D_ASSERT(txi->txi_magic == UMEM_TX_DATA_MAGIC);
120         D_ASSERT(txi->txi_fn != NULL);
121
122         /* When 'noop' is true, callback will only free txi_data */
123         txi->txi_fn(txi->txi_data, noop);
124     }
125
126 TF     if (*cnt != 0) {

128         *cnt = 0;

132 /*
133  * This callback will be called on the outermost transaction commit (stage
134  * == TX_STAGE_ONCOMMIT), abort (stage == TX_STAGE_ONABORT) and end (stage
135  * == TX_STAGE_NONE).
136  */
137 static void
138 pmem_stage_callback(PMEMobjpool *pop, int stage, void *data)
139 {
140     struct umem_tx_stage_data   *txd = data;
141     struct umem_tx_stage_item   *vec;
142     unsigned int            *cnt;
143
144     D_ASSERT(stage >= TX_STAGE_NONE && stage < MAX_TX_STAGE);
145     D_ASSERT(txd != NULL);
146     D_ASSERT(txd->txd_magic == UMEM_TX_DATA_MAGIC);
147
148     switch (stage) {
149 T     case TX_STAGE_ONCOMMIT:
150         vec = txd->txd_commit_vec;
151         cnt = &txd->txd_commit_cnt;
152         /* Abandon the abort callbacks */
153         pmem_process_cb_vec(txd->txd_abort_vec, &txd->txd_abort_cnt,
154                     true);
155         break;
156 T     case TX_STAGE_ONABORT:
157         vec = txd->txd_abort_vec;
158         cnt = &txd->txd_abort_cnt;
159         /* Abandon the commit callbacks */
160         pmem_process_cb_vec(txd->txd_commit_vec, &txd->txd_commit_cnt,
161                     true);
162         break;
163 T     case TX_STAGE_NONE:
164         D_ASSERT(txd->txd_commit_cnt == 0);
165         D_ASSERT(txd->txd_abort_cnt == 0);
166         vec = txd->txd_end_vec;
167         cnt = &txd->txd_end_cnt;
168         break;
169 T     default:
170         /* Ignore all other stages */
171         return;
172     }
173
174     pmem_process_cb_vec(vec, cnt, false);
175 }

233 static int
234 pmem_tx_add_callback(struct umem_instance *umm, struct umem_tx_stage_data *txd,
235              int stage, umem_tx_cb_t cb, void *data)
236 {
237     struct umem_tx_stage_item   *txi, **pvec;
238     unsigned int            *cnt, *cnt_max;
239
240     D_ASSERT(txd != NULL);
241     D_ASSERT(txd->txd_magic == UMEM_TX_DATA_MAGIC);
242     D_ASSERT(pmemobj_tx_stage() == TX_STAGE_WORK);
243
244 TF     if (cb == NULL)
245         return -DER_INVAL;
246
247     switch (stage) {
248 T     case TX_STAGE_ONCOMMIT:
249         pvec = &txd->txd_commit_vec;
250         cnt = &txd->txd_commit_cnt;
251         cnt_max = &txd->txd_commit_max;
252         break;
253 T     case TX_STAGE_ONABORT:
254         pvec = &txd->txd_abort_vec;
255         cnt = &txd->txd_abort_cnt;
256         cnt_max = &txd->txd_abort_max;
257         break;
258 T     case TX_STAGE_NONE:
259         pvec = &txd->txd_end_vec;
260         cnt = &txd->txd_end_cnt;
261         cnt_max = &txd->txd_end_max;
262         break;
263 T     default:
264         D_ERROR("Invalid stage %d\n", stage);
265         return -DER_INVAL;
266     }
267
268     D_ASSERT(*cnt <= TXD_CB_MAX);
269 TF     if (*cnt == *cnt_max) {
270         unsigned int new_max;
271
272 TF         if (*cnt_max == TXD_CB_MAX) {
273             D_ERROR("Too many transaction callbacks "
274                 "cnt:%u, stage:%d\n", *cnt, stage);
275             return -DER_OVERFLOW;
276         }
277
278         new_max = min((*cnt_max) << 1, TXD_CB_MAX);
279         D_REALLOC(txi, *pvec,
280               new_max * sizeof(struct umem_tx_stage_item));
281
282 TF         if (txi == NULL)
283             return -DER_NOMEM;
284
285         *pvec = txi;
286         *cnt_max = new_max;
287     }
288
289     txi = &(*pvec)[*cnt];
290     (*cnt)++;
291     txi->txi_magic = UMEM_TX_DATA_MAGIC;
292     txi->txi_fn = cb;
293     txi->txi_data = data;
294
295     return 0;
296 }

316 umem_tx_errno(int err)
317 {
318 TF     if (err == 0) {

320 TF         if (err == ENOMEM) /* pmdk returns ENOMEM for out of space */
