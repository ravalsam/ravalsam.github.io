445 /**
446  * Parse pool buffer and construct domain+target array (tree) based on
447  * the information in pool buffer.
448  *
449  * \param buf       [IN]    pool buffer to be parsed
450  * \param tree_pp   [OUT]   the returned domain+target tree.
451  */
452 static int
453 pool_buf_parse(struct pool_buf *buf, struct pool_domain **tree_pp)
454 {
455     struct pool_domain *tree;
456     struct pool_domain *domain;
457     struct pool_domain *parent;
458     struct pool_target *targets;
459     pool_comp_type_t    type;
460     int         size;
461     int         i;
462     int         rc = 0;
463
464 TF     if (buf->pb_target_nr == 0 || buf->pb_node_nr == 0 ||
465         buf->pb_domain_nr + buf->pb_node_nr + buf->pb_target_nr !=
466                                 buf->pb_nr) {
467         D_DEBUG(DB_MGMT, "Invalid number of components: %d/%d/%d/%d\n",
468             buf->pb_nr, buf->pb_domain_nr, buf->pb_node_nr,
469             buf->pb_target_nr);
470         return -DER_INVAL;
471     }
472
473     size = sizeof(struct pool_domain) * (buf->pb_domain_nr + 1) + /* root */
474            sizeof(struct pool_domain) * (buf->pb_node_nr) +
475            sizeof(struct pool_target) * (buf->pb_target_nr);
476
477     D_DEBUG(DB_MGMT, "domain %d node %d target %d\n", buf->pb_domain_nr,
478         buf->pb_node_nr, buf->pb_target_nr);
479
480     D_ALLOC(tree, size);
481 TF     if (tree == NULL)
482         return -DER_NOMEM;
483
484     targets = (struct pool_target *)&tree[buf->pb_domain_nr +
485                           buf->pb_node_nr + 1];
486 TF     for (i = 0; i < buf->pb_target_nr; i++)
487         targets[i].ta_comp = buf->pb_comps[buf->pb_domain_nr +
488                         buf->pb_node_nr + i];
489
490     /* Initialize the root */
491     parent = &tree[0]; /* root */
492     parent->do_comp.co_type   = PO_COMP_TP_ROOT;
493     parent->do_comp.co_status = PO_COMP_ST_UP;
494 TF     if (buf->pb_domain_nr == 0) {
495         /* nodes are directly attached under the root */
496         parent->do_target_nr = buf->pb_node_nr;
497         parent->do_child_nr = buf->pb_node_nr;
498     } else {
499         parent->do_child_nr = buf->pb_domain_nr;
500     }
501     parent->do_children = &tree[1];
502
503     parent++;
504     type = buf->pb_comps[0].co_type;
505
506     for (i = 1;; i++) {
507         struct pool_component *comp = &tree[i].do_comp;
508         int            nr = 0;
509
510         *comp = buf->pb_comps[i - 1];
511 TF         if (comp->co_type > PO_COMP_TP_TARGET) {
512             D_DEBUG(DB_MGMT, "Invalid type %d/%d\n",
513                 type, comp->co_type);
514             rc = -DER_INVAL;
515             goto out;
516         }
517
518         D_DEBUG(DB_MGMT, "Parse %s[%d] i %d nr %d\n",
519             pool_comp_type2str(comp->co_type), comp->co_id,
520             i, comp->co_nr);
521
522 TF         if (comp->co_type == type)
523             continue;
524
525         type = comp->co_type;
526
527 TF         for (; parent < &tree[i]; parent++) {
528 TF             if (type != PO_COMP_TP_TARGET) {
529                 D_DEBUG(DB_MGMT, "Setup children for %s[%d]"
530                     " child nr %d\n",
531                     pool_domain_name(parent),
532                     parent->do_comp.co_id,
533                     parent->do_child_nr);
534
535                 parent->do_children = &tree[i + nr];
536                 nr += parent->do_child_nr;
537             } else {
538                 /* parent is the last level domain */
539                 D_DEBUG(DB_MGMT, "Setup targets for %s[%d]\n",
540                     pool_domain_name(parent),
541                     parent->do_comp.co_id);
542
543                 parent->do_target_nr  = parent->do_comp.co_nr;
544                 parent->do_comp.co_nr = 0;
545                 parent->do_targets    = targets;
546                 targets += parent->do_target_nr;
547
548                 D_DEBUG(DB_MGMT, "%s[%d] has %d targets\n",
549                     pool_domain_name(parent),
550                     parent->do_comp.co_id,
551                     parent->do_target_nr);
552             }
553         }
554
555 TF         if (type == PO_COMP_TP_TARGET)
556             break;
557     }
558
559     D_DEBUG(DB_MGMT, "Build children and targets pointers\n");
560
561 TF     for (domain = &tree[0]; domain->do_targets == NULL;
562          domain = &tree[0]) {
563 TF         while (domain->do_targets == NULL) {
564             parent = domain;
565             D_ASSERTF(domain->do_children != NULL,
566                   "%s[%d]: %d/%d\n",
567                   pool_domain_name(domain),
568                   domain->do_comp.co_id,
569                   domain->do_child_nr, domain->do_target_nr);
570             domain = &domain->do_children[0];
571         }
572
573         type = parent->do_comp.co_type;
574 TF         for (; parent->do_comp.co_type == type; parent++) {
575             parent->do_targets = domain->do_targets;
576 TF             for (i = 0; i < parent->do_child_nr; i++, domain++)
577                 parent->do_target_nr += domain->do_target_nr;
578
579             D_DEBUG(DB_MGMT, "Set %d target for %s[%d]\n",
580                 parent->do_target_nr,
581                 pool_comp_type2str(parent->do_comp.co_type),
582                 parent->do_comp.co_id);
583         }
584     }
585
586 out:
587 TF     if (rc)
588         D_FREE(tree);
589
590     *tree_pp = &tree[0];
591     return rc;
592 }

594 /**
595  * Extract pool buffer from a pool map.
596  *
597  * \param map       [IN]    The pool map to extract from.
598  * \param buf_pp    [OUT]   The returned pool buffer, should be freed
599  *              by pool_buf_free.
600  */
601 int
602 pool_buf_extract(struct pool_map *map, struct pool_buf **buf_pp)
603 {
604     struct pool_buf     *buf;
605     struct pool_domain  *tree;
606     struct pool_comp_cntr    cntr;
607     unsigned int         dom_nr;
608     int          i;
609     int          rc;
610
611     D_ASSERT(map->po_tree != NULL);
612     tree = &map->po_tree[1]; /* skip the root */
613     pool_tree_count(tree, &cntr);
614
615 TF     if (cntr.cc_domains + cntr.cc_targets == 0) {
616         D_DEBUG(DB_MGMT, "Empty pool map.\n");
617         return -DER_NONEXIST;
618     }
619
620     buf = pool_buf_alloc(cntr.cc_domains + cntr.cc_targets);
621 TF     if (buf == NULL)
622         return -DER_NOMEM;
623
624 TF     for (dom_nr = cntr.cc_top_doms; dom_nr != 0;
625          tree = tree[0].do_children) {
626         int     child_nr;
627
628 TF         for (i = child_nr = 0; i < dom_nr; i++) {
629             struct pool_component   comp;
630
631             comp = tree[i].do_comp;
632 TF             if (tree[i].do_children != NULL) {
633                 /* intermediate domain */
634                 child_nr += tree[i].do_child_nr;
635             } else {
636                 /* the last level domain */
637                 comp.co_nr = tree[i].do_target_nr;
638             }
639             pool_buf_attach(buf, &comp, 1);
640         }
641         dom_nr = child_nr;
642     }
643
644     tree = &map->po_tree[0];
645 TF     for (i = 0; i < cntr.cc_targets; i++)
646         pool_buf_attach(buf, &tree->do_targets[i].ta_comp, 1);
647
648 TF     if (buf->pb_nr != buf->pb_target_nr + buf->pb_domain_nr +
649               buf->pb_node_nr) {
650         D_DEBUG(DB_MGMT, "Invalid pool map format.\n");
651         D_GOTO(failed, rc = -DER_INVAL);
652     }
653
654     *buf_pp = buf;
655     return 0;
656  failed:
657     pool_buf_free(buf);
658     return rc;
659 }

661 /**
662  * Count number of domains, targets, and layers of domains etc in the
663  * component tree.
664  */
665 static void
666 pool_tree_count(struct pool_domain *tree, struct pool_comp_cntr *cntr)
667 {
668     unsigned int    dom_nr;
669
670     memset(cntr, 0, sizeof(*cntr));
671 TF     if (tree[0].do_children != NULL) {
672         dom_nr = tree[0].do_children - tree;
673     } else {
674         D_ASSERT(tree[0].do_targets != NULL);
675         dom_nr = (struct pool_domain *)tree[0].do_targets - tree;
676     }
677
678     cntr->cc_top_doms = dom_nr;
679     cntr->cc_domains  = dom_nr;
680
681 TF     for (; tree != NULL; tree = tree[0].do_children, cntr->cc_layers++) {
682         int      child_nr;
683         int      i;
684
685         D_DEBUG(DB_MGMT, "%s, nr = %d\n", pool_domain_name(&tree[0]),
686             dom_nr);
687 TF         for (i = child_nr = 0; i < dom_nr; i++) {
688 TF             if (tree[i].do_children != NULL) {
689                 cntr->cc_domains += tree[i].do_child_nr;
690                 child_nr += tree[i].do_child_nr;
691             } else {
692                 cntr->cc_targets += tree[i].do_target_nr;
693             }
694         }
695         dom_nr = child_nr;
696     }
697 }

754 /** Check if component buffer is sane */
755 static bool
756 pool_tree_sane(struct pool_domain *tree, uint32_t version)
757 {
758     struct pool_domain  *parent = NULL;
759     struct pool_target  *targets = tree[0].do_targets;
760     struct pool_comp_cntr    cntr;
761     int          dom_nr;
762     int          i;
763
764     D_DEBUG(DB_MGMT, "Sanity check of component buffer\n");
765     pool_tree_count(tree, &cntr);
766 TF     if (cntr.cc_targets == 0) {
767         D_DEBUG(DB_MGMT, "Buffer has no target\n");
768         return false;
769     }
770
771 TF     for (dom_nr = cntr.cc_top_doms; tree != NULL;
772          tree = tree[0].do_children) {
773         struct pool_domain *prev = &tree[0];
774         int         child_nr = 0;
775
776 TF         if (parent != NULL &&
777             parent->do_comp.co_type >= tree[0].do_comp.co_type) {
778             D_DEBUG(DB_MGMT,
779                 "Type of parent domain %d(%s) should be "
780                 "smaller than child domain %d(%s)\n",
781                 parent->do_comp.co_type,
782                 pool_domain_name(parent),
783                 tree[0].do_comp.co_type,
784                 pool_domain_name(&tree[0]));
785             return false;
786         }
787
788 TF         for (i = 0; i < dom_nr; i++) {
789 TF             if (tree[i].do_comp.co_ver > version) {
790                 D_DEBUG(DB_MGMT, "Invalid version %u/%u\n",
791                     tree[i].do_comp.co_ver, version);
792                 return false;
793             }
794
795 TF             if (prev->do_comp.co_type != tree[i].do_comp.co_type) {
796                 D_DEBUG(DB_MGMT,
797                     "Unmatched domain type %d/%d\n",
798                     tree[i].do_comp.co_type,
799                     prev->do_comp.co_type);
800                 return false;
801             }
802
803 TF             if (tree[i].do_targets == NULL ||
804                 tree[i].do_target_nr == 0) {
805                 D_DEBUG(DB_MGMT, "No target found\n");
806                 return false; /* always has targets */
807             }
808
809 TF             if ((prev->do_children == NULL) ^
810                 (tree[i].do_children == NULL)) {
811                 D_DEBUG(DB_MGMT, "Invalid child tree\n");
812                 return false;
813             }
814
815 TF             if ((prev->do_targets == NULL) ^
816                 (tree[i].do_targets == NULL)) {
817                 D_DEBUG(DB_MGMT, "Invalid target tree\n");
818                 return false;
819             }
820
821 TF             if (prev != &tree[i] &&
822                 prev->do_children != NULL &&
823                 prev->do_children + prev->do_child_nr !=
824                 tree[i].do_children) {
825                 D_DEBUG(DB_MGMT, "Invalid children pointer\n");
826                 return false;
827             }
828
829 TF             if (prev != &tree[i] &&
830                 prev->do_targets != NULL &&
831                 prev->do_targets + prev->do_target_nr !=
832                 tree[i].do_targets) {
833                 D_DEBUG(DB_MGMT, "Invalid children pointer i"
834                     " %d target nr %d\n", i,
835                     prev->do_target_nr);
836                 return false;
837             }
838
839 TF             if (tree[i].do_child_nr != 0)
840                 child_nr += tree[i].do_child_nr;
841
842             prev = &tree[i];
843         }
844         parent = &tree[0];
845         dom_nr = child_nr;
846     }
847
848 TF     for (i = 0; i < cntr.cc_targets; i++) {
849 TF         if (targets[i].ta_comp.co_type != PO_COMP_TP_TARGET) {
850             D_DEBUG(DB_MGMT, "Invalid leaf type %d(%s) i %d\n",
851                 targets[i].ta_comp.co_type,
852                 pool_comp_name(&targets[i].ta_comp), i);
853             return false;
854         }
855
856 TF         if (targets[i].ta_comp.co_ver > version) {
857             D_DEBUG(DB_MGMT, "Invalid version %u/%u i %d\n",
858                 targets[i].ta_comp.co_ver, version, i);
859             return false;
860         }
861     }
862     D_DEBUG(DB_MGMT, "Component buffer is sane\n");
863     return true;
864 }

905 /**
906  * Install a component tree to a pool map.
907  *
908  * \param map       [IN]    The pool map to be initialised.

912 static int
913 pool_map_initialise(struct pool_map *map, bool activate,
