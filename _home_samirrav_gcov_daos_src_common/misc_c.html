456 TF     if (idx == size)
457         return -1;
458
459 TF     for (off = 0; off < 32; off++)
460 TF         if (isclr(bits, off))
461             break;
462
463     return idx * 32 + off;
464 }

568 #define CRT_SOCKET_PROV     "ofi+sockets"
569 /**
570  * a helper to get the needed crt_init_opt.
571  * When using SEP (scalable endpoint), if user set un-reasonable CRT_CTX_NUM it
572  * possibly cause failure when creating cart context. So in this helper it will
573  * check the ENV setting, if SEP is used then will set the crt_init_opttions_t
574  * and the caller can pass it to crt_init_opt().
575  *
576  * \param server [IN]   true for server
577  * \param ctx_nr [IN]   number of contexts
578  *
579  * \return      the pointer to crt_init_options_t (NULL if not needed)
580  */
581 crt_init_options_t daos_crt_init_opt;
582 crt_init_options_t *
583 daos_crt_init_opt_get(bool server, int ctx_nr)
584 {
585     crt_phy_addr_t  addr_env;
586     bool        sep = false;
587
588     d_getenv_bool("CRT_CTX_SHARE_ADDR", &sep);
589 TF     if (!sep)
590         return NULL;
591
592     daos_crt_init_opt.cio_crt_timeout = 0;
593     daos_crt_init_opt.cio_sep_override = 1;
594
595     /* for socket provider, force it to use regular EP rather than SEP for:
596      * 1) now sockets provider cannot create more than 16 contexts for SEP
597      * 2) some problems if SEP communicates with regular EP.
598      */
599     addr_env = (crt_phy_addr_t)getenv(CRT_PHY_ADDR_ENV);
600 TF     if (addr_env != NULL &&
601         strncmp(addr_env, CRT_SOCKET_PROV, strlen(CRT_SOCKET_PROV)) == 0) {
602         D_INFO("for sockets provider force it to use regular EP.\n");
603         daos_crt_init_opt.cio_use_sep = 0;
604         return &daos_crt_init_opt;
605     }
606
607     /* for psm2 provider, set a reasonable cio_ctx_max_num for cart */
608     daos_crt_init_opt.cio_use_sep = 1;
609 TF     if (!server) {
610         /* to workaround a bug in mercury/ofi, that the basic EP cannot
611          * communicate with SEP. Setting 2 for client to make it to use
612          * SEP for client.
613          */
614         daos_crt_init_opt.cio_ctx_max_num = 2;
615     } else {
616         daos_crt_init_opt.cio_ctx_max_num = ctx_nr;
617     }
618
619     return &daos_crt_init_opt;
620 }

684 /**
685  * Check if the input daos_prop_t parameter is valid
686  * \a pool true for pool properties, false for container properties.
687  * \a input true for input properties that should with reasonable value,
688  *          false for output that need not check the value.
689  */
690 bool
691 daos_prop_valid(daos_prop_t *prop, bool pool, bool input)
692 {
693     uint32_t    type;
694     uint64_t    val;
695     struct daos_acl *acl_ptr;
696     int     i;
697
698 TF     if (prop == NULL) {
699         D_ERROR("NULL properties\n");
700         return false;
701     }
702 TF     if (prop->dpp_nr > DAOS_PROP_ENTRIES_MAX_NR) {
703         D_ERROR("invalid ddp_nr %d (> %d).\n",
704             prop->dpp_nr, DAOS_PROP_ENTRIES_MAX_NR);
705         return false;
706     }
707 TF     if (prop->dpp_nr == 0) {
708 TF         if (prop->dpp_entries != NULL)
709             D_ERROR("invalid properties, NON-NULL dpp_entries with "
710                 "zero dpp_nr.\n");
711         return prop->dpp_entries == NULL;
712     }
713 TF     if (prop->dpp_entries == NULL) {
714         D_ERROR("invalid properties, NULL dpp_entries with non-zero "
715             "dpp_nr.\n");
716         return false;
717     }
718 TF     for (i = 0; i < prop->dpp_nr; i++) {
719         type = prop->dpp_entries[i].dpe_type;
720 TF         if (pool) {
721 TF             if (type <= DAOS_PROP_PO_MIN ||
722                 type >= DAOS_PROP_PO_MAX) {
723                 D_ERROR("invalid type %d for pool.\n", type);
724                 return false;
725             }
726         } else {
727 TF             if (type <= DAOS_PROP_CO_MIN ||
728                 type >= DAOS_PROP_CO_MAX) {
729                 D_ERROR("invalid type %d for container.\n",
730                     type);
731                 return false;
732             }
733         }
734         /* for output parameter need not check entry value */
735 TF         if (!input)
736             continue;
737         switch (type) {
738         /* pool properties */
739 T         case DAOS_PROP_PO_LABEL:
740 TF             if (!daos_prop_label_valid(
741                 prop->dpp_entries[i].dpe_str))
742                 return false;
743             break;
744 T         case DAOS_PROP_PO_ACL:
745             acl_ptr = prop->dpp_entries[i].dpe_val_ptr;
746 TF             if (daos_acl_validate(acl_ptr) != 0)
747                 return false;
748             break;
749 T         case DAOS_PROP_CO_ACL:
750             /* TODO: Implement container ACL */
751             break;
752 T         case DAOS_PROP_PO_SPACE_RB:
753             val = prop->dpp_entries[i].dpe_val;
754 TF             if (val > 100) {
755                 D_ERROR("invalid space_rb "DF_U64".\n", val);
756                 return false;
757             }
758             break;
759 T         case DAOS_PROP_PO_SELF_HEAL:
760             break;
761 T         case DAOS_PROP_PO_RECLAIM:
762             val = prop->dpp_entries[i].dpe_val;
763 TF             if (val != DAOS_RECLAIM_SNAPSHOT &&
764                 val != DAOS_RECLAIM_BATCH &&
765                 val != DAOS_RECLAIM_TIME) {
766                 D_ERROR("invalid reclaim "DF_U64".\n", val);
767                 return false;
768             }
769             break;
770         /* container properties */
771 T         case DAOS_PROP_CO_LABEL:
772 TF             if (!daos_prop_label_valid(
773                 prop->dpp_entries[i].dpe_str))
774                 return false;
775             break;
776 T         case DAOS_PROP_CO_LAYOUT_TYPE:
777             val = prop->dpp_entries[i].dpe_val;
778 TF             if (val != DAOS_PROP_CO_LAYOUT_UNKOWN &&
779                 val != DAOS_PROP_CO_LAYOUT_POSIX &&
780                 val != DAOS_PROP_CO_LAYOUT_HDF5) {
781                 D_ERROR("invalid layout type "DF_U64".\n", val);
782                 return false;
783             }
784             break;
785 T         case DAOS_PROP_CO_LAYOUT_VER:
786             break;
787 T         case DAOS_PROP_CO_CSUM:
788             val = prop->dpp_entries[i].dpe_val;
789 TF             if (val != DAOS_PROP_CO_CSUM_OFF &&
790                 val != DAOS_PROP_CO_CSUM_CRC16 &&
791                 val != DAOS_PROP_CO_CSUM_CRC32 &&
792                 val != DAOS_PROP_CO_CSUM_SHA1 &&
793                 val != DAOS_PROP_CO_CSUM_SHA2) {
794                 D_ERROR("invalid checksum type "DF_U64".\n",
795                     val);
796                 return false;
797             }
798             break;
799 T         case DAOS_PROP_CO_REDUN_FAC:
800             val = prop->dpp_entries[i].dpe_val;
801 TF             if (val != DAOS_PROP_CO_REDUN_RF1 &&
802                 val != DAOS_PROP_CO_REDUN_RF3) {
803                 D_ERROR("invalid redundancy factor "DF_U64".\n",
804                     val);
805                 return false;
806             }
807             break;
808 T         case DAOS_PROP_CO_REDUN_LVL:
809             val = prop->dpp_entries[i].dpe_val;
810 TF             if (val != DAOS_PROP_CO_REDUN_RACK &&
811                 val != DAOS_PROP_CO_REDUN_NODE) {
812                 D_ERROR("invalid redundancy level "DF_U64".\n",
813                     val);
814                 return false;
815             }
816             break;
817 T         case DAOS_PROP_CO_SNAPSHOT_MAX:
818 T         case DAOS_PROP_CO_COMPRESS:
819 T         case DAOS_PROP_CO_ENCRYPT:
820             break;
821 T         default:
822             D_ERROR("invaid dpe_type %d.\n", type);
823             return false;
824         }
825     }
826     return true;
827 }

829 /**
830  * duplicate the properties
831  * \a pool true for pool properties, false for container properties.
832  */
833 daos_prop_t *
834 daos_prop_dup(daos_prop_t *prop, bool pool)
835 {
836     daos_prop_t     *prop_dup;
837     struct daos_prop_entry  *entry, *entry_dup;
838     int          i;
839     struct daos_acl     *acl_ptr;
840
841 TF     if (!daos_prop_valid(prop, pool, true))
842         return NULL;
843
844     prop_dup = daos_prop_alloc(prop->dpp_nr);
845 TF     if (prop_dup == NULL)
846         return NULL;
847
848 TF     for (i = 0; i < prop->dpp_nr; i++) {
849         entry = &prop->dpp_entries[i];
850         entry_dup = &prop_dup->dpp_entries[i];
851         entry_dup->dpe_type = entry->dpe_type;
852         switch (entry->dpe_type) {
853 T         case DAOS_PROP_PO_LABEL:
854 T         case DAOS_PROP_CO_LABEL:
855             entry_dup->dpe_str = strndup(entry->dpe_str,
856                              DAOS_PROP_LABEL_MAX_LEN);
857 TF             if (entry_dup->dpe_str == NULL) {
858                 D_ERROR("failed to dup label.\n");
859                 daos_prop_free(prop_dup);
860                 return NULL;
861             }
862             break;
863 T         case DAOS_PROP_PO_ACL:
864             acl_ptr = entry->dpe_val_ptr;
865             entry_dup->dpe_val_ptr = daos_acl_dup(acl_ptr);
866 TF             if (entry_dup->dpe_val_ptr == NULL) {
867                 D_ERROR("failed to dup ACL\n");
868                 daos_prop_free(prop_dup);
869                 return NULL;
870             }
871             break;
872 T         case DAOS_PROP_CO_ACL:
873             /* TODO: Implement container ACL */
874             break;
875 T         default:
876             entry_dup->dpe_val = entry->dpe_val;
877             break;
878         }
879     }
880
881     return prop_dup;
882 }

902 /**
903  * Copy properties from \a prop_reply to \a prop_req.
904  * Used to copy the properties from pool query or container query to user's
905  * properties. If user provided \a prop_req with zero dpp_nr (and NULL
906  * dpp_entries), it will allocate needed buffer and assign to user's daos_prop_t
907  * struct, the needed buffer to store label will be allocated as well.
908  * User can free properties buffer by calling daos_prop_free().
909  */
910 int
911 daos_prop_copy(daos_prop_t *prop_req, daos_prop_t *prop_reply)
912 {
913     struct daos_prop_entry  *entry_req, *entry_reply;
914     struct daos_prop_entry  *entries_alloc = NULL;
915     d_string_t       label_alloc = NULL;
916     void            *acl_alloc = NULL;
917     struct daos_acl     *acl;
918     uint32_t         type;
919     int          i;
920     int          rc = 0;
921
922 TF     if (prop_reply == NULL || prop_reply->dpp_nr == 0 ||
923         prop_reply->dpp_entries == NULL) {
924         D_ERROR("no prop or empty prop in reply.\n");
925         return -DER_PROTO;
926     }
927 TF     if (prop_req->dpp_nr == 0) {
928         prop_req->dpp_nr = prop_reply->dpp_nr;
929         D_ALLOC_ARRAY(prop_req->dpp_entries, prop_req->dpp_nr);
930 TF         if (prop_req->dpp_entries == NULL)
931             return -DER_NOMEM;
932         entries_alloc = prop_req->dpp_entries;
933     }
934
935 TF     for (i = 0; i < prop_req->dpp_nr; i++) {
936         entry_req = &prop_req->dpp_entries[i];
937         type = entry_req->dpe_type;
938 TF         if (type == 0) {
939             /* this is the case that dpp_entries allocated above */
940             D_ASSERT(prop_req->dpp_nr == prop_reply->dpp_nr);
941             type = prop_reply->dpp_entries[i].dpe_type;
942             entry_req->dpe_type = type;
943         }
