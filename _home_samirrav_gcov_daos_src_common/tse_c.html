407     d_list_for_each_entry_safe(dtc, tmp, &dtp->dtp_prep_cb_list, dtc_list) {
408         d_list_del(&dtc->dtc_list);
409         /** no need to call if task was completed in one of the cbs */
410 TF         if (!dtp->dtp_completed) {
411             rc = dtc->dtc_cb(task, dtc->dtc_arg);
412 TF             if (task->dt_result == 0)
413                 task->dt_result = rc;
414         }
415
416         D_FREE(dtc);
417
418         /** Task was re-initialized; break */
419 TF         if (!dtp->dtp_running && !dtp->dtp_completing)
420             return false;
421     }
422
423     return true;
424 }
425
426 /*
427  * Execute the callback of the task and returns true if all CBs were executed
428  * and non re-init the task. If the task is re-initialized by the user, it means
429  * it's in-flight again, so we break at the current CB that re-initialized it,
430  * and return false, meaning the task is not completed. All the remaining CBs
431  * that haven't been executed remain attached, but the ones that have executed
432  * already have been removed from the list at this point.
433  */
434 static bool
435 tse_task_complete_callback(tse_task_t *task)
436 {
437     struct tse_task_private *dtp = tse_task2priv(task);
438     struct tse_task_cb  *dtc;
439     struct tse_task_cb  *tmp;
440
441     d_list_for_each_entry_safe(dtc, tmp, &dtp->dtp_comp_cb_list, dtc_list) {
442         int ret;
443
444         d_list_del(&dtc->dtc_list);
445         ret = dtc->dtc_cb(task, dtc->dtc_arg);
446 TF         if (task->dt_result == 0)
447             task->dt_result = ret;
448
449         D_FREE(dtc);
450
451         /** Task was re-initialized; break */
452 TF         if (!dtp->dtp_completing) {
453             D_DEBUG(DB_TRACE, "re-init task %p\n", task);
454             return false;
455         }
456     }
457
458     return true;
459 }
460
461 /*
462  * Process the task in the init list of the scheduler. This executes all the
463  * body function of all tasks with no dependencies in the scheduler's init
464  * list.
465  */
466 static int
467 tse_sched_process_init(struct tse_sched_private *dsp)
468 {
469     struct tse_task_private     *dtp;
470     struct tse_task_private     *tmp;
471     d_list_t            list;
472     int             processed = 0;
473
474     D_INIT_LIST_HEAD(&list);
475     D_MUTEX_LOCK(&dsp->dsp_lock);
476     d_list_for_each_entry_safe(dtp, tmp, &dsp->dsp_init_list,
477                       dtp_list) {
478 TF         if (dtp->dtp_dep_cnt == 0 || dsp->dsp_cancelling) {
479             d_list_move_tail(&dtp->dtp_list, &list);
480             dsp->dsp_inflight++;
481         }
482     }
483     D_MUTEX_UNLOCK(&dsp->dsp_lock);
484
485 TF     while (!d_list_empty(&list)) {
486         tse_task_t *task;
487         bool bumped = false;
488
489         dtp = d_list_entry(list.next, struct tse_task_private,
490                    dtp_list);
491
492         task = tse_priv2task(dtp);
493
494         D_MUTEX_LOCK(&dsp->dsp_lock);
495 TF         if (dsp->dsp_cancelling) {
496             tse_task_complete_locked(dtp, dsp);
497         } else {
498             dtp->dtp_running = 1;
499             d_list_move_tail(&dtp->dtp_list,
500                      &dsp->dsp_running_list);
501             /** +1 in case prep cb calls task_complete() */
502             tse_task_addref_locked(dtp);
503             bumped = true;
504         }
505         D_MUTEX_UNLOCK(&dsp->dsp_lock);
506
507 TF         if (!dsp->dsp_cancelling) {
508             /** if task is reinitialized in prep cb, skip over it */
509 TF             if (!tse_task_prep_callback(task)) {
510                 tse_task_decref(task);
511                 continue;
512             }
513             D_ASSERT(dtp->dtp_func != NULL);
514 TF             if (!dtp->dtp_completed)
515                 dtp->dtp_func(task);
516         }
517 TF         if (bumped)
518             tse_task_decref(task);
519
520         processed++;
521     }
522     return processed;
523 }

525 /**
526  * Check the task in the complete list, dependent task
527  * status check, schedule status update etc. The task
528  * will be moved to fini list after this
529  **/
530 static int
531 tse_task_post_process(tse_task_t *task)
532 {
533     struct tse_task_private  *dtp = tse_task2priv(task);
534     struct tse_sched_private *dsp = dtp->dtp_sched;
535     int rc = 0;
536
537     D_ASSERT(dtp->dtp_completed == 1);
538
539     /* set scheduler result */
540 TF     if (tse_priv2sched(dsp)->ds_result == 0)
541         tse_priv2sched(dsp)->ds_result = task->dt_result;
542
543     /* Check dependent list */
544     D_MUTEX_LOCK(&dsp->dsp_lock);
545 TF     while (!d_list_empty(&dtp->dtp_dep_list)) {
546         struct tse_task_link    *tlink;
547         tse_task_t      *task_tmp;
548         struct tse_task_private *dtp_tmp;
549
550         tlink = d_list_entry(dtp->dtp_dep_list.next,
551                      struct tse_task_link, tl_link);
552         d_list_del(&tlink->tl_link);
553         task_tmp = tlink->tl_task;
554         dtp_tmp = tse_task2priv(task_tmp);
555         D_FREE(tlink);
556
557         /* propagate dep task's failure */
558 TF         if (task_tmp->dt_result == 0)
559             task_tmp->dt_result = task->dt_result;
560
561         /* see if the dependent task is ready to be scheduled */
562         D_ASSERT(dtp_tmp->dtp_dep_cnt > 0);
563         dtp_tmp->dtp_dep_cnt--;
564         D_DEBUG(DB_TRACE, "daos task %p dep_cnt %d\n", dtp_tmp,
565             dtp_tmp->dtp_dep_cnt);
566 TF         if (!dsp->dsp_cancelling && dtp_tmp->dtp_dep_cnt == 0 &&
567             dtp_tmp->dtp_running) {
568             bool done;
569
570             /*
571              * If the task is already running, let's mark it
572              * complete. This happens when we create subtasks in the
573              * body function of the main task. So the task function
574              * is done, but it will stay in the running state until
575              * all the tasks that it depends on are completed, then
576              * it is completed when they completed in this code
577              * block.
578              */
579
580             dtp_tmp->dtp_completing = 1;
581             /** release lock for CB */
582             D_MUTEX_UNLOCK(&dsp->dsp_lock);
583             done = tse_task_complete_callback(task_tmp);
584             D_MUTEX_LOCK(&dsp->dsp_lock);
585
586             /*
587              * task reinserted itself in scheduler by
588              * calling tse_task_reinit().
589              */
590 TF             if (!done) {
591                 /* -1 for tlink (addref by add_dependent) */
592                 tse_task_decref_locked(dtp_tmp);
593                 continue;
594             }
595
596             tse_task_complete_locked(dtp_tmp, dsp);
597         }
598
599         /* -1 for tlink (addref by add_dependent) */
600         tse_task_decref_locked(dtp_tmp);
601     }
602
603     D_ASSERT(dsp->dsp_inflight > 0);
604     dsp->dsp_inflight--;
605     D_MUTEX_UNLOCK(&dsp->dsp_lock);
606
607 TF     if (task->dt_result == 0)
608         task->dt_result = rc;
609
610     return rc;
611 }
612
613 int
614 tse_sched_process_complete(struct tse_sched_private *dsp)
615 {
616     struct tse_task_private *dtp;
617     struct tse_task_private *tmp;
618     d_list_t comp_list;
619     int processed = 0;
620
621     /* pick tasks from complete_list */
622     D_INIT_LIST_HEAD(&comp_list);
623     D_MUTEX_LOCK(&dsp->dsp_lock);
624     d_list_splice_init(&dsp->dsp_complete_list, &comp_list);
625     D_MUTEX_UNLOCK(&dsp->dsp_lock);
626
627     d_list_for_each_entry_safe(dtp, tmp, &comp_list, dtp_list) {
628         tse_task_t *task = tse_priv2task(dtp);
629
630         tse_task_post_process(task);
631         d_list_del_init(&dtp->dtp_list);
632         tse_task_decref(task);  /* drop final ref */
633         processed++;
634     }
635     return processed;
636 }

638 bool
639 tse_sched_check_complete(tse_sched_t *sched)
640 {
641     struct tse_sched_private *dsp = tse_sched2priv(sched);
642     bool completed;
643
644     /* check if all tasks are done */
645     D_MUTEX_LOCK(&dsp->dsp_lock);
646     completed = (d_list_empty(&dsp->dsp_init_list) &&
647              dsp->dsp_inflight == 0);
648     D_MUTEX_UNLOCK(&dsp->dsp_lock);
649
650     return completed;
651 }
652
653 /* Run tasks for this schedule */
654 static void
655 tse_sched_run(tse_sched_t *sched)
656 {
657     struct tse_sched_private *dsp = tse_sched2priv(sched);
658
659     while (1) {
660         int processed = 0;
661         bool    completed;
662
663         processed += tse_sched_process_init(dsp);
664         processed += tse_sched_process_complete(dsp);
665         completed = tse_sched_check_complete(sched);
666 TF         if (completed || processed == 0)
667             break;
668     };
669
670     /* drop reference of tse_sched_init() */
671     tse_sched_decref(dsp);
672 }

674 /*
675  * Poke the scheduler to run tasks in the init list if ready, finish tasks that
676  * have completed.
677  */
678 void
679 tse_sched_progress(tse_sched_t *sched)
680 {
681     struct tse_sched_private *dsp = tse_sched2priv(sched);
682
683 TF     if (dsp->dsp_cancelling)
684         return;
685
686     D_MUTEX_LOCK(&dsp->dsp_lock);
687     /** +1 for tse_sched_run() */
688     tse_sched_addref_locked(dsp);
689     D_MUTEX_UNLOCK(&dsp->dsp_lock);
690
691 TF     if (!dsp->dsp_cancelling)
692         tse_sched_run(sched);
693     /** If another thread canceled, drop the ref count */
694     else
695         tse_sched_decref(dsp);
696 }

698 static int
699 tse_sched_complete_inflight(struct tse_sched_private *dsp)
700 {
701     struct tse_task_private *dtp;
702     struct tse_task_private *tmp;
703     int           processed = 0;
704
705     D_MUTEX_LOCK(&dsp->dsp_lock);
706     d_list_for_each_entry_safe(dtp, tmp, &dsp->dsp_running_list,
707                       dtp_list)
708 TF         if (dtp->dtp_dep_cnt == 0) {
709             d_list_del(&dtp->dtp_list);
710             tse_task_complete_locked(dtp, dsp);
711             processed++;
712         }
713     D_MUTEX_UNLOCK(&dsp->dsp_lock);
714
715     return processed;
716 }

718 void
719 tse_sched_complete(tse_sched_t *sched, int ret, bool cancel)
720 {
721     struct tse_sched_private *dsp = tse_sched2priv(sched);
722
723 TF     if (sched->ds_result == 0)
724         sched->ds_result = ret;
725
726     D_MUTEX_LOCK(&dsp->dsp_lock);
727 TF     if (dsp->dsp_cancelling || dsp->dsp_completing) {
728         D_MUTEX_UNLOCK(&dsp->dsp_lock);
729         return;
730     }
731
732 TF     if (cancel)
733         dsp->dsp_cancelling = 1;
734     else
735         dsp->dsp_completing = 1;
736
737     /** Wait for all in-flight tasks */
738     while (1) {
739         /** +1 for tse_sched_run */
740         tse_sched_addref_locked(dsp);
741         D_MUTEX_UNLOCK(&dsp->dsp_lock);
742
743         tse_sched_run(sched);
744 TF         if (dsp->dsp_inflight == 0)
745             break;
746 TF         if (dsp->dsp_cancelling)
747             tse_sched_complete_inflight(dsp);
748
749         D_MUTEX_LOCK(&dsp->dsp_lock);
750     }
751
752     tse_sched_complete_cb(sched);
753     sched->ds_udata = NULL;
754     tse_sched_decref(dsp);
755 }
756
757 void
758 tse_task_complete(tse_task_t *task, int ret)
759 {
760     struct tse_task_private     *dtp    = tse_task2priv(task);
761     struct tse_sched_private    *dsp    = dtp->dtp_sched;
762     bool                bumped  = false;
763     bool                done;
764
765 TF     if (dtp->dtp_completed)
766         return;
767
768 TF     if (task->dt_result == 0)
769         task->dt_result = ret;
770
771     dtp->dtp_completing = 1;
772     /** Execute task completion callbacks first. */
773     done = tse_task_complete_callback(task);
774
775     D_MUTEX_LOCK(&dsp->dsp_lock);
776
777 TF     if (!dsp->dsp_cancelling) {
778         /** +1 for tse_sched_run() */
779         tse_sched_addref_locked(dsp);
780         /** track in case another thread cancels */
781         bumped = true;
782
783         /** if task reinserted itself in scheduler, don't complete */
784 TF         if (done)
785             tse_task_complete_locked(dtp, dsp);
786     } else {
787         tse_task_decref_locked(dtp);
788     }
789     D_MUTEX_UNLOCK(&dsp->dsp_lock);
790
791     /** update task in scheduler lists. */
792 TF     if (!dsp->dsp_cancelling && done)
793         tse_sched_process_complete(dsp);
794     /** If another thread canceled, make sure we drop the ref count */
795 TF     else if (bumped)
796         tse_sched_decref(dsp);
797
798     /** -1 from tse_task_create() if it has not been reinitialized */
799 TF     if (done)
800         tse_sched_decref(dsp);
801 }

803 /**
804  * If one task dependents on other tasks, only if the dependent task
805  * is done, then the task can be added to the scheduler list
806  **/
807 static int
808 tse_task_add_dependent(tse_task_t *task, tse_task_t *dep)
809 {
810     struct tse_task_private  *dtp = tse_task2priv(task);
811     struct tse_task_private  *dep_dtp = tse_task2priv(dep);
812     struct tse_task_link      *tlink;
813
814 TF     if (dtp->dtp_sched != dep_dtp->dtp_sched) {
815         D_ERROR("Two tasks should belong to the same scheduler.\n");
816         return -DER_NO_PERM;
817     }
818
819 TF     if (dtp->dtp_completed) {
820         D_ERROR("Can't add a depedency for a completed task (%p)\n",
821             task);
822         return -DER_NO_PERM;
823     }
824
825     /** if task to depend on has completed already, do nothing */
826 TF     if (dep_dtp->dtp_completed)
827         return 0;
828
829     D_ALLOC_PTR(tlink);
830 TF     if (tlink == NULL)
831         return -DER_NOMEM;
832
833     D_DEBUG(DB_TRACE, "Add dependent %p ---> %p\n", dep_dtp, dtp);
834
835     D_MUTEX_LOCK(&dtp->dtp_sched->dsp_lock);
836
837     tse_task_addref_locked(dtp);
838     tlink->tl_task = task;
839
840     d_list_add_tail(&tlink->tl_link, &dep_dtp->dtp_dep_list);
841     dtp->dtp_dep_cnt++;
842
843     D_MUTEX_UNLOCK(&dtp->dtp_sched->dsp_lock);
844
845     return 0;
846 }

848 int
849 tse_task_register_deps(tse_task_t *task, int num_deps,
850                tse_task_t *dep_tasks[])
851 {
852     int i;
853     int rc;
854
855 TF     for (i = 0; i < num_deps; i++) {
856         rc = tse_task_add_dependent(task, dep_tasks[i]);
857 TF         if (rc)

863 int
864 tse_task_create(tse_task_func_t task_func, tse_sched_t *sched, void *priv,
865         tse_task_t **taskp)
866 {
867     struct tse_sched_private *dsp = tse_sched2priv(sched);
868     struct tse_task_private  *dtp;
869     tse_task_t       *task;
870
871     D_ALLOC_PTR(task);
872 TF     if (task == NULL)
873         return -DER_NOMEM;
874
875     dtp = tse_task2priv(task);
876     D_CASSERT(sizeof(task->dt_private) >= sizeof(*dtp));
877

879     D_INIT_LIST_HEAD(&dtp->dtp_task_list);

885     dtp->dtp_func     = task_func;
886     dtp->dtp_priv     = priv;

889     *taskp = task;
890     return 0;
891 }

893 int
894 tse_task_schedule(tse_task_t *task, bool instant)
895 {
896     struct tse_task_private  *dtp = tse_task2priv(task);
897     struct tse_sched_private *dsp = dtp->dtp_sched;
898     int rc = 0;
899
900     D_ASSERT(!instant || dtp->dtp_func);
901
902     /* Add task to scheduler */
903     D_MUTEX_LOCK(&dsp->dsp_lock);
904 TF     if (dtp->dtp_func == NULL || instant) {
905         /** If task has no body function, mark it as running */
906         dsp->dsp_inflight++;
907         dtp->dtp_running = 1;
908         d_list_add_tail(&dtp->dtp_list, &dsp->dsp_running_list);
909
910         /** +1 in case task is completed in body function */
911 TF         if (instant)
912             tse_task_addref_locked(dtp);
913     } else {
914         /** Otherwise, scheduler will process it from init list */
915         d_list_add_tail(&dtp->dtp_list, &dsp->dsp_init_list);
